001package com.quorum.tessera.config.migration;
002
003import com.moandjiezana.toml.Toml;
004import com.quorum.tessera.config.ArgonOptions;
005import com.quorum.tessera.config.SslAuthenticationMode;
006import com.quorum.tessera.config.builder.ConfigBuilder;
007import com.quorum.tessera.config.builder.JdbcConfigFactory;
008import com.quorum.tessera.config.builder.KeyDataBuilder;
009import com.quorum.tessera.config.builder.SslTrustModeFactory;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013import java.io.InputStream;
014import java.net.MalformedURLException;
015import java.net.URL;
016import java.util.List;
017import java.util.Objects;
018import java.util.Optional;
019
020import static java.util.Collections.emptyList;
021
022public class TomlConfigFactory {
023
024    private static final Logger LOGGER = LoggerFactory.getLogger(TomlConfigFactory.class);
025
026    public ConfigBuilder create(InputStream configData, ArgonOptions options, String... filenames) {
027        Objects.requireNonNull(configData, "No config data provided. ");
028        if (filenames.length != 0) {
029            throw new UnsupportedOperationException("keyConfigData arg is not implemented for TomlConfigFactory");
030        }
031
032        Toml toml = new Toml().read(configData);
033        toml.toMap().forEach((key, value) -> LOGGER.debug("Found entry in toml file : {} {}", key, value));
034
035        final String urlWithoutPort = Optional
036            .ofNullable(toml.getString("url"))
037            .map(url -> {
038                try {
039                    return new URL(url);
040                } catch (final MalformedURLException e) {
041                    throw new RuntimeException("Bad server url given: " + e.getMessage());
042                }
043            }).map(uri -> uri.getProtocol() + "://" + uri.getHost())
044            .orElse(null);
045
046        final Integer port = Optional.ofNullable(toml.getLong("port"))
047            .map(Long::intValue)
048            .orElse(null);
049
050        final String workdir = toml.getString("workdir", "");
051        final String socket = toml.getString("socket");
052
053        final String tls = toml.getString("tls", "off").toUpperCase();
054
055        final List<String> othernodes = toml.getList("othernodes", emptyList());
056
057        final List<String> alwaysSendToKeyPaths = toml.getList("alwayssendto", emptyList());
058
059        final String storage = toml.getString("storage", "memory");
060
061        final List<String> ipwhitelist = toml.getList("ipwhitelist", emptyList());
062        final boolean useWhiteList = !ipwhitelist.isEmpty();
063
064        //Server side
065        final String tlsservertrust = toml.getString("tlsservertrust", "tofu");
066        final Optional<String> tlsserverkey = Optional.ofNullable(toml.getString("tlsserverkey"));
067        final Optional<String> tlsservercert = Optional.ofNullable(toml.getString("tlsservercert"));
068        final Optional<List<String>> tlsserverchainnames = Optional.ofNullable(toml.getList("tlsserverchain", emptyList()));
069        final Optional<String> tlsknownclients = Optional.ofNullable(toml.getString("tlsknownclients"));
070
071        //Client side
072        final String tlsclienttrust = toml.getString("tlsclienttrust", "tofu");
073        final Optional<String> tlsclientkey = Optional.ofNullable(toml.getString("tlsclientkey"));
074        final Optional<String> tlsclientcert = Optional.ofNullable(toml.getString("tlsclientcert"));
075        final Optional<List<String>> tlsclientchainnames = Optional.ofNullable(toml.getList("tlsclientchain", emptyList()));
076        final Optional<String> tlsknownservers = Optional.ofNullable(toml.getString("tlsknownservers"));
077
078        ConfigBuilder configBuilder = ConfigBuilder.create()
079            .serverPort(port)
080            .serverHostname(urlWithoutPort)
081            .unixSocketFile(socket)
082            .sslAuthenticationMode(SslAuthenticationMode.valueOf(tls))
083            .sslServerTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsservertrust))
084            .sslClientTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsclienttrust))
085            .peers(othernodes)
086            .alwaysSendTo(alwaysSendToKeyPaths)
087            .useWhiteList(useWhiteList)
088            .workdir(workdir);
089
090        tlsserverkey.ifPresent(configBuilder::sslServerTlsKeyPath);
091        tlsservercert.ifPresent(configBuilder::sslServerTlsCertificatePath);
092        tlsserverchainnames.ifPresent(configBuilder::sslServerTrustCertificates);
093        tlsknownclients.ifPresent(configBuilder::sslKnownClientsFile);
094        tlsclientkey.ifPresent(configBuilder::sslClientTlsKeyPath);
095        tlsclientcert.ifPresent(configBuilder::sslClientTlsCertificatePath);
096        tlsclientchainnames.ifPresent(configBuilder::sslClientTrustCertificates);
097        tlsknownservers.ifPresent(configBuilder::sslKnownServersFile);
098
099        Optional.ofNullable(storage)
100            .map(JdbcConfigFactory::fromLegacyStorageString)
101            .ifPresent(configBuilder::jdbcConfig);
102
103        return configBuilder;
104    }
105
106    KeyDataBuilder createKeyDataBuilder(InputStream configData) {
107        final Toml toml = new Toml().read(configData);
108
109        final List<String> publicKeyList = toml.getList("publickeys", emptyList());
110
111        final List<String> privateKeyList = toml.getList("privatekeys", emptyList());
112
113        final String pwd = toml.getString("passwords");
114
115        final String workdir = toml.getString("workdir");
116
117        return KeyDataBuilder.create()
118            .withPublicKeys(publicKeyList)
119            .withPrivateKeys(privateKeyList)
120            .withPrivateKeyPasswordFile(pwd)
121            .withWorkingDirectory(workdir);
122    }
123
124}
