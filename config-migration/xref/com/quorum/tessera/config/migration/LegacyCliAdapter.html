37      public LegacyCliAdapter() {

41      @Override
42      public CliResult execute(String... args) throws Exception {
43  
44          Options options = buildOptions();
45          final List<String> argsList = Arrays.asList(args);
46          if (argsList.isEmpty() || argsList.contains("help")) {
47              String header = "Generate Tessera JSON config file from a Constellation TOML config file";
48              HelpFormatter formatter = new HelpFormatter();
49  
50              PrintWriter pw = new PrintWriter(sys().out());
51              formatter.printHelp(pw,
52                      200, "tessera-config-migration",
53                      header, options, formatter.getLeftPadding(),
54                      formatter.getDescPadding(), null, false);
55              pw.flush();
56  
57              final int exitCode = argsList.isEmpty() ? 1 : 0;
58              return new CliResult(exitCode, true, null);
59          }
60  
61          CommandLineParser parser = new DefaultParser();
62  
63          CommandLine line = parser.parse(options, args);
64  
65          final ConfigBuilder configBuilder = Optional.ofNullable(line.getOptionValue("tomlfile"))
66                  .map(Paths::get)
67                  .map(fileDelegate::newInputStream)
68                  .map(stream -> this.configFactory.create(stream, null))
69                  .orElse(ConfigBuilder.create());
70  
71          final KeyDataBuilder keyDataBuilder = Optional.ofNullable(line.getOptionValue("tomlfile"))
72                  .map(Paths::get)
73                  .map(fileDelegate::newInputStream)
74                  .map(configFactory::createKeyDataBuilder)
75                  .orElse(KeyDataBuilder.create());
76  
77          ConfigBuilder adjustedConfig = applyOverrides(line, configBuilder, keyDataBuilder);
78  
79          Config config = adjustedConfig.build();
80  
81          String outputfilevalue = line.getOptionValue("outputfile", "tessera-config.json");
82  
83          Path outputPath = Paths.get(outputfilevalue).toAbsolutePath();
84  
85          return writeToOutputFile(config, outputPath);
86      }
87  
88      static CliResult writeToOutputFile(Config config, Path outputPath) throws IOException {
89          SystemAdapter systemAdapter = SystemAdapter.INSTANCE;
90          systemAdapter.out().printf("Saving config to %s", outputPath);
91          systemAdapter.out().println();
92          JaxbUtil.marshalWithNoValidation(config, systemAdapter.out());
93          systemAdapter.out().println();
94  
95          try (OutputStream outputStream = Files.newOutputStream(outputPath)){
96              JaxbUtil.marshal(config, outputStream);
97              systemAdapter.out().printf("Saved config to  %s", outputPath);
98              systemAdapter.out().println();
99              return new CliResult(0, false, config);
100         } catch (ConstraintViolationException validationException) {
101             validationException.getConstraintViolations()
102                     .stream()
103                     .map(cv -> "Warning: " + cv.getMessage() + " on property " + cv.getPropertyPath())
104                     .forEach(systemAdapter.err()::println);
105 
106             Files.write(outputPath, JaxbUtil.marshalToStringNoValidation(config).getBytes());
107             systemAdapter.out().printf("Saved config to  %s", outputPath);
108             systemAdapter.out().println();
109             return new CliResult(2, false, config);
110         }
111     }

113     static ConfigBuilder applyOverrides(CommandLine line, ConfigBuilder configBuilder, KeyDataBuilder keyDataBuilder) {
114 
115         Optional.ofNullable(line.getOptionValue("workdir"))
116                 .ifPresent(configBuilder::workdir);
117 
118         Optional.ofNullable(line.getOptionValue("workdir"))
119                 .ifPresent(keyDataBuilder::withWorkingDirectory);
120 
121         Optional.ofNullable(line.getOptionValue("url"))
122                 .map(url -> {
123                     try{
124                         return new URL(url);
125                     } catch (MalformedURLException e) {
126                         throw new RuntimeException("Bad server url given: " + e.getMessage());
127                     }
128                 }).map(uri -> uri.getProtocol() + "://" + uri.getHost())
129                 .ifPresent(configBuilder::serverHostname);
130 
131         Optional.ofNullable(line.getOptionValue("port"))
132                 .map(Integer::valueOf)
133                 .ifPresent(configBuilder::serverPort);
134 
135         Optional.ofNullable(line.getOptionValue("socket"))
136                 .ifPresent(configBuilder::unixSocketFile);
137 
138         Optional.ofNullable(line.getOptionValues("othernodes"))
139                 .map(Arrays::asList)
140                 .ifPresent(configBuilder::peers);
141 
142         Optional.ofNullable(line.getOptionValues("publickeys"))
143                 .map(Arrays::asList)
144                 .ifPresent(keyDataBuilder::withPublicKeys);
145 
146         Optional.ofNullable(line.getOptionValues("privatekeys"))
147                 .map(Arrays::asList)
148                 .ifPresent(keyDataBuilder::withPrivateKeys);
149 
150         Optional.ofNullable(line.getOptionValues("alwayssendto"))
151                 .map(Arrays::asList)
152                 .ifPresent(configBuilder::alwaysSendTo);
153 
154         Optional.ofNullable(line.getOptionValue("passwords"))
155                 .ifPresent(keyDataBuilder::withPrivateKeyPasswordFile);
156 
157         Optional.ofNullable(line.getOptionValue("storage"))
158                 .map(JdbcConfigFactory::fromLegacyStorageString)
159                 .ifPresent(configBuilder::jdbcConfig);
160 
161         if (line.hasOption("ipwhitelist")) {
162             configBuilder.useWhiteList(true);
163         }
164 
165         Optional.ofNullable(line.getOptionValue("tls"))
166                 .map(String::toUpperCase)
167                 .map(SslAuthenticationMode::valueOf)
168                 .ifPresent(configBuilder::sslAuthenticationMode);
169 
170         Optional.ofNullable(line.getOptionValue("tlsservertrust"))
171                 .map(SslTrustModeFactory::resolveByLegacyValue)
172                 .ifPresent(configBuilder::sslServerTrustMode);
173 
174         Optional.ofNullable(line.getOptionValue("tlsclienttrust"))
175                 .map(SslTrustModeFactory::resolveByLegacyValue)
176                 .ifPresent(configBuilder::sslClientTrustMode);
177 
178         Optional.ofNullable(line.getOptionValue("tlsservercert"))
179                 .ifPresent(configBuilder::sslServerTlsCertificatePath);
180 
181         Optional.ofNullable(line.getOptionValue("tlsclientcert"))
182                 .ifPresent(configBuilder::sslClientTlsCertificatePath);
183 
184         Optional.ofNullable(line.getOptionValues("tlsserverchain"))
185                 .map(Arrays::asList)
186                 .ifPresent(configBuilder::sslServerTrustCertificates);
187 
188         Optional.ofNullable(line.getOptionValues("tlsclientchain"))
189                 .map(Arrays::asList)
190                 .ifPresent(configBuilder::sslClientTrustCertificates);
191 
192         Optional.ofNullable(line.getOptionValue("tlsserverkey"))
193                 .ifPresent(configBuilder::sslServerTlsKeyPath);
194 
195         Optional.ofNullable(line.getOptionValue("tlsclientkey"))
196                 .ifPresent(configBuilder::sslClientTlsKeyPath);
197 
198         Optional.ofNullable(line.getOptionValue("tlsknownservers"))
199                 .ifPresent(configBuilder::sslKnownServersFile);
200 
201         Optional.ofNullable(line.getOptionValue("tlsknownclients"))
202                 .ifPresent(configBuilder::sslKnownClientsFile);
203 
204         final KeyConfiguration keyConfiguration = keyDataBuilder.build();
205 
206         if (!keyConfiguration.getKeyData().isEmpty()) {
207             configBuilder.keyData(keyConfiguration);
208         } else {
209             if (Optional.ofNullable(line.getOptionValue("passwords")).isPresent()) {
210                 SystemAdapter.INSTANCE.err().println("Info: Public/Private key data not provided in overrides.  Overriden password file has not been added to config.");
211             }
212         }
213 
214         return configBuilder;
215     }
