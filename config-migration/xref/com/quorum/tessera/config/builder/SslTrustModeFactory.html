10  public interface SslTrustModeFactory {
11      
12       Map<String, SslTrustMode> TRUST_MODE_LOOKUP = 
13               Collections.unmodifiableMap(new HashMap<String, SslTrustMode>() {
14          {
15              put("ca", SslTrustMode.CA);
16              put("tofu", SslTrustMode.TOFU);
17              put("ca-or-tofu", SslTrustMode.CA_OR_TOFU);
18              put("whitelist", SslTrustMode.WHITELIST);
19              put("insecure-no-validation",SslTrustMode.NONE);
20              put("none", SslTrustMode.NONE);
21          }
22      });
23      
24      static SslTrustMode resolveByLegacyValue(String value) {
25          return TRUST_MODE_LOOKUP.getOrDefault(value, SslTrustMode.NONE);
26      }
