001package com.quorum.tessera.keypairconverter;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.KeyVaultType;
005import com.quorum.tessera.config.keypairs.AzureVaultKeyPair;
006import com.quorum.tessera.config.keypairs.ConfigKeyPair;
007import com.quorum.tessera.config.keypairs.HashicorpVaultKeyPair;
008import com.quorum.tessera.config.util.EnvironmentVariableProvider;
009import com.quorum.tessera.config.vault.data.AzureGetSecretData;
010import com.quorum.tessera.config.vault.data.GetSecretData;
011import com.quorum.tessera.config.vault.data.HashicorpGetSecretData;
012import com.quorum.tessera.encryption.KeyPair;
013import com.quorum.tessera.encryption.PrivateKey;
014import com.quorum.tessera.encryption.PublicKey;
015import com.quorum.tessera.key.vault.KeyVaultService;
016import com.quorum.tessera.key.vault.KeyVaultServiceFactory;
017
018import java.util.Base64;
019import java.util.Collection;
020import java.util.List;
021import java.util.Objects;
022import java.util.stream.Collectors;
023
024public class KeyPairConverter {
025
026    private final Config config;
027
028    private final EnvironmentVariableProvider envProvider;
029
030    public KeyPairConverter(Config config, EnvironmentVariableProvider envProvider) {
031        this.config = config;
032        this.envProvider = envProvider;
033    }
034
035    public Collection<KeyPair> convert(Collection<ConfigKeyPair> configKeyPairs) {
036        return configKeyPairs
037            .stream()
038            .map(this::convert)
039            .collect(Collectors.toList());
040    }
041
042    private KeyPair convert(ConfigKeyPair configKeyPair) {
043        String base64PublicKey;
044        String base64PrivateKey;
045
046        if (configKeyPair instanceof AzureVaultKeyPair) {
047
048            KeyVaultServiceFactory keyVaultServiceFactory = KeyVaultServiceFactory.getInstance(KeyVaultType.AZURE);
049
050            KeyVaultService keyVaultService = keyVaultServiceFactory.create(config, envProvider);
051
052            AzureVaultKeyPair akp = (AzureVaultKeyPair) configKeyPair;
053
054            GetSecretData getPublicKeyData = new AzureGetSecretData(akp.getPublicKeyId(), akp.getPublicKeyVersion());
055            GetSecretData getPrivateKeyData = new AzureGetSecretData(akp.getPrivateKeyId(), akp.getPrivateKeyVersion());
056
057            base64PublicKey = keyVaultService.getSecret(getPublicKeyData);
058            base64PrivateKey = keyVaultService.getSecret(getPrivateKeyData);
059        } else if (configKeyPair instanceof HashicorpVaultKeyPair) {
060
061            KeyVaultServiceFactory keyVaultServiceFactory = KeyVaultServiceFactory.getInstance(KeyVaultType.HASHICORP);
062
063            KeyVaultService keyVaultService = keyVaultServiceFactory.create(config, envProvider);
064
065            HashicorpVaultKeyPair hkp = (HashicorpVaultKeyPair) configKeyPair;
066
067            GetSecretData getPublicKeyData = new HashicorpGetSecretData(hkp.getSecretEngineName(), hkp.getSecretName(), hkp.getPublicKeyId(), hkp.getSecretVersionAsInt());
068            GetSecretData getPrivateKeyData = new HashicorpGetSecretData(hkp.getSecretEngineName(), hkp.getSecretName(), hkp.getPrivateKeyId(), hkp.getSecretVersionAsInt());
069
070            base64PublicKey = keyVaultService.getSecret(getPublicKeyData);
071            base64PrivateKey = keyVaultService.getSecret(getPrivateKeyData);
072        } else {
073
074            base64PublicKey = configKeyPair.getPublicKey();
075            base64PrivateKey = configKeyPair.getPrivateKey();
076
077        }
078
079        return new KeyPair(
080            PublicKey.from(Base64.getDecoder().decode(base64PublicKey.trim())),
081            PrivateKey.from(Base64.getDecoder().decode(base64PrivateKey.trim()))
082        );
083    }
084
085    public List<PublicKey> convert(List<String> values) {
086        return Objects.requireNonNull(values, "Key values cannot be null")
087            .stream()
088            .map(v -> Base64.getDecoder().decode(v))
089            .map(PublicKey::from)
090            .collect(Collectors.toList());
091    }
092
093}
