001package com.quorum.tessera.nio.unix;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URI;
007import java.nio.channels.AsynchronousFileChannel;
008import java.nio.channels.FileChannel;
009import java.nio.channels.SeekableByteChannel;
010import java.nio.file.AccessMode;
011import java.nio.file.CopyOption;
012import java.nio.file.DirectoryStream;
013import java.nio.file.FileStore;
014import java.nio.file.FileSystem;
015import java.nio.file.FileSystems;
016import java.nio.file.LinkOption;
017import java.nio.file.OpenOption;
018import java.nio.file.Path;
019import java.nio.file.attribute.BasicFileAttributes;
020import java.nio.file.attribute.FileAttribute;
021import java.nio.file.attribute.FileAttributeView;
022import java.nio.file.spi.FileSystemProvider;
023import java.util.Map;
024import java.util.Set;
025import java.util.concurrent.ExecutorService;
026
027/**
028 * Implementation of FileSystemProvider that handles URIs with unix scheme
029 *
030 */
031public class UnixSocketFileSystemProvider extends FileSystemProvider {
032
033    private final FileSystemProvider delegate;
034
035    public UnixSocketFileSystemProvider() {
036        this(FileSystems.getDefault().provider());
037    }
038
039    public UnixSocketFileSystemProvider(FileSystemProvider delegate) {
040        this.delegate = delegate;
041    }
042
043    @Override
044    public String getScheme() {
045        return "unix";
046    }
047
048    private static URI convert(URI uri) {
049        return UriCallback.execute(() -> {
050            return new URI("file",uri.getUserInfo(),uri.getHost(),uri.getPort(),uri.getPath(),uri.getQuery(),uri.getFragment());
051        });
052
053    }
054
055    @Override
056    public FileSystem newFileSystem(URI uri, Map<String, ?> env) throws IOException {
057        return delegate.newFileSystem(convert(uri), env);
058    }
059
060    @Override
061    public FileSystem getFileSystem(URI uri) {
062        URI adjusted = convert(uri);
063        return delegate.getFileSystem(adjusted);
064    }
065
066    @Override
067    public Path getPath(URI uri) {
068        return delegate.getPath(convert(uri));
069    }
070
071    @Override
072    public SeekableByteChannel newByteChannel(Path path, Set<? extends OpenOption> options, FileAttribute<?>... attrs) throws IOException {
073        return delegate.newByteChannel(path, options, attrs);
074    }
075
076    @Override
077    public DirectoryStream<Path> newDirectoryStream(Path dir, DirectoryStream.Filter<? super Path> filter) throws IOException {
078        return delegate.newDirectoryStream(dir, filter);
079    }
080
081    @Override
082    public void createDirectory(Path dir, FileAttribute<?>... attrs) throws IOException {
083        delegate.createDirectory(dir, attrs);
084    }
085
086    @Override
087    public void delete(Path path) throws IOException {
088        delegate.delete(path);
089    }
090
091    @Override
092    public void copy(Path source, Path target, CopyOption... options) throws IOException {
093        delegate.copy(source, target, options);
094    }
095
096    @Override
097    public FileSystem newFileSystem(Path path, Map<String, ?> env) throws IOException {
098        return delegate.newFileSystem(path, env);
099    }
100
101    @Override
102    public InputStream newInputStream(Path path, OpenOption... options) throws IOException {
103        return delegate.newInputStream(path, options);
104    }
105
106    @Override
107    public OutputStream newOutputStream(Path path, OpenOption... options) throws IOException {
108        return delegate.newOutputStream(path, options);
109    }
110
111    @Override
112    public FileChannel newFileChannel(Path path, Set<? extends OpenOption> options, FileAttribute<?>... attrs) throws IOException {
113        return delegate.newFileChannel(path, options, attrs);
114    }
115
116    @Override
117    public AsynchronousFileChannel newAsynchronousFileChannel(Path path, Set<? extends OpenOption> options, ExecutorService executor, FileAttribute<?>... attrs) throws IOException {
118        return delegate.newAsynchronousFileChannel(path, options, executor, attrs);
119    }
120
121    @Override
122    public void createSymbolicLink(Path link, Path target, FileAttribute<?>... attrs) throws IOException {
123        delegate.createSymbolicLink(link, target, attrs);
124    }
125
126    @Override
127    public void createLink(Path link, Path existing) throws IOException {
128        delegate.createLink(link, existing);
129    }
130
131    @Override
132    public boolean deleteIfExists(Path path) throws IOException {
133        return delegate.deleteIfExists(path);
134    }
135
136    @Override
137    public Path readSymbolicLink(Path link) throws IOException {
138        return delegate.readSymbolicLink(link);
139    }
140
141    @Override
142    public void move(Path source, Path target, CopyOption... options) throws IOException {
143        delegate.move(source, target, options);
144    }
145
146    @Override
147    public boolean isSameFile(Path path, Path path2) throws IOException {
148        return delegate.isSameFile(path, path2);
149    }
150
151    @Override
152    public boolean isHidden(Path path) throws IOException {
153        return delegate.isHidden(path);
154    }
155
156    @Override
157    public FileStore getFileStore(Path path) throws IOException {
158        return delegate.getFileStore(path);
159    }
160
161    @Override
162    public void checkAccess(Path path, AccessMode... modes) throws IOException {
163        delegate.checkAccess(path, modes);
164    }
165
166    @Override
167    public <V extends FileAttributeView> V getFileAttributeView(Path path, Class<V> type, LinkOption... options) {
168        return delegate.getFileAttributeView(path, type, options);
169    }
170
171    @Override
172    public <A extends BasicFileAttributes> A readAttributes(Path path, Class<A> type, LinkOption... options) throws IOException {
173        return delegate.readAttributes(path, type, options);
174    }
175
176    @Override
177    public Map<String, Object> readAttributes(Path path, String attributes, LinkOption... options) throws IOException {
178        return delegate.readAttributes(path, attributes, options);
179    }
180
181    @Override
182    public void setAttribute(Path path, String attribute, Object value, LinkOption... options) throws IOException {
183        delegate.setAttribute(path, attribute, value, options);
184    }
185
186}
