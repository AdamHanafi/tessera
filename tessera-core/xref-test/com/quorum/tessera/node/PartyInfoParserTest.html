116     @Test
117     public void toUsingSameInfoFromFixture() {
118 
119         final PartyInfo partyInfo = partyInfoParser.from(dataOne);
120         final byte[] result = partyInfoParser.to(partyInfo);
121 
122         final ByteBuffer byteBuffer = ByteBuffer.wrap(result);
123 
124         assertThat(result).isNotEmpty();
125         assertThat(byteBuffer.getLong()).isEqualTo(21L);
126 
127         byte[] urlData = new byte[21];
128         byteBuffer.get(urlData);
129 
130         final String url = new String(urlData);
131         assertThat(url).isEqualTo(partyInfo.getUrl());
132 
133         long numberOfRecipients = byteBuffer.getLong();
134         assertThat(numberOfRecipients).isEqualTo(1L);
135 
136         long keyByteLength = byteBuffer.getLong();
137         assertThat(keyByteLength).isEqualTo(32L);
138 
139         final byte[] keyData = new byte[32];
140         byteBuffer.get(keyData);
141         assertThat(keyData).hasSize(32)
142                 .isEqualTo(partyInfo.getRecipients().iterator().next().getKey().getKeyBytes());
143 
144         long recipientUrlLength = byteBuffer.getLong();
145         assertThat(recipientUrlLength).isEqualTo(21L);
146 
147         byte[] recipientUrlData = new byte[21];
148         byteBuffer.get(recipientUrlData);
149 
150         String recipientUrl = new String(recipientUrlData);
151         assertThat(recipientUrl).isEqualTo(partyInfo.getRecipients().iterator().next().getUrl());
152 
153         long partyCount = byteBuffer.getLong();
154         assertThat(partyCount).isEqualTo(1L);
155 
156         long partyUrlLength = byteBuffer.getLong();
157         assertThat(partyUrlLength).isEqualTo(21L);
158 
159         byte[] partyUrlData = new byte[21];
160         byteBuffer.get(partyUrlData);
161 
162         String partyUrl = new String(partyUrlData);
163         assertThat(partyUrl).isEqualTo(partyInfo.getParties().iterator().next().getUrl());
164
