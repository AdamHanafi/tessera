3   import com.quorum.tessera.core.config.ConfigService;
4   import com.quorum.tessera.node.model.Party;
5   import com.quorum.tessera.node.model.PartyInfo;
6   import com.quorum.tessera.node.model.Recipient;
7   
8   import java.time.Instant;
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  import static java.util.Collections.unmodifiableSet;
13  
14  /**
15   * Stores a list of all discovered nodes and public keys
16   */
17  public class PartyInfoStore {
18  
19      private final String advertisedUrl;
20  
21      private final Set<Recipient> recipients;
22  
23      private final Set<Party> parties;
24  
25      public PartyInfoStore(final ConfigService configService) {
26  
27          //TODO: remove the extra "/" when we deprecate backwards compatibility
28          this.advertisedUrl = configService.getServerUri().toString() + "/";
29  
30          this.recipients = new HashSet<>();
31          this.parties = new HashSet<>();
32          this.parties.add(new Party(this.advertisedUrl));
33      }
34  
35      /**
36       * Merge an incoming {@link PartyInfo} into the current one, adding any
37       * new keys or parties to the current store
38       * @param newInfo the incoming information that may contain new nodes/keys
39       */
40      public synchronized void store(final PartyInfo newInfo) {
41          recipients.addAll(newInfo.getRecipients());
42          parties.addAll(newInfo.getParties());
43  
44          //update the sender to have been seen recently
45          final Party sender = new Party(newInfo.getUrl());
46          sender.setLastContacted(Instant.now());
47          parties.remove(sender);
48          parties.add(sender);
49      }
50  
51      /**
52       * Fetch a copy of all the currently discovered nodes/keys
53       *
54       * @return an immutable copy of the current state of the store
55       */
56      public synchronized PartyInfo getPartyInfo() {
57          return new PartyInfo(
58              advertisedUrl,
59              unmodifiableSet(new HashSet<>(recipients)),
60              unmodifiableSet(new HashSet<>(parties))
61          );
62      }
63  
64  }
