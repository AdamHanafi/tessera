11  import javax.transaction.Transactional;
12  import java.util.Arrays;
13  import java.util.Optional;
14  
15  public class ResendManagerImpl implements ResendManager {
16  
17      private final EncryptedTransactionDAO encryptedTransactionDAO;
18  
19      private final PayloadEncoder payloadEncoder;
20  
21      private final Enclave enclave;
22  
23      private final MessageHashFactory messageHashFactory = MessageHashFactory.create();
24  
25      public ResendManagerImpl(final EncryptedTransactionDAO dao, final PayloadEncoder encoder, final Enclave enclave) {
26          this.encryptedTransactionDAO = dao;
27          this.payloadEncoder = encoder;
28          this.enclave = enclave;
29      }
30  
31      //TODO: synchronize based on messagehash, so different message don't lock each other
32      @Transactional
33      public synchronized void acceptOwnMessage(final byte[] message) {
34  
35          final EncodedPayload payload = payloadEncoder.decode(message);
36  
37          //check the payload can be decrpyted to ensure it isn't rubbish being sent to us
38          final byte[] newDecrypted = enclave.unencryptTransaction(payload, null);
39  
40          final MessageHash transactionHash = Optional.of(payload)
41              .map(EncodedPayload::getCipherText)
42              .map(messageHashFactory::createFromCipherText)
43              .get();
44  
45          final PublicKey sender = payload.getSenderKey();
46          if (!enclave.getPublicKeys().contains(sender)) {
47              throw new IllegalArgumentException(
48                  "Message " + transactionHash.toString() + " does not have one the nodes own keys as a sender"
49              );
50          }
51  
52          //this is a tx which we created
53          final Optional<EncryptedTransaction> tx = this.encryptedTransactionDAO.retrieveByHash(transactionHash);
54  
55          if (tx.isPresent()) {
56  
57              //we just need to add the recipient
58              final byte[] encodedPayload = tx.get().getEncodedPayload();
59              final EncodedPayload existing = payloadEncoder.decode(encodedPayload);
60  
61              if (!existing.getRecipientKeys().contains(payload.getRecipientKeys().get(0))) {
62                  //lets compare it against another message received before
63                  final byte[] oldDecrypted = enclave.unencryptTransaction(existing, null);
64                  final boolean same = Arrays.equals(newDecrypted, oldDecrypted)
65                      && Arrays.equals(payload.getCipherText(), existing.getCipherText());
66  
67                  if (!same) {
68                      throw new IllegalArgumentException("Invalid payload provided");
69                  }
70  
71                  existing.getRecipientKeys().add(payload.getRecipientKeys().get(0));
72                  existing.getRecipientBoxes().add(payload.getRecipientBoxes().get(0));
73  
74                  tx.get().setEncodedPayload(payloadEncoder.encode(existing));
75  
76                  this.encryptedTransactionDAO.save(tx.get());
77              }
78  
79          } else {
80  
81              //we need to recreate this
82              payload.getRecipientKeys().add(sender);
83              byte[] newbox = enclave.createNewRecipientBox(payload, sender);
84              payload.getRecipientBoxes().add(newbox);
85  
86              final byte[] encoded = payloadEncoder.encode(payload);
87  
88              this.encryptedTransactionDAO.save(new EncryptedTransaction(transactionHash, encoded));
89  
90          }
91  
92      }
93  
94  }
