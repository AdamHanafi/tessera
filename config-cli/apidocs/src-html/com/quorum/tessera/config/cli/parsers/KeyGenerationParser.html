001package com.quorum.tessera.config.cli.parsers;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.cli.CliException;
005import com.quorum.tessera.config.keypairs.ConfigKeyPair;
006import com.quorum.tessera.config.util.JaxbUtil;
007import com.quorum.tessera.key.generation.KeyGenerator;
008import com.quorum.tessera.key.generation.KeyGeneratorFactory;
009import com.quorum.tessera.key.generation.KeyVaultOptions;
010import org.apache.commons.cli.CommandLine;
011
012import javax.validation.ConstraintViolation;
013import javax.validation.ConstraintViolationException;
014import javax.validation.Validation;
015import javax.validation.Validator;
016import java.io.IOException;
017import java.io.InputStream;
018import java.nio.file.Files;
019import java.nio.file.Path;
020import java.nio.file.Paths;
021import java.util.ArrayList;
022import java.util.List;
023import java.util.Optional;
024import java.util.Set;
025import java.util.stream.Collectors;
026import java.util.stream.Stream;
027
028import static java.util.Collections.singletonList;
029
030public class KeyGenerationParser implements Parser<List<ConfigKeyPair>> {
031
032    private final KeyGeneratorFactory factory = KeyGeneratorFactory.newFactory();
033
034    private final Validator validator = Validation.byDefaultProvider()
035        .configure()
036        .ignoreXmlConfiguration()
037        .buildValidatorFactory()
038        .getValidator();
039
040    public List<ConfigKeyPair> parse(final CommandLine commandLine) throws IOException {
041
042        final ArgonOptions argonOptions = this.argonOptions(commandLine).orElse(null);
043        final KeyVaultOptions keyVaultOptions = this.keyVaultOptions(commandLine).orElse(null);
044        final KeyVaultConfig keyVaultConfig = this.keyVaultConfig(commandLine).orElse(null);
045
046        final KeyGenerator generator = factory.create(keyVaultConfig);
047
048        if (commandLine.hasOption("keygen")) {
049            return this.filenames(commandLine)
050                .stream()
051                .map(name -> generator.generate(name, argonOptions, keyVaultOptions))
052                .collect(Collectors.toList());
053        }
054
055        return new ArrayList<>();
056    }
057
058    private Optional<ArgonOptions> argonOptions(final CommandLine commandLine) throws IOException {
059
060        if (commandLine.hasOption("keygenconfig")) {
061            final String pathName = commandLine.getOptionValue("keygenconfig");
062            final InputStream configStream = Files.newInputStream(Paths.get(pathName));
063
064            final ArgonOptions argonOptions = JaxbUtil.unmarshal(configStream, ArgonOptions.class);
065            return Optional.of(argonOptions);
066        }
067
068        return Optional.empty();
069    }
070
071    private Optional<KeyVaultOptions> keyVaultOptions(final CommandLine commandLine) {
072        Optional<String> secretEngineName = Optional.ofNullable(commandLine.getOptionValue("keygenvaultsecretengine"));
073
074        return secretEngineName.map(KeyVaultOptions::new);
075    }
076
077    private List<String> filenames(final CommandLine commandLine) {
078
079        if (commandLine.hasOption("filename")) {
080
081            final String keyNames = commandLine.getOptionValue("filename");
082            if (keyNames != null) {
083                return Stream.of(keyNames.split(",")).collect(Collectors.toList());
084            }
085
086        }
087
088        return singletonList("");
089
090    }
091
092    private Optional<KeyVaultConfig> keyVaultConfig(CommandLine commandLine) {
093        if(!commandLine.hasOption("keygenvaulttype") && !commandLine.hasOption("keygenvaulturl")) {
094            return Optional.empty();
095        }
096
097        final String t = commandLine.getOptionValue("keygenvaulttype");
098
099        KeyVaultType keyVaultType;
100        try {
101            keyVaultType = KeyVaultType.valueOf(
102                t.trim()
103                 .toUpperCase()
104            );
105        } catch(IllegalArgumentException | NullPointerException e) {
106            throw new CliException("Key vault type either not provided or not recognised");
107        }
108
109        String keyVaultUrl = commandLine.getOptionValue("keygenvaulturl");
110
111        KeyVaultConfig keyVaultConfig;
112
113        if(keyVaultType.equals(KeyVaultType.AZURE)) {
114            keyVaultConfig = new AzureKeyVaultConfig(keyVaultUrl);
115
116            Set<ConstraintViolation<AzureKeyVaultConfig>> violations = validator.validate((AzureKeyVaultConfig)keyVaultConfig);
117
118            if(!violations.isEmpty()) {
119                throw new ConstraintViolationException(violations);
120            }
121        } else {
122            if(!commandLine.hasOption("filename")) {
123                throw new CliException("At least one -filename must be provided when saving generated keys in a Hashicorp Vault");
124            }
125
126            String approlePath = commandLine.getOptionValue("keygenvaultapprole");
127
128            Optional<Path> tlsKeyStorePath = Optional.ofNullable(commandLine.getOptionValue("keygenvaultkeystore"))
129                                                     .map(Paths::get);
130
131            Optional<Path> tlsTrustStorePath = Optional.ofNullable(commandLine.getOptionValue("keygenvaulttruststore"))
132                                                       .map(Paths::get);
133
134            keyVaultConfig = new HashicorpKeyVaultConfig(
135                keyVaultUrl,
136                approlePath,
137                tlsKeyStorePath.orElse(null),
138                tlsTrustStorePath.orElse(null)
139            );
140
141            Set<ConstraintViolation<HashicorpKeyVaultConfig>> violations = validator.validate((HashicorpKeyVaultConfig)keyVaultConfig);
142
143            if(!violations.isEmpty()) {
144                throw new ConstraintViolationException(violations);
145            }
146        }
147
148        return Optional.of(keyVaultConfig);
149    }
150
151}
