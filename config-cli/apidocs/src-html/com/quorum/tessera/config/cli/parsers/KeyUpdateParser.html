001package com.quorum.tessera.config.cli.parsers;
002
003import com.quorum.tessera.config.ArgonOptions;
004import com.quorum.tessera.config.KeyDataConfig;
005import com.quorum.tessera.config.PrivateKeyData;
006import com.quorum.tessera.config.PrivateKeyType;
007import com.quorum.tessera.config.keys.KeyEncryptor;
008import com.quorum.tessera.config.util.JaxbUtil;
009import com.quorum.tessera.config.util.PasswordReader;
010import com.quorum.tessera.encryption.PrivateKey;
011import com.quorum.tessera.io.SystemAdapter;
012import org.apache.commons.cli.CommandLine;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016import java.io.IOException;
017import java.nio.file.Files;
018import java.nio.file.Path;
019import java.nio.file.Paths;
020import java.util.List;
021import java.util.Objects;
022import java.util.Optional;
023
024import static java.nio.charset.StandardCharsets.UTF_8;
025import java.util.Base64;
026import static java.util.Collections.emptyList;
027import static java.util.Collections.singletonList;
028
029public class KeyUpdateParser implements Parser<Optional> {
030
031    private static final Logger LOGGER = LoggerFactory.getLogger(KeyUpdateParser.class);
032
033    private final KeyEncryptor keyEncryptor;
034
035    private final PasswordReader passwordReader;
036
037    public KeyUpdateParser(final KeyEncryptor keyEncryptor, final PasswordReader passwordReader) {
038        this.keyEncryptor = Objects.requireNonNull(keyEncryptor);
039        this.passwordReader = Objects.requireNonNull(passwordReader);
040    }
041
042    @Override
043    public Optional parse(final CommandLine commandLine) throws IOException {
044        if (commandLine.hasOption("updatepassword")) {
045            final ArgonOptions argonOptions = argonOptions(commandLine);
046            final List<String> passwords = passwords(commandLine);
047            final Path keypath = privateKeyPath(commandLine);
048
049            final KeyDataConfig keyDataConfig = JaxbUtil.unmarshal(Files.newInputStream(keypath), KeyDataConfig.class);
050            final PrivateKey privateKey = this.getExistingKey(keyDataConfig, passwords);
051
052            final String newPassword = passwordReader.requestUserPassword();
053
054            final KeyDataConfig updatedKey;
055            if(newPassword.isEmpty()) {
056                final PrivateKeyData privateKeyData = new PrivateKeyData(privateKey.encodeToBase64(), null, null, null, null);
057                updatedKey = new KeyDataConfig(privateKeyData, PrivateKeyType.UNLOCKED);
058            } else {
059                final PrivateKeyData privateKeyData = keyEncryptor.encryptPrivateKey(privateKey, newPassword, argonOptions);
060                updatedKey = new KeyDataConfig(privateKeyData, PrivateKeyType.LOCKED);
061            }
062
063            //write the key to file
064            Files.write(keypath, JaxbUtil.marshalToString(updatedKey).getBytes(UTF_8));
065            SystemAdapter.INSTANCE.out().println("Private key at " + keypath.toString() + " updated.");
066        }
067
068        return Optional.empty();
069    }
070
071    PrivateKey getExistingKey(final KeyDataConfig kdc, final List<String> passwords) {
072
073        if (kdc.getType() == PrivateKeyType.UNLOCKED) {
074            byte[] privateKeyData = Base64.getDecoder().decode(kdc.getValue().getBytes(UTF_8));
075            return PrivateKey.from(privateKeyData);
076        } else {
077
078            for (final String pass : passwords) {
079                try {
080                    return PrivateKey.from(keyEncryptor.decryptPrivateKey(kdc.getPrivateKeyData(), pass).getKeyBytes());
081                } catch (final Exception e) {
082                    LOGGER.debug("Password failed to decrypt. Trying next if available.");
083                }
084            }
085
086            throw new IllegalArgumentException("Locked key but no valid password given");
087        }
088    }
089
090    static Path privateKeyPath(final CommandLine commandLine) {
091        final String privateKeyPath = commandLine.getOptionValue("keys.keyData.privateKeyPath");
092
093        if (privateKeyPath == null) {
094            throw new IllegalArgumentException("Private key path cannot be null when updating key password");
095        }
096
097        final Path keypath = Paths.get(privateKeyPath);
098        if (Files.notExists(keypath)) {
099            throw new IllegalArgumentException("Private key path must exist when updating key password");
100        }
101
102        return keypath;
103    }
104
105    static List<String> passwords(final CommandLine commandLine) throws IOException {
106        final String password = commandLine.getOptionValue("keys.passwords");
107        final String passwordFile = commandLine.getOptionValue("keys.passwordFile");
108
109        if (password != null) {
110            return singletonList(password);
111        } else if (passwordFile != null) {
112            return Files.readAllLines(Paths.get(passwordFile));
113        } else {
114            return emptyList();
115        }
116
117    }
118
119    static ArgonOptions argonOptions(final CommandLine commandLine) {
120        final String algorithm = commandLine.getOptionValue("keys.keyData.config.data.aopts.algorithm", "i");
121        final String iterations = commandLine.getOptionValue("keys.keyData.config.data.aopts.iterations", "10");
122        final String memory = commandLine.getOptionValue("keys.keyData.config.data.aopts.memory", "1048576");
123        final String parallelism = commandLine.getOptionValue("keys.keyData.config.data.aopts.parallelism", "4");
124
125        return new ArgonOptions(
126            algorithm, Integer.valueOf(iterations), Integer.valueOf(memory), Integer.valueOf(parallelism)
127        );
128    }
129
130}
