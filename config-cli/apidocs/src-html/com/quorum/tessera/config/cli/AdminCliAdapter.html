001package com.quorum.tessera.config.cli;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.cli.parsers.ConfigurationParser;
005import com.quorum.tessera.jaxrs.client.ClientFactory;
006import java.io.PrintWriter;
007import java.net.URI;
008import java.util.Arrays;
009import java.util.List;
010import java.util.Objects;
011import java.util.Optional;
012import javax.ws.rs.client.Client;
013import javax.ws.rs.client.Entity;
014import javax.ws.rs.core.MediaType;
015import javax.ws.rs.core.Response;
016import javax.ws.rs.core.UriBuilder;
017import org.apache.commons.cli.CommandLine;
018import org.apache.commons.cli.DefaultParser;
019import org.apache.commons.cli.HelpFormatter;
020import org.apache.commons.cli.Option;
021import org.apache.commons.cli.Options;
022
023/**
024 * Cli Adapter to be used for runtime updates
025 */
026public class AdminCliAdapter implements CliAdapter {
027
028    private final ClientFactory clientFactory;
029
030    public AdminCliAdapter(final ClientFactory clientFactory) {
031        this.clientFactory = Objects.requireNonNull(clientFactory);
032    }
033
034    /**
035     *
036     * @param args
037     * @return CliResult with config object always null.
038     * @throws Exception
039     */
040    @Override
041    public CliResult execute(String... args) throws Exception {
042
043        final Options options = new Options();
044
045        options.addOption(
046                Option.builder("configfile")
047                        .desc("Path to node configuration file")
048                        .hasArg(true)
049                        .required()
050                        .optionalArg(false)
051                        .numberOfArgs(1)
052                        .argName("PATH")
053                        .build());
054
055        options.addOption(Option.builder("addpeer")
056                .desc("Add peer to running node")
057                .hasArg(true)
058                .optionalArg(false)
059                .numberOfArgs(1)
060                .argName("URL")
061                .build());
062
063        final List<String> argsList = Arrays.asList(args);
064
065        if (argsList.contains("help") || argsList.isEmpty()) {
066            HelpFormatter formatter = new HelpFormatter();
067            PrintWriter pw = new PrintWriter(sys().out());
068            formatter.printHelp(pw,
069                    200, "tessera admin",
070                    null, options, formatter.getLeftPadding(),
071                    formatter.getDescPadding(), null, false);
072            pw.flush();
073            
074            return new CliResult(0, true, null);
075        }
076
077        final CommandLine line = new DefaultParser().parse(options, args);
078        if (!line.hasOption("addpeer")) {
079            sys().out().println("No peer defined");
080            return new CliResult(1, true, null);
081        }
082
083        Config config = new ConfigurationParser().parse(line);
084
085        //TODO revisit - maybe the admin stuff should be reached via unix socket - in order to avoid security concerns
086        ServerConfig serverConfig = config.getServerConfigs().stream()
087                .filter(c -> c.getApp() == AppType.ADMIN)
088                .findFirst().orElse(config.getServerConfigs().stream().findAny().get());
089
090        Client restClient = clientFactory.buildFrom(serverConfig);
091
092        String peerUrl = line.getOptionValue("addpeer");
093
094        final Peer peer = new Peer(peerUrl);
095
096        String scheme = Optional.of(serverConfig)
097                .map(ServerConfig::getBindingUri)
098                .map(URI::getScheme)
099                .orElse("http");
100
101        Integer port = Optional.of(serverConfig)
102                .map(ServerConfig::getServerUri)
103                .map(URI::getPort)
104                .orElse(80);
105
106        URI uri = UriBuilder.fromUri(serverConfig.getBindingUri())
107                .port(port)
108                .scheme(scheme)
109                .build();
110
111        Response response = restClient.target(uri)
112                .path("config")
113                .path("peers")
114                .request(MediaType.APPLICATION_JSON)
115                .accept(MediaType.APPLICATION_JSON)
116                .put(Entity.entity(peer, MediaType.APPLICATION_JSON));
117
118        if (response.getStatus() == Response.Status.CREATED.getStatusCode()) {
119
120            sys().out().printf("Peer %s added.", response.getLocation());
121            sys().out().println();
122
123            return new CliResult(0, true, null);
124        }
125
126        sys().err().println("Unable to create peer");
127
128        return new CliResult(1, true, null);
129    }
130
131}
