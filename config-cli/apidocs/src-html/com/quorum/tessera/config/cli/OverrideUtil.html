052    static Map<String, Class> buildConfigOptions() {
053        final Map<String, Class> fields = fields(null, Config.class);
054
055        //add key overrides separately as they cannot be determined from the type directly
056        fields.put("keys.keyData.privateKeyPath", Path.class);
057        fields.put("keys.keyData.config.data.aopts.parallelism", String.class);
058        fields.put("keys.keyData.config.data.aopts.memory", String.class);
059        fields.put("keys.keyData.config.data.aopts.iterations", String.class);
060        fields.put("keys.keyData.config.data.aopts.algorithm", String.class);
061
062        return fields;

065    static String resolveName(Field field) {
066        if (!field.isAnnotationPresent(XmlElement.class)) {
067            return field.getName();
068        }
069        final XmlElement xmlElement = field.getAnnotation(XmlElement.class);
070        String name = xmlElement.name();
071        return Objects.equals("##default", name) ? field.getName() : name;
072    }
073
074    static Class resolveCollectionParameter(Type type) {
075
076        return Stream.of(type)
077                .map(ParameterizedType.class::cast)
078                .map(pt -> pt.getActualTypeArguments()[0])
079                .map(Type::getTypeName)
080                .map(n -> ReflectCallback.execute(() -> Class.forName(n))).findAny().get();
081
082    }
083
084    static boolean isSimple(Field field) {
085        return isSimple(field.getType());
086    }
087
088    static boolean isSimple(Class type) {
089
090        if (type.isPrimitive()) {
091            return true;
092        }
093
094        if (type.getPackage() == null) {
095            return true;
096        }
097
098        if (SIMPLE_TYPES.contains(type)) {
099            return true;
100        }
101
102        return type.isEnum();
103    }
104
105    static Map<String, Class> fields(String prefix, Class type) {
106
107        String amendedPrefix = Optional.ofNullable(prefix)
108                .map(s -> s.concat("."))
109                .orElse("");
110
111        Map<String, Class> list = new HashMap<>();
112        for (Field field : type.getDeclaredFields()) {
113
114            if (isSimple(field)) {
115
116                list.put(amendedPrefix + resolveName(field), field.getType());
117                continue;
118            }
119
120            if (field.getType().getPackage().equals(Config.class.getPackage())) {
121                list.putAll(fields(amendedPrefix + resolveName(field), field.getType()));
122            }
123
124            if (Collection.class.isAssignableFrom(field.getType())) {
125                Class t = resolveCollectionParameter(field.getGenericType());
126
127                if (isSimple(t)) {
128                    final Class arrayType = toArrayType(t);
129                    list.put(amendedPrefix + resolveName(field), arrayType);
130
131                } else {
132                    list.putAll(fields(amendedPrefix + resolveName(field), t));
133                }
134            }
135        }
136
137        return list;
138
139    }
140
141    static <T> Class<T[]> toArrayType(Class<T> t) {
142        return (Class<T[]>) Array.newInstance(t, 0).getClass();
143    }

145    /**
146     * Directly set field values using reflection.
147     *
148     * @param root
149     * @param path
150     * @param value
151     */
152    static void setValue(Object root, String path, String... value) {
153
154        if(root == null) {
155            return;
156        }
157
158        final ListIterator<String> pathTokens = Arrays.asList(path.split("\\.")).listIterator();
159
160        final Class rootType = root.getClass();
161
162        while (pathTokens.hasNext()) {
163
164            final String token = pathTokens.next();
165            final Field field = resolveField(rootType, token);
166            field.setAccessible(true);
167
168            final Class fieldType = field.getType();
169
170            if (Collection.class.isAssignableFrom(fieldType)) {
171
172                final Class genericType = resolveCollectionParameter(field.getGenericType());
173
174                List list = (List) Optional.ofNullable(getValue(root, field))
175                        .orElse(new ArrayList<>());
176                if (isSimple(genericType)) {
177
178                    List convertedValues = (List) Stream.of(value)
179                            .map(v -> convertTo(genericType, v))
180                            .collect(Collectors.toList());
181
182                    List merged = new ArrayList(list);
183                    merged.addAll(convertedValues);
184
185                    setValue(root, field, merged);
186
187                } else {
188
189                    List<String> builder = new ArrayList<>();
190                    pathTokens.forEachRemaining(builder::add);
191                    String nestedPath = builder.stream().collect(Collectors.joining("."));
192                    
193                    final Object[] newList;
194                    if(ADDITIVE_COLLECTION_FIELDS.contains(field.getName())) {
195                        newList = new Object[value.length];
196                    } else {
197                        newList = Arrays.copyOf(list.toArray(), value.length);
198                    }
199
200                    for (int i = 0; i < value.length; i++) {
201                        final String v = value[i];
202
203                        final Object nestedObject = Optional.ofNullable(newList[i])
204                                .orElse(createInstance(genericType));
205
206                        initialiseNestedObjects(nestedObject);
207
208                        setValue(nestedObject, nestedPath, v);
209                        newList[i] = nestedObject;
210                    }
211                    List merged = new ArrayList();
212                    if(ADDITIVE_COLLECTION_FIELDS.contains(field.getName())) {
213                        merged.addAll(list);
214                    }
215                    merged.addAll(Arrays.asList(newList));
216                    setValue(root, field, merged);
217
218                }
219
220            } else if (isSimple(fieldType)) {
221                Class convertedType = PRIMATIVE_LOOKUP.getOrDefault(fieldType, fieldType);
222                Object convertedValue = convertTo(convertedType, value[0]);
223                setValue(root, field, convertedValue);
224
225            } else {
226
227                Object nestedObject = getOrCreate(root, field);
228                List<String> builder = new ArrayList<>();
229                pathTokens.forEachRemaining(builder::add);
230                String nestedPath = builder.stream().collect(Collectors.joining("."));
231                setValue(nestedObject, nestedPath, value);
232
233                setValue(root, field, nestedObject);
234            }
235
236        }
237
238    }

240    static <T> T getOrCreate(Object from, Field field) {
241        T value = getValue(from, field);
242        return Optional.ofNullable(value)
243                .orElse((T) createInstance(field.getType()));
244    }
245
246    static <T> T getValue(Object from, Field field) {
247        return ReflectCallback.execute(() -> {
248            return (T) field.get(from);
249        });
250    }
251
252    static void setValue(Object obj, Field field, Object value) {
253        ReflectCallback.execute(() -> {
254            field.set(obj, value);
255            return null;
256        });
257    }
258
259    static Field resolveField(Class type, String name) {
260        LOGGER.debug("Resolving {}#{}", type, name);
261
262        return Stream.of(type.getDeclaredFields())
263                .filter(f -> f.isAnnotationPresent(XmlElement.class))
264                .filter(f -> f.getAnnotation(XmlElement.class).name().equals(name))
265                .findAny()
266                .orElseGet(() -> ReflectCallback.execute(() -> type.getDeclaredField(name)));
267    }
268
269    static <T> T createInstance(Class<T> type) {
270
271        if(type.isInterface()) {
272            return null;
273        }
274
275        return ReflectCallback.execute(() -> {
276            final T instance = type.newInstance();
277            initialiseNestedObjects(instance);
278            return instance;
279        });
280
281    }
282
283    static Class classForName(String classname) {
284        return ReflectCallback.execute(() -> Class.forName(classname));
285    }
286
287    static void initialiseNestedObjects(Object obj) {
288        if (obj == null) {
289            return;
290        }
291        ReflectCallback.execute(() -> {
292            Class type = obj.getClass();
293            Field[] fields = type.getDeclaredFields();
294
295            for (Field field : fields) {
296                field.setAccessible(true);
297                Class fieldType = field.getType();
298                if (isSimple(fieldType)) {
299                    continue;
300                }
301
302                if (Collection.class.isAssignableFrom(fieldType)) {
303                    setValue(obj, field, new ArrayList<>());
304                    continue;
305                }
306
307                Object nestedObject = createInstance(fieldType);
308                initialiseNestedObjects(nestedObject);
309                setValue(obj, field, nestedObject);
310
311            }
312            return null;
313        });
314
315    }
316
317    static <T> T convertTo(Class<T> type, String value) {
318
319        if (Objects.isNull(value)) {
320            return null;
321        }
322
323        if (String.class.equals(type)) {
324            return (T) value;
325        }
326
327        if (Path.class.equals(type)) {
328            return (T) Paths.get(value);
329        }
330
331        if (type.isEnum()) {
332            return (T) Enum.valueOf(type.asSubclass(Enum.class), value);
333        }
334
335        if (byte[].class.isAssignableFrom(type)) {
336            return (T) value.getBytes(StandardCharsets.UTF_8);
337        }
338
339        return SIMPLE_TYPES.stream()
340                .filter(t -> t.equals(type))
341                .findFirst()
342                .map(c -> ReflectCallback.execute(() -> c.getDeclaredMethod("valueOf", String.class)))
343                .map(m -> ReflectCallback.execute(() -> m.invoke(null, value)))
344                .map(type::cast)
345                .get();
346
347    }
