001package com.quorum.tessera.config.cli;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.cli.parsers.ConfigurationParser;
005import com.quorum.tessera.config.cli.parsers.KeyGenerationParser;
006import com.quorum.tessera.config.cli.parsers.KeyUpdateParser;
007import com.quorum.tessera.config.cli.parsers.PidFileParser;
008import com.quorum.tessera.config.keypairs.ConfigKeyPair;
009import com.quorum.tessera.config.keys.KeyEncryptorFactory;

011import org.apache.commons.cli.*;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015import javax.validation.ConstraintViolation;
016import javax.validation.ConstraintViolationException;
017import javax.validation.Validation;
018import javax.validation.Validator;
019import java.io.IOException;
020import java.io.PrintWriter;
021import java.util.*;
022
023public class DefaultCliAdapter implements CliAdapter {
024
025    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultCliAdapter.class);
026
027    private final Validator validator = Validation.byDefaultProvider()
028        .configure()
029        .ignoreXmlConfiguration()
030        .buildValidatorFactory()
031        .getValidator();
032
033    @Override
034    public CliResult execute(String... args) throws Exception {
035
036        Options options = this.buildBaseOptions();
037
038        Map<String, Class> overrideOptions = OverrideUtil.buildConfigOptions();
039
040        overrideOptions.forEach((optionName, optionType) -> {
041
042            final boolean isCollection = optionType.isArray();
043
044            Option.Builder optionBuilder = Option.builder()
045                .longOpt(optionName)
046                .desc(String.format("Override option for %s , type: %s", optionName, optionType.getSimpleName()));
047
048            if (isCollection) {
049                optionBuilder.hasArgs().argName(optionType.getSimpleName().toUpperCase() + "...");
050            } else {
051                optionBuilder.hasArg().argName(optionType.getSimpleName().toUpperCase());
052            }
053
054            options.addOption(optionBuilder.build());
055
056        });
057
058        final List<String> argsList = Arrays.asList(args);
059        if (argsList.contains("help") || argsList.isEmpty()) {
060            HelpFormatter formatter = new HelpFormatter();
061            PrintWriter pw = new PrintWriter(sys().out());
062            formatter.printHelp(pw, 
063                    200, "tessera -configfile <PATH> [-keygen <PATH>] [-pidfile <PATH>]", 
064                    null, options, formatter.getLeftPadding(), 
065                    formatter.getDescPadding(), null, false);
066            pw.flush();
067            return new CliResult(0, true, null);
068        }
069
070        try {
071
072            final CommandLine line = new DefaultParser().parse(options, args);
073
074            final Config config = parseConfig(line);
075
076            if (Objects.nonNull(config)) {
077
078                overrideOptions.forEach((optionName, value) -> {
079                    if (line.hasOption(optionName)) {
080                        String[] values = line.getOptionValues(optionName);
081                        LOGGER.debug("Setting : {} with value(s) {}", optionName, values);
082                        OverrideUtil.setValue(config, optionName, values);
083                        LOGGER.debug("Set : {} with value(s) {}", optionName, values);
084                    }
085                });
086
087                Set<ConstraintViolation<Config>> violations = validator.validate(config);
088                if (!violations.isEmpty()) {
089                    throw new ConstraintViolationException(violations);
090                }
091            }
092
093            new PidFileParser().parse(line);
094
095            boolean suppressStartup = line.hasOption("keygen") && Objects.isNull(config);
096
097            return new CliResult(0, suppressStartup, config);
098
099        } catch (ParseException exp) {
100            throw new CliException(exp.getMessage());
101        }
102
103    }
104
105    private Config parseConfig(CommandLine commandLine) throws IOException {
106
107        if(commandLine.hasOption("updatepassword")) {
108            new KeyUpdateParser(
109                KeyEncryptorFactory.create(),
110                PasswordReaderFactory.create()
111            ).parse(commandLine);
112
113            //return early so other options don't get processed
114            return null;
115        }
116
117        final List<ConfigKeyPair> newKeys = new KeyGenerationParser().parse(commandLine);
118
119        final Config config = new ConfigurationParser().withNewKeys(newKeys).parse(commandLine);
120
121        if (!commandLine.hasOption("configfile") && !commandLine.hasOption("keygen") && !commandLine.hasOption("updatepassword")) {
122            throw new CliException("One or more: -configfile or -keygen or -updatepassword options are required.");
123        }
124
125        return config;
126    }
127
128    private Options buildBaseOptions() {
129
130        final Options options = new Options();
131
132        options.addOption(
133            Option.builder("configfile")
134                .desc("Path to node configuration file")
135                .hasArg(true)
136                .optionalArg(false)
137                .numberOfArgs(1)
138                .argName("PATH")
139                .build());
140
141        options.addOption(
142            Option.builder("keygen")
143                .desc("Use this option to generate public/private keypair")
144                .hasArg(false)
145                .build());
146
147        options.addOption(
148            Option.builder("filename")
149                .desc("Comma-separated list of paths to save generated key files. Can also be used with keyvault. Number of args equals number of key-pairs generated.")
150                .hasArgs()
151                .optionalArg(false)
152                .argName("PATH")
153                .build());
154
155        options.addOption(
156            Option.builder("keygenconfig")
157                .desc("Path to private key config for generation of missing key files")
158                .hasArg(true)
159                .optionalArg(false)
160                .argName("PATH")
161                .build());
162
163        options.addOption(
164            Option.builder("output")
165                .desc("Generate updated config file with generated keys")
166                .hasArg(true)
167                .numberOfArgs(1)
168                .build());
169
170        options.addOption(
171            Option.builder("keygenvaulttype")
172                .desc("Type of key vault the generated key is to be saved in")
173                .hasArg()
174                .optionalArg(false)
175                .argName("KEYVAULTTYPE")
176                .build()
177        );
178
179        options.addOption(
180            Option.builder("keygenvaulturl")
181                .desc("Base url for key vault")
182                .hasArg()
183                .optionalArg(false)
184                .argName("STRING")
185                .build()
186        );
187
188        options.addOption(
189            Option.builder("keygenvaultapprole")
190                  .desc("AppRole path for Hashicorp Vault authentication (defaults to 'approle')")
191                  .hasArg()
192                  .optionalArg(false)
193                  .argName("STRING")
194                  .build()
195        );
196
197        options.addOption(
198            Option.builder("keygenvaultkeystore")
199                .desc("Path to JKS keystore for TLS Hashicorp Vault communication")
200                .hasArg()
201                .optionalArg(false)
202                .argName("PATH")
203                .build()
204        );
205
206        options.addOption(
207            Option.builder("keygenvaulttruststore")
208                  .desc("Path to JKS truststore for TLS Hashicorp Vault communication")
209                  .hasArg()
210                  .optionalArg(false)
211                  .argName("PATH")
212                  .build()
213        );
214
215        options.addOption(
216            Option.builder("keygenvaultsecretengine")
217                  .desc("Name of already enabled Hashicorp v2 kv secret engine")
218                  .hasArg()
219                  .optionalArg(false)
220                  .argName("STRING")
221                  .build()
222        );
223
224        options.addOption(
225            Option.builder("pidfile")
226                .desc("Path to pid file")
227                .hasArg(true)
228                .optionalArg(false)
229                .numberOfArgs(1)
230                .argName("PATH")
231                .build());
232
233        options.addOption(
234            Option.builder("updatepassword")
235                .desc("Update the password for a locked key")
236                .hasArg(false)
237                .build()
238        );
239
240        return options;
241    }
