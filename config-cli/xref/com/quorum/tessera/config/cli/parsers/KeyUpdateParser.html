29  public class KeyUpdateParser implements Parser<Optional> {
30  
31      private static final Logger LOGGER = LoggerFactory.getLogger(KeyUpdateParser.class);
32  
33      private final KeyEncryptor keyEncryptor;
34  
35      private final PasswordReader passwordReader;
36  
37      public KeyUpdateParser(final KeyEncryptor keyEncryptor, final PasswordReader passwordReader) {
38          this.keyEncryptor = Objects.requireNonNull(keyEncryptor);
39          this.passwordReader = Objects.requireNonNull(passwordReader);
40      }
41  
42      @Override
43      public Optional parse(final CommandLine commandLine) throws IOException {
44          if (commandLine.hasOption("updatepassword")) {
45              final ArgonOptions argonOptions = argonOptions(commandLine);
46              final List<String> passwords = passwords(commandLine);
47              final Path keypath = privateKeyPath(commandLine);
48  
49              final KeyDataConfig keyDataConfig = JaxbUtil.unmarshal(Files.newInputStream(keypath), KeyDataConfig.class);
50              final PrivateKey privateKey = this.getExistingKey(keyDataConfig, passwords);
51  
52              final String newPassword = passwordReader.requestUserPassword();
53  
54              final KeyDataConfig updatedKey;
55              if(newPassword.isEmpty()) {
56                  final PrivateKeyData privateKeyData = new PrivateKeyData(privateKey.encodeToBase64(), null, null, null, null);
57                  updatedKey = new KeyDataConfig(privateKeyData, PrivateKeyType.UNLOCKED);
58              } else {
59                  final PrivateKeyData privateKeyData = keyEncryptor.encryptPrivateKey(privateKey, newPassword, argonOptions);
60                  updatedKey = new KeyDataConfig(privateKeyData, PrivateKeyType.LOCKED);
61              }
62  
63              //write the key to file
64              Files.write(keypath, JaxbUtil.marshalToString(updatedKey).getBytes(UTF_8));
65              SystemAdapter.INSTANCE.out().println("Private key at " + keypath.toString() + " updated.");
66          }
67  
68          return Optional.empty();
69      }
70  
71      PrivateKey getExistingKey(final KeyDataConfig kdc, final List<String> passwords) {
72  
73          if (kdc.getType() == PrivateKeyType.UNLOCKED) {
74              byte[] privateKeyData = Base64.getDecoder().decode(kdc.getValue().getBytes(UTF_8));
75              return PrivateKey.from(privateKeyData);
76          } else {
77  
78              for (final String pass : passwords) {
79                  try {
80                      return PrivateKey.from(keyEncryptor.decryptPrivateKey(kdc.getPrivateKeyData(), pass).getKeyBytes());
81                  } catch (final Exception e) {
82                      LOGGER.debug("Password failed to decrypt. Trying next if available.");
83                  }
84              }
85  
86              throw new IllegalArgumentException("Locked key but no valid password given");
87          }
88      }
89  
90      static Path privateKeyPath(final CommandLine commandLine) {
91          final String privateKeyPath = commandLine.getOptionValue("keys.keyData.privateKeyPath");
92  
93          if (privateKeyPath == null) {
94              throw new IllegalArgumentException("Private key path cannot be null when updating key password");
95          }
96  
97          final Path keypath = Paths.get(privateKeyPath);
98          if (Files.notExists(keypath)) {
99              throw new IllegalArgumentException("Private key path must exist when updating key password");
100         }
101 
102         return keypath;
103     }
104 
105     static List<String> passwords(final CommandLine commandLine) throws IOException {
106         final String password = commandLine.getOptionValue("keys.passwords");
107         final String passwordFile = commandLine.getOptionValue("keys.passwordFile");
108 
109         if (password != null) {
110             return singletonList(password);
111         } else if (passwordFile != null) {
112             return Files.readAllLines(Paths.get(passwordFile));
113         } else {
114             return emptyList();
115         }
116 
117     }
118 
119     static ArgonOptions argonOptions(final CommandLine commandLine) {
120         final String algorithm = commandLine.getOptionValue("keys.keyData.config.data.aopts.algorithm", "i");
121         final String iterations = commandLine.getOptionValue("keys.keyData.config.data.aopts.iterations", "10");
122         final String memory = commandLine.getOptionValue("keys.keyData.config.data.aopts.memory", "1048576");
123         final String parallelism = commandLine.getOptionValue("keys.keyData.config.data.aopts.parallelism", "4");
124 
125         return new ArgonOptions(
126             algorithm, Integer.valueOf(iterations), Integer.valueOf(memory), Integer.valueOf(parallelism)
127         );
128     }
