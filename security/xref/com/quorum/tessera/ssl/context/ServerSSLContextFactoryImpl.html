10  import org.bouncycastle.operator.OperatorCreationException;
11  
12  import javax.net.ssl.SSLContext;
13  import java.io.IOException;
14  import java.nio.file.Path;
15  import java.nio.file.Paths;
16  import java.security.GeneralSecurityException;
17  import java.util.Optional;
18  
19  public class ServerSSLContextFactoryImpl implements ServerSSLContextFactory {
20  
21      private static final String DEFAULT_KNOWN_CLIENT_FILEPATH = "knownClients";
22  
23      private static final EnvironmentVariableProvider envVarProvider = EnvironmentVariableProviderFactory.load().create();
24  
25      @Override
26      public SSLContext from(String address, SslConfig sslConfig) {
27  
28          TrustMode trustMode = TrustMode
29              .getValueIfPresent(sslConfig.getServerTrustMode().name())
30              .orElse(TrustMode.NONE);
31  
32          final Path knownClientsFile = Optional.ofNullable(sslConfig.getKnownClientsFile())
33              .orElse(Paths.get(DEFAULT_KNOWN_CLIENT_FILEPATH));
34  
35          final SSLContextProperties properties = new SSLContextProperties(
36              address,
37              sslConfig.getServerKeyStore(),
38              getServerKeyStorePassword(sslConfig),
39              sslConfig.getServerTlsKeyPath(),
40              sslConfig.getServerTlsCertificatePath(),
41              sslConfig.getServerTrustStore(),
42              getServerTrustStorePassword(sslConfig),
43              sslConfig.getServerTrustCertificates(),
44              knownClientsFile
45          );
46  
47          try {
48              return trustMode.createSSLContext(properties);
49          } catch (IOException | OperatorCreationException | GeneralSecurityException ex) {
50              throw new TesseraSecurityException(ex);
51          }
52      }
53  
54      // TODO - Package private for testing, refactor so this can be made private
55      String getServerKeyStorePassword(SslConfig sslConfig) {
56          return getPreferredPassword(sslConfig.getServerKeyStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.SERVER_KEYSTORE_PWD);
57      }
58  
59      // TODO - Package private for testing, refactor so this can be made private
60      String getServerTrustStorePassword(SslConfig sslConfig) {
61          return getPreferredPassword(sslConfig.getServerTrustStorePassword(), sslConfig.getEnvironmentVariablePrefix(), EnvironmentVariables.SERVER_TRUSTSTORE_PWD);
62      }
63  
64      // Return the prefixed env var value if set, else return the config value, else return the global env var value
65      private String getPreferredPassword(String configPassword, String envVarPrefix, String envVar) {
66          String password = envVarProvider.getEnv(envVarPrefix + "_" + envVar);
67  
68          if(password != null) {
69              return password;
70          } else if(configPassword != null) {
71              return configPassword;
72          }
73  
74          return envVarProvider.getEnv(envVar);
75      }
76  }
