27  public interface TlsUtils {

29  
30      String ENCRYPTION = "RSA";
31      String COMMON_NAME_STRING = "CN=";
32      String SIGNATURE_ALGORITHM = "SHA512WithRSAEncryption";
33      String KEYSTORE_TYPE = "JKS";
34  
35      // These localhost values are added in SAN names for dev. They can be removed for production
36      // if necessary
37      String LOCALHOST = "localhost";
38      String LOCALHOST_IP = "127.0.0.1";
39      String LOCALHOST_IP_2 = "0.0.0.0";
40  
41      Provider provider = new BouncyCastleProvider();
42  
43      default void generateKeyStoreWithSelfSignedCertificate(String address, Path privateKeyFile, String password)
44          throws NoSuchAlgorithmException, IOException, OperatorCreationException,
45          CertificateException, InvalidKeyException, NoSuchProviderException, SignatureException, KeyStoreException {
46  
47          final SecureRandom secureRandom = new SecureRandom();
48  
49          KeyPairGenerator keyGen = KeyPairGenerator.getInstance(ENCRYPTION);
50          keyGen.initialize(2048, secureRandom);
51          KeyPair keypair = keyGen.generateKeyPair();
52          final PublicKey publicKey = keypair.getPublic();
53          final PrivateKey privateKey = keypair.getPrivate();
54          final String cnString = address.replaceFirst("^(http[s]?://www\\.|http[s]?://|www\\.)","");
55          final X500Name commonName = new X500Name(COMMON_NAME_STRING + cnString);
56          Date startDate = new Date(System.currentTimeMillis());
57          Calendar calendar = Calendar.getInstance();
58          calendar.setTime(startDate);
59          calendar.add(Calendar.YEAR, 1);
60          Date endDate = calendar.getTime();
61  
62          X509v3CertificateBuilder builder = new JcaX509v3CertificateBuilder(
63              commonName,
64              new BigInteger(64, secureRandom),
65              startDate,
66              endDate,
67              commonName,
68              publicKey);
69  
70          GeneralName[] subjectAlternativeNames = new GeneralName[] {
71              new GeneralName(GeneralName.dNSName, LOCALHOST),
72              new GeneralName(GeneralName.dNSName, HostnameUtil.create().getHostName()),
73              new GeneralName(GeneralName.iPAddress, LOCALHOST_IP),
74              new GeneralName(GeneralName.iPAddress, LOCALHOST_IP_2),
75              new GeneralName(GeneralName.iPAddress, HostnameUtil.create().getHostIpAddress())
76          };
77  
78          builder.addExtension(
79              Extension.subjectAlternativeName,
80              false,
81              new GeneralNames(subjectAlternativeNames));
82  
83          ContentSigner contentSigner = new JcaContentSignerBuilder(SIGNATURE_ALGORITHM).build(privateKey);
84          X509CertificateHolder certHolder = builder.build(contentSigner);
85          X509Certificate certificate =
86              new JcaX509CertificateConverter().setProvider(provider).getCertificate(certHolder);
87  
88          certificate.verify(publicKey);
89  
90          KeyStore keyStore = KeyStore.getInstance(KEYSTORE_TYPE);
91          keyStore.load(null, null);
92          keyStore.setKeyEntry("tessera",privateKey, password.toCharArray(), new X509Certificate[]{certificate});
93  
94          try(OutputStream keyStoreFile = Files.newOutputStream(privateKeyFile)) {
95              keyStore.store(keyStoreFile, password.toCharArray());
96          }
97      }
98  
99      static TlsUtils create(){
100         Security.addProvider(new BouncyCastleProvider());
101         return new TlsUtils() {
102         };
103     }
