001package com.quorum.tessera.server;
002
003import com.quorum.tessera.config.CommunicationType;
004import com.quorum.tessera.config.ServerConfig;
005
006import javax.ws.rs.core.Application;
007import java.util.Set;
008
009/**
010 * Creates Grizzly and Jersey implementations of the {@link TesseraServer}
011 */
012public class JerseyServerFactory implements TesseraServerFactory<Object> {
013
014    @Override
015    public TesseraServer createServer(ServerConfig serverConfig, Set<Object> services) {
016        Application application = services.stream()
017                .filter(Application.class::isInstance)
018                .filter(serverConfig.getApp().getIntf()::isInstance)
019                .findFirst()
020                .map(Application.class::cast)
021                .get();
022
023        return new JerseyServer(serverConfig, application);
024    }
025
026    @Override
027    public CommunicationType communicationType() {
028        return CommunicationType.REST;
029    }
030
031}
