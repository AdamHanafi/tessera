001package com.quorum.tessera.server;
002
003import com.jpmorgan.quorum.server.utils.ServerUtils;
004import com.quorum.tessera.config.InfluxConfig;
005import com.quorum.tessera.config.ServerConfig;
006import com.quorum.tessera.server.monitoring.InfluxDbClient;
007import com.quorum.tessera.server.monitoring.InfluxDbPublisher;
008import com.quorum.tessera.server.monitoring.MetricsResource;
009import org.glassfish.jersey.server.ResourceConfig;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012import org.slf4j.bridge.SLF4JBridgeHandler;
013import javax.ws.rs.core.Application;
014import java.net.URI;
015import java.util.HashMap;
016import java.util.Map;
017import java.util.Objects;
018import java.util.concurrent.ScheduledExecutorService;
019import java.util.concurrent.TimeUnit;
020
021import static java.util.concurrent.Executors.newSingleThreadScheduledExecutor;
022import org.eclipse.jetty.servlet.ServletContextHandler;
023import org.eclipse.jetty.servlet.ServletHolder;
024import org.glassfish.jersey.servlet.ServletContainer;
025
026/**
027 * Implementation of a RestServer using Jersey and Jetty.
028 */
029public class JerseyServer implements TesseraServer {
030
031    private static final Logger LOGGER = LoggerFactory.getLogger(JerseyServer.class);
032
033    private org.eclipse.jetty.server.Server server;
034
035    private final URI uri;
036
037    private final Application application;
038
039    private final ScheduledExecutorService executor;
040
041    private final InfluxConfig influxConfig;
042
043    private final ServerConfig serverConfig;
044
045    public JerseyServer(final ServerConfig serverConfig, final Application application) {
046        this.uri = serverConfig.getBindingUri();
047        this.application = Objects.requireNonNull(application);
048        this.serverConfig = serverConfig;
049
050        this.executor = newSingleThreadScheduledExecutor();
051
052        if (serverConfig.getInfluxConfig() != null) {
053            this.influxConfig = serverConfig.getInfluxConfig();
054        } else {
055            this.influxConfig = null;
056        }
057    }
058
059    @Override
060    public void start() throws Exception {
061
062        SLF4JBridgeHandler.removeHandlersForRootLogger();
063        SLF4JBridgeHandler.install();
064
065        //https://jersey.github.io/documentation/latest/appendix-properties.html
066        final Map<String, Object> initParams = new HashMap<>();
067        initParams.put("jersey.config.server.application.name", application.getClass().getSimpleName());
068        initParams.put("jersey.config.server.tracing.type", "ON_DEMAND");
069        initParams.put("jersey.config.server.tracing.threshold", "SUMMARY");
070        initParams.put("jersey.config.logging.verbosity", "PAYLOAD_ANY");
071        initParams.put("jersey.config.beanValidation.enableOutputValidationErrorEntity.server", "true");
072        initParams.put("jersey.config.server.monitoring.statistics.enabled", "true");
073        initParams.put("jersey.config.server.monitoring.enabled", "true");
074        initParams.put("jersey.config.server.monitoring.statistics.mbeans.enabled", "true");
075
076        final ResourceConfig config = ResourceConfig.forApplication(application);
077
078        config.addProperties(initParams)
079                .register(MetricsResource.class);
080
081        this.server = ServerUtils.buildWebServer(serverConfig);
082
083        ServletContextHandler context = new ServletContextHandler(server, "/");
084        ServletContainer servletContainer = new ServletContainer(config);
085        ServletHolder jerseyServlet = new ServletHolder(servletContainer);
086
087        context.addServlet(jerseyServlet, "/*");
088
089        LOGGER.info("Starting {}", uri);
090
091        this.server.start();
092
093        LOGGER.info("Started {}", uri);
094        LOGGER.info("WADL {}/application.wadl", uri);
095
096        if (influxConfig != null) {
097            startInfluxMonitoring();
098        }
099
100    }
101
102    private void startInfluxMonitoring() {
103        InfluxDbClient influxDbClient = new InfluxDbClient(this.uri, influxConfig);
104        Runnable publisher = new InfluxDbPublisher(influxDbClient);
105
106        final Runnable exceptionSafePublisher = () -> {
107            try{
108                publisher.run();
109            } catch (final Throwable ex) {
110                LOGGER.error("Error when executing action {}", publisher.getClass().getSimpleName());
111                LOGGER.error("Error when executing action", ex);
112            }
113        };
114
115        final long delayInSecs = influxConfig.getPushIntervalInSecs();
116        this.executor.scheduleWithFixedDelay(exceptionSafePublisher, delayInSecs, delayInSecs, TimeUnit.SECONDS);
117    }
118
119    @Override
120    public void stop() {
121        LOGGER.info("Stopping Jersey server at {}", uri);
122
123        if (influxConfig != null) {
124            this.executor.shutdown();
125        }
126
127        if (Objects.nonNull(this.server)) {
128            try{
129                this.server.stop();
130            } catch (Exception ex) {
131                LOGGER.warn(null, ex);
132            }
133        }
134
135        LOGGER.info("Stopped Jersey server at {}", uri);
136    }
137
138}
