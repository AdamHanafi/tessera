001package com.quorum.tessera.jaxrs.unixsocket;
002
003import java.io.ByteArrayInputStream;
004import java.io.ByteArrayOutputStream;
005import java.io.IOException;
006import java.net.URI;
007import java.nio.file.Path;
008import java.nio.file.Paths;
009import java.util.Objects;
010import java.util.concurrent.CompletableFuture;
011import java.util.concurrent.Future;
012
013import javax.ws.rs.ProcessingException;
014import javax.ws.rs.core.MultivaluedMap;
015import javax.ws.rs.core.Response;
016import javax.ws.rs.core.UriBuilder;
017import org.glassfish.jersey.client.ClientRequest;
018import org.glassfish.jersey.client.ClientResponse;
019import org.glassfish.jersey.client.spi.AsyncConnectorCallback;
020import org.glassfish.jersey.client.spi.Connector;
021import org.eclipse.jetty.client.HttpClient;
022import org.eclipse.jetty.client.api.ContentProvider;
023import org.eclipse.jetty.client.api.ContentResponse;
024import org.eclipse.jetty.client.api.Request;
025import org.eclipse.jetty.client.util.BytesContentProvider;
026import org.eclipse.jetty.http.HttpMethod;
027import org.eclipse.jetty.unixsocket.client.HttpClientTransportOverUnixSockets;
028import org.glassfish.jersey.message.internal.Statuses;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032public class JerseyUnixSocketConnector implements Connector {
033
034    private static final Logger LOGGER = LoggerFactory.getLogger(JerseyUnixSocketConnector.class);
035
036    private HttpClient httpClient;
037
038    private URI unixfile;
039
040    public JerseyUnixSocketConnector(URI unixfile) {
041        this.unixfile = unixfile;
042        String unixFilePath = Paths.get(unixfile).toFile().getAbsolutePath();
043
044        httpClient = new HttpClient(new HttpClientTransportOverUnixSockets(unixFilePath), null);
045        try{
046            httpClient.start();
047        } catch (Exception ex) {
048            throw new RuntimeException(ex);
049        }
050    }
051
052    @Override
053    public ClientResponse apply(ClientRequest request) {
054
055        try{
056            return doApply(request);
057        } catch (Exception ex) {
058            throw new ProcessingException(ex);
059        }
060
061    }
062
063    private ClientResponse doApply(ClientRequest request) throws Exception {
064
065        HttpMethod httpMethod = HttpMethod.valueOf(request.getMethod());
066        final URI originalUri = request.getUri();
067        final URI uri;
068        Path basePath = Paths.get(unixfile);
069
070        if (originalUri.getScheme().startsWith("unix")) {
071            
072            String path = originalUri.getRawPath()
073                    .replaceFirst(basePath.toString(), "");
074
075            LOGGER.trace("Extracted path {} from {}",path, originalUri.getRawPath());
076
077            uri = UriBuilder.fromUri(originalUri)
078                    .replacePath(path)
079                    .scheme("http")
080                    .port(99)
081                    .host("localhost")
082                    .build();
083                        
084            LOGGER.trace("Created psuedo uri {} for originalUri {}", uri, originalUri);
085        } else {
086            uri = originalUri;
087        }
088
089        Request clientRequest = httpClient.newRequest(uri)
090                .method(httpMethod);
091
092        MultivaluedMap<String, Object> headers = request.getHeaders();
093
094        headers.keySet().stream().forEach(name -> {
095            headers.get(name).forEach(value -> {
096                clientRequest.header(name, Objects.toString(value));
097            });
098
099        });
100
101        if (request.hasEntity()) {
102            final long length = request.getLengthLong();
103
104            try (ByteArrayOutputStream bout = new ByteArrayOutputStream()){
105
106                request.setStreamProvider((int contentLength) -> bout);
107                request.writeEntity();
108
109                ContentProvider content = new BytesContentProvider(bout.toByteArray());
110                clientRequest.content(content);
111            }
112
113        }
114        final ContentResponse contentResponse = clientRequest.send();
115
116        int statusCode = contentResponse.getStatus();
117        String reason = contentResponse.getReason();
118
119        LOGGER.trace("uri {}, method: {},statusCode:{},reason: {} ", uri, httpMethod, statusCode, reason);
120
121        final Response.StatusType status = Statuses.from(statusCode, reason);
122
123        ClientResponse response = new ClientResponse(status, request);
124        contentResponse.getHeaders().stream()
125                .forEach(header -> {
126                    response.headers(header.getName(), (Object[]) header.getValues());
127                });
128
129        response.setEntityStream(new ByteArrayInputStream(contentResponse.getContent()));
130        return response;
131
132    }
133
134    @Override
135    public Future<?> apply(final ClientRequest request, final AsyncConnectorCallback callback) {
136
137        try{
138            callback.response(doApply(request));
139        } catch (IOException ex) {
140            callback.failure(new ProcessingException(ex));
141        } catch (Throwable t) {
142            callback.failure(t);
143        }
144
145        return CompletableFuture.completedFuture(null);
146    }
147
148    @Override
149    public String getName() {
150        return getClass().getSimpleName();
151    }
152
153    @Override
154    public void close() {
155        try{
156            httpClient.stop();
157        } catch (Exception ex) {
158
159        }
160    }
161
162}
