49      @Test
50      public void ping() {
51  
52          Response result = newClient(unixfile)
53                  .target(URI.create("http://localhost:88"))
54                  .path("ping")
55                  .request()
56                  .get();
57  
58          assertThat(result.getStatus()).isEqualTo(200);
59          assertThat(result.readEntity(String.class)).isEqualTo("HEllow");

62      @Test
63      public void create() {
64  
65          SamplePayload payload = new SamplePayload();
66          payload.setValue("Hellow");
67  
68          Response result = newClient(unixfile)
69                  .target(unixfile)
70                  .path("create")
71                  .request()
72                  .post(Entity.entity(payload, MediaType.APPLICATION_JSON));
73  
74          assertThat(result.getStatus()).isEqualTo(201);
75          assertThat(result.getLocation()).isNotNull();
76  
77          Response result2 = newClient(unixfile)
78                  .target(result.getLocation())
79                  .request(MediaType.APPLICATION_JSON)
80                  .get();
81  
82          SamplePayload p = result2.readEntity(SamplePayload.class);
83          assertThat(p).isNotNull();
84          assertThat(p.getValue()).isEqualTo("Hellow");
85  
86          Response result3 = newClient(unixfile)
87                  .target(unixfile)
88                  .path(p.getId()).request().delete();
89  
90          assertThat(result3.getStatus()).isEqualTo(200);
91          SamplePayload deleted = result3.readEntity(SamplePayload.class);
92          assertThat(deleted.getValue()).isEqualTo("Hellow");
93      }
