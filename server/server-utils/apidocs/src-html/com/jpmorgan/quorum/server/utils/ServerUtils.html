001package com.jpmorgan.quorum.server.utils;
002
003import com.quorum.tessera.config.ServerConfig;
004import com.quorum.tessera.ssl.context.ServerSSLContextFactory;
005import java.net.URI;
006import java.nio.file.Paths;
007import javax.net.ssl.SSLContext;
008import org.eclipse.jetty.server.Connector;
009import org.eclipse.jetty.server.HttpConfiguration;
010import org.eclipse.jetty.server.HttpConnectionFactory;
011import org.eclipse.jetty.server.SecureRequestCustomizer;
012import org.eclipse.jetty.server.Server;
013import org.eclipse.jetty.server.ServerConnector;
014import org.eclipse.jetty.server.SslConnectionFactory;
015import org.eclipse.jetty.util.ssl.SslContextFactory;
016import org.eclipse.jetty.unixsocket.UnixSocketConnector;
017
018public class ServerUtils {
019
020    public static Server buildWebServer(ServerConfig serverConfig) {
021        Server server = new Server();
022        URI uri = serverConfig.getBindingUri();
023
024        if (serverConfig.isUnixSocket()) {
025
026            HttpConnectionFactory httpConnectionFactory = new HttpConnectionFactory();
027
028            UnixSocketConnector connector = new UnixSocketConnector(server, httpConnectionFactory);
029            connector.setAcceptQueueSize(128);
030            
031            String path = Paths.get(serverConfig.getServerUri()).toAbsolutePath().toString();
032            connector.setUnixSocket(path);
033
034            server.setConnectors(new Connector[]{connector});
035
036            return server;
037        }
038
039        if (serverConfig.isSsl()) {
040            HttpConfiguration https = new HttpConfiguration();
041            https.addCustomizer(new SecureRequestCustomizer());
042
043            SSLContext sslContext = ServerSSLContextFactory.create()
044                    .from(uri.toString(), serverConfig.getSslConfig());
045
046            SslContextFactory sslContextFactory = new SslContextFactory();
047            sslContextFactory.setSslContext(sslContext);
048            sslContextFactory.setNeedClientAuth(true);
049            ServerConnector connector = new ServerConnector(server,
050                    new SslConnectionFactory(sslContextFactory, "http/1.1"),
051                    new HttpConnectionFactory(https));
052            connector.setPort(uri.getPort());
053            server.setConnectors(new Connector[]{connector});
054            return server;
055        }
056
057        ServerConnector connector = new ServerConnector(server);
058        connector.setPort(uri.getPort());
059        server.setConnectors(new Connector[]{connector});
060
061        return server;
062    }
063}
