001package com.quorum.tessera.grpc.server;
002
003import com.quorum.tessera.server.TesseraServer;
004import io.grpc.Server;
005import org.slf4j.LoggerFactory;
006
007import java.io.IOException;
008import java.net.URI;
009
010public class GrpcServer implements TesseraServer {
011
012    private static final org.slf4j.Logger LOGGER = LoggerFactory.getLogger(GrpcServer.class);
013
014    private final URI uri;
015
016    private final Server server;
017
018    public GrpcServer(URI uri, Server server) {
019        this.uri = uri;
020        this.server = server;
021    }
022
023    @Override
024    public void start() throws IOException {
025        try {
026            server.start();
027            LOGGER.info("gRPC server started, listening on " + uri.getPort());
028            Runtime.getRuntime().addShutdownHook(new Thread() {
029                @Override
030                public void run() {
031                    LOGGER.info("*** Shutting down gRPC server");
032                    GrpcServer.this.stop();
033                    LOGGER.info("*** gRPC server shut down");
034                }
035            });
036        } catch (IOException ex) {
037            LOGGER.error("Cannot start gRPC server. See cause ", ex);
038            throw ex;
039        }
040    }
041
042    @Override
043    public void stop() {
044        if (server != null) {
045            server.shutdown();
046        }
047    }
048
049}
