1   package com.quorum.tessera.config.keypairs;

3   import com.quorum.tessera.config.KeyDataConfig;
4   import com.quorum.tessera.config.adapters.PathAdapter;
5   import com.quorum.tessera.config.constraints.ValidBase64;
6   import com.quorum.tessera.config.constraints.ValidContent;
7   import com.quorum.tessera.config.constraints.ValidPath;
8   import com.quorum.tessera.config.util.JaxbUtil;
9   import com.quorum.tessera.io.IOCallback;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  import javax.validation.constraints.NotNull;
14  import javax.validation.constraints.Pattern;
15  import javax.validation.constraints.Size;
16  import javax.xml.bind.annotation.XmlElement;
17  import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
18  import java.nio.file.Files;
19  import java.nio.file.Path;
20  
21  import static java.nio.charset.StandardCharsets.UTF_8;
22  
23  public class FilesystemKeyPair implements ConfigKeyPair {
24  
25      private static final Logger LOGGER = LoggerFactory.getLogger(FilesystemKeyPair.class);
26  
27      @ValidContent(minLines = 1, maxLines = 1, message = "file expected to contain a single non empty value")
28      @NotNull
29      @ValidPath(checkExists = true, message = "File does not exist")
30      @XmlElement
31      @XmlJavaTypeAdapter(PathAdapter.class)
32      private final Path publicKeyPath;
33  
34      @ValidContent(minLines = 1, message = "file expected to contain at least one line")
35      @NotNull
36      @ValidPath(checkExists = true, message = "File does not exist")
37      @XmlElement
38      @XmlJavaTypeAdapter(PathAdapter.class)
39      private final Path privateKeyPath;
40  
41      private InlineKeypair inlineKeypair;
42  
43      private String password;
44  
45      public FilesystemKeyPair(final Path publicKeyPath, final Path privateKeyPath) {
46          this.publicKeyPath = publicKeyPath;
47          this.privateKeyPath = privateKeyPath;
48  
49          try {
50              loadKeys();
51          } catch (final Exception ex) {
52              //silently discard errors as these get picked up by the validator
53              LOGGER.debug("Unable to read key files", ex);
54          }
55      }
56  
57      @Override
58      @Size(min = 1)
59      @ValidBase64(message = "Invalid Base64 key provided")
60      public String getPublicKey() {
61          if (this.inlineKeypair == null) {
62              return null;
63          }
64          return this.inlineKeypair.getPublicKey();
65      }
66  
67      @Override
68      @Size(min = 1)
69      @ValidBase64(message = "Invalid Base64 key provided")
70      @Pattern(regexp = "^((?!NACL_FAILURE).)*$", message = "Could not decrypt the private key with the provided password, please double check the passwords provided")
71      @Pattern(regexp = "^((?!MISSING_PASSWORD).)*$", message = "{InlineKeyData.missingPassword.message}")
72      public String getPrivateKey() {
73          if (this.inlineKeypair == null) {
74              return null;
75          }
76          return this.inlineKeypair.getPrivateKey();
77      }
78  
79      @Override
80      public void withPassword(final String password) {
81          this.password = password;
82          if (this.inlineKeypair != null) {
83              this.inlineKeypair.withPassword(this.password);
84          }
85      }
86  
87      @Override
88      public String getPassword() {
89          return this.password;
90      }
91  
92      public Path getPublicKeyPath() {
93          return publicKeyPath;
94      }
95  
96      public Path getPrivateKeyPath() {
97          return privateKeyPath;
98      }
99  
100     public InlineKeypair getInlineKeypair() {
101         return inlineKeypair;
102     }
103 
104     private void loadKeys() {
105         this.inlineKeypair = new InlineKeypair(
106             IOCallback.execute(() -> new String(Files.readAllBytes(this.publicKeyPath), UTF_8)),
107             JaxbUtil.unmarshal(
108                 IOCallback.execute(() -> Files.newInputStream(privateKeyPath)),
109                 KeyDataConfig.class
110             )
111         );
112     }
113 
114 }
