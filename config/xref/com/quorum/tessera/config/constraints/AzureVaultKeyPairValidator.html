9   public class AzureVaultKeyPairValidator implements ConstraintValidator<ValidAzureVaultKeyPair, AzureVaultKeyPair> {
10  
11      @Override
12      public boolean isValid(final AzureVaultKeyPair keypair, final ConstraintValidatorContext cvc) {
13          if (keypair == null) {
14              return true;
15          }
16  
17          return Objects.isNull(keypair.getPublicKeyVersion()) == Objects.isNull(keypair.getPrivateKeyVersion());
18      }
