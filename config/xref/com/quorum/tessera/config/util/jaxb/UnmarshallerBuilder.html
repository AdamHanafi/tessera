1   package com.quorum.tessera.config.util.jaxb;
2   
3   import com.quorum.tessera.jaxb.JaxbCallback;
4   import com.quorum.tessera.config.util.JaxbUtil;
5   import javax.xml.bind.JAXBContext;
6   import javax.xml.bind.Unmarshaller;
7   
8   public class UnmarshallerBuilder {
9   
10      public static UnmarshallerBuilder create() {
11          return new UnmarshallerBuilder();
12      }
13  
14      private boolean beanvalidation = true;
15  
16      private MediaType mediaType = MediaType.JSON;
17  
18      public UnmarshallerBuilder withoutBeanValidation() {
19          this.beanvalidation = false;
20          return this;
21      }
22  
23      public UnmarshallerBuilder withXmlMediaType() {
24          this.mediaType = MediaType.XML;
25          return this;
26      }
27  
28      public Unmarshaller build() {
29  
30          return JaxbCallback.execute(() -> {
31              JAXBContext jAXBContext = JAXBContext.newInstance(JaxbUtil.JAXB_CLASSES);
32  
33              Unmarshaller unmarshaller = jAXBContext.createUnmarshaller();
34              if (!beanvalidation) {
35                  Enum enu = Enum.valueOf(Class.class.cast(unmarshaller
36                          .getProperty("eclipselink.beanvalidation.mode")
37                          .getClass()), "NONE");
38  
39                  unmarshaller.setProperty("eclipselink.beanvalidation.mode", enu);
40              }
41              unmarshaller.setProperty("eclipselink.media-type", mediaType.getValue());
42  
43              if (mediaType == MediaType.JSON) {
44                  unmarshaller.setProperty("eclipselink.json.include-root", false);
45              }
46              return unmarshaller;
47          });
48      }
49  }
