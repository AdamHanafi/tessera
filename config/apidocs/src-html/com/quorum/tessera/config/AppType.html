001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.apps.*;
004
005import javax.xml.bind.annotation.XmlEnumValue;
006import java.util.Arrays;
007import java.util.HashSet;
008import java.util.Set;
009
010import static java.util.Collections.singleton;
011
012public enum AppType {
013    P2P(P2PApp.class,
014        new HashSet<>(Arrays.asList(CommunicationType.GRPC, CommunicationType.REST))),
015    Q2T(Q2TApp.class,
016        new HashSet<>(Arrays.asList(CommunicationType.GRPC, CommunicationType.REST))
017        ),
018    @XmlEnumValue("ThirdParty")
019    THIRD_PARTY(ThirdPartyApp.class,
020        singleton(CommunicationType.REST)),
021    ENCLAVE(EnclaveApp.class,
022        singleton(CommunicationType.REST)
023    ),
024    ADMIN(AdminApp.class,
025        singleton(CommunicationType.REST)
026    );
027
028    private final Class<? extends TesseraApp> intf;
029    private final Set<CommunicationType> allowedCommunicationTypes;
030
031    AppType(Class<? extends TesseraApp> intf,
032            Set<CommunicationType> allowedCommunicationTypes) {
033        this.intf = intf;
034        this.allowedCommunicationTypes = allowedCommunicationTypes;
035    }
036
037    public Class<? extends TesseraApp> getIntf() {
038        return intf;
039    }
040
041    public Set<CommunicationType> getAllowedCommunicationTypes() {
042        return allowedCommunicationTypes;
043    }
044
045
046}
