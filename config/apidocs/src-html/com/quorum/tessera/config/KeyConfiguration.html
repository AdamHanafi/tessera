001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.adapters.KeyDataAdapter;
004import com.quorum.tessera.config.adapters.PathAdapter;
005import com.quorum.tessera.config.constraints.ValidPath;
006import com.quorum.tessera.config.keypairs.ConfigKeyPair;
007
008import javax.validation.Valid;
009import javax.validation.constraints.NotNull;
010import javax.validation.constraints.Size;
011import javax.xml.bind.annotation.XmlAccessType;
012import javax.xml.bind.annotation.XmlAccessorType;
013import javax.xml.bind.annotation.XmlElement;
014import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
015import java.nio.file.Path;
016import java.util.List;
017
018@XmlAccessorType(XmlAccessType.FIELD)
019public class KeyConfiguration extends ConfigItem {
020
021    @ValidPath(checkExists = true, message = "Password file does not exist")
022    @XmlElement(type = String.class)
023    @XmlJavaTypeAdapter(PathAdapter.class)
024    private Path passwordFile;
025
026    private List<String> passwords;
027
028    @Valid
029    @NotNull
030    @Size(min = 1, message = "At least 1 public/private key pair must be provided")
031    @XmlJavaTypeAdapter(KeyDataAdapter.class)
032    private List<@Valid ConfigKeyPair> keyData;
033
034    @Valid
035    @XmlElement
036    private AzureKeyVaultConfig azureKeyVaultConfig;
037
038    @Valid
039    @XmlElement
040    private HashicorpKeyVaultConfig hashicorpKeyVaultConfig;
041
042    public KeyConfiguration(final Path passwordFile, final List<String> passwords, final List<ConfigKeyPair> keyData, final AzureKeyVaultConfig azureKeyVaultConfig, final HashicorpKeyVaultConfig hashicorpKeyVaultConfig) {
043        this.passwordFile = passwordFile;
044        this.passwords = passwords;
045        this.keyData = keyData;
046        this.azureKeyVaultConfig = azureKeyVaultConfig;
047        this.hashicorpKeyVaultConfig = hashicorpKeyVaultConfig;
048    }
049
050    public KeyConfiguration() {
051    }
052
053    public Path getPasswordFile() {
054        return this.passwordFile;
055    }
056
057    public List<String> getPasswords() {
058        return this.passwords;
059    }
060
061    public List<ConfigKeyPair> getKeyData() {
062        return this.keyData;
063    }
064
065    public AzureKeyVaultConfig getAzureKeyVaultConfig() {
066        return this.azureKeyVaultConfig;
067    }
068
069    public HashicorpKeyVaultConfig getHashicorpKeyVaultConfig() {
070        return hashicorpKeyVaultConfig;
071    }
072
073    public void setPasswordFile(Path passwordFile) {
074        this.passwordFile = passwordFile;
075    }
076
077    public void setPasswords(List<String> passwords) {
078        this.passwords = passwords;
079    }
080
081    public void setKeyData(List<ConfigKeyPair> keyData) {
082        this.keyData = keyData;
083    }
084
085    public void setAzureKeyVaultConfig(AzureKeyVaultConfig azureKeyVaultConfig) {
086        this.azureKeyVaultConfig = azureKeyVaultConfig;
087    }
088
089    public void setHashicorpKeyVaultConfig(HashicorpKeyVaultConfig hashicorpKeyVaultConfig) {
090        this.hashicorpKeyVaultConfig = hashicorpKeyVaultConfig;
091    }
092
093}
