001package com.quorum.tessera.config.util;

003import com.quorum.tessera.io.IOCallback;
004import com.quorum.tessera.config.*;
005import com.quorum.tessera.config.util.jaxb.MarshallerBuilder;
006import com.quorum.tessera.config.util.jaxb.UnmarshallerBuilder;
007import java.io.ByteArrayOutputStream;
008
009import javax.xml.bind.JAXBException;
010import javax.xml.bind.Marshaller;
011import javax.xml.bind.Unmarshaller;
012import javax.xml.transform.stream.StreamSource;
013import java.io.InputStream;
014import java.io.OutputStream;
015import java.io.StringReader;
016import java.io.StringWriter;
017import java.util.Objects;
018import java.util.Optional;
019import javax.validation.ConstraintViolationException;
020import javax.xml.transform.Transformer;
021import javax.xml.transform.TransformerFactory;
022import javax.xml.transform.stream.StreamResult;
023
024public final class JaxbUtil {
025
026    public static final Class[] JAXB_CLASSES = new Class[]{
027        Config.class,
028        KeyDataConfig.class,
029        PrivateKeyData.class,
030        ArgonOptions.class,
031        JdbcConfig.class,
032        KeyData.class,
033        Peer.class,
034        PrivateKeyType.class,
035        ServerConfig.class,
036        DeprecatedServerConfig.class,
037        SslAuthenticationMode.class,
038        SslConfig.class,
039        SslTrustMode.class
040    };
041
042    private JaxbUtil() {
043    }
044
045    public static <T> T unmarshal(InputStream inputStream, Class<T> type) {
046
047        try {
048            return UnmarshallerBuilder.create()
049                    .build().unmarshal(new StreamSource(inputStream), type).getValue();
050        } catch (JAXBException ex) {
051            throw new ConfigException(ex);
052        }
053    }
054
055    public static void marshal(Object object, OutputStream outputStream) {
056        try {
057            MarshallerBuilder.create().build()
058                    .marshal(object, outputStream);
059        } catch (Throwable ex) {
060            Optional<ConstraintViolationException> validationException = unwrapConstraintViolationException(ex);
061            if (validationException.isPresent()) {
062                throw validationException.get();
063            }
064            throw new ConfigException(ex);
065        }
066
067    }
068
069    public static void marshalWithNoValidation(Object object, OutputStream outputStream) {
070        try {
071            MarshallerBuilder.create()
072                    .withoutBeanValidation()
073                    .build()
074                    .marshal(object, outputStream);
075
076        } catch (JAXBException ex) {
077            throw new ConfigException(ex);
078        }
079    }
080
081    public static String marshalToString(Object object) {
082        return IOCallback.execute(() -> {
083            try (OutputStream out = new ByteArrayOutputStream()) {
084                marshal(object, out);
085                return out.toString();
086            }
087        });
088    }
089
090    public static String marshalToStringNoValidation(Object object) {
091        return IOCallback.execute(() -> {
092            try (OutputStream out = new ByteArrayOutputStream()) {
093                marshalWithNoValidation(object, out);
094                return out.toString();
095            }
096        });
097    }
098
099    protected static Optional<ConstraintViolationException> unwrapConstraintViolationException(Throwable ex) {
100        return Optional.of(ex)
101                .map(Throwable::getCause)
102                .filter(Objects::nonNull)
103                .filter(ConstraintViolationException.class::isInstance)
104                .map(ConstraintViolationException.class::cast);
105    }
106
107    public static void marshalMasked(Config object, OutputStream outputStream) {
108
109        XmlProcessingCallback.execute(() -> {
110
111            Marshaller marshaller = MarshallerBuilder.create()
112                    .withXmlMediaType()
113                    .withoutBeanValidation()
114                    .build();
115
116            String xmlData;
117            try (StringWriter writer = new StringWriter()) {
118                marshaller.marshal(object, writer);
119                xmlData = writer.toString();
120            }
121
122            StreamSource xmlSource = new StreamSource(new StringReader(xmlData));
123            try (StringWriter writer = new StringWriter()) {
124                StreamResult xmlResult = new StreamResult(writer);
125                createMaskingXslTransformer().transform(xmlSource, xmlResult);
126                writer.flush();
127
128                Unmarshaller unmarshaller = UnmarshallerBuilder.create()
129                        .withXmlMediaType()
130                        .withoutBeanValidation()
131                        .build();
132
133                Config masked = (Config) unmarshaller.unmarshal(new StringReader(writer.toString()));
134
135                marshalWithNoValidation(masked, outputStream);
136                return null;
137            }
138
139        });
140
141    }
142
143    private static Transformer createMaskingXslTransformer() {
144        return XmlProcessingCallback.execute(() -> {
145            try (InputStream inputStream = JaxbUtil.class.getResourceAsStream("/xsl/mask-config.xsl")) {
146                return TransformerFactory.newInstance().newTransformer(new StreamSource(inputStream));
147            }
148        });
149
150    }
