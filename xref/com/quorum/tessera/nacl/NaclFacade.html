1   package com.quorum.tessera.nacl;
2   
3   import com.quorum.tessera.encryption.KeyPair;
4   import com.quorum.tessera.encryption.MasterKey;
5   import com.quorum.tessera.encryption.PrivateKey;
6   import com.quorum.tessera.encryption.PublicKey;
7   import com.quorum.tessera.encryption.SharedKey;
8   
9   /**
10   * The API provided to the application that all implementation of this API
11   * module should extend
12   * <p>
13   * Provides all function relating to encrypting and decrypting messages
14   * using public/private and symmetric keys.
15   */
16  public interface NaclFacade {
17  
18      /**
19       * Compute the shared key from a public/private key combination
20       * The keys must be from different keysets.
21       * Providing the public key for the corresponding private key (and vice versa) results in an error
22       * <p>
23       * The shared key for a public/private key combo is the same as if the private/public corresponding keys
24       * were provided.
25       * i.e. public1/private2 == private1/public2
26       *
27       * @param publicKey  A public key from the first keyset
28       * @param privateKey A private key from the second keyset
29       * @return The shared key for this key pair.
30       */
31      SharedKey computeSharedKey(PublicKey publicKey, PrivateKey privateKey);
32  
33      /**
34       * Encrypt a payload directly using the given public/private key pair for the sender/recipient
35       *
36       * @param message    The payload to be encrypted
37       * @param nonce      A unique nonce for this public/private pair
38       * @param publicKey  The key from either sender or recipient
39       * @param privateKey The other key from either sender or recipient
40       * @return The encrypted payload
41       */
42      byte[] seal(byte[] message, Nonce nonce, PublicKey publicKey, PrivateKey privateKey);
43  
44      /**
45       * Decrypt a payload directly using the given public/private key pair for the sender/recipient
46       *
47       * @param cipherText The payload to be encrypted
48       * @param nonce      A unique nonce for this public/private pair
49       * @param publicKey  The key from either sender or recipient
50       * @param privateKey The other key from either sender or recipient
51       * @return The encrypted payload
52       */
53      byte[] open(byte[] cipherText, Nonce nonce, PublicKey publicKey, PrivateKey privateKey);
54  
55      /**
56       * Encrypt a payload using the given public/private key pair for the sender/recipient
57       *
58       * @param message   The payload to be encrypted
59       * @param nonce     A unique nonce for this public/private pair
60       * @param sharedKey The shared key between the sender and recipient of the payload
61       * @return The encrypted payload
62       */
63      byte[] sealAfterPrecomputation(byte[] message, Nonce nonce, SharedKey sharedKey);
64  
65      default byte[] sealAfterPrecomputation(byte[] message, Nonce nonce, MasterKey masterKey) {
66          SharedKey sharedKey = SharedKey.from(masterKey.getKeyBytes());
67          return sealAfterPrecomputation(message, nonce, sharedKey);
68      }
69  
70      /**
71       * Decrypts a payload using the shared key between the sender and recipient
72       *
73       * @param cipherText The encrypted payload
74       * @param nonce      The nonce that was used to encrypt this payload
75       * @param sharedKey  The shared key for the sender and recipient
76       * @return The decrypted payload
77       */
78      byte[] openAfterPrecomputation(byte[] cipherText, Nonce nonce, SharedKey sharedKey);
79  
80      /**
81       * Generates a new random nonce of the correct size
82       *
83       * @return a {@link Nonce} containing random data to be used as a nonce
84       */
85      Nonce randomNonce();
86  
87      /**
88       * Generates a new public and private keypair
89       *
90       * @return A pair of public and private keys
91       */
92      KeyPair generateNewKeys();
93  
94      /**
95       * Creates a single standalone key
96       *
97       * @return The randomly generated key
98       */
99      SharedKey createSingleKey();
100 
101     /**
102      * Create a randomly generated {@link MasterKey}
103      *
104      * @return a random {@link MasterKey}
105      */
106     default MasterKey createMasterKey() {
107         SharedKey sharedKey = createSingleKey();
108         return MasterKey.from(sharedKey.getKeyBytes());
109     }
110 
111     /**
112      * Decrypts a payload using the given {@link MasterKey}
113      *
114      * @param cipherText      the ciphertext to decrypt
115      * @param cipherTextNonce the nonce that was used to encrypt the payload
116      * @param masterKey       the key used to encrypt the payload
117      * @return the decrypted payload
118      * @see NaclFacade#openAfterPrecomputation(byte[], Nonce, SharedKey)
119      */
120     default byte[] openAfterPrecomputation(byte[] cipherText, Nonce cipherTextNonce, MasterKey masterKey) {
121         SharedKey sharedKey = SharedKey.from(masterKey.getKeyBytes());
122         return openAfterPrecomputation(cipherText, cipherTextNonce, sharedKey);
123     }
124 
125 }
