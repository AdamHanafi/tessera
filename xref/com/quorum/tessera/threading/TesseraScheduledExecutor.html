1   package com.quorum.tessera.threading;
2   
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   import javax.annotation.PostConstruct;
7   import javax.annotation.PreDestroy;
8   import java.util.Objects;
9   import java.util.concurrent.ScheduledExecutorService;
10  import java.util.concurrent.TimeUnit;
11  
12  /**
13   * Schedules a continuous running task as an alternative to
14   * a {@link Thread} running a {@code while(true)} loop
15   *
16   * Also allows delays if required between each execution of the loop
17   */
18  public class TesseraScheduledExecutor {
19  
20      private static final Logger LOGGER = LoggerFactory.getLogger(TesseraScheduledExecutor.class);
21  
22      private final ScheduledExecutorService executor;
23  
24      private final Runnable action;
25  
26      private final long rate;
27      
28      private final long initialDelay;
29  
30      public TesseraScheduledExecutor(final ScheduledExecutorService executor,
31                                      final Runnable action,
32                                      final long rate,
33                                      final long delay) {
34          this.executor = Objects.requireNonNull(executor);
35          this.action = Objects.requireNonNull(action);
36          this.rate = rate;
37          this.initialDelay = delay;
38      }
39  
40      /**
41       * Starts the submitted task and schedules it to run every given time frame
42       * Catches any Throwable and logs it so that the scheduling doesn't break
43       */
44      @PostConstruct
45      public void start() {
46          LOGGER.info("Starting {}", this.action.getClass().getSimpleName());
47  
48          final Runnable exceptionSafeRunnable = () -> {
49              try {
50                  this.action.run();
51              } catch (final Throwable ex) {
52                  LOGGER.error("Error when executing action {}", action.getClass().getSimpleName());
53                  LOGGER.error("Error when executing action", ex);
54              }
55          };
56  
57          this.executor.scheduleWithFixedDelay(exceptionSafeRunnable, initialDelay, rate, TimeUnit.MILLISECONDS);
58  
59          LOGGER.info("Started {}", this.action.getClass().getSimpleName());
60      }
61  
62      /**
63       * Stops any more executions of the submitted task from running
64       * Does not cancel the currently running task, which may be blocking
65       */
66      @PreDestroy
67      public void stop() {
68          LOGGER.info("Stopping {}", this.action.getClass().getSimpleName());
69  
70          this.executor.shutdown();
71  
72          LOGGER.info("Stopped {}", this.action.getClass().getSimpleName());
73      }
74  
75  }
