12  
13  import java.util.HashMap;
14  import java.util.Map;
15  import java.util.Objects;
16  
17  public class HashicorpVaultKeyGenerator implements KeyGenerator {
18      private static final Logger LOGGER = LoggerFactory.getLogger(HashicorpVaultKeyGenerator.class);
19  
20      private final NaclFacade nacl;
21      private final KeyVaultService keyVaultService;
22  
23      public HashicorpVaultKeyGenerator(final NaclFacade nacl, KeyVaultService keyVaultService) {
24          this.nacl = nacl;
25          this.keyVaultService = keyVaultService;
26      }
27  
28      @Override
29      public HashicorpVaultKeyPair generate(String filename, ArgonOptions encryptionOptions, KeyVaultOptions keyVaultOptions) {
30          Objects.requireNonNull(filename);
31          Objects.requireNonNull(keyVaultOptions, "-keygenvaultsecretengine must be provided if using the Hashicorp vault type");
32          Objects.requireNonNull(keyVaultOptions.getSecretEngineName(), "-keygenvaultsecretengine must be provided if using the Hashicorp vault type");
33  
34          final KeyPair keys = this.nacl.generateNewKeys();
35  
36          String pubId = "publicKey";
37          String privId = "privateKey";
38          Map<String, Object> keyPairData = new HashMap<>();
39          keyPairData.put(pubId, keys.getPublicKey().encodeToBase64());
40          keyPairData.put(privId, keys.getPrivateKey().encodeToBase64());
41  
42          SetSecretData setSecretData = new HashicorpSetSecretData(keyVaultOptions.getSecretEngineName(), filename, keyPairData);
43  
44          keyVaultService.setSecret(setSecretData);
45          LOGGER.debug("Key {} saved to vault secret engine {} with name {} and id {}", keyPairData.get(pubId), keyVaultOptions.getSecretEngineName(), filename, pubId);
46          LOGGER.info("Key saved to vault secret engine {} with name {} and id {}", keyVaultOptions.getSecretEngineName(), filename, pubId);
47          LOGGER.debug("Key {} saved to vault secret engine {} with name {} and id {}", keyPairData.get(privId), keyVaultOptions.getSecretEngineName(), filename, privId);
48          LOGGER.info("Key saved to vault secret engine {} with name {} and id {}", keyVaultOptions.getSecretEngineName(), filename, privId);
49  
50          return new HashicorpVaultKeyPair(pubId, privId, keyVaultOptions.getSecretEngineName(), filename, null);
51      }
52  }
