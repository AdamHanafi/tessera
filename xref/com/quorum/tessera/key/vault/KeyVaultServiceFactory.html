18          List<KeyVaultServiceFactory> providers = new ArrayList<>();
19          ServiceLoader.load(KeyVaultServiceFactory.class).forEach(providers::add);
20  
21          return providers.stream()
22              .filter(factory -> factory.getType() == keyVaultType)
23              .findFirst()
24              .orElseThrow(() -> new NoKeyVaultServiceFactoryException(keyVaultType + " implementation of KeyVaultServiceFactory was not found on the classpath"));
25      }
