8   import java.util.Map;
9   
10  class KeyValueOperationsDelegate {
11  
12      private final VaultVersionedKeyValueOperations keyValueOperations;
13  
14      KeyValueOperationsDelegate(VaultVersionedKeyValueOperations keyValueOperations) {
15          this.keyValueOperations = keyValueOperations;
16      }
17  
18      Versioned<Map<String, Object>> get(HashicorpGetSecretData getSecretData) {
19          //if version 0 then latest version retrieved
20          return keyValueOperations.get(getSecretData.getSecretName(), Versioned.Version.from(getSecretData.getSecretVersion()));
21      }
22  
23      Versioned.Metadata set(HashicorpSetSecretData setSecretData) {
24          return keyValueOperations.put(setSecretData.getSecretName(), setSecretData.getNameValuePairs());
25      }
26  
27  }
