1   package com.quorum.tessera.p2p;
2   
3   import io.swagger.annotations.Api;
4   import io.swagger.annotations.ApiResponse;
5   import io.swagger.annotations.ApiResponses;
6   
7   import javax.ws.rs.GET;
8   import javax.ws.rs.Path;
9   import javax.ws.rs.Produces;
10  import javax.ws.rs.core.Context;
11  import javax.ws.rs.core.Request;
12  import javax.ws.rs.core.Response;
13  import javax.ws.rs.core.Response.Status;
14  import javax.ws.rs.core.Variant;
15  import java.io.IOException;
16  import java.net.URL;
17  import java.util.List;
18  
19  import static javax.ws.rs.core.MediaType.*;
20  
21  /**
22   * Provides HTTP endpoints for accessing the OpenAPI schema
23   */
24  @Path("/api")
25  @Api("Provides access to OpenAPI schema documentation.")
26  public class ApiResource {
27  
28      private static final List<Variant> VARIANTS = Variant.mediaTypes(APPLICATION_JSON_TYPE, TEXT_HTML_TYPE).build();
29  
30      @GET
31      @Produces({APPLICATION_JSON, TEXT_HTML})
32      @ApiResponses({@ApiResponse(code = 200, message = "Returns JSON or HTML OpenAPI document")})
33      public Response api(@Context final Request request) throws IOException {
34  
35          final Variant variant = request.selectVariant(VARIANTS);
36  
37          final URL url;
38          if (variant.getMediaType() == APPLICATION_JSON_TYPE) {
39              url = getClass().getResource("/swagger.json");
40  
41          } else if (variant.getMediaType() == TEXT_HTML_TYPE) {
42              url = getClass().getResource("/swagger.html");
43  
44          } else {
45  
46              return Response.status(Status.BAD_REQUEST).build();
47          }
48  
49          return Response.ok(url.openStream(), variant.getMediaType()).build();
50  
51      }
52  
53  }
