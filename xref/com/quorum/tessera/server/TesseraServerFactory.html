13  public interface TesseraServerFactory<T> {
14  
15      TesseraServer createServer(ServerConfig config, Set<T> services);
16  
17      Logger LOGGER = LoggerFactory.getLogger(TesseraServerFactory.class);
18      
19      static TesseraServerFactory create(CommunicationType communicationType) {
20  
21          if(communicationType == CommunicationType.WEB_SOCKET) {
22              throw new UnsupportedOperationException("Websockets are not supported yet");
23          }
24  
25          List<TesseraServerFactory> all = new ArrayList<>();
26          ServiceLoader.load(TesseraServerFactory.class).forEach(all::add);
27  
28          return all.stream()
29                  .filter(f -> f.communicationType() == communicationType)
30                  .findFirst()
31                  .orElseThrow(() -> new IllegalStateException("No server factory found for "+ communicationType));
32      }
33  
34      CommunicationType communicationType();
35      
36  }
