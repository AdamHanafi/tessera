1   package com.quorum.tessera.encryption;
2   
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   import java.util.Collection;
7   import java.util.HashSet;
8   import java.util.Objects;
9   import java.util.Set;
10  import java.util.stream.Collectors;
11  
12  public class KeyManagerImpl implements KeyManager {
13  
14      private static final Logger LOGGER = LoggerFactory.getLogger(KeyManagerImpl.class);
15  
16      /**
17       * A list of all pub/priv keys that are attached to this node
18       */
19      private final Set<KeyPair> localKeys;
20  
21      private final KeyPair defaultKeys;
22  
23      private final Set<PublicKey> forwardingPublicKeys;
24  
25      public KeyManagerImpl(final Collection<KeyPair> keys, Collection<PublicKey> forwardKeys) {
26  
27          this.localKeys = new HashSet<>(keys);
28  
29          this.defaultKeys = localKeys.iterator().next();
30  
31          this.forwardingPublicKeys = new HashSet<>(forwardKeys);
32      }
33  
34      @Override
35      public PublicKey getPublicKeyForPrivateKey(final PrivateKey privateKey) {
36          LOGGER.debug("Attempting to find public key for the private key {}", privateKey);
37  
38          final PublicKey publicKey = localKeys
39              .stream()
40              .filter(keypair -> Objects.equals(keypair.getPrivateKey(), privateKey))
41              .findFirst()
42              .map(KeyPair::getPublicKey)
43              .orElseThrow(() -> new KeyNotFoundException(
44                  "Private key " + privateKey.encodeToBase64() + " not found when searching for public key"
45              ));
46  
47          LOGGER.debug("Found public key {} for private key {}", publicKey, privateKey);
48  
49          return publicKey;
50      }
51  
52  
53      @Override
54      public PrivateKey getPrivateKeyForPublicKey(final PublicKey publicKey) {
55          LOGGER.debug("Attempting to find private key for the public key {}", publicKey);
56  
57          final PrivateKey privateKey = localKeys
58              .stream()
59              .filter(keypair -> Objects.equals(keypair.getPublicKey(), publicKey))
60              .findFirst()
61              .map(KeyPair::getPrivateKey)
62              .orElseThrow(() -> new KeyNotFoundException(
63                  "Public key " + publicKey.encodeToBase64() + " not found when searching for private key"
64              ));
65  
66          LOGGER.debug("Found private key {} for public key {}", privateKey, publicKey);
67  
68          return privateKey;
69      }
70  
71      @Override
72      public Set<PublicKey> getPublicKeys() {
73          return localKeys
74              .stream()
75              .map(KeyPair::getPublicKey)
76              .collect(Collectors.toSet());
77      }
78  
79      @Override
80      public PublicKey defaultPublicKey() {
81          return defaultKeys.getPublicKey();
82      }
83  
84      @Override
85      public Set<PublicKey> getForwardingKeys() {
86          return this.forwardingPublicKeys;
87      }
88  
89  }
