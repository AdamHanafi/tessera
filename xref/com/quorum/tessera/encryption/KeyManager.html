1   package com.quorum.tessera.encryption;
2   
3   
4   import java.util.Set;
5   
6   /**
7    * Manages local keys for the running node, include key lookups and key loading/generation
8    */
9   public interface KeyManager {
10  
11      /**
12       * Fetches the public key that corresponds to the given private key
13       * This means that the returned public key is under control of the current node
14       * (i.e. it is one of our keys)
15       *
16       * @param privateKey the private key for which to find the public key
17       * @return the corresponding public key
18       * @throws RuntimeException if the private key could not be found
19       */
20      PublicKey getPublicKeyForPrivateKey(PrivateKey privateKey);
21  
22      /**
23       * Fetches the private key that corresponds to the given public key
24       * This means that the returned private key is under control of the current node
25       * (i.e. it is one of our keys)
26       *
27       * @param publicKey the public key for which to find the private key
28       * @return the corresponding private key
29       * @throws RuntimeException if the public key could not be found
30       */
31      PrivateKey getPrivateKeyForPublicKey(PublicKey publicKey);
32  
33      /**
34       * Return a list of all recipients public keys of this node
35       *
36       * @return the set of all public keys
37       */
38      Set<PublicKey> getPublicKeys();
39  
40      /**
41       * Returns the default public key of this node if a transaction is
42       * sent that doesn't have a sender.
43       * <p>
44       * Must return the same key each time during a single run,
45       * but no guarantee is made about which key is returned if the
46       * application is restarted.
47       *
48       * @return The public key to use as the default sender
49       */
50      PublicKey defaultPublicKey();
51  
52      /**
53       * Returns a set of public keys that all transactions should be sent to in
54       * addition to the declared recipients
55       *
56       * @return the set of keys to forward to
57       */
58      Set<PublicKey> getForwardingKeys();
59  
60  }
