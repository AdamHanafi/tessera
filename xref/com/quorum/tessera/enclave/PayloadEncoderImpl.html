4   import com.quorum.tessera.nacl.Nonce;

6   import java.nio.ByteBuffer;
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  import static java.util.Collections.emptyList;
11  import static java.util.Collections.singletonList;
12  import static java.util.stream.Collectors.toList;
13  
14  public class PayloadEncoderImpl implements PayloadEncoder, BinaryEncoder {
15  
16      @Override
17      public byte[] encode(final EncodedPayload payload) {
18  
19          final byte[] senderKey = encodeField(payload.getSenderKey().getKeyBytes());
20          final byte[] cipherText = encodeField(payload.getCipherText());
21          final byte[] nonce = encodeField(payload.getCipherTextNonce().getNonceBytes());
22          final byte[] recipientNonce = encodeField(payload.getRecipientNonce().getNonceBytes());
23          final byte[] recipients = encodeArray(payload.getRecipientBoxes());
24          final byte[] recipientBytes
25              = encodeArray(payload.getRecipientKeys().stream().map(PublicKey::getKeyBytes).collect(toList()));
26  
27          return ByteBuffer
28              .allocate(senderKey.length + cipherText.length + nonce.length + recipients.length + recipientNonce.length + recipientBytes.length)
29              .put(senderKey)
30              .put(cipherText)
31              .put(nonce)
32              .put(recipients)
33              .put(recipientNonce)
34              .put(recipientBytes)
35              .array();
36      }
37  
38      @Override
39      public EncodedPayload decode(final byte[] input) {
40          final ByteBuffer buffer = ByteBuffer.wrap(input);
41  
42          final long senderSize = buffer.getLong();
43          final byte[] senderKey = new byte[Math.toIntExact(senderSize)];
44          buffer.get(senderKey);
45  
46          final long cipherTextSize = buffer.getLong();
47          final byte[] cipherText = new byte[Math.toIntExact(cipherTextSize)];
48          buffer.get(cipherText);
49  
50          final long nonceSize = buffer.getLong();
51          final byte[] nonce = new byte[Math.toIntExact(nonceSize)];
52          buffer.get(nonce);
53  
54          final long numberOfRecipients = buffer.getLong();
55          final List<byte[]> recipientBoxes = new ArrayList<>();
56          for (long i = 0; i < numberOfRecipients; i++) {
57              final long boxSize = buffer.getLong();
58              final byte[] box = new byte[Math.toIntExact(boxSize)];
59              buffer.get(box);
60              recipientBoxes.add(box);
61          }
62  
63          final long recipientNonceSize = buffer.getLong();
64          final byte[] recipientNonce = new byte[Math.toIntExact(recipientNonceSize)];
65          buffer.get(recipientNonce);
66  
67          //this means there are no recipients in the payload (which we receive when we are a participant)
68          if (!buffer.hasRemaining()) {
69              return new EncodedPayload(
70                  PublicKey.from(senderKey), cipherText, new Nonce(nonce),
71                  recipientBoxes, new Nonce(recipientNonce), emptyList()
72              );
73          }
74  
75          final long recipientLength = buffer.getLong();
76  
77          final List<byte[]> recipientKeys = new ArrayList<>();
78          for (long i = 0; i < recipientLength; i++) {
79              final long boxSize = buffer.getLong();
80              final byte[] box = new byte[Math.toIntExact(boxSize)];
81              buffer.get(box);
82              recipientKeys.add(box);
83          }
84  
85          return new EncodedPayload(
86              PublicKey.from(senderKey), cipherText, new Nonce(nonce), recipientBoxes, new Nonce(recipientNonce),
87              recipientKeys.stream().map(PublicKey::from).collect(toList())
88          );
89      }
90  
91      @Override
92      public EncodedPayload forRecipient(final EncodedPayload payload, final PublicKey recipient) {
93  
94          if (!payload.getRecipientKeys().contains(recipient)) {
95              throw new InvalidRecipientException("Recipient " + recipient.encodeToBase64() + " is not a recipient of transaction ");
96          }
97  
98          final int recipientIndex = payload.getRecipientKeys().indexOf(recipient);
99          final byte[] recipientBox = payload.getRecipientBoxes().get(recipientIndex);
100 
101         return new EncodedPayload(
102             payload.getSenderKey(), payload.getCipherText(), payload.getCipherTextNonce(),
103             singletonList(recipientBox), payload.getRecipientNonce(), emptyList()
104         );
105     }
106 
107 }
