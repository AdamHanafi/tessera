3   import com.quorum.tessera.io.SystemAdapter;
4   import java.util.Objects;
5   
6   /**
7    * Allows a password to be read from the System console if it is available
8    * otherwise reads from the provided input stream
9    */
10  public interface PasswordReader {
11  
12      /**
13       * Read a password from either system console or the given stream
14       * @return the read password, which may be empty if no password is given
15       */
16      String readPasswordFromConsole();
17  
18      /**
19       * Requests user input for a password until two matching consecutive entries are made
20       *
21       * @return The password that the user has input
22       */
23      default String requestUserPassword() {
24  
25          for(;;) {
26  
27              sys().out().println("Enter a password if you want to lock the private key or leave blank");
28              final String password = this.readPasswordFromConsole();
29  
30              sys().out().println("Please re-enter the password (or lack of) to confirm");
31              final String passwordCheck = this.readPasswordFromConsole();
32  
33              if(Objects.equals(password, passwordCheck)) {
34                  return password;
35              } else {
36                  sys().out().println("Passwords did not match, try again...");
37              }
38  
39          }
40  
41      }
42      
43      static SystemAdapter sys() {
44          return SystemAdapter.INSTANCE;
45      }
46  
47  }
