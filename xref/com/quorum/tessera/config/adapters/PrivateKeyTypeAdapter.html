1   package com.quorum.tessera.config.adapters;
2   
3   import com.quorum.tessera.config.PrivateKeyType;
4   
5   import javax.xml.bind.annotation.adapters.XmlAdapter;
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   public class PrivateKeyTypeAdapter extends XmlAdapter<String, PrivateKeyType> {
10  
11      private static final Map<String, PrivateKeyType> MAPPING = new HashMap<String, PrivateKeyType>() {
12          {
13              put("unlocked", PrivateKeyType.UNLOCKED);
14              put("argon2sbox", PrivateKeyType.LOCKED);
15          }
16      };
17  
18      @Override
19      public PrivateKeyType unmarshal(final String v) {
20          return MAPPING.get(v);
21      }
22  
23      @Override
24      public String marshal(final PrivateKeyType v) {
25          return MAPPING
26              .entrySet()
27              .stream()
28              .filter(e -> e.getValue() == v)
29              .map(Map.Entry::getKey)
30              .findAny()
31              .orElse(null);
32      }
33  
34  }
