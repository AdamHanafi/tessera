18      @Valid
19      @NotNull
20      @XmlAttribute
21      private String url;

24      @XmlElement

32  
33      @Valid
34      @ValidPath(checkExists = true, message = "File does not exist")
35      @XmlElement(type = String.class)

39      public HashicorpKeyVaultConfig(String url, String approlePath, Path tlsKeyStorePath, Path tlsTrustStorePath) {
40          this.url = url;
41          this.approlePath = approlePath;
42          this.tlsKeyStorePath = tlsKeyStorePath;
43          this.tlsTrustStorePath = tlsTrustStorePath;
44      }

47      }

62          this.tlsKeyStorePath = tlsKeyStorePath;
63      }
64

66          return tlsTrustStorePath;
67      }
68  
69      public void setTlsTrustStorePath(Path tlsTrustStorePath) {
70          this.tlsTrustStorePath = tlsTrustStorePath;
71      }

78      }

81          this.approlePath = approlePath;
82      }
