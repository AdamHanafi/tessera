6   import java.io.PrintWriter;
7   import java.net.URI;
8   import java.util.Arrays;
9   import java.util.List;
10  import java.util.Objects;
11  import java.util.Optional;

13  import javax.ws.rs.client.Entity;

15  import javax.ws.rs.core.Response;
16  import javax.ws.rs.core.UriBuilder;
17  import org.apache.commons.cli.CommandLine;
18  import org.apache.commons.cli.DefaultParser;
19  import org.apache.commons.cli.HelpFormatter;
20  import org.apache.commons.cli.Option;
21  import org.apache.commons.cli.Options;
22  
23  /**
24   * Cli Adapter to be used for runtime updates
25   */
26  public class AdminCliAdapter implements CliAdapter {
27  
28      private final ClientFactory clientFactory;
29  
30      public AdminCliAdapter(final ClientFactory clientFactory) {
31          this.clientFactory = Objects.requireNonNull(clientFactory);
32      }
33  
34      /**
35       *
36       * @param args
37       * @return CliResult with config object always null.
38       * @throws Exception
39       */
40      @Override
41      public CliResult execute(String... args) throws Exception {
42  
43          final Options options = new Options();
44  
45          options.addOption(
46                  Option.builder("configfile")
47                          .desc("Path to node configuration file")
48                          .hasArg(true)
49                          .required()
50                          .optionalArg(false)
51                          .numberOfArgs(1)
52                          .argName("PATH")
53                          .build());
54  
55          options.addOption(Option.builder("addpeer")
56                  .desc("Add peer to running node")
57                  .hasArg(true)
58                  .optionalArg(false)
59                  .numberOfArgs(1)
60                  .argName("URL")
61                  .build());
62  
63          final List<String> argsList = Arrays.asList(args);
64  
65          if (argsList.contains("help") || argsList.isEmpty()) {
66              HelpFormatter formatter = new HelpFormatter();
67              PrintWriter pw = new PrintWriter(sys().out());
68              formatter.printHelp(pw,
69                      200, "tessera admin",
70                      null, options, formatter.getLeftPadding(),
71                      formatter.getDescPadding(), null, false);
72              pw.flush();
73              
74              return new CliResult(0, true, null);
75          }
76  
77          final CommandLine line = new DefaultParser().parse(options, args);
78          if (!line.hasOption("addpeer")) {
79              sys().out().println("No peer defined");
80              return new CliResult(1, true, null);
81          }
82  
83          Config config = new ConfigurationParser().parse(line);
84  
85          //TODO revisit - maybe the admin stuff should be reached via unix socket - in order to avoid security concerns
86          ServerConfig serverConfig = config.getServerConfigs().stream()
87                  .filter(c -> c.getApp() == AppType.ADMIN)
88                  .findFirst().orElse(config.getServerConfigs().stream().findAny().get());
89  
90          Client restClient = clientFactory.buildFrom(serverConfig);
91  
92          String peerUrl = line.getOptionValue("addpeer");
93  
94          final Peer peer = new Peer(peerUrl);
95  
96          String scheme = Optional.of(serverConfig)
97                  .map(ServerConfig::getBindingUri)
98                  .map(URI::getScheme)
99                  .orElse("http");
100 
101         Integer port = Optional.of(serverConfig)
102                 .map(ServerConfig::getServerUri)
103                 .map(URI::getPort)
104                 .orElse(80);
105 
106         URI uri = UriBuilder.fromUri(serverConfig.getBindingUri())
107                 .port(port)
108                 .scheme(scheme)
109                 .build();
110 
111         Response response = restClient.target(uri)
112                 .path("config")
113                 .path("peers")
114                 .request(MediaType.APPLICATION_JSON)
115                 .accept(MediaType.APPLICATION_JSON)
116                 .put(Entity.entity(peer, MediaType.APPLICATION_JSON));
117 
118         if (response.getStatus() == Response.Status.CREATED.getStatusCode()) {
119 
120             sys().out().printf("Peer %s added.", response.getLocation());
121             sys().out().println();
122 
123             return new CliResult(0, true, null);
124         }
125 
126         sys().err().println("Unable to create peer");
127 
128         return new CliResult(1, true, null);
129     }
130 
131 }
