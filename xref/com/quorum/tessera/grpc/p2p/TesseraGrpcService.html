1   package com.quorum.tessera.grpc.p2p;
2   
3   import com.google.protobuf.Empty;
4   import com.quorum.tessera.grpc.p2p.UpCheckMessage;
5   import com.quorum.tessera.grpc.p2p.VersionMessage;
6   import com.quorum.tessera.grpc.StreamObserverTemplate;
7   import io.grpc.stub.StreamObserver;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  public class TesseraGrpcService extends TesseraGrpc.TesseraImplBase {
12  
13      private static final Logger LOGGER = LoggerFactory.getLogger(TesseraGrpcService.class);
14  
15      private static final String UPCHECK_RESPONSE = "I'm up!";
16  
17      private static final String VERSION = "0.6";
18  
19      @Override
20      public void getVersion(Empty request, StreamObserver<VersionMessage> responseObserver) {
21          LOGGER.info("GET version");
22  
23          StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
24  
25          template.handle(() -> {
26              return VersionMessage.newBuilder()
27                      .setVersion(VERSION)
28                      .build();
29          });
30  
31      }
32  
33      @Override
34      public void getUpCheck(Empty request, StreamObserver<UpCheckMessage> responseObserver) {
35          LOGGER.info("GET upcheck");
36          StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
37  
38          template.handle(() -> {
39              return UpCheckMessage.newBuilder()
40                      .setUpCheck(UPCHECK_RESPONSE)
41                      .build();
42          });
43  
44      }
45  
46  }
