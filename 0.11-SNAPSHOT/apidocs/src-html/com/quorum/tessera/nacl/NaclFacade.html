001package com.quorum.tessera.nacl;
002
003import com.quorum.tessera.encryption.KeyPair;
004import com.quorum.tessera.encryption.MasterKey;
005import com.quorum.tessera.encryption.PrivateKey;
006import com.quorum.tessera.encryption.PublicKey;
007import com.quorum.tessera.encryption.SharedKey;
008
009/**
010 * The API provided to the application that all implementation of this API
011 * module should extend
012 * <p>
013 * Provides all function relating to encrypting and decrypting messages
014 * using public/private and symmetric keys.
015 */
016public interface NaclFacade {
017
018    /**
019     * Compute the shared key from a public/private key combination
020     * The keys must be from different keysets.
021     * Providing the public key for the corresponding private key (and vice versa) results in an error
022     * <p>
023     * The shared key for a public/private key combo is the same as if the private/public corresponding keys
024     * were provided.
025     * i.e. public1/private2 == private1/public2
026     *
027     * @param publicKey  A public key from the first keyset
028     * @param privateKey A private key from the second keyset
029     * @return The shared key for this key pair.
030     */
031    SharedKey computeSharedKey(PublicKey publicKey, PrivateKey privateKey);
032
033    /**
034     * Encrypt a payload directly using the given public/private key pair for the sender/recipient
035     *
036     * @param message    The payload to be encrypted
037     * @param nonce      A unique nonce for this public/private pair
038     * @param publicKey  The key from either sender or recipient
039     * @param privateKey The other key from either sender or recipient
040     * @return The encrypted payload
041     */
042    byte[] seal(byte[] message, Nonce nonce, PublicKey publicKey, PrivateKey privateKey);
043
044    /**
045     * Decrypt a payload directly using the given public/private key pair for the sender/recipient
046     *
047     * @param cipherText The payload to be encrypted
048     * @param nonce      A unique nonce for this public/private pair
049     * @param publicKey  The key from either sender or recipient
050     * @param privateKey The other key from either sender or recipient
051     * @return The encrypted payload
052     */
053    byte[] open(byte[] cipherText, Nonce nonce, PublicKey publicKey, PrivateKey privateKey);
054
055    /**
056     * Encrypt a payload using the given public/private key pair for the sender/recipient
057     *
058     * @param message   The payload to be encrypted
059     * @param nonce     A unique nonce for this public/private pair
060     * @param sharedKey The shared key between the sender and recipient of the payload
061     * @return The encrypted payload
062     */
063    byte[] sealAfterPrecomputation(byte[] message, Nonce nonce, SharedKey sharedKey);
064
065    default byte[] sealAfterPrecomputation(byte[] message, Nonce nonce, MasterKey masterKey) {
066        SharedKey sharedKey = SharedKey.from(masterKey.getKeyBytes());
067        return sealAfterPrecomputation(message, nonce, sharedKey);
068    }
069
070    /**
071     * Decrypts a payload using the shared key between the sender and recipient
072     *
073     * @param cipherText The encrypted payload
074     * @param nonce      The nonce that was used to encrypt this payload
075     * @param sharedKey  The shared key for the sender and recipient
076     * @return The decrypted payload
077     */
078    byte[] openAfterPrecomputation(byte[] cipherText, Nonce nonce, SharedKey sharedKey);
079
080    /**
081     * Generates a new random nonce of the correct size
082     *
083     * @return a {@link Nonce} containing random data to be used as a nonce
084     */
085    Nonce randomNonce();
086
087    /**
088     * Generates a new public and private keypair
089     *
090     * @return A pair of public and private keys
091     */
092    KeyPair generateNewKeys();
093
094    /**
095     * Creates a single standalone key
096     *
097     * @return The randomly generated key
098     */
099    SharedKey createSingleKey();
100
101    /**
102     * Create a randomly generated {@link MasterKey}
103     *
104     * @return a random {@link MasterKey}
105     */
106    default MasterKey createMasterKey() {
107        SharedKey sharedKey = createSingleKey();
108        return MasterKey.from(sharedKey.getKeyBytes());
109    }
110
111    /**
112     * Decrypts a payload using the given {@link MasterKey}
113     *
114     * @param cipherText      the ciphertext to decrypt
115     * @param cipherTextNonce the nonce that was used to encrypt the payload
116     * @param masterKey       the key used to encrypt the payload
117     * @return the decrypted payload
118     * @see NaclFacade#openAfterPrecomputation(byte[], Nonce, SharedKey)
119     */
120    default byte[] openAfterPrecomputation(byte[] cipherText, Nonce cipherTextNonce, MasterKey masterKey) {
121        SharedKey sharedKey = SharedKey.from(masterKey.getKeyBytes());
122        return openAfterPrecomputation(cipherText, cipherTextNonce, sharedKey);
123    }
124
125}
