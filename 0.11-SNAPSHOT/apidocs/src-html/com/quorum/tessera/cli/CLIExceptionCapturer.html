001package com.quorum.tessera.cli;
002
003import picocli.CommandLine;
004
005// This method usually returns an exit code for a given exception whilst parsing and handling commandline args
006// however, we have a different system in place to do this.
007// So here a dummy exit code is returned with access to the exception thrown
008public class CLIExceptionCapturer implements CommandLine.IExecutionExceptionHandler {
009
010    private Exception thrown;
011
012    @Override
013    public int handleExecutionException(
014            final Exception ex, final CommandLine cmd, final CommandLine.ParseResult result) {
015        this.thrown = ex;
016        return 0;
017    }
018
019    public Exception getThrown() {
020        return thrown;
021    }
022}
