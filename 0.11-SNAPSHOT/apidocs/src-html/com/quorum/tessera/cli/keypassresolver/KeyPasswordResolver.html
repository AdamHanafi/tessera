001package com.quorum.tessera.cli.keypassresolver;
002
003import com.quorum.tessera.config.Config;
004
005public interface KeyPasswordResolver {
006
007    /**
008     * Attempts to resolve all the locked keys using/fetching the required
009     * password.
010     *
011     * The passwords may be provided in the configuration, or they may be
012     * fetched from other sources as required.
013     *
014     * @param config the configuration that contains the keys to resolve, as
015     *               well as some predefined passwords
016     */
017    void resolveKeyPasswords(Config config);
018
019}
