001package com.quorum.tessera.cli.keypassresolver;
002
003import com.quorum.tessera.config.Config;
004import com.quorum.tessera.config.KeyConfiguration;
005import com.quorum.tessera.config.PrivateKeyType;
006import com.quorum.tessera.config.keypairs.ConfigKeyPair;
007import com.quorum.tessera.config.keypairs.FilesystemKeyPair;
008import com.quorum.tessera.config.keypairs.InlineKeypair;
009import com.quorum.tessera.io.SystemAdapter;
010import com.quorum.tessera.passwords.PasswordReader;
011import com.quorum.tessera.passwords.PasswordReaderFactory;
012import org.apache.commons.lang3.StringUtils;
013
014import java.io.IOException;
015import java.nio.charset.StandardCharsets;
016import java.nio.file.Files;
017import java.util.ArrayList;
018import java.util.List;
019import java.util.Objects;
020import java.util.stream.IntStream;
021
022public class CliKeyPasswordResolver implements KeyPasswordResolver {
023
024    private static final int MAX_PASSWORD_ATTEMPTS = 2;
025
026    private final PasswordReader passwordReader;
027
028    public CliKeyPasswordResolver() {
029        this(PasswordReaderFactory.create());
030    }
031
032    public CliKeyPasswordResolver(final PasswordReader passwordReader) {
033        this.passwordReader = Objects.requireNonNull(passwordReader);
034    }
035
036    public void resolveKeyPasswords(final Config config) {
037
038        final KeyConfiguration input = config.getKeys();
039        if (input == null) {
040            //invalid config, but gets picked up by validation later
041            return;
042        }
043
044        final List<String> allPasswords = new ArrayList<>();
045        if (input.getPasswords() != null) {
046            allPasswords.addAll(input.getPasswords());
047        } else if (input.getPasswordFile() != null) {
048            try {
049                allPasswords.addAll(Files.readAllLines(input.getPasswordFile(), StandardCharsets.UTF_8));
050            } catch (final IOException ex) {
051                //dont do anything, if any keys are locked validation will complain that
052                //locked keys were provided without passwords
053                SystemAdapter.INSTANCE.err().println("Could not read the password file");
054            }
055        }
056
057        IntStream
058            .range(0, input.getKeyData().size())
059            .forEachOrdered(i -> {
060                if(i < allPasswords.size()) {
061                    input.getKeyData().get(i).withPassword(allPasswords.get(i));
062                }
063            });
064
065        //decrypt the keys, either using provided passwords or read from CLI
066        IntStream
067            .range(0, input.getKeyData().size())
068            .forEachOrdered(keyNumber -> getSingleKeyPassword(keyNumber, input.getKeyData().get(keyNumber)));
069    }
070
071    //TODO: make private
072    //@VisibleForTesting
073    public void getSingleKeyPassword(final int keyNumber, final ConfigKeyPair keyPair) {
074        final boolean isInline = keyPair instanceof InlineKeypair;
075        final boolean isFilesystem = keyPair instanceof FilesystemKeyPair;
076
077        if (!isInline && !isFilesystem) {
078            //some other key type that doesn't use passwords, skip
079            return;
080        }
081
082        final InlineKeypair inlineKey = isInline ? (InlineKeypair)keyPair : ((FilesystemKeyPair)keyPair).getInlineKeypair();
083
084        if(inlineKey == null) {
085            //filesystem key pair that couldn't load the keys, catch in validation later
086            return;
087        }
088
089        final boolean isLocked = inlineKey.getPrivateKeyConfig().getType() == PrivateKeyType.LOCKED;
090
091        if (isLocked) {
092            int currentAttemptNumber = MAX_PASSWORD_ATTEMPTS;
093            while (currentAttemptNumber > 0) {
094                if (StringUtils.isEmpty(keyPair.getPassword()) || keyPair.getPrivateKey() == null || keyPair.getPrivateKey().contains("NACL_FAILURE")) {
095                    final String attemptOutput = "Attempt " + (MAX_PASSWORD_ATTEMPTS-currentAttemptNumber+1) + " of " + MAX_PASSWORD_ATTEMPTS + ".";
096                    System.out.println("Password for key[" + keyNumber + "] missing or invalid.");
097                    System.out.println(attemptOutput + " Enter a password for the key");
098                    final String pass = passwordReader.readPasswordFromConsole();
099                    keyPair.withPassword(pass);
100                }
101                currentAttemptNumber--;
102            }
103        }
104    }
105}
