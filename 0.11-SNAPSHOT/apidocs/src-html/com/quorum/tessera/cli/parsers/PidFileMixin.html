001package com.quorum.tessera.cli.parsers;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005import picocli.CommandLine;
006
007import java.io.OutputStream;
008import java.lang.management.ManagementFactory;
009import java.nio.file.Files;
010import java.nio.file.Path;
011import java.util.concurrent.Callable;
012
013import static java.nio.charset.StandardCharsets.UTF_8;
014import static java.nio.file.StandardOpenOption.CREATE;
015import static java.nio.file.StandardOpenOption.TRUNCATE_EXISTING;
016
017public class PidFileMixin implements Callable<Boolean> {
018
019    private static final Logger LOGGER = LoggerFactory.getLogger(PidFileMixin.class);
020
021    @CommandLine.Option(names = "-pidfile", arity = "1", description = "the path to write the PID to")
022    private Path pidFilePath = null;
023
024    @Override
025    public Boolean call() throws Exception {
026        if (pidFilePath == null) {
027            return true;
028        }
029
030        if (Files.exists(pidFilePath)) {
031            LOGGER.info("File already exists {}", pidFilePath);
032        } else {
033            LOGGER.info("Created pid file {}", pidFilePath);
034        }
035
036        final String pid = ManagementFactory.getRuntimeMXBean().getName().split("@")[0];
037
038        try (OutputStream stream = Files.newOutputStream(pidFilePath, CREATE, TRUNCATE_EXISTING)) {
039            stream.write(pid.getBytes(UTF_8));
040        }
041
042        return true;
043    }
044
045    public void setPidFilePath(final Path pidFilePath) {
046        this.pidFilePath = pidFilePath;
047    }
048}
