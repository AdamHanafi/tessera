001package com.quorum.tessera.nio.unix;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URI;
007import java.nio.channels.AsynchronousFileChannel;
008import java.nio.channels.FileChannel;
009import java.nio.channels.SeekableByteChannel;
010import java.nio.file.*;
011import java.nio.file.attribute.BasicFileAttributes;
012import java.nio.file.attribute.FileAttribute;
013import java.nio.file.attribute.FileAttributeView;
014import java.nio.file.spi.FileSystemProvider;
015import java.util.Map;
016import java.util.Objects;
017import java.util.Set;
018import java.util.concurrent.ExecutorService;
019
020public abstract class DelegatingFileSystemProvider extends FileSystemProvider {
021
022    private final FileSystemProvider delegate;
023
024    public DelegatingFileSystemProvider(final FileSystemProvider delegate) {
025        this.delegate = Objects.requireNonNull(delegate);
026    }
027
028    @Override
029    public abstract String getScheme();
030
031    @Override
032    public FileSystem newFileSystem(final URI uri, final Map<String, ?> env) throws IOException {
033        return delegate.newFileSystem(uri, env);
034    }
035
036    @Override
037    public FileSystem getFileSystem(final URI uri) {
038        return delegate.getFileSystem(uri);
039    }
040
041    @Override
042    public Path getPath(final URI uri) {
043        return delegate.getPath(uri);
044    }
045
046    @Override
047    public SeekableByteChannel newByteChannel(final Path path,
048                                              final Set<? extends OpenOption> options,
049                                              final FileAttribute<?>... attrs) throws IOException {
050        return delegate.newByteChannel(path, options, attrs);
051    }
052
053    @Override
054    public DirectoryStream<Path> newDirectoryStream(
055        final Path dir,
056        final DirectoryStream.Filter<? super Path> filter
057    ) throws IOException {
058        return delegate.newDirectoryStream(dir, filter);
059    }
060
061    @Override
062    public void createDirectory(final Path dir, final FileAttribute<?>... attrs) throws IOException {
063        delegate.createDirectory(dir, attrs);
064    }
065
066    @Override
067    public void delete(final Path path) throws IOException {
068        delegate.delete(path);
069    }
070
071    @Override
072    public void copy(final Path source, final Path target, final CopyOption... options) throws IOException {
073        delegate.copy(source, target, options);
074    }
075
076    @Override
077    public FileSystem newFileSystem(final Path path, final Map<String, ?> env) throws IOException {
078        return delegate.newFileSystem(path, env);
079    }
080
081    @Override
082    public InputStream newInputStream(final Path path, final OpenOption... options) throws IOException {
083        return delegate.newInputStream(path, options);
084    }
085
086    @Override
087    public OutputStream newOutputStream(final Path path, final OpenOption... options) throws IOException {
088        return delegate.newOutputStream(path, options);
089    }
090
091    @Override
092    public FileChannel newFileChannel(final Path path,
093                                      final Set<? extends OpenOption> options,
094                                      final FileAttribute<?>... attrs) throws IOException {
095        return delegate.newFileChannel(path, options, attrs);
096    }
097
098    @Override
099    public AsynchronousFileChannel newAsynchronousFileChannel(final Path path,
100                                                              final Set<? extends OpenOption> options,
101                                                              final ExecutorService executor,
102                                                              final FileAttribute<?>... attrs) throws IOException {
103        return delegate.newAsynchronousFileChannel(path, options, executor, attrs);
104    }
105
106    @Override
107    public void createSymbolicLink(final Path link,
108                                   final Path target,
109                                   final FileAttribute<?>... attrs) throws IOException {
110        delegate.createSymbolicLink(link, target, attrs);
111    }
112
113    @Override
114    public void createLink(final Path link, final Path existing) throws IOException {
115        delegate.createLink(link, existing);
116    }
117
118    @Override
119    public boolean deleteIfExists(final Path path) throws IOException {
120        return delegate.deleteIfExists(path);
121    }
122
123    @Override
124    public Path readSymbolicLink(final Path link) throws IOException {
125        return delegate.readSymbolicLink(link);
126    }
127
128    @Override
129    public void move(final Path source, final Path target, final CopyOption... options) throws IOException {
130        delegate.move(source, target, options);
131    }
132
133    @Override
134    public boolean isSameFile(final Path path, final Path path2) throws IOException {
135        return delegate.isSameFile(path, path2);
136    }
137
138    @Override
139    public boolean isHidden(final Path path) throws IOException {
140        return delegate.isHidden(path);
141    }
142
143    @Override
144    public FileStore getFileStore(final Path path) throws IOException {
145        return delegate.getFileStore(path);
146    }
147
148    @Override
149    public void checkAccess(final Path path, final AccessMode... modes) throws IOException {
150        delegate.checkAccess(path, modes);
151    }
152
153    @Override
154    public <V extends FileAttributeView> V getFileAttributeView(final Path path,
155                                                                final Class<V> type,
156                                                                final LinkOption... options) {
157        return delegate.getFileAttributeView(path, type, options);
158    }
159
160    @Override
161    public <A extends BasicFileAttributes> A readAttributes(final Path path,
162                                                            final Class<A> type,
163                                                            final LinkOption... options) throws IOException {
164        return delegate.readAttributes(path, type, options);
165    }
166
167    @Override
168    public Map<String, Object> readAttributes(final Path path,
169                                              final String attributes,
170                                              final LinkOption... options) throws IOException {
171        return delegate.readAttributes(path, attributes, options);
172    }
173
174    @Override
175    public void setAttribute(final Path path,
176                             final String attribute,
177                             final Object value,
178                             final LinkOption... options) throws IOException {
179        delegate.setAttribute(path, attribute, value, options);
180    }
181
182}
