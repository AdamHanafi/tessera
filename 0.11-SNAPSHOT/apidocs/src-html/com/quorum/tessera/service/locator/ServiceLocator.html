001package com.quorum.tessera.service.locator;
002
003import com.quorum.tessera.ServiceLoaderUtil;
004
005import java.util.Set;
006
007/** Creates a set of services that are configured */
008public interface ServiceLocator {
009
010    /**
011     * Retrieves all the services specified in the configuration file
012     *
013     * @return the set of all configuration services
014     */
015    Set<Object> getServices();
016
017    /**
018     * Returns an implementation of the {@link ServiceLocator} from the service loader
019     *
020     * @return the {@link ServiceLocator} instance
021     */
022    static ServiceLocator create() {
023        // TODO: return the stream and let the caller deal with it
024        return ServiceLoaderUtil.loadAll(ServiceLocator.class).findAny().get();
025    }
026}
