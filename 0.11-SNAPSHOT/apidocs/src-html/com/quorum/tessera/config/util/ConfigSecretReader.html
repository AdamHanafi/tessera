001package com.quorum.tessera.config.util;
002
003import com.quorum.tessera.io.SystemAdapter;
004import com.quorum.tessera.passwords.PasswordReaderFactory;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.IOException;
009import java.nio.file.Files;
010import java.nio.file.Path;
011import java.nio.file.Paths;
012import java.util.Optional;
013
014public final class ConfigSecretReader {
015
016    private static final Logger LOGGER = LoggerFactory.getLogger(ConfigSecretReader.class);
017
018    private ConfigSecretReader() {}
019
020    public static Optional<String> readSecretFromFile() {
021
022        final EnvironmentVariableProvider envProvider = new EnvironmentVariableProvider();
023
024        if (envProvider.hasEnv(EnvironmentVariables.CONFIG_SECRET_PATH)) {
025            final Path secretPath = Paths.get(envProvider.getEnv(EnvironmentVariables.CONFIG_SECRET_PATH));
026            if (Files.exists(secretPath)) {
027                try {
028                    return Optional.of(new String(Files.readAllBytes(secretPath)).trim());
029                } catch (IOException ex) {
030                    LOGGER.error("Error while reading secret from file");
031                }
032            }
033        }
034
035        LOGGER.warn("Not able to find or read any secret for decrypting sensitive values in config.");
036
037        return Optional.empty();
038    }
039
040    public static String readSecretFromConsole() {
041        SystemAdapter.INSTANCE.out().println("Please enter the secret/password used to decrypt config value");
042        return PasswordReaderFactory.create().readPasswordFromConsole();
043    }
044}
