001package com.quorum.tessera.config;
002
003import javax.xml.bind.annotation.XmlEnumValue;
004import java.util.Arrays;
005import java.util.Collections;
006import java.util.Set;
007import java.util.stream.Collectors;
008
009public enum AppType {
010    P2P(CommunicationType.REST, CommunicationType.GRPC),
011
012    Q2T(CommunicationType.REST, CommunicationType.GRPC),
013
014    @XmlEnumValue("ThirdParty")
015    THIRD_PARTY(CommunicationType.REST),
016
017    ENCLAVE(CommunicationType.REST),
018
019    ADMIN(CommunicationType.REST);
020
021    private final Set<CommunicationType> allowedCommunicationTypes;
022
023    AppType(CommunicationType... allowedCommunicationTypes) {
024        this.allowedCommunicationTypes =
025                Collections.unmodifiableSet(Arrays.stream(allowedCommunicationTypes).collect(Collectors.toSet()));
026    }
027
028    public Set<CommunicationType> getAllowedCommunicationTypes() {
029        return allowedCommunicationTypes;
030    }
031}
