001package com.quorum.tessera.config.adapters;
002
003import com.quorum.tessera.config.KeyData;
004import com.quorum.tessera.config.keypairs.*;
005
006import javax.xml.bind.annotation.adapters.XmlAdapter;
007import java.util.Objects;
008
009public class KeyDataAdapter extends XmlAdapter<KeyData, ConfigKeyPair> {
010
011    public static final String NACL_FAILURE_TOKEN = "NACL_FAILURE";
012
013    @Override
014    public ConfigKeyPair unmarshal(final KeyData keyData) {
015
016        //case 1, the keys are provided inline
017        if (Objects.nonNull(keyData.getPrivateKey()) && Objects.nonNull(keyData.getPublicKey())) {
018            return new DirectKeyPair(keyData.getPublicKey(), keyData.getPrivateKey());
019        }
020
021        //case 2, the config is provided inline
022        if (keyData.getPublicKey() != null && keyData.getConfig() != null) {
023            return new InlineKeypair(keyData.getPublicKey(), keyData.getConfig());
024        }
025
026        //case 3, the Azure Key Vault data is provided
027        if(keyData.getAzureVaultPublicKeyId() != null && keyData.getAzureVaultPrivateKeyId() != null) {
028            return new AzureVaultKeyPair(keyData.getAzureVaultPublicKeyId(), keyData.getAzureVaultPrivateKeyId(), keyData.getAzureVaultPublicKeyVersion(), keyData.getAzureVaultPrivateKeyVersion());
029        }
030
031        //case 4, the Hashicorp Vault data is provided
032        if(keyData.getHashicorpVaultPublicKeyId() != null && keyData.getHashicorpVaultPrivateKeyId() != null
033            && keyData.getHashicorpVaultSecretEngineName() != null && keyData.getHashicorpVaultSecretName() != null) {
034            return new HashicorpVaultKeyPair(keyData.getHashicorpVaultPublicKeyId(), keyData.getHashicorpVaultPrivateKeyId(), keyData.getHashicorpVaultSecretEngineName(), keyData.getHashicorpVaultSecretName(), keyData.getHashicorpVaultSecretVersion());
035        }
036
037        //case 5, the keys are provided inside a file
038        if(keyData.getPublicKeyPath() != null && keyData.getPrivateKeyPath() != null) {
039            return new FilesystemKeyPair(keyData.getPublicKeyPath(), keyData.getPrivateKeyPath());
040        }
041
042        //case 6, the key config specified is invalid
043        return new UnsupportedKeyPair(
044            keyData.getConfig(),
045            keyData.getPrivateKey(),
046            keyData.getPublicKey(),
047            keyData.getPrivateKeyPath(),
048            keyData.getPublicKeyPath(),
049            keyData.getAzureVaultPublicKeyId(),
050            keyData.getAzureVaultPrivateKeyId(),
051            keyData.getAzureVaultPublicKeyVersion(),
052            keyData.getAzureVaultPrivateKeyVersion(),
053            keyData.getHashicorpVaultPublicKeyId(),
054            keyData.getHashicorpVaultPrivateKeyId(),
055            keyData.getHashicorpVaultSecretEngineName(),
056            keyData.getHashicorpVaultSecretName(),
057            keyData.getHashicorpVaultSecretVersion()
058        );
059    }
060
061    @Override
062    public KeyData marshal(final ConfigKeyPair keyPair) {
063
064        KeyData keyData = new KeyData();
065
066        if(keyPair instanceof DirectKeyPair) {
067            DirectKeyPair kp = (DirectKeyPair) keyPair;
068
069            keyData.setPublicKey(kp.getPublicKey());
070            keyData.setPrivateKey(kp.getPrivateKey());
071            return keyData;
072        }
073
074        if(keyPair instanceof InlineKeypair) {
075            InlineKeypair kp = (InlineKeypair) keyPair;
076
077            keyData.setPublicKey(kp.getPublicKey());
078            keyData.setConfig(kp.getPrivateKeyConfig());
079            return keyData;
080        }
081
082        if(keyPair instanceof AzureVaultKeyPair) {
083            AzureVaultKeyPair kp = (AzureVaultKeyPair) keyPair;
084
085            keyData.setAzureVaultPublicKeyId(kp.getPublicKeyId());
086            keyData.setAzureVaultPrivateKeyId(kp.getPrivateKeyId());
087            keyData.setAzureVaultPublicKeyVersion(kp.getPublicKeyVersion());
088            keyData.setAzureVaultPrivateKeyVersion(kp.getPrivateKeyVersion());
089            return keyData;
090        }
091
092        if(keyPair instanceof HashicorpVaultKeyPair) {
093            HashicorpVaultKeyPair kp = (HashicorpVaultKeyPair) keyPair;
094
095            keyData.setHashicorpVaultPublicKeyId(kp.getPublicKeyId());
096            keyData.setHashicorpVaultPrivateKeyId(kp.getPrivateKeyId());
097            keyData.setHashicorpVaultSecretEngineName(kp.getSecretEngineName());
098            keyData.setHashicorpVaultSecretName(kp.getSecretName());
099            return keyData;
100        }
101
102        if(keyPair instanceof FilesystemKeyPair) {
103            FilesystemKeyPair kp = (FilesystemKeyPair) keyPair;
104
105            keyData.setPublicKeyPath(kp.getPublicKeyPath());
106            keyData.setPrivateKeyPath(kp.getPrivateKeyPath());
107            return keyData;
108        }
109
110        if(keyPair instanceof UnsupportedKeyPair) {
111            UnsupportedKeyPair kp = (UnsupportedKeyPair) keyPair;
112            return new KeyData(
113                kp.getConfig(),
114                kp.getPrivateKey(),
115                kp.getPublicKey(),
116                kp.getPrivateKeyPath(),
117                kp.getPublicKeyPath(),
118                kp.getAzureVaultPrivateKeyId(),
119                kp.getAzureVaultPublicKeyId(),
120                kp.getAzureVaultPublicKeyVersion(),
121                kp.getAzureVaultPrivateKeyVersion(),
122                kp.getHashicorpVaultPrivateKeyId(),
123                kp.getHashicorpVaultPublicKeyId(),
124                kp.getHashicorpVaultSecretEngineName(),
125                kp.getHashicorpVaultSecretName(),
126                kp.getHashicorpVaultSecretVersion());
127        }
128
129        throw new UnsupportedOperationException("The keypair type " + keyPair.getClass() + " is not allowed");
130    }
131}
