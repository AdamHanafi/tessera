001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.keypairs.ConfigKeyPair;
004import com.quorum.tessera.config.util.JaxbUtil;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.nio.file.Files;
009import java.nio.file.Path;
010import java.nio.file.Paths;
011import java.nio.file.attribute.PosixFilePermission;
012import java.util.List;
013import java.util.Objects;
014import java.util.Set;
015import java.util.stream.Collectors;
016import java.util.stream.Stream;
017
018import static java.nio.file.StandardOpenOption.APPEND;
019
020public class JaxbConfigFactory implements ConfigFactory {
021
022    private static final Set<PosixFilePermission> NEW_PASSWORD_FILE_PERMS = Stream
023        .of(PosixFilePermission.OWNER_READ, PosixFilePermission.OWNER_WRITE)
024        .collect(Collectors.toSet());
025    
026    @Override
027    public Config create(final InputStream configData, final List<ConfigKeyPair> newKeys) {
028
029        final Config config = JaxbUtil.unmarshal(configData, Config.class);
030
031        boolean createdNewPasswordFile = false;
032
033        if (Objects.nonNull(config.getKeys()) && !newKeys.isEmpty()) {
034            try {
035                final List<String> newPasswords = newKeys
036                    .stream()
037                    .map(ConfigKeyPair::getPassword)
038                    .collect(Collectors.toList());
039
040                if (config.getKeys().getPasswords() != null) {
041                    config.getKeys().getPasswords().addAll(newPasswords);
042                } else if (config.getKeys().getPasswordFile() != null) {
043                    this.createFile(config.getKeys().getPasswordFile());
044                    Files.write(config.getKeys().getPasswordFile(), newPasswords, APPEND);
045                } else if (!newPasswords.stream().allMatch(Objects::isNull)) {
046                    final List<String> existingPasswords = config
047                        .getKeys()
048                        .getKeyData()
049                        .stream()
050                        .map(k -> "")
051                        .collect(Collectors.toList());
052                    existingPasswords.addAll(newPasswords);
053
054                    this.createFile(Paths.get("passwords.txt"));
055                    Files.write(Paths.get("passwords.txt"), existingPasswords, APPEND);
056                    createdNewPasswordFile = true;
057                }
058            } catch (final IOException ex) {
059                throw new RuntimeException("Could not store new passwords: " + ex.getMessage());
060            }
061
062            config.getKeys().getKeyData().addAll(newKeys);
063
064        }
065
066        if(createdNewPasswordFile) {
067            //return a new object with the password file set
068            return new Config(
069                    config.getJdbcConfig(),
070                    config.getServerConfigs(),
071                    config.getPeers(),
072                    new KeyConfiguration(Paths.get("passwords.txt"), null, config.getKeys().getKeyData(), config.getKeys().getAzureKeyVaultConfig(), config.getKeys().getHashicorpKeyVaultConfig()),
073                    config.getAlwaysSendTo(),
074                    config.getUnixSocketFile(),
075                    config.isUseWhiteList(),
076                    config.isDisablePeerDiscovery()
077            );
078        } else {
079            //leave config untouched since it wasn't needed to make a new one
080            return config;
081        }
082    }
083
084    //create a file if it doesn't exist and set the permissions to be only
085    // read/write for the creator
086    private void createFile(final Path fileToMake) throws IOException {
087        if(Files.notExists(fileToMake)) {
088            Files.createFile(fileToMake);
089            Files.setPosixFilePermissions(fileToMake, NEW_PASSWORD_FILE_PERMS);
090        }
091    }
092
093}
