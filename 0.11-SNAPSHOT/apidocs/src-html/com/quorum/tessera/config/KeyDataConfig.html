001package com.quorum.tessera.config;
002
003import com.quorum.tessera.config.adapters.PrivateKeyTypeAdapter;
004
005import javax.validation.constraints.NotNull;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlAttribute;
009import javax.xml.bind.annotation.XmlElement;
010import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
011import java.util.Optional;
012
013@XmlAccessorType(XmlAccessType.FIELD)
014public class KeyDataConfig {
015
016    @NotNull
017    @XmlElement(name = "data")
018    private PrivateKeyData privateKeyData;
019
020    @NotNull
021    @XmlAttribute
022    @XmlJavaTypeAdapter(PrivateKeyTypeAdapter.class)
023    private PrivateKeyType type;
024
025    public KeyDataConfig(PrivateKeyData privateKeyData, PrivateKeyType type) {
026        this.privateKeyData = privateKeyData;
027        this.type = type;
028    }
029
030    public KeyDataConfig() {
031
032    }
033
034    public PrivateKeyType getType() {
035        return type;
036    }
037
038    public PrivateKeyData getPrivateKeyData() {
039        return privateKeyData;
040    }
041
042    public String getValue() {
043        return Optional.ofNullable(privateKeyData)
044                .map(PrivateKeyData::getValue)
045                .orElse(null);  
046    }
047
048    public String getSnonce() {
049        return Optional.ofNullable(privateKeyData)
050                .map(PrivateKeyData::getSnonce)
051                .orElse(null);   
052    }
053
054    public String getAsalt() {
055        return Optional.ofNullable(privateKeyData)
056                .map(PrivateKeyData::getAsalt)
057                .orElse(null);   
058    }
059
060    public String getSbox() {
061        return Optional.ofNullable(privateKeyData)
062                .map(PrivateKeyData::getSbox)
063                .orElse(null);   
064    }
065
066    public ArgonOptions getArgonOptions() {
067        return Optional.ofNullable(privateKeyData)
068                .map(PrivateKeyData::getArgonOptions)
069                .orElse(null);   
070    }
071
072}
