001package com.quorum.tessera.config.constraints;
002
003import java.lang.annotation.Documented;
004import static java.lang.annotation.ElementType.*;
005import java.lang.annotation.Repeatable;
006import java.lang.annotation.Retention;
007import static java.lang.annotation.RetentionPolicy.RUNTIME;
008import java.lang.annotation.Target;
009import javax.validation.Constraint;
010import javax.validation.Payload;
011
012@Documented
013@Constraint(validatedBy = Base64Validator.class)
014@Target({FIELD, METHOD, PARAMETER, ANNOTATION_TYPE, TYPE_USE})
015@Retention(RUNTIME)
016@Repeatable(value = ValidBase64.List.class)
017public @interface ValidBase64 {
018
019    String message() default "{ValidBase64.message}";
020
021    Class<?>[] groups() default {};
022
023    Class<? extends Payload>[] payload() default {};
024
025    @Target(value = {METHOD, FIELD, ANNOTATION_TYPE, PARAMETER, TYPE_USE})
026    @Retention(value = RUNTIME)
027    @Documented
028    @interface List {
029        ValidBase64[] value();
030    }
031
032}
