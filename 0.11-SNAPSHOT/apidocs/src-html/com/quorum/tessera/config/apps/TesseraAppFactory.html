001package com.quorum.tessera.config.apps;
002
003import com.quorum.tessera.ServiceLoaderUtil;
004import com.quorum.tessera.config.AppType;
005import com.quorum.tessera.config.CommunicationType;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import java.util.ArrayList;
010import java.util.List;
011import java.util.Optional;
012
013public class TesseraAppFactory {
014
015    private static final Logger LOGGER = LoggerFactory.getLogger(TesseraAppFactory.class);
016
017    private final List<TesseraApp> cache = new ArrayList<>();
018
019    private static final TesseraAppFactory INSTANCE = new TesseraAppFactory();
020
021    public static Optional<TesseraApp> create(CommunicationType communicationType, AppType appType) {
022        return INSTANCE.createApp(communicationType, appType);
023    }
024
025    private TesseraAppFactory() {
026        ServiceLoaderUtil.loadAll(TesseraApp.class).peek(app -> LOGGER.info("Loaded app {}", app)).forEach(cache::add);
027    }
028
029    private Optional<TesseraApp> createApp(CommunicationType communicationType, AppType appType) {
030        LOGGER.info("Creating application type {} for {}", appType, communicationType);
031        return cache.stream()
032                .filter(a -> a.getAppType() == appType)
033                .filter(a -> a.getCommunicationType() == communicationType)
034                .findAny();
035    }
036}
