001package com.quorum.tessera.encryption;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006import java.util.Collection;
007import java.util.HashSet;
008import java.util.Objects;
009import java.util.Set;
010import java.util.stream.Collectors;
011
012public class KeyManagerImpl implements KeyManager {
013
014    private static final Logger LOGGER = LoggerFactory.getLogger(KeyManagerImpl.class);
015
016    /**
017     * A list of all pub/priv keys that are attached to this node
018     */
019    private final Set<KeyPair> localKeys;
020
021    private final KeyPair defaultKeys;
022
023    private final Set<PublicKey> forwardingPublicKeys;
024
025    public KeyManagerImpl(final Collection<KeyPair> keys, Collection<PublicKey> forwardKeys) {
026
027        this.localKeys = new HashSet<>(keys);
028
029        this.defaultKeys = localKeys.iterator().next();
030
031        this.forwardingPublicKeys = new HashSet<>(forwardKeys);
032    }
033
034    @Override
035    public PublicKey getPublicKeyForPrivateKey(final PrivateKey privateKey) {
036        LOGGER.debug("Attempting to find public key for the private key {}", privateKey);
037
038        final PublicKey publicKey = localKeys
039            .stream()
040            .filter(keypair -> Objects.equals(keypair.getPrivateKey(), privateKey))
041            .findFirst()
042            .map(KeyPair::getPublicKey)
043            .orElseThrow(() -> new KeyNotFoundException(
044                "Private key " + privateKey.encodeToBase64() + " not found when searching for public key"
045            ));
046
047        LOGGER.debug("Found public key {} for private key {}", publicKey, privateKey);
048
049        return publicKey;
050    }
051
052
053    @Override
054    public PrivateKey getPrivateKeyForPublicKey(final PublicKey publicKey) {
055        LOGGER.debug("Attempting to find private key for the public key {}", publicKey);
056
057        final PrivateKey privateKey = localKeys
058            .stream()
059            .filter(keypair -> Objects.equals(keypair.getPublicKey(), publicKey))
060            .findFirst()
061            .map(KeyPair::getPrivateKey)
062            .orElseThrow(() -> new KeyNotFoundException(
063                "Public key " + publicKey.encodeToBase64() + " not found when searching for private key"
064            ));
065
066        LOGGER.debug("Found private key {} for public key {}", privateKey, publicKey);
067
068        return privateKey;
069    }
070
071    @Override
072    public Set<PublicKey> getPublicKeys() {
073        return localKeys
074            .stream()
075            .map(KeyPair::getPublicKey)
076            .collect(Collectors.toSet());
077    }
078
079    @Override
080    public PublicKey defaultPublicKey() {
081        return defaultKeys.getPublicKey();
082    }
083
084    @Override
085    public Set<PublicKey> getForwardingKeys() {
086        return this.forwardingPublicKeys;
087    }
088
089}
