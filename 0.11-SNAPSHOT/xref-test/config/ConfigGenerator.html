24      public List<ConfigDescriptor> generateConfigs(ExecutionContext executionContext) {
25  
26          Path path = calculatePath(executionContext);
27          try {
28              Files.createDirectories(path);
29          } catch (IOException ex) {
30              throw new UncheckedIOException(ex);
31          }
32  
33          List<Config> configs = createConfigs(executionContext);
34  
35          List<Config> enclaveConfigs;
36          if (executionContext.getEnclaveType() == EnclaveType.REMOTE) {
37              enclaveConfigs = configs.stream().map(this::createEnclaveConfig).collect(Collectors.toList());
38              configs.forEach(c -> c.setKeys(null));
39          } else {
40              enclaveConfigs = Collections.emptyList();
41          }
42          // Remove keys
43          List<ConfigDescriptor> configList = new ArrayList<>();
44  
45          for (NodeAlias alias : NodeAlias.values()) {
46              int i = alias.ordinal();
47              final Config config = configs.get(i);
48  
49              final String filename = String.format("config%d.json", (i + 1));
50              final Path ouputFile = path.resolve(filename);
51  
52              try (OutputStream out = Files.newOutputStream(ouputFile)) {
53                  JaxbUtil.marshalWithNoValidation(config, out);
54  
55              } catch (IOException ex) {
56                  throw new UncheckedIOException(ex);
57              }
58  
59              final Config enclaveConfig;
60              final Path enclaveOuputFile;
61              if (!enclaveConfigs.isEmpty()) {
62                  enclaveConfig = enclaveConfigs.get(i);
63                  String enclaveFilename = String.format("enclave%d.json", (i + 1));
64                  enclaveOuputFile = path.resolve(enclaveFilename);
65                  try (OutputStream enclaveout = Files.newOutputStream(enclaveOuputFile)) {
66                      JaxbUtil.marshalWithNoValidation(enclaveConfig, enclaveout);
67  
68                  } catch (IOException ex) {
69                      throw new UncheckedIOException(ex);
70                  }
71              } else {
72                  enclaveConfig = null;
73                  enclaveOuputFile = null;
74              }
75  
76              configList.add(new ConfigDescriptor(alias, ouputFile, config, enclaveConfig, enclaveOuputFile));
77          }
78  
79          return configList;
80      }
81  
82      Config createEnclaveConfig(Config config) {
83  
84          final Config enclaveConfig = new Config();
85  
86          ServerConfig serverConfig =
87                  config.getServerConfigs().stream().filter(s -> s.getApp() == AppType.ENCLAVE).findAny().get();
88  
89          enclaveConfig.setServerConfigs(Arrays.asList(serverConfig));
90  
91          enclaveConfig.setKeys(config.getKeys());
92          enclaveConfig.setAlwaysSendTo(config.getAlwaysSendTo());
93  
94          return enclaveConfig;
95      }

97      public static Path calculatePath(ExecutionContext executionContext) {
98          try {
99              URI baseUri = ConfigGenerator.class.getResource("/").toURI();
100 
101             return executionContext
102                     .getPrefix()
103                     .map(v -> Paths.get(baseUri).resolve(v))
104                     .orElse(Paths.get(baseUri))
105                     .resolve(executionContext.getCommunicationType().name().toLowerCase())
106                     .resolve(executionContext.getSocketType().name().toLowerCase())
107                     .resolve(executionContext.getDbType().name().toLowerCase())
108                     .resolve("enclave-" + executionContext.getEnclaveType().name().toLowerCase());
109         } catch (URISyntaxException ex) {
110             throw new RuntimeException(ex);
111         }
112     }

162     public List<Config> createConfigs(ExecutionContext executionContext) {
163 
164         PortUtil port = new PortUtil(50520);
165         String nodeId = NodeId.generate(executionContext);
166         final FeatureToggles toggles = new FeatureToggles();
167         toggles.setEnableRemoteKeyValidation(true);
168 
169         Config first =
170                 new ConfigBuilder()
171                         .withNodeId(nodeId)
172                         .withNodeNumber(1)
173                         .withExecutionContext(executionContext)
174                         .withQt2Port(port.nextPort())
175                         .withP2pPort(port.nextPort())
176                         .withAdminPort(port.nextPort())
177                         .withEnclavePort(port.nextPort())
178                         .withKeys(keyLookUp.get(1))
179                         .withFeatureToggles(toggles)
180                         .build();
181 
182         Config second =
183                 new ConfigBuilder()
184                         .withNodeId(nodeId)
185                         .withNodeNumber(2)
186                         .withExecutionContext(executionContext)
187                         .withQt2Port(port.nextPort())
188                         .withP2pPort(port.nextPort())
189                         .withAdminPort(port.nextPort())
190                         .withEnclavePort(port.nextPort())
191                         .withKeys(keyLookUp.get(2))
192                         .withFeatureToggles(toggles)
193                         .build();
194 
195         Config third =
196                 new ConfigBuilder()
197                         .withNodeId(nodeId)
198                         .withNodeNumber(3)
199                         .withExecutionContext(executionContext)
200                         .withQt2Port(port.nextPort())
201                         .withP2pPort(port.nextPort())
202                         .withAdminPort(port.nextPort())
203                         .withEnclavePort(port.nextPort())
204                         .withAlwaysSendTo(keyLookUp.get(1).keySet().iterator().next())
205                         .withKeys(keyLookUp.get(3))
206                         .withFeatureToggles(toggles)
207                         .build();
208 
209         Config fourth =
210                 new ConfigBuilder()
211                         .withNodeId(nodeId)
212                         .withNodeNumber(4)
213                         .withExecutionContext(executionContext)
214                         .withQt2Port(port.nextPort())
215                         .withP2pPort(port.nextPort())
216                         .withAdminPort(port.nextPort())
217                         .withEnclavePort(port.nextPort())
218                         .withKeys(keyLookUp.get(4))
219                         .withFeatureToggles(toggles)
220                         .build();
221 
222         first.addPeer(new Peer(second.getP2PServerConfig().getServerAddress()));
223         second.addPeer(new Peer(third.getP2PServerConfig().getServerAddress()));
224         third.addPeer(new Peer(fourth.getP2PServerConfig().getServerAddress()));
225         fourth.addPeer(new Peer(first.getP2PServerConfig().getServerAddress()));
226 
227         return Arrays.asList(first, second, third, fourth);
228     }

230     public static void main(String[] args) throws Exception {
231 
232         System.setProperty("javax.xml.bind.JAXBContextFactory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
233         System.setProperty("javax.xml.bind.context.factory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
234 
235         ExecutionContext executionContext =
236                 ExecutionContext.Builder.create()
237                         .with(CommunicationType.REST)
238                         .with(DBType.H2)
239                         .with(SocketType.UNIX)
240                         .with(EnclaveType.LOCAL)
241                         .prefix("p2p")
242                         .build();
243 
244         Path path = ConfigGenerator.calculatePath(executionContext);
245 
246         List<Config> configs = new ConfigGenerator().createConfigs(executionContext);
247 
248         for (int i = 1; i <= configs.size(); i++) {
249             String filename = String.format("config%d.json", i);
250             Path ouputFile = path.resolve(filename);
251             System.out.println(ouputFile);
252         }
