107     public Config build() {
108         final Config config = new Config();
109 
110         JdbcConfig jdbcConfig = new JdbcConfig();
111         jdbcConfig.setUrl(executionContext.getDbType().createUrl(nodeId, nodeNumber));
112         jdbcConfig.setUsername("sa");
113         jdbcConfig.setPassword("");
114         config.setJdbcConfig(jdbcConfig);
115 
116         ServerConfig q2tServerConfig = new ServerConfig();
117         q2tServerConfig.setApp(AppType.Q2T);
118         q2tServerConfig.setEnabled(true);
119         q2tServerConfig.setCommunicationType(executionContext.getCommunicationType());
120 
121         if (executionContext.getCommunicationType() == CommunicationType.REST
122                 && (q2tSocketType != null || executionContext.getSocketType() == SocketType.UNIX)) {
123             q2tServerConfig.setServerAddress(String.format("unix:/tmp/q2t-rest-unix-%d.ipc", nodeNumber));
124         } else {
125             q2tServerConfig.setServerAddress("http://localhost:" + q2tPort);
126             q2tServerConfig.setBindingAddress("http://0.0.0.0:" + q2tPort);
127         }
128 
129         List<ServerConfig> servers = new ArrayList<>();
130 
131         servers.add(q2tServerConfig);
132 
133         ServerConfig p2pServerConfig = new ServerConfig();
134         p2pServerConfig.setApp(AppType.P2P);
135         p2pServerConfig.setEnabled(true);
136         p2pServerConfig.setCommunicationType(executionContext.getCommunicationType());
137         p2pServerConfig.setServerAddress("http://localhost:" + p2pPort);
138         p2pServerConfig.setBindingAddress("http://0.0.0.0:" + p2pPort);
139         servers.add(p2pServerConfig);
140 
141         if (executionContext.getCommunicationType() == CommunicationType.REST && Objects.nonNull(adminPort)) {
142             ServerConfig adminServerConfig = new ServerConfig();
143             adminServerConfig.setApp(AppType.ADMIN);
144             adminServerConfig.setEnabled(true);
145             adminServerConfig.setServerAddress("http://localhost:" + adminPort);
146             adminServerConfig.setBindingAddress("http://0.0.0.0:" + adminPort);
147             adminServerConfig.setCommunicationType(CommunicationType.REST);
148 
149             servers.add(adminServerConfig);
150         }
151 
152         if (executionContext.getEnclaveType() == EnclaveType.REMOTE) {
153             ServerConfig enclaveServerConfig = new ServerConfig();
154             enclaveServerConfig.setApp(AppType.ENCLAVE);
155             enclaveServerConfig.setEnabled(true);
156             SslConfig sslConfig =
157                     new SslConfig(
158                             SslAuthenticationMode.STRICT,
159                             false,
160                             Paths.get(
161                                     getClass().getResource("/certificates/localhost-with-san-keystore.jks").getFile()),
162                             "testtest",
163                             Paths.get(getClass().getResource("/certificates/truststore.jks").getFile()),
164                             "testtest",
165                             SslTrustMode.CA,
166                             Paths.get(getClass().getResource("/certificates/quorum-client-keystore.jks").getFile()),
167                             "testtest",
168                             Paths.get(getClass().getResource("/certificates/truststore.jks").getFile()),
169                             "testtest",
170                             SslTrustMode.CA,
171                             null,
172                             null,
173                             null,
174                             null,
175                             null,
176                             null,
177                             null,
178                             null,
179                             null);
180             enclaveServerConfig.setBindingAddress("http://0.0.0.0:" + enclavePort);
181             enclaveServerConfig.setServerAddress("http://localhost:" + enclavePort);
182             // enclaveServerConfig.setSslConfig(sslConfig);
183             enclaveServerConfig.setCommunicationType(CommunicationType.REST);
184 
185             servers.add(enclaveServerConfig);
186         }
187 
188         config.setServerConfigs(servers);
189 
190         peerUrls.stream().map(Peer::new).forEach(config::addPeer);
191 
192         config.setKeys(new KeyConfiguration());
193 
194         final List<ConfigKeyPair> pairs =
195                 keys.entrySet().stream()
196                         .map(e -> new DirectKeyPair(e.getKey(), e.getValue()))
197                         .collect(Collectors.toList());
198 
199         config.getKeys().setKeyData(pairs);
200 
201         config.setAlwaysSendTo(alwaysSendTo);
202 
203         config.setFeatures(featureToggles);
204 
205         return config;
206     }

213         ExecutionContext executionContext =
214                 ExecutionContext.Builder.create()
215                         .with(CommunicationType.REST)

220 
221         Config config =
222                 new ConfigBuilder()
223                         .withExecutionContext(executionContext)
224                         .withNodeId("mynode")
225                         .withNodeNumber(1)
226                         .withPeer("http://localhost:999")
227                         .withKeys(
228                                 "/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=",

232                         .withEnclavePort(989)
