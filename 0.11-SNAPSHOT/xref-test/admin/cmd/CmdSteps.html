18  public class CmdSteps implements En {
19  
20      private final PartyHelper partyHelper = PartyHelper.create();
21  
22      private final RestUtils restUtils = new RestUtils();
23  
24  
25      public CmdSteps() {
26  
27          Party subjectNode = partyHelper.getParties().findAny().get();
28          Client client = ClientBuilder.newClient();
29          Given("any node is running", () -> {
30              assertThat(Stream.of(subjectNode)
31                  .map(Party::getP2PUri)
32                  .map(client::target)
33                  .map(t -> t.path("upcheck"))
34                  .map(WebTarget::request)
35                  .map(Invocation.Builder::get)
36                  .allMatch(r -> r.getStatus() == 200))
37                  .isTrue();
38  
39          });
40  
41          When("admin user executes add peer", () -> {
42              int exitcode = Utils.addPeer(subjectNode, "bogus");
43              assertThat(exitcode).isEqualTo(0);
44          });
45  
46          Then("a peer is added to party", () -> {
47  
48              Response response = Stream.of(subjectNode)
49                  .map(Party::getAdminUri)
50                  .map(client::target)
51                  .map(t -> t.path("config"))
52                  .map(t -> t.path("peers"))
53                  .map(WebTarget::request)
54                  .map(Invocation.Builder::get).findAny().get();
55  
56              assertThat(response.getStatus()).isEqualTo(200);
57              Peer[] peers = response.readEntity(Peer[].class);
58  
59              List<String> urls = Stream.of(peers)
60                  .map(Peer::getUrl)
61                  .collect(Collectors.toList());
62  
63              assertThat(urls).contains("bogus");
64  
65          });
66      }
67  
68  }
