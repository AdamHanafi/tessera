3   import com.quorum.tessera.config.ServerConfig;
4   import com.quorum.tessera.jaxrs.client.ClientFactory;
5   import com.quorum.tessera.test.DefaultPartyHelper;
6   import com.quorum.tessera.test.Party;
7   import com.quorum.tessera.test.PartyHelper;
8   import java.util.Arrays;
9   import java.util.List;
10  import java.util.stream.Collectors;
11  import javax.json.JsonObject;
12  import javax.ws.rs.client.Client;
13  import javax.ws.rs.core.Response;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  
17  public class RestPartyInfoChecker implements PartyInfoChecker {
18  
19      private static final Logger LOGGER = LoggerFactory.getLogger(RestPartyInfoChecker.class);
20  
21      private PartyHelper partyHelper = new DefaultPartyHelper();
22  
23      @Override
24      public boolean hasSynced() {
25          LOGGER.info("hasSynced {}", this);
26  
27          List<Party> parties = partyHelper.getParties().collect(Collectors.toList());
28  
29          Boolean[] results = new Boolean[parties.size()];
30  
31          for (int i = 0; i < parties.size(); i++) {
32              Party p = parties.get(i);
33              ServerConfig p2pConfig = p.getConfig().getP2PServerConfig();
34              Client client = new ClientFactory().buildFrom(p2pConfig);
35  
36              LOGGER.debug("Request party info for {}. On {}", p.getAlias(), p2pConfig.getServerUri());
37              Response response = client.target(p2pConfig.getServerUri()).path("partyinfo").request().get();
38  
39              LOGGER.debug("Requested party info for {} . {}", p.getAlias(), response.getStatus());
40              if (response.getStatus() == 200) {
41                  final JsonObject result = response.readEntity(JsonObject.class);
42                  final long peerCount =
43                          result.getJsonArray("peers").stream()
44                                  .map(val -> (JsonObject) val)
45                                  .filter(peer -> !peer.isNull("lastContact"))
46                                  .count();
47  
48                  LOGGER.debug("Found {} peers of {} on {}", peerCount, parties.size(), p.getAlias());
49                  results[i] = peerCount == parties.size();
50              } else {
51                  results[i] = false;
52              }
53          }
54  
55          return Arrays.stream(results).allMatch(p -> p);
56      }
57  }
