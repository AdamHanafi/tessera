3   import java.io.File;
4   import java.io.IOException;
5   import java.net.URI;
6   import java.nio.file.Files;
7   import java.nio.file.Paths;
8   import jnr.unixsocket.UnixSocketAddress;
9   import jnr.unixsocket.UnixSocketChannel;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  public class UnixSocketServerStatusCheck implements ServerStatusCheck {
14  
15      private static final Logger LOGGER = LoggerFactory.getLogger(UnixSocketServerStatusCheck.class);
16  
17      private final File file;
18  
19      public UnixSocketServerStatusCheck(URI uri) {
20          this.file = Paths.get(uri).toFile();
21      }
22  
23      @Override
24      public boolean checkStatus() {
25          if(!Files.exists(file.toPath())) {
26              LOGGER.debug("File does not exist yet {}",file);
27              return false;
28          }
29          
30          if(!file.canRead()) {
31              LOGGER.debug("Cannot read file {} yet",file);
32              return false;
33          }
34          
35          if(!file.canWrite()) {
36              LOGGER.debug("Cannot write to file {} yet",file);
37              return false;
38          }
39          
40          LOGGER.debug("File {} exists can get read and write.",file);
41          UnixSocketAddress unixAddress = new UnixSocketAddress(file);
42          LOGGER.debug("Open socket address {}",unixAddress);
43          
44          try (UnixSocketChannel channel = UnixSocketChannel.open(unixAddress)){
45              LOGGER.debug("Opened channel on socket address {}",unixAddress);
46              return channel.isConnected();
47          } catch (IOException ex) {
48             LOGGER.debug("Exception connecting to {}. {}",unixAddress,ex.getMessage());
49              return false;
50          }
51      }
52  
53      @Override
54      public String toString() {
55          return "UnixSocketServerStatusCheck{" + "file=" + file + '}';
56      }
57  
58  }
