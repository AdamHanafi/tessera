1   package suite;

3   import com.google.protobuf.Empty;
4   import com.quorum.tessera.config.AppType;
5   import com.quorum.tessera.grpc.api.APITransactionGrpc;
6   import com.quorum.tessera.grpc.p2p.P2PTransactionGrpc;
7   import com.quorum.tessera.grpc.p2p.TesseraGrpc;
8   import com.quorum.tessera.grpc.p2p.UpCheckMessage;
9   import io.grpc.CallOptions;
10  import io.grpc.ManagedChannel;
11  import io.grpc.ManagedChannelBuilder;
12  import java.net.URL;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  public class GrpcServerStatusCheck implements ServerStatusCheck {
17  
18      private static final Logger LOGGER = LoggerFactory.getLogger(GrpcServerStatusCheck.class);
19  
20      private final AppType appType;
21  
22      private final URL url;
23  
24      public GrpcServerStatusCheck(URL url, AppType appType) {
25          this.url = url;
26          this.appType = appType;
27      }
28  
29      @Override
30      public boolean checkStatus() {
31  
32          ManagedChannel channel = ManagedChannelBuilder
33                  .forAddress(url.getHost(), url.getPort())
34                  .usePlaintext()
35                  .build();
36  
37          try{
38  
39  
40              if (appType == AppType.Q2T) {
41  
42                  CallOptions callOptions = APITransactionGrpc.newBlockingStub(channel)
43                          .getCallOptions();
44                  
45                  LOGGER.info("{} callOptions result {} ", appType, callOptions);
46                 return true;
47              }
48  
49  
50              if (appType == AppType.P2P) {
51  
52                  UpCheckMessage result = TesseraGrpc.newBlockingStub(channel)
53                          .getUpCheck(Empty.getDefaultInstance());
54                  LOGGER.info("{} Upcheck result {} ", appType, result.getUpCheck());
55  
56                  CallOptions callOptions = P2PTransactionGrpc.newBlockingStub(channel)
57                          .getCallOptions();
58                  
59                  LOGGER.info("{} callOptions result {} ", appType, callOptions);
60  
61                  return true;
62              }
63  
64              throw new UnsupportedOperationException(appType + " not supported");
65          } catch (Exception ex) {
66              LOGGER.info("cannot connect to {}. {}", this, ex.getMessage());
67              return false;
68          } finally {
69              channel.shutdown();
70          }
71  
72      }
73  
74      @Override
75      public String toString() {
76          return "GrpcServerStatusCheck{" + "appType=" + appType + ", url=" + url + '}';
77      }
78  
79  }
