26  import static org.assertj.core.api.Assertions.*;
27  import com.quorum.tessera.test.PartyHelper;

30  public class GrpcSteps implements En {
31  
32      private PartyHelper partyFactory = PartyHelper.create();
33  
34      public GrpcSteps() {
35  
36          Collection<Party> senderHolder = new ArrayList<>();
37  
38          Set<Party> recipients = new HashSet<>();
39  
40          final Set<String> storedHashes = new TreeSet<>();
41  
42          final byte[] txnData = Utils.generateTransactionData();
43  
44          Given("^Sender party (.+)$", (String pty) -> {
45              partyFactory.getParties()
46                  .filter(p -> p.getAlias().equals(pty))
47                  .findAny()
48                  .ifPresent(senderHolder::add);
49  
50          });
51  
52          And("^Recipient part(?:y|ies) (.+)$", (String alias) -> {
53              parseAliases(alias).stream()
54                  .map(partyFactory::findByAlias)
55                  .forEach(recipients::add);
56  
57              assertThat(recipients).isNotEmpty();
58          });
59  
60          And("^all parties are running$", () -> {
61  
62              partyFactory.getParties()
63                  .map(p -> createP2PChannel(p))
64                  .map(TesseraGrpc::newBlockingStub)
65                  .map(stub -> stub.getUpCheck(Empty.getDefaultInstance()))
66                  .forEach(mesage -> {
67                      assertThat(mesage.getUpCheck()).isNotNull().isEqualTo("I'm up!");
68                  });
69  
70          });
71  
72          When("sender party receives transaction from Quorum peer", () -> {
73  
74              Party sender = senderHolder.stream().findAny().get();
75  
76              List<String> recipientKeys = recipients.stream()
77                  .map(Party::getPublicKey)
78                  .collect(Collectors.toList());
79  
80              SendRequest sendRequest = SendRequest.newBuilder()
81                  .setFrom(sender.getPublicKey())
82                  .addAllTo(recipientKeys)
83                  .setPayload(ByteString.copyFrom(txnData))
84                  .build();
85  
86              SendResponse sendResponse = senderHolder.stream()
87                  .map(p -> createQ2TChannel(p))
88                  .map(APITransactionGrpc::newBlockingStub)
89                  .map(stub -> stub.send(sendRequest))
90                  .findAny().get();
91  
92              storedHashes.add(sendResponse.getKey());
93  
94          });
95  
96          When("sender party receives transaction with an unknown party from Quorum peer", () -> {
97  
98              Party sender = senderHolder.stream().findAny().get();
99  
100             SendRequest sendRequest = SendRequest.newBuilder()
101                 .setFrom(sender.getPublicKey())
102                 .addTo("8SjRHlUBe4hAmTk3KDeJ96RhN+s10xRrHDrxEi1O5W0=")
103                 .setPayload(ByteString.copyFrom(txnData))
104                 .build();
105 
106             APITransactionGrpc.APITransactionBlockingStub stub = senderHolder.stream()
107                 .map(p -> createQ2TChannel(p))
108                 .map(APITransactionGrpc::newBlockingStub)
109                 .findAny().get();
110 
111             try {
112                 stub.send(sendRequest);
113                 failBecauseExceptionWasNotThrown(io.grpc.StatusRuntimeException.class);
114             } catch (io.grpc.StatusRuntimeException ex) {
115                 //FIXME: Should be invalid arg
116                 assertThat(ex.getStatus())
117                         .describedAs("Expected  io.grpc.Status.UNKNOWN staus. recieved: "+ ex.getStatus())
118                         .isEqualTo(io.grpc.Status.UNKNOWN);
119             }
120 
121         });
122 
123         Then("an invalid request error is raised", () -> {
124             //FIXME: handled in requesting line
125         });
126 
127         Then("^sender party stores the transaction$", () -> {
128             Party sender = senderHolder.iterator().next();
129             try (PreparedStatement statement
130                 = sender.getDatabaseConnection().prepareStatement("SELECT COUNT(*) FROM ENCRYPTED_TRANSACTION WHERE HASH = ?")) {
131                 statement.setBytes(1, Base64.getDecoder().decode(storedHashes.iterator().next()));
132 
133                 try (ResultSet results = statement.executeQuery()) {
134                     assertThat(results.next()).isTrue();
135                     assertThat(results.getLong(1)).isEqualTo(1);
136                 }
137 
138             }
139         });
140 
141         Then("^forwards the transaction to recipient part(?:y|ies)$", () -> {
142 
143             String hash = storedHashes.stream().findAny().get();
144 
145             recipients.forEach(recipient -> {
146 
147                 ReceiveRequest receiveRequest = ReceiveRequest.newBuilder()
148                     .setTo(recipient.getPublicKey())
149                     .setKey(hash)
150                     .build();
151 
152                 ManagedChannel channel = createQ2TChannel(recipient);
153 
154                 ReceiveResponse receiveResponse = APITransactionGrpc.newBlockingStub(channel).receive(receiveRequest);
155                 assertThat(receiveResponse.getPayload().toByteArray()).isNotNull().isEqualTo(txnData);
156 
157                 channel.shutdownNow();
158             });
159 
160         });
161 
162         After(() -> {
163             channels.forEach(ManagedChannel::shutdownNow);
164             channels.clear();
165 
166         });
167 
168     }
169 
170     static List<String> parseAliases(String alias) {
171         return Arrays.asList(alias.split(",| and "));
172     }
173 
174     static List<ManagedChannel> channels = new ArrayList<>();
175 
176     static ManagedChannel createP2PChannel(Party party) {
177 
178         ManagedChannel channel = ManagedChannelBuilder.forAddress("127.0.0.1", party.getGrpcPort())
179             .usePlaintext()
180             .build();
181 
182         channels.add(channel);
183 
184         return channel;
185     }
186 
187     static ManagedChannel createQ2TChannel(Party party) {
188 
189         ManagedChannel channel = ManagedChannelBuilder.forAddress("127.0.0.1", party.getQ2TUri().getPort())
190             .usePlaintext()
191             .build();
192 
193         channels.add(channel);
194 
195         return channel;
196     }
