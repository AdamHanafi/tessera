24      /**
25       * This rule manages automatic graceful shutdown for the registered servers and channels at the
26       * end of test.
27       */
28      @Rule
29      public final GrpcCleanupRule grpcCleanup = new GrpcCleanupRule();
30  
31      private final PartyInfoGrpc.PartyInfoImplBase partyInfoService =
32          mock(PartyInfoGrpc.PartyInfoImplBase.class,
33              delegatesTo(new PartyInfoGrpcServiceDelegate() {
34              }));
35  
36      private final P2PTransactionGrpc.P2PTransactionImplBase p2pTransactionService =
37          mock(P2PTransactionGrpc.P2PTransactionImplBase.class,
38              delegatesTo(new TransactionGrpcServiceDelegate() {
39              }));
40  
41      private GrpcClientImpl client;
42  
43      @Before
44      public void setUp() throws Exception {
45          String serverName = InProcessServerBuilder.generateName();
46  
47          grpcCleanup.register(InProcessServerBuilder
48              .forName(serverName)
49              .directExecutor()
50              .addService(partyInfoService)
51              .addService(p2pTransactionService)
52              .build()
53              .start());
54  
55          final ManagedChannel channel = grpcCleanup.register(
56              InProcessChannelBuilder.forName(serverName).directExecutor().usePlaintext().build());
57  
58          client = new GrpcClientImpl(channel);
59      }
60  
61      @Test
62      public void testGetPartyInfo() {
63  
64          ArgumentCaptor<PartyInfoMessage> requestCaptor = ArgumentCaptor.forClass(PartyInfoMessage.class);
65          final byte[] data = "REQUEST".getBytes();
66  
67          client.sendPartyInfo(data);
68  
69          verify(partyInfoService).getPartyInfo(requestCaptor.capture(), any());
70  
71          assertEquals(ByteString.copyFrom(data), requestCaptor.getValue().getPartyInfo());
72  
73      }
74  
75      @Test
76      public void testGetPartyInfoFailed() throws InterruptedException {
77          client.shutdown();
78          final byte[] data = "REQUEST".getBytes();
79          byte[] response = client.sendPartyInfo(data);
80          assertThat(response).isNull();
81      }

125     /**
126      * Delegate to mock the PartyInfoGrpcService class
127      * The focus here is to ensure the client fires requests correctly.
128      * The real logic for these services are being tests at service test classes
129      */
130     private class PartyInfoGrpcServiceDelegate extends PartyInfoGrpc.PartyInfoImplBase {
131         @Override
132         public void getPartyInfo(PartyInfoMessage request, StreamObserver<PartyInfoMessage> responseObserver) {
133             byte[] responseData = "RESPONSE".getBytes();
134             final PartyInfoMessage response = PartyInfoMessage.newBuilder()
135                 .setPartyInfo(ByteString.copyFrom(responseData))
136                 .build();
137             responseObserver.onNext(response);
138             responseObserver.onCompleted();
139         }
140     }
