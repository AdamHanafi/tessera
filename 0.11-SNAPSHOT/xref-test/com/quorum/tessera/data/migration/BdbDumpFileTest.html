28      @Test
29      public void loadOnNonexistentFileFails() throws IOException {
30          final Path randomFile = Files.createTempFile("other", ".txt").resolveSibling("non-existent.txt");
31  
32          final Throwable throwable = catchThrowable(() -> loader.load(randomFile));
33          assertThat(throwable)
34              .isNotNull()
35              .isInstanceOf(IllegalArgumentException.class)
36              .hasMessageContaining("doesn't exist or is not a file");
37      }

39      @Test
40      public void nextReturnsEntryWhenResultsAreLeft() throws URISyntaxException, IOException {
41          final Path inputFile = Paths.get(getClass().getResource("/bdb/bdb-sample.txt").toURI());
42          loader.load(inputFile);
43  
44          final DataEntry nextEntry = loader.nextEntry();
45  
46          assertThat(nextEntry).isNotNull();
47      }
48  
49      @Test
50      public void nextReturnsNullWhenNoResultsAreLeft() throws IOException, URISyntaxException {
51          final Path directory = Paths.get(getClass().getResource("/bdb/single-entry.txt").toURI());
52  
53          this.loader.load(directory);
54  
55          //There is one result in the database
56          final DataEntry nextNotNull = this.loader.nextEntry();
57          assertThat(nextNotNull).isNotNull();
58  
59          //There should be 0 results left in the database
60          final DataEntry next = this.loader.nextEntry();
61  
62          assertThat(next).isNull();
63      }
