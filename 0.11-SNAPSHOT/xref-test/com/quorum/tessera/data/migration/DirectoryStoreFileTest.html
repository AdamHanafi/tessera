21      @Before

23          this.loader = new DirectoryStoreFile();

26      @Test
27      public void loadOnNondirectory() throws IOException {
28          final Path randomFile = Files.createTempFile("other", ".txt");
29  
30          final Throwable throwable = catchThrowable(() -> loader.load(randomFile));
31          assertThat(throwable)
32              .isNotNull()
33              .isInstanceOf(IllegalArgumentException.class)
34              .hasMessageContaining("doesn't exist or is not a directory");
35      }
36  
37      @Test
38      public void loadOnNonexistentFolder() throws IOException {
39          final Path randomFile = Files.createTempFile("other", ".txt").getParent().resolve("unknown/other").getParent();
40  
41          final Throwable throwable = catchThrowable(() -> loader.load(randomFile));
42          assertThat(throwable)
43              .isNotNull()
44              .isInstanceOf(IllegalArgumentException.class)
45              .hasMessageContaining("doesn't exist or is not a directory");
46      }

48      @Test
49      public void loadProperDatabaseHasNoError() throws URISyntaxException {
50          final Path directory = Paths.get(getClass().getResource("/dir/").toURI());
51  
52          final Throwable throwable = catchThrowable(() -> loader.load(directory));
53  
54          assertThat(throwable).isNull();
55      }

57      @Test
58      public void nextReturnsEntryWhenResultsAreLeft() throws URISyntaxException, IOException {
59          final Path directory = Paths.get(getClass().getResource("/dir/").toURI());
60  
61          this.loader.load(directory);
62  
63          //There should be 22 results left in the database
64          final DataEntry next = this.loader.nextEntry();
65  
66          assertThat(next).isNotNull();
67      }

69      @Test
70      public void hasNextReturnsFalseWhenNoResultsAreLeft() throws IOException, URISyntaxException {
71          final Path directory = Paths.get(getClass().getResource("/dir/").toURI());
72  
73          this.loader.load(directory);
74  
75          for (int i = 0; i < 22; i++) {
76              final DataEntry next = this.loader.nextEntry();
77              assertThat(next).isNotNull();
78          }
79  
80          //There should be 0 results left in the database
81          final DataEntry next = this.loader.nextEntry();
82  
83          assertThat(next).isNull();
