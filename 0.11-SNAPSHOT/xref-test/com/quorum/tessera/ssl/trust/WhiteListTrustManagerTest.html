26  public class WhiteListTrustManagerTest {
27  
28      private WhiteListTrustManager trustManager;
29  
30      Path knownHosts;
31  
32      @Mock
33      X509Certificate certificate;
34  
35      @Before
36      public void setUp() throws IOException, CertificateException {
37          MockitoAnnotations.initMocks(this);
38          when(certificate.getEncoded()).thenReturn("thumbprint".getBytes(UTF_8));
39          X500Principal cn = new X500Principal("CN=localhost");
40          when(certificate.getSubjectX500Principal()).thenReturn(cn);
41          knownHosts = Files.createTempFile("test", "knownHosts");
42  
43          try (BufferedWriter writer = Files.newBufferedWriter(knownHosts, StandardOpenOption.APPEND)) {
44              writer.write("someaddress somethumbprint");
45              writer.newLine();
46              writer.write("localhost" + " " + CertificateUtil.create().thumbPrint(certificate));
47              writer.newLine();
48          }
49  
50          trustManager = new WhiteListTrustManager(knownHosts);
51  
52      }
53  
54      @After
55      public void after() {
56          verifyNoMoreInteractions(certificate);
57      }
58  
59      @Test
60      public void testLoadCertificatesFromWhiteListFile() throws CertificateException {
61  
62          trustManager.checkServerTrusted(new X509Certificate[]{certificate},"str");
63          trustManager.checkClientTrusted(new X509Certificate[]{certificate},"str");
64  
65          verify(certificate, times(3)).getEncoded();
66          verify(certificate, times(2)).getSubjectX500Principal();
67  
68      }
69  
70      @Test
71      public void testCertificatesNotInWhiteList() throws CertificateException {
72          when(certificate.getEncoded()).thenReturn("some-other-thumbprint".getBytes(UTF_8));
73          try {
74              trustManager.checkClientTrusted(new X509Certificate[]{certificate}, "str");
75              failBecauseExceptionWasNotThrown(Exception.class);
76          }
77          catch (Exception ex){
78              assertThat(ex)
79                  .isInstanceOf(CertificateException.class)
80                  .hasMessage("Connections not allowed");
81          }
82          verify(certificate, times(2)).getEncoded();
83          verify(certificate).getSubjectX500Principal();
84      }
85  
86      @Test
87      public void testGetAcceptIssuers() throws CertificateEncodingException {
88          assertThat(trustManager.getAcceptedIssuers()).isEmpty();
89          verify(certificate).getEncoded();
90      }
