4   import com.quorum.tessera.client.GrpcP2pClient;
5   import com.quorum.tessera.enclave.EncodedPayload;
6   import com.quorum.tessera.enclave.PayloadEncoder;
7   import static org.assertj.core.api.Assertions.*;
8   import org.junit.After;
9   import org.junit.Before;
10  import org.junit.Test;
11  import static org.mockito.Mockito.*;
12  
13  public class GrpcPayloadPublisherTest {
14      
15      private GrpcPayloadPublisher publisher;
16      
17      private GrpcP2pClient p2pClient;
18      
19      private PayloadEncoder payloadEncoder; 
20      
21      @Before
22      public void onSetUp() {
23          payloadEncoder = mock(PayloadEncoder.class);
24          p2pClient = mock(GrpcP2pClient.class);
25          publisher = new GrpcPayloadPublisher(p2pClient,payloadEncoder);
26      }
27      
28      @After
29      public void onTearDown() {
30          verifyNoMoreInteractions(p2pClient,payloadEncoder);
31      }
32      
33      @Test
34      public void publish() {
35  
36          EncodedPayload encodedPayload = mock(EncodedPayload.class);
37          byte[] encodedData = "I LOVE SPARROWS".getBytes();
38          
39          when(payloadEncoder.encode(encodedPayload)).thenReturn(encodedData);
40          
41          String url = "http://somedomain.com";
42          
43          publisher.publishPayload(encodedPayload, url);
44          
45          verify(p2pClient).push(url, encodedData);
46          
47          verify(payloadEncoder).encode(encodedPayload);
48      }
49      
50      @Test
51     public void constructWithDefaultConstructor() {
52         assertThat(new GrpcPayloadPublisher()).isNotNull();
53     }
54      
55  }
