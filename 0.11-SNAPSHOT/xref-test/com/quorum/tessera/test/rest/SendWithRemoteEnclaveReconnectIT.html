126     @Test
127     public void sendTransactionToSelfWhenEnclaveIsDown() {
128 
129         enclaveExecManager.stop();
130 
131         RestUtils utils = new RestUtils();
132         byte[] transactionData = utils.createTransactionData();
133         final SendRequest sendRequest = new SendRequest();
134         sendRequest.setFrom(party.getPublicKey());
135         sendRequest.setPayload(transactionData);
136 
137         Client client = ClientBuilder.newClient();
138 
139         final Response response =
140                 client.target(party.getQ2TUri())
141                         .path("send")
142                         .request()
143                         .post(Entity.entity(sendRequest, MediaType.APPLICATION_JSON));
144 
145         assertThat(response.getStatus()).isEqualTo(503);
146 
147         this.enclaveExecManager = new EnclaveExecManager(this.configDescriptor);
148         enclaveExecManager.start();
149 
150         final Response secondresponse =
151                 client.target(party.getQ2TUri())
152                         .path("send")
153                         .request()
154                         .post(Entity.entity(sendRequest, MediaType.APPLICATION_JSON));
155 
156         assertThat(secondresponse.getStatus()).isEqualTo(201);
157     }

168         enclaveConfig.getKeys().getKeyData().add(addedKeypair);
169         this.writeConfig(enclaveConfig, this.configDescriptor.getEnclavePath());
