16  public class DeleteIT {
17  
18      private static final String COUNT_ALL = "select count(*) from ENCRYPTED_TRANSACTION where hash = ?";
19  
20      private final PartyHelper partyHelper = PartyHelper.create();
21  
22      @Test
23      public void deleteTransactionThatExists() throws Exception {
24          //setup (sending in a tx)
25  
26          Party sender = partyHelper.getParties().findAny().get();
27  
28          Party recipient = partyHelper.getParties()
29                  .filter(p -> !p.getPublicKey().equals(sender.getPublicKey()))
30                  .findAny().get();
31  
32          RestUtils utils = new RestUtils();
33          byte[] txnData = utils.createTransactionData();
34  
35          Response response = utils.send(sender, txnData, Collections.singleton(recipient));
36          assertThat(response.getStatus()).isEqualTo(201);
37  
38          final SendResponse sendResponse = response.readEntity(SendResponse.class);
39  
40          final String encodedHash = URLEncoder.encode(sendResponse.getKey(), UTF_8.toString());
41  
42          try(PreparedStatement statement = sender.getDatabaseConnection().prepareStatement(COUNT_ALL)) {
43              statement.setBytes(1, Base64.getDecoder().decode(sendResponse.getKey()));
44              try(ResultSet rs = statement.executeQuery()) {
45                  assertThat(rs.next()).isTrue();
46                  assertThat(rs.getLong(1)).isEqualTo(1);
47              }
48          }
49  
50          //delete it
51          final Response resp = sender.getRestClientWebTarget()
52                  .path("transaction")
53                  .path(encodedHash)
54                  .request()
55                  .delete();
56  
57          //validate result
58          assertThat(resp).isNotNull();
59          assertThat(resp.getStatus()).isEqualTo(204);
60  
61      }
62  
63      @Test
64      public void deleteTransactionThatDoesntExist() {
65  
66          final String madeupHash = Base64.getUrlEncoder().encodeToString("madeup".getBytes());
67  
68  
69          Party party = partyHelper.getParties().findAny().get();
70  
71          final Response response = party.getRestClientWebTarget()
72                  .path("transaction")
73                  .path(madeupHash)
74                  .request()
75                  .buildDelete()
76                  .invoke();
77  
78          //validate result
79          assertThat(response).isNotNull();
80          assertThat(response.getStatus()).isEqualTo(404);
81  
82      }
