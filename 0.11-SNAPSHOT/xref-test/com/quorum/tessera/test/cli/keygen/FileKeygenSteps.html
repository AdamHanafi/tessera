8   import com.quorum.tessera.nacl.NaclFacade;
9   import com.quorum.tessera.nacl.NaclFacadeFactory;
10  import cucumber.api.java8.En;
11  
12  import java.nio.file.Files;
13  import java.nio.file.Path;
14  import java.nio.file.Paths;
15  import java.util.*;
16  import java.util.concurrent.ExecutorService;
17  import java.util.concurrent.Executors;
18  
19  import static org.assertj.core.api.Assertions.assertThat;
20  
21  public class FileKeygenSteps implements En {
22  
23      private static final Base64.Decoder DECODER = Base64.getDecoder();
24  
25      private final ExecutorService executorService = Executors.newFixedThreadPool(2);
26  
27      // this is used to check the generated keys against, to produce the SharedKey
28      private final KeyPair knownGoodKeypair =
29              new KeyPair(
30                      PublicKey.from(DECODER.decode("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=")),
31                      PrivateKey.from(DECODER.decode("yAWAJjwPqUtNVlqGjSrBmr1/iIkghuOh1803Yzx9jLM=")));
32  
33      private Path buildDir;
34  
35      private String password;
36  
37      private List<String> args;
38  
39      private Integer exitCode;
40  
41      private Path publicKeyPath;
42  
43      private Path privateKeyPath;
44  
45      public FileKeygenSteps() {
46  
47          Given(
48                  "the application is available",
49                  () -> {
50                      this.exitCode = null;
51                      this.buildDir = Files.createTempDirectory(UUID.randomUUID().toString());
52  
53                      final String appPath = System.getProperty("application.jar");
54  
55                      if (Objects.equals("", appPath)) {
56                          throw new IllegalStateException("No application.jar system property defined");
57                      }
58  
59                      final Path applicationJar = Paths.get(appPath);
60  
61                      this.args = new ArrayList<>(Arrays.asList("java", "-jar", applicationJar.toString(), "-keygen"));
62                  });
63  
64          Given("no file exists at {string}", (String path) -> Files.deleteIfExists(Paths.get(path)));
65  
66          Given("no password", () -> this.password = "");
67  
68          Given("a password of {string}", (String password) -> this.password = password);
69  
70          // here to explicitly state we are doing nothing
71          Given("no file path is provided", () -> {});
72  
73          Given("a file path of {string}", (String path) -> this.args.addAll(Arrays.asList("-filename", path)));
74  
75          When(
76                  "new keys are generated",
77                  () -> {
78                      final ProcessBuilder processBuilder = new ProcessBuilder(args);
79                      processBuilder.directory(buildDir.toFile());
80  
81                      final Process process = processBuilder.start();
82                      executorService.submit(
83                              new FileKeygenStreamConsumer(
84                                      process.getInputStream(), process.getOutputStream(), password));
85                      executorService.submit(
86                              new FileKeygenStreamConsumer(
87                                      process.getErrorStream(), process.getOutputStream(), password));
88  
89                      this.exitCode = process.waitFor();
90                  });
91  
92          Then(
93                  "the application has an exit code of {int}",
94                  (Integer expectedExitCode) -> assertThat(exitCode).isEqualTo(expectedExitCode));
95  
96          And(
97                  "public key file {string} exists and private key file {string} exists",
98                  (String publicKeyPath, String privateKeyPath) -> {
99                      this.publicKeyPath = buildDir.resolve(publicKeyPath);
100                     this.privateKeyPath = buildDir.resolve(privateKeyPath);
101 
102                     assertThat(this.publicKeyPath).exists();
103                     assertThat(this.privateKeyPath).exists();
104                 });
105 
106         And(
107                 "the generated keys are valid",
108                 () -> {
109                     final FilesystemKeyPair generatedKeys =
110                             new FilesystemKeyPair(this.publicKeyPath, this.privateKeyPath);
111                     generatedKeys.withPassword(this.password);
112 
113                     final PublicKey publicKey = PublicKey.from(DECODER.decode(generatedKeys.getPublicKey()));
114                     final PrivateKey privateKey = PrivateKey.from(DECODER.decode(generatedKeys.getPrivateKey()));
115 
116                     final NaclFacade naclFacade = NaclFacadeFactory.newFactory().create();
117                     final SharedKey firstKey = naclFacade.computeSharedKey(publicKey, knownGoodKeypair.getPrivateKey());
118                     final SharedKey secondKey =
119                             naclFacade.computeSharedKey(knownGoodKeypair.getPublicKey(), privateKey);
120 
121                     assertThat(firstKey).isEqualTo(secondKey);
122                 });
123     }
124 }
