109         Given(
110                 "^the mock AKV server has stubs for the endpoints used to get secrets$",
111                 () -> {
112                     // --- Authentication flow for the AKV client ---
113                     // The AKV Client makes a request to GET a secret from the AKV.  If the response is 200 Success then
114                     // no authentication is required and the secret is returned to the caller.  If the response is 401
115                     // Unauthorized then the client proceeds to authenticate itself.  Any other response code results in
116                     // error being returned and termination of the GET request.
117                     // See https://docs.microsoft.com/en-us/rest/api/keyvault/getsecret/getsecret for the AKV HTTP
118                     // GetSecret API docs
119                     //
120                     // The 401 Unauthorized response will contain a WWW-Authenticate header which the AKV Client uses to
121                     // authenticate itself.
122                     // The 'authorization' component of the header provides the url of the authorization server to be
123                     // used.
124                     // The 'resource' component provides the Azure resource (i.e. key vault) to request authentication
125                     // for.
126                     //
127                     // To authenticate, the client POSTs to the {authorization}/oauth2/token endpoint, where
128                     // {authorization}
129                     // is the authorization server url provided in the WWW-Authenticate header.
130                     // See
131                     // https://docs.microsoft.com/en-us/azure/active-directory/develop/v1-protocols-oauth-code#use-the-authorization-code-to-request-an-access-token
132                     // for more info.
133                     //
134                     // The WWW-Authenticate header is used in subsequent GETs to the same Azure resource to make
135                     // authentication requests immediately instead of having to first make 401 GETs.
136 
137                     final String respFormat = "{ \"value\": \"%s\" }";
138 
139                     final String authScenario = "AUTH";
140                     final String received401 = "RECEIVED_401";
141

347     private void startTessera(List<String> args, Path verifyConfig) throws Exception {
348         System.out.println(String.join(" ", args));
349 
350         ProcessBuilder tesseraProcessBuilder = new ProcessBuilder(args);
351 
352         Map<String, String> tesseraEnvironment = tesseraProcessBuilder.environment();
353         tesseraEnvironment.put(AZURE_CLIENT_ID, "my-client-id");
354         tesseraEnvironment.put(AZURE_CLIENT_SECRET, "my-client-secret");
355 
356         try {
357             tesseraProcess.set(tesseraProcessBuilder.redirectErrorStream(true).start());
358         } catch (NullPointerException ex) {
359             throw new NullPointerException("Check that application.jar property has been set");
360         }
361 
362         executorService.submit(
363                 () -> {
364                     try (BufferedReader reader =
365                             Stream.of(tesseraProcess.get().getInputStream())
366                                     .map(InputStreamReader::new)
367                                     .map(BufferedReader::new)
368                                     .findAny()
369                                     .get()) {
370 
371                         String line;
372                         while ((line = reader.readLine()) != null) {
373                             System.out.println(line);
374                         }
375 
376                     } catch (IOException ex) {
377                         throw new UncheckedIOException(ex);
378                     }
379                 });
380 
381         CountDownLatch startUpLatch = new CountDownLatch(1);
382 
383         if (Objects.nonNull(verifyConfig)) {
384             final Config config = JaxbUtil.unmarshal(Files.newInputStream(verifyConfig), Config.class);
385 
386             final URL bindingUrl =
387                     UriBuilder.fromUri(config.getP2PServerConfig().getBindingUri()).path("upcheck").build().toURL();
388 
389             executorService.submit(
390                     () -> {
391                         while (true) {
392                             try {
393                                 HttpURLConnection conn = (HttpURLConnection) bindingUrl.openConnection();
394                                 conn.connect();
395 
396                                 System.out.println(bindingUrl + " started." + conn.getResponseCode());
397 
398                                 startUpLatch.countDown();
399                                 return;
400                             } catch (IOException ex) {
401                                 try {
402                                     TimeUnit.MILLISECONDS.sleep(200L);
403                                 } catch (InterruptedException ex1) {
404                                 }
405                             }
406                         }
407                     });
408 
409             boolean started = startUpLatch.await(30, TimeUnit.SECONDS);
410 
411             if (!started) {
412                 System.err.println(bindingUrl + " Not started. ");
413             }
414         }
415 
416         executorService.submit(
417                 () -> {
418                     try {
419                         int exitCode = tesseraProcess.get().waitFor();
420                         startUpLatch.countDown();
421                         if (0 != exitCode) {
422                             System.err.println("Tessera node exited with code " + exitCode);
423                         }
424                     } catch (InterruptedException ex) {
425                         ex.printStackTrace();
426                     }
427                 });
428 
429         startUpLatch.await(30, TimeUnit.SECONDS);
430     }
