9   import org.junit.Before;
10  import org.junit.Test;
11  import org.mockito.ArgumentCaptor;
12  
13  import java.util.Set;
14  
15  import java.util.Collections;
16  import java.util.stream.Collectors;
17  import java.util.stream.Stream;
18  import static org.assertj.core.api.Assertions.assertThat;
19  import static org.mockito.Mockito.*;
20  
21  public class TransactionRequesterTest {
22  
23      private static final PublicKey KEY_ONE = PublicKey.from(new byte[] {1});
24  
25      private static final PublicKey KEY_TWO = PublicKey.from(new byte[] {2});
26  
27      private Enclave enclave;
28  
29      private P2pClient p2pClient;
30  
31      private TransactionRequester transactionRequester;
32  
33      @Before
34      public void init() {
35  
36          this.enclave = mock(Enclave.class);
37          this.p2pClient = mock(P2pClient.class);
38  
39          doReturn(true).when(p2pClient).makeResendRequest(anyString(), any(ResendRequest.class));
40  
41          this.transactionRequester = new TransactionRequesterImpl(enclave, p2pClient);
42      }
43  
44      @After
45      public void after() {
46          verifyNoMoreInteractions(enclave, p2pClient);
47      }
48  
49      @Test
50      public void noPublicKeysMakesNoCalls() {
51          when(enclave.getPublicKeys()).thenReturn(Collections.emptySet());
52  
53          this.transactionRequester.requestAllTransactionsFromNode("fakeurl.com");
54  
55          verifyZeroInteractions(p2pClient);
56          verify(enclave).getPublicKeys();
57      }
58  
59      @Test
60      public void multipleKeysMakesCorrectCalls() {
61  
62          final Set<PublicKey> allKeys = Stream.of(KEY_ONE, KEY_TWO).collect(Collectors.toSet());
63  
64          when(enclave.getPublicKeys()).thenReturn(allKeys);
65  
66          this.transactionRequester.requestAllTransactionsFromNode("fakeurl1.com");
67  
68          final ArgumentCaptor<ResendRequest> captor = ArgumentCaptor.forClass(ResendRequest.class);
69          verify(p2pClient, times(2)).makeResendRequest(eq("fakeurl1.com"), captor.capture());
70          verify(enclave).getPublicKeys();
71  
72          String encodedKeyOne = Base64.getEncoder().encodeToString(KEY_ONE.getKeyBytes());
73          String encodedKeyTwo = Base64.getEncoder().encodeToString(KEY_TWO.getKeyBytes());
74  
75          assertThat(captor.getAllValues())
76                  .hasSize(2)
77                  .extracting("publicKey")
78                  .containsExactlyInAnyOrder(encodedKeyOne, encodedKeyTwo);
79      }
80  
81      @Test
82      public void failedCallRetries() {
83          when(enclave.getPublicKeys()).thenReturn(Collections.singleton(KEY_ONE));
84  
85          when(p2pClient.makeResendRequest(anyString(), any(ResendRequest.class))).thenReturn(false);
86  
87          this.transactionRequester.requestAllTransactionsFromNode("fakeurl.com");
88  
89          verify(p2pClient, times(5)).makeResendRequest(eq("fakeurl.com"), any(ResendRequest.class));
90          verify(enclave).getPublicKeys();
91      }
92  
93      @Test
94      public void calltoPostDelegateThrowsException() {
95  
96          when(enclave.getPublicKeys()).thenReturn(Collections.singleton(KEY_ONE));
97          when(p2pClient.makeResendRequest(anyString(), any(ResendRequest.class))).thenThrow(RuntimeException.class);
98  
99          this.transactionRequester.requestAllTransactionsFromNode("fakeurl.com");
100 
101         verify(p2pClient, times(5)).makeResendRequest(eq("fakeurl.com"), any(ResendRequest.class));
102         verify(enclave).getPublicKeys();
103     }
104 }
