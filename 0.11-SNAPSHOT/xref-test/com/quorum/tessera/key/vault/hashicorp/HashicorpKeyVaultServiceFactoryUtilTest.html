36      @Test
37      public void configureSslUsesKeyStoreAndTrustStoreIfBothProvided() throws Exception {
38          HashicorpKeyVaultConfig keyVaultConfig = mock(HashicorpKeyVaultConfig.class);
39          EnvironmentVariableProvider envProvider = mock(EnvironmentVariableProvider.class);
40  
41          Path path = Files.createTempFile(UUID.randomUUID().toString(), ".tmp");
42          path.toFile().deleteOnExit();
43  
44          when(keyVaultConfig.getTlsKeyStorePath()).thenReturn(path);
45          when(keyVaultConfig.getTlsTrustStorePath()).thenReturn(path);
46  
47          SslConfiguration result = util.configureSsl(keyVaultConfig, envProvider);
48  
49          assertThat(result.getKeyStoreConfiguration().isPresent()).isTrue();
50          assertThat(result.getTrustStoreConfiguration().isPresent()).isTrue();
51      }

139     public void configureClientAuthenticationIfOnlyRoleIdSetThenException() {
140         HashicorpKeyVaultConfig keyVaultConfig = mock(HashicorpKeyVaultConfig.class);
141         EnvironmentVariableProvider envProvider = mock(EnvironmentVariableProvider.class);
142         ClientHttpRequestFactory clientHttpRequestFactory = mock(ClientHttpRequestFactory.class);
143         VaultEndpoint vaultEndpoint = mock(VaultEndpoint.class);
144 
145         when(envProvider.getEnv(HASHICORP_ROLE_ID)).thenReturn("role-id");
146         when(envProvider.getEnv(HASHICORP_SECRET_ID)).thenReturn(null);
147         when(envProvider.getEnv(HASHICORP_TOKEN)).thenReturn(null);
148 
149         Throwable ex = catchThrowable(() -> util.configureClientAuthentication(keyVaultConfig, envProvider, clientHttpRequestFactory, vaultEndpoint));
150 
151         assertThat(ex).isExactlyInstanceOf(HashicorpCredentialNotSetException.class);
152         assertThat(ex.getMessage()).isEqualTo("Both " + HASHICORP_ROLE_ID + " and " + HASHICORP_SECRET_ID + " environment variables must be set to use the AppRole authentication method");
153     }
