3   import com.quorum.tessera.encryption.PublicKey;
4   import org.junit.Test;
5   
6   import static org.assertj.core.api.Assertions.assertThat;
7   
8   public class RawTransactionBuilderTest {
9   
10      @Test
11      public void build() {
12  
13          final byte[] encryptedKey = "KEY".getBytes();
14  
15          final byte[] encryptedPayload = "PAYLOAD".getBytes();
16  
17          final byte[] nonce = "NONCE".getBytes();
18  
19          final PublicKey from = PublicKey.from("SOMEKEY".getBytes());
20  
21          final RawTransaction txn = RawTransactionBuilder.create()
22              .withEncryptedKey(encryptedKey)
23              .withEncryptedPayload(encryptedPayload)
24              .withNonce(nonce)
25              .withFrom(from)
26              .build();
27  
28          assertThat(txn).isNotNull();
29          assertThat(txn.getEncryptedKey()).isEqualTo(encryptedKey);
30          assertThat(txn.getEncryptedPayload()).isEqualTo(encryptedPayload);
31          assertThat(txn.getFrom()).isEqualTo(from);
32          assertThat(txn.getNonce().getNonceBytes()).isEqualTo(nonce);
33  
34      }
