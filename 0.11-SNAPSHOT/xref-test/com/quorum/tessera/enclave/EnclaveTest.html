162     @Test
163     public void encryptPayload() {
164 
165         byte[] message = "MESSAGE".getBytes();
166 
167         PublicKey senderPublicKey = mock(PublicKey.class);
168         PublicKey recipientPublicKey = mock(PublicKey.class);
169 
170         byte[] masterKeyBytes = "masterKeyBytes".getBytes();
171         MasterKey masterKey = MasterKey.from(masterKeyBytes);
172         Nonce cipherNonce = mock(Nonce.class);
173         Nonce recipientNonce = mock(Nonce.class);
174 
175         byte[] cipherText = "cipherText".getBytes();
176 
177         when(nacl.createMasterKey()).thenReturn(masterKey);
178         when(nacl.randomNonce()).thenReturn(cipherNonce, recipientNonce);
179 
180         when(nacl.sealAfterPrecomputation(message, cipherNonce, masterKey)).thenReturn(cipherText);
181 
182         PrivateKey senderPrivateKey = mock(PrivateKey.class);
183         when(keyManager.getPrivateKeyForPublicKey(senderPublicKey)).thenReturn(senderPrivateKey);
184 
185         SharedKey sharedKey = mock(SharedKey.class);
186         when(nacl.computeSharedKey(recipientPublicKey, senderPrivateKey)).thenReturn(sharedKey);
187 
188         byte[] encryptedMasterKeys = "encryptedMasterKeys".getBytes();
189         when(nacl.sealAfterPrecomputation(masterKeyBytes, recipientNonce, sharedKey)).thenReturn(encryptedMasterKeys);
190 
191         EncodedPayload result = enclave.encryptPayload(message, senderPublicKey, Arrays.asList(recipientPublicKey));
192 
193         assertThat(result).isNotNull();
194         assertThat(result.getRecipientKeys()).containsExactly(recipientPublicKey);
195         assertThat(result.getCipherText()).isEqualTo(cipherText);
196         assertThat(result.getCipherTextNonce()).isEqualTo(cipherNonce);
197         assertThat(result.getSenderKey()).isEqualTo(senderPublicKey);
198         assertThat(result.getRecipientBoxes()).containsExactly(encryptedMasterKeys);
199 
200         verify(nacl).createMasterKey();
201         verify(nacl, times(2)).randomNonce();
202         verify(nacl).sealAfterPrecomputation(message, cipherNonce, masterKey);
203         verify(nacl).sealAfterPrecomputation(masterKeyBytes, recipientNonce, sharedKey);
204         verify(nacl).computeSharedKey(recipientPublicKey, senderPrivateKey);
205         verify(keyManager).getPrivateKeyForPublicKey(senderPublicKey);
206     }

329     @Test
330     public void createNewRecipientBoxGivesBackSuccessfulEncryptedKey() {
331 
332         final PublicKey publicKey = PublicKey.from("recipient".getBytes());
333         final PublicKey senderKey = PublicKey.from("sender".getBytes());
334         final PrivateKey privateKey = PrivateKey.from("sender-priv".getBytes());
335         final SharedKey recipientSenderShared = SharedKey.from("shared-one".getBytes());
336         final SharedKey senderShared = SharedKey.from("shared-two".getBytes());
337         final byte[] closedbox = "closed".getBytes();
338         final byte[] openbox = "open".getBytes();
339         final Nonce nonce = new Nonce("nonce".getBytes());
340 
341         final EncodedPayload payload
342             = new EncodedPayload(senderKey, null, null, singletonList(closedbox), nonce, singletonList(publicKey));
343 
344         when(nacl.computeSharedKey(publicKey, privateKey)).thenReturn(recipientSenderShared);
345         when(nacl.computeSharedKey(senderKey, privateKey)).thenReturn(senderShared);
346         when(nacl.openAfterPrecomputation(closedbox, nonce, recipientSenderShared)).thenReturn(openbox);
347         when(nacl.sealAfterPrecomputation(openbox, nonce, senderShared)).thenReturn("newbox".getBytes());
348         when(keyManager.getPrivateKeyForPublicKey(senderKey)).thenReturn(privateKey);
349 
350         final byte[] newRecipientBox = enclave.createNewRecipientBox(payload, senderKey);
351 
352         assertThat(newRecipientBox).containsExactly("newbox".getBytes());
353 
354         verify(nacl).computeSharedKey(publicKey, privateKey);
355         verify(nacl).computeSharedKey(senderKey, privateKey);
356         verify(nacl).openAfterPrecomputation(closedbox, nonce, recipientSenderShared);
357         verify(nacl).sealAfterPrecomputation(openbox, nonce, senderShared);
358         verify(keyManager, times(2)).getPrivateKeyForPublicKey(senderKey);
359     }
