3   import org.junit.Before;
4   import org.junit.Test;
5   
6   import java.nio.file.Path;
7   import java.nio.file.Paths;
8   
9   import static org.assertj.core.api.Assertions.assertThat;
10  
11  public class HashicorpKeyVaultConfigTest {
12  
13      private HashicorpKeyVaultConfig vaultConfig;
14  
15      @Before
16      public void setUp() {
17          vaultConfig = new HashicorpKeyVaultConfig();
18      }
19  
20      @Test
21      public void multiArgConstructor() {
22          String url = "url";
23          String approle = "approle";
24          Path keyStore = Paths.get("keystore");
25          Path trustStore = Paths.get("truststore");
26  
27          HashicorpKeyVaultConfig conf = new HashicorpKeyVaultConfig(url, approle, keyStore, trustStore);
28  
29          assertThat(conf.getUrl()).isEqualTo(url);
30          assertThat(conf.getApprolePath()).isEqualTo("approle");
31          assertThat(conf.getTlsKeyStorePath()).isEqualTo(keyStore);
32          assertThat(conf.getTlsTrustStorePath()).isEqualTo(trustStore);
33      }
34  
35      @Test
36      public void gettersAndSetters() {
37          assertThat(vaultConfig.getUrl()).isEqualTo(null);
38          assertThat(vaultConfig.getTlsKeyStorePath()).isEqualTo(null);
39          assertThat(vaultConfig.getTlsTrustStorePath()).isEqualTo(null);
40  
41          String url = "url";
42          Path keyStore = Paths.get("keystore");
43          Path trustStore = Paths.get("truststore");
44  
45          vaultConfig.setUrl(url);
46          vaultConfig.setTlsKeyStorePath(keyStore);
47          vaultConfig.setTlsTrustStorePath(trustStore);
48  
49          assertThat(vaultConfig.getUrl()).isEqualTo(url);
50          assertThat(vaultConfig.getTlsKeyStorePath()).isEqualTo(keyStore);
51          assertThat(vaultConfig.getTlsTrustStorePath()).isEqualTo(trustStore);
52      }
53  
54      @Test
55      public void getType() {
56          assertThat(vaultConfig.getKeyVaultType()).isEqualTo(KeyVaultType.HASHICORP);
57      }
58  
59      @Test
60      public void getApprolePathReturnsDefaultIfNotSet() {
61          assertThat(vaultConfig.getApprolePath()).isEqualTo("approle");
62      }
63  
64      @Test
65      public void getApprolePath() {
66          vaultConfig.setApprolePath("notdefault");
67          assertThat(vaultConfig.getApprolePath()).isEqualTo("notdefault");
68      }
69  
70  }
