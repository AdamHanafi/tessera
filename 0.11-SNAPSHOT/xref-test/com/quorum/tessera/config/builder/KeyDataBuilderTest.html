22      @Test
23      public void buildThreeLocked() throws IOException {
24  
25          final Path passwordFile = Files.createTempFile("tessera-passwords", ".txt");
26  
27          List<Path> privateKeyPaths = Arrays.asList(
28              Files.createTempFile("buildThreeLocked1", ".txt"),
29              Files.createTempFile("buildThreeLocked2", ".txt"),
30              Files.createTempFile("buildThreeLocked3", ".txt")
31          );
32  
33          final byte[] privateKeyData = FixtureUtil.createLockedPrivateKey().toString().getBytes();
34          for (Path p : privateKeyPaths) {
35              Files.write(p, privateKeyData);
36          }
37  
38          List<String> publicKeys = Arrays.asList("PUB1", "PUB2", "PUB3");
39          List<String> privateKeys = privateKeyPaths.stream().map(Path::toString).collect(Collectors.toList());
40  
41          Files.write(passwordFile, Arrays.asList("SECRET1", "SECRET2", "SECRET3"));
42  
43          List<ConfigKeyPair> result = KeyDataBuilder.create()
44              .withPrivateKeys(privateKeys)
45              .withPublicKeys(publicKeys)
46              .withPrivateKeyPasswordFile(passwordFile.toString())
47              .build()
48              .getKeyData();
49  
50          assertThat(result).hasSize(3);
51  
52      }

54      @Test
55      public void differentAmountOfKeysThrowsError() {
56  
57          final KeyDataBuilder keyDataBuilder = KeyDataBuilder.create()
58              .withPrivateKeys(Collections.emptyList())
59              .withPublicKeys(Collections.singletonList("keyfile.txt"))
60              .withPrivateKeyPasswordFile("pwfile.txt");
61  
62          final Throwable throwable = catchThrowable(() -> keyDataBuilder.build());
63  
64          assertThat(throwable)
65              .isInstanceOf(ConfigException.class)
66              .hasCauseExactlyInstanceOf(RuntimeException.class);
67  
68          assertThat(throwable.getCause()).hasMessage("Different amount of public and private keys supplied");
69  
70      }
