28      @Test
29      public void publicPrivateInlineUnlocked() {
30          final Config config
31              = JaxbUtil.unmarshal(getClass().getResourceAsStream("/keytests/pubPrivInlineUnlocked.json"), Config.class);
32          adapter.resolveKeyPasswords(config);
33  
34          assertThat(config).isNotNull();
35          assertThat(config.getKeys()).isNotNull();
36          assertThat(config.getKeys().getKeyData()).isNotNull().hasSize(1);
37          assertThat(config.getKeys().getKeyData().get(0).getPublicKey()).isEqualTo("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=");
38          assertThat(config.getKeys().getKeyData().get(0).getPrivateKey()).isEqualTo("Wl+xSyXVuuqzpvznOS7dOobhcn4C5auxkFRi7yLtgtA=");
39      }
