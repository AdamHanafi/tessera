613         String urlOverride = "http://junit.com:8989";
614         int portOverride = 9999;
615         String unixSocketFileOverride = "unixSocketFileOverride.ipc";
616         String workdirOverride = "workdirOverride";

620 
621         //TODO check all CLI options have assertions here
622         when(commandLine.getOptionValue("url")).thenReturn(urlOverride);
623         when(commandLine.getOptionValue("port")).thenReturn(String.valueOf(portOverride));
624         when(commandLine.getOptionValue("workdir")).thenReturn(workdirOverride);
625         when(commandLine.getOptionValue("socket")).thenReturn(unixSocketFileOverride);
626 
627         when(commandLine.getOptionValues("othernodes"))
628                 .thenReturn(
629                         overridePeers.stream().map(Peer::getUrl).toArray(String[]::new)
630                 );
631 
632         when(commandLine.getOptionValues("publickeys"))
633                 .thenReturn(new String[]{"ONE", "TWO"});
634 
635         when(commandLine.getOptionValue("storage")).thenReturn("sqlite:somepath");
636 
637         when(commandLine.getOptionValue("tlsservertrust")).thenReturn("whitelist");
638 
639         when(commandLine.getOptionValue("tlsclienttrust")).thenReturn("ca");
640 
641         when(commandLine.getOptionValue("tlsservercert")).thenReturn("tlsservercert.cert");
642 
643         when(commandLine.getOptionValue("tlsclientcert")).thenReturn("tlsclientcert.cert");
644 
645         when(commandLine.getOptionValues("tlsserverchain")).thenReturn(new String[]{
646             "server1.crt", "server2.crt", "server3.crt"
647         });
648 
649         when(commandLine.getOptionValues("tlsclientchain")).thenReturn(new String[]{
650             "client1.crt", "client2.crt", "client3.crt"
651         });
652 
653         when(commandLine.getOptionValue("tlsserverkey"))
654                 .thenReturn("tlsserverkey.key");
655 
656         when(commandLine.getOptionValue("tlsclientkey"))
657                 .thenReturn("tlsclientkey.key");
658 
659         doReturn(new String[]{}).when(commandLine).getOptionValues("alwayssendto");
660 
661         List<Path> privateKeyPaths = Arrays.asList(
662                 Files.createTempFile("applyOverrides1", ".txt"),
663                 Files.createTempFile("applyOverrides2", ".txt")
664         );
665 
666         when(commandLine.getOptionValue("tlsknownservers")).thenReturn("tlsknownservers.file");
667 
668         when(commandLine.getOptionValue("tlsknownclients")).thenReturn("tlsknownclients.file");
669 
670         final byte[] privateKeyData = FixtureUtil.createLockedPrivateKey().toString().getBytes();
671         for (Path p : privateKeyPaths) {
672             Files.write(p, privateKeyData);
673         }
674 
675         final String[] privateKeyPathStrings = privateKeyPaths
676                 .stream()
677                 .map(Path::toString)
678                 .toArray(String[]::new);
679 
680         when(commandLine.getOptionValues("privatekeys")).thenReturn(privateKeyPathStrings);
681 
682         final List<String> privateKeyPasswords = Arrays.asList("SECRET1", "SECRET2");
683 
684         final Path privateKeyPasswordFile = Files.createTempFile("applyOverridesPasswords", ".txt");
685         Files.write(privateKeyPasswordFile, privateKeyPasswords);
686 
687         when(commandLine.getOptionValue("passwords")).thenReturn(privateKeyPasswordFile.toString());
688 
689         Config result = LegacyCliAdapter.applyOverrides(commandLine, builderWithValidValues, KeyDataBuilder.create()).build();
690 
691         assertThat(result).isNotNull();
692         assertThat(result.getServer().getHostName()).isEqualTo("http://junit.com");
693         assertThat(result.getServer().getPort()).isEqualTo(portOverride);
694         assertThat(result.getServer().getBindingAddress()).isEqualTo("http://junit.com:" + portOverride);
695         assertThat(result.getUnixSocketFile()).isEqualTo(Paths.get(workdirOverride, unixSocketFileOverride));
696         assertThat(result.getPeers()).containsExactly(overridePeers.toArray(new Peer[0]));
697         assertThat(result.getKeys().getKeyData()).hasSize(2);
698         assertThat(result.getJdbcConfig()).isNotNull();
699         assertThat(result.getJdbcConfig().getUrl()).isEqualTo("jdbc:sqlite:somepath");
700 
701         assertThat(result.getServer().getSslConfig().getServerTrustMode()).isEqualTo(SslTrustMode.WHITELIST);
702         assertThat(result.getServer().getSslConfig().getClientTrustMode()).isEqualTo(SslTrustMode.CA);
703 
704         assertThat(result.getServer().getSslConfig().getClientTlsCertificatePath()).isEqualTo(Paths.get("workdirOverride/tlsclientcert.cert"));
705 
706         assertThat(result.getServer().getSslConfig().getServerTlsCertificatePath()).isEqualTo(Paths.get("workdirOverride/tlsservercert.cert"));
707 
708         assertThat(result.getServer().getSslConfig().getServerTrustCertificates())
709                 .containsExactly(Paths.get(workdirOverride, "server1.crt"), Paths.get(workdirOverride, "server2.crt"), Paths.get(workdirOverride, "server3.crt"));
710 
711         assertThat(result.getServer().getSslConfig().getClientTrustCertificates())
712                 .containsExactly(Paths.get(workdirOverride, "client1.crt"), Paths.get(workdirOverride, "client2.crt"), Paths.get(workdirOverride, "client3.crt"));
713 
714         assertThat(result.getServer().getSslConfig().getServerKeyStore())
715                 .isEqualTo(Paths.get("workdirOverride/sslServerKeyStorePath"));
716         assertThat(result.getServer().getSslConfig().getClientKeyStore())
717                 .isEqualTo(Paths.get("workdirOverride/sslClientKeyStorePath"));
718         assertThat(result.getServer().getSslConfig().getKnownServersFile())
719                 .isEqualTo(Paths.get("workdirOverride/tlsknownservers.file"));
720         assertThat(result.getServer().getSslConfig().getKnownClientsFile())
721                 .isEqualTo(Paths.get(workdirOverride, "tlsknownclients.file"));
722     }
