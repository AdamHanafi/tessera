26  @ServerEndpoint(
27          value = "/sync",
28          decoders = {SyncRequestMessageCodec.class, SyncResponseMessageCodec.class},
29          encoders = {SyncRequestMessageCodec.class, SyncResponseMessageCodec.class},
30          configurator = PartyInfoEndpointConfigurator.class)
31  public class PartyInfoEndpoint {
32  
33      private static final Logger LOGGER = LoggerFactory.getLogger(PartyInfoEndpoint.class);
34  
35      private final Map<String, Session> sessions = new HashMap<>();
36  
37      private final PartyInfoService partyInfoService;
38  
39      private final PayloadEncoder payloadEncoder = PayloadEncoder.create();
40  
41      private final TransactionManager transactionManager;
42  
43      private final Enclave enclave;
44  
45      public PartyInfoEndpoint(
46              PartyInfoService partyInfoService, TransactionManager transactionManager, Enclave enclave) {
47          this.partyInfoService = partyInfoService;
48          this.enclave = enclave;
49          this.transactionManager = transactionManager;
50      }
51  
52      @OnOpen
53      public void onOpen(Session session) {
54          LOGGER.info("Open session : {}, {}", session.getId());
55          sessions.put(session.getId(), session);
56      }
57  
58      @OnMessage
59      public void onSync(Session session, SyncRequestMessage syncRequestMessage) throws IOException, EncodeException {
60  
61          if (syncRequestMessage.getType() == SyncRequestMessage.Type.TRANSACTION_PUSH) {
62              EncodedPayload payload = syncRequestMessage.getTransactions();
63              transactionManager.storePayload(payloadEncoder.encode(payload));
64              return;
65          }
66  
67          if (syncRequestMessage.getType() == SyncRequestMessage.Type.TRANSACTION_SYNC) {
68  
69              PublicKey recipientPublicKey = syncRequestMessage.getRecipientKey();
70  
71              ResendRequest resendRequest = new ResendRequest();
72              resendRequest.setType(ResendRequestType.ALL);
73              resendRequest.setPublicKey(recipientPublicKey.encodeToBase64());
74  
75              transactionManager.resend(resendRequest);
76              return;
77          }
78  
79          PartyInfo partyInfo = syncRequestMessage.getPartyInfo();
80  
81          LOGGER.info("Message {}", partyInfo.getUrl());
82  
83          PartyInfo mergedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
84  
85          SyncResponseMessage partyInfoResponseMessage =
86                  SyncResponseMessage.Builder.create(SyncResponseMessage.Type.PARTY_INFO)
87                          .withPartyInfo(mergedPartyInfo)
88                          .build();
89  
90          session.getBasicRemote().sendObject(partyInfoResponseMessage);
91      }
92  
93      @OnClose
94      public void onClose(Session session) {
95          partyInfoService.removeRecipient(session.getRequestURI().toString());
96          LOGGER.info("Close session: {}", session.getId());
97          sessions.remove(session.getId());
98      }
99  
100     public Collection<Session> getSessions() {
101         return Collections.unmodifiableCollection(sessions.values());
102     }
103 }
