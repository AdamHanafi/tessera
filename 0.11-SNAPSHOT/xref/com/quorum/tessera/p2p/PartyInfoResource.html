76      /**
77       * Allows node information to be retrieved in a specific encoded form including other node URLS and public key to
78       * URL mappings
79       *
80       * @param payload The encoded node information from the requester
81       * @return the merged node information from this node, which may contain new information
82       */
83      @POST
84      @Consumes(MediaType.APPLICATION_OCTET_STREAM)
85      @Produces(MediaType.APPLICATION_OCTET_STREAM)
86      @ApiOperation(value = "Request public key/url of other nodes", produces = "public keylist/url")
87      @ApiResponses({@ApiResponse(code = 200, message = "Encoded PartyInfo Data", response = byte[].class)})
88      public Response partyInfo(@ApiParam(required = true) final byte[] payload) {
89  
90          final PartyInfo partyInfo = partyInfoParser.from(payload);
91  
92          if (!enableKeyValidation) {
93              partyInfoService.updatePartyInfo(partyInfo);
94  
95              // create an empty party info object with our URL to send back
96              // this is used by older versions (before 0.10.0), but we don't want to give any info back
97              final PartyInfo emptyInfo = new PartyInfo(partyInfoService.getPartyInfo().getUrl(), emptySet(), emptySet());
98              final byte[] returnData = partyInfoParser.to(emptyInfo);
99              return Response.ok(returnData).build();
100         }
101 
102         // Start validation stuff
103         final PublicKey sender = enclave.defaultPublicKey();
104 
105         final String url = partyInfo.getUrl();
106 
107         final String dataToEncrypt = UUID.randomUUID().toString();
108 
109         final Predicate<Recipient> isValidRecipientKey =
110                 r -> {
111                     try {
112 
113                         final PublicKey key = r.getKey();
114                         final EncodedPayload encodedPayload =
115                                 enclave.encryptPayload(dataToEncrypt.getBytes(), sender, Arrays.asList(key));
116 
117                         final byte[] encodedPayloadData = payloadEncoder.encode(encodedPayload);
118 
119                         try (Response response =
120                                 restClient
121                                         .target(r.getUrl())
122                                         .path("partyinfo")
123                                         .path("validate")
124                                         .request()
125                                         .post(Entity.entity(encodedPayloadData, MediaType.APPLICATION_OCTET_STREAM))) {
126 
127                             String unencodedValidationData = response.readEntity(String.class);
128 
129                             boolean isValid = Objects.equals(unencodedValidationData, dataToEncrypt);
130                             if (!isValid) {
131                                 LOGGER.warn("Invalid key found {} recipient will be ignored.", r.getUrl());
132                             }
133 
134                             return isValid;
135                         }
136                         // Assume any all exceptions to mean invalid. enclave bubbles up nacl array out of
137                         // bounds when calculating shared key from invalid data
138                     } catch (Exception ex) {
139                         LOGGER.debug(null, ex);
140                         return false;
141                     }
142                 };
143 
144         final Predicate<Recipient> isSendingUrl = r -> r.getUrl().equalsIgnoreCase(url);
145 
146         // Validate caller and treat no valid certs as security issue.
147         final Set<Recipient> recipients =
148                 partyInfo.getRecipients().stream()
149                         .filter(isSendingUrl.and(isValidRecipientKey))
150                         .collect(Collectors.toSet());
151 
152         if (recipients.isEmpty()) {
153             throw new SecurityException("No key found for url " + url);
154         }
155 
156         final PartyInfo modifiedPartyInfo = new PartyInfo(url, recipients, partyInfo.getParties());
157 
158         // End validation stuff
159         partyInfoService.updatePartyInfo(modifiedPartyInfo);
160 
161         return Response.ok().build();
162     }

164     @GET
165     @Produces(MediaType.APPLICATION_JSON)
166     @ApiOperation(value = "Fetch network/peer information", produces = "public list of peers/publickey mappings")
167     @ApiResponses({@ApiResponse(code = 200, message = "Peer/Network information", response = PartyInfo.class)})
168     public Response getPartyInfo() {
169 
170         final PartyInfo current = this.partyInfoService.getPartyInfo();
171 
172         // TODO: remove the filter when URIs don't need to end with a /
173         final JsonArrayBuilder peersBuilder = Json.createArrayBuilder();
174         current.getParties().stream()
175                 .filter(p -> p.getUrl().endsWith("/"))
176                 .map(
177                         party -> {
178                             final JsonObjectBuilder builder = Json.createObjectBuilder();
179                             builder.add("url", party.getUrl());
180                             if (party.getLastContacted() != null) {
181                                 builder.add("lastContact", party.getLastContacted().toString());
182                             } else {
183                                 builder.addNull("lastContact");
184                             }
185                             return builder.build();
186                         })
187                 .forEach(peersBuilder::add);
188 
189         final JsonArrayBuilder recipientBuilder = Json.createArrayBuilder();
190         current.getRecipients().stream()
191                 .map(
192                         recipient ->
193                                 Json.createObjectBuilder()
194                                         .add("key", recipient.getKey().encodeToBase64())
195                                         .add("url", recipient.getUrl())
196                                         .build())
197                 .forEach(recipientBuilder::add);
198 
199         final String output =
200                 Json.createObjectBuilder()
201                         .add("url", current.getUrl())
202                         .add("peers", peersBuilder.build())
203                         .add("keys", recipientBuilder.build())
204                         .build()
205                         .toString();
206 
207         return Response.status(Response.Status.OK).entity(output).build();
208     }

210     @POST
211     @Path("validate")
212     @Consumes(MediaType.APPLICATION_OCTET_STREAM)
213     @Produces(MediaType.TEXT_PLAIN)
214     public Response validate(byte[] payloadData) {
215         final EncodedPayload payload = payloadEncoder.decode(payloadData);
216 
217         final PublicKey mykey = payload.getRecipientKeys().iterator().next();
218 
219         final byte[] result = enclave.unencryptTransaction(payload, mykey);
220 
221         return Response.ok(new String(result)).build();
222     }
