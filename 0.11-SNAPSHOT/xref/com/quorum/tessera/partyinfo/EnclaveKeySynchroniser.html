3   import com.quorum.tessera.admin.ConfigService;
4   import com.quorum.tessera.enclave.Enclave;

8   import java.util.Objects;
9   import java.util.Set;
10  
11  import static java.util.Collections.emptySet;
12  import static java.util.stream.Collectors.toSet;
13  
14  /**
15   * Fetches local public keys from the Enclave and adds them to the local store. This is useful when the Enclave is
16   * remote and can restart with new keys independently of the Transaction Manager
17   */
18  public class EnclaveKeySynchroniser implements Runnable {
19  
20      private final Enclave enclave;
21  
22      private PartyInfoStore partyInfoStore;
23  
24      private final String advertisedUrl;
25  
26      public EnclaveKeySynchroniser(
27              final Enclave enclave, final PartyInfoStore partyInfoStore, final ConfigService configService) {
28          this.enclave = Objects.requireNonNull(enclave);
29          this.partyInfoStore = Objects.requireNonNull(partyInfoStore);
30          this.advertisedUrl = URLNormalizer.create().normalize(configService.getServerUri().toString());
31      }
32  
33      @Override
34      public void run() {
35          // fetch keys and create recipients
36          final Set<Recipient> ourKeys =
37                  this.enclave.getPublicKeys().stream()
38                          .map(key -> new Recipient(key, this.advertisedUrl))
39                          .collect(toSet());
40  
41          // add to store
42          this.partyInfoStore.store(new PartyInfo(this.advertisedUrl, ourKeys, emptySet()));
43      }
44  }
