3   import com.quorum.tessera.admin.ConfigService;
4   import com.quorum.tessera.enclave.Enclave;
5   import com.quorum.tessera.service.locator.ServiceLocator;

7   public class PartyInfoServiceFactoryImpl implements PartyInfoServiceFactory {
8   
9       private final ServiceLocator serviceLocator = ServiceLocator.create();
10  
11      public <T> T find(final Class<T> type) {
12          return serviceLocator.getServices().stream()
13                  .filter(type::isInstance)
14                  .map(type::cast)
15                  .findAny()
16                  .orElseThrow(() -> new IllegalStateException("Unable to find service type: " + type));
17      }
18  
19      @Override
20      public ResendManager resendManager() {
21          return find(ResendManager.class);
22      }
23  
24      @Override
25      public PartyInfoService partyInfoService() {
26          return find(PartyInfoService.class);
27      }
28  
29      @Override
30      public ConfigService configService() {
31          return find(ConfigService.class);
32      }
33  
34      @Override
35      public Enclave enclave() {
36          return find(Enclave.class);
37      }
38  
39      @Override
40      public PayloadPublisher payloadPublisher() {
41          return find(PayloadPublisher.class);
42      }
43  
44      @Override
45      public PartyInfoStore partyInfoStore() {
46          return find(PartyInfoStore.class);
47      }
48  }
