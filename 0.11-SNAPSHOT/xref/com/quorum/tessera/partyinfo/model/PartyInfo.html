1   package com.quorum.tessera.partyinfo.model;
2   
3   import java.util.Collections;
4   import java.util.HashSet;
5   import java.util.Objects;
6   import java.util.Set;
7   
8   /**
9    * Contains all information that is transferred between two nodes on the
10   * network, including:
11   * - the external URL of this node
12   * - all known {@link Recipient} that contains public key to URL mappings
13   * - all known URLs on the network
14   */
15  public class PartyInfo {
16      
17      private final String url;
18  
19      private Set<Recipient> recipients;
20  
21      private Set<Party> parties;
22  
23      public PartyInfo(final String url, final Set<Recipient> recipients, final Set<Party> parties) {
24          this.url = Objects.requireNonNull(url);
25          this.recipients = Collections.unmodifiableSet(new HashSet<>(recipients));
26          this.parties = Collections.unmodifiableSet(new HashSet<>(parties));
27      }
28  
29      public String getUrl() {
30          return url;
31      }
32  
33      public Set<Recipient> getRecipients() {
34          return recipients;
35      }
36  
37      public Set<Party> getParties() {
38          return parties;
39      }
40  
41  }
