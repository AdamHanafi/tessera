1   package com.quorum.tessera.argon2;
2   
3   import de.mkammerer.argon2.Argon2Advanced;
4   import de.mkammerer.argon2.Argon2Factory;
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   import java.util.Arrays;
9   
10  public class Argon2Impl implements Argon2 {
11  
12      private static final Logger LOGGER = LoggerFactory.getLogger(Argon2Impl.class);
13  
14      private static final ArgonOptions DEFAULT_OPTIONS = new ArgonOptions("i", 10, 1048576, 4);
15  
16      @Override
17      public ArgonResult hash(final ArgonOptions options, final String password, final byte[] salt) {
18          final Argon2Advanced argon2 = this.getArgon2Instance(options.getAlgorithm());
19  
20          final byte[] hash = argon2.rawHash(
21              options.getIterations(), options.getMemory(), options.getParallelism(), password, salt
22          );
23  
24          LOGGER.debug("Argon2 hash produced the array {}", Arrays.toString(hash));
25  
26          return new ArgonResult(options, salt, hash);
27      }
28  
29      @Override
30      public ArgonResult hash(final String password, final byte[] salt) {
31          return this.hash(DEFAULT_OPTIONS, password, salt);
32      }
33  
34      /**
35       * The string form of the algorithm to use.
36       * If an invalid algorithm is chosen, a default of Argon2i is chosen.
37       *
38       * @param algorithm the algorithm to use
39       * @return an instance of the chosen algorithm
40       */
41      private Argon2Advanced getArgon2Instance(final String algorithm) {
42          LOGGER.debug("Searching for the Argon2 algorithm {}", algorithm);
43  
44          switch (algorithm) {
45              case "d":
46                  return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2d);
47              case "id":
48                  return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2id);
49              case "i":
50                  return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2i);
51              default:
52                  throw new IllegalArgumentException("Invalid Argon2 algorithm " + algorithm);
53          }
54      }
55  
56  }
