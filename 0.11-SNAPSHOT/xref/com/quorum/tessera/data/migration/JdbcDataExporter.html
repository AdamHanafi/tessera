3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.nio.file.Path;
6   import java.sql.*;
7   import java.util.List;

9   public class JdbcDataExporter implements DataExporter {
10  
11      private final String jdbcUrl;
12  
13      private final String insertRow;
14  
15      private final List<String> createTables;
16  
17      public JdbcDataExporter(final String jdbcUrl, final String insertRow, final List<String> createTables) {
18          this.jdbcUrl = jdbcUrl;
19          this.insertRow = insertRow;
20          this.createTables = createTables;
21      }
22  
23      @Override
24      public void export(final StoreLoader loader,
25                         final Path output,
26                         final String username,
27                         final String password) throws SQLException, IOException {
28  
29          try (Connection conn = DriverManager.getConnection(jdbcUrl, username, password)) {
30  
31              try (Statement stmt = conn.createStatement()) {
32                  for (final String createTable : createTables) {
33                      stmt.executeUpdate(createTable);
34                  }
35              }
36  
37              try (PreparedStatement insertStatement = conn.prepareStatement(insertRow)) {
38                  DataEntry next;
39                  while ((next = loader.nextEntry()) != null) {
40                      try (InputStream data = next.getValue()) {
41                          insertStatement.setBytes(1, next.getKey());
42                          insertStatement.setBinaryStream(2, data);
43                          insertStatement.execute();
44                      }
45                  }
46              }
47  
48          }
49      }
50  
51  }
