2   
3   import com.quorum.tessera.enclave.RawTransaction;
4   import com.quorum.tessera.encryption.PublicKey;
5   import com.quorum.tessera.nacl.Nonce;
6   
7   import javax.persistence.*;
8   import java.io.Serializable;
9   import java.util.Objects;
10  
11  /** The JPA entity that contains the raw transaction information */
12  @Entity
13  @Table(name = "ENCRYPTED_RAW_TRANSACTION")
14  public class EncryptedRawTransaction implements Serializable {
15  
16      @EmbeddedId
17      @AttributeOverride(
18              name = "hashBytes",
19              column = @Column(name = "HASH", nullable = false, unique = true, updatable = false))
20      private MessageHash hash;
21  
22      @Lob
23      @Column(name = "ENCRYPTED_PAYLOAD", nullable = false)
24      private byte[] encryptedPayload;
25  
26      @Lob
27      @Column(name = "ENCRYPTED_KEY", nullable = false)
28      private byte[] encryptedKey;
29  
30      @Lob
31      @Column(name = "NONCE", nullable = false)
32      private byte[] nonce;
33  
34      @Lob
35      @Column(name = "SENDER", nullable = false)
36      private byte[] sender;
37  
38      @Column(name = "TIMESTAMP", updatable = false)
39      private long timestamp;
40  
41      public EncryptedRawTransaction(
42              final MessageHash hash,
43              final byte[] encryptedPayload,
44              final byte[] encryptedKey,
45              final byte[] nonce,
46              final byte[] sender) {
47          this.hash = hash;
48          this.encryptedPayload = encryptedPayload;
49          this.encryptedKey = encryptedKey;
50          this.nonce = nonce;
51          this.sender = sender;
52      }
53  
54      public EncryptedRawTransaction() {}
55  
56      @PrePersist
57      public void onPersist() {
58          this.timestamp = System.currentTimeMillis();
59      }
60  
61      public MessageHash getHash() {
62          return this.hash;
63      }
64  
65      public void setHash(final MessageHash hash) {
66          this.hash = hash;
67      }
68  
69      public byte[] getEncryptedPayload() {
70          return encryptedPayload;
71      }
72  
73      public void setEncryptedPayload(byte[] encryptedPayload) {
74          this.encryptedPayload = encryptedPayload;
75      }
76  
77      public byte[] getEncryptedKey() {
78          return encryptedKey;
79      }
80  
81      public void setEncryptedKey(byte[] encryptedKey) {
82          this.encryptedKey = encryptedKey;
83      }
84  
85      public byte[] getNonce() {
86          return nonce;
87      }
88  
89      public void setNonce(byte[] nonce) {
90          this.nonce = nonce;
91      }
92  
93      public long getTimestamp() {
94          return this.timestamp;
95      }
96  
97      public byte[] getSender() {
98          return sender;
99      }
100 
101     public void setSender(byte[] sender) {
102         this.sender = sender;
103     }
104 
105     @Override
106     public int hashCode() {
107         return 47 * 3 + Objects.hashCode(this.hash);
108     }
109 
110     @Override
111     public boolean equals(final Object obj) {
112 
113         return (obj instanceof EncryptedRawTransaction)
114                 && Objects.equals(this.hash, ((EncryptedRawTransaction) obj).hash);
115     }
116 
117     public RawTransaction toRawTransaction() {
118         return new RawTransaction(
119                 this.encryptedPayload, this.encryptedKey, new Nonce(this.nonce), PublicKey.from(this.sender));
120     }
121 }
