5   /**
6    * A SyncableParty is a {@link Party} that is to be contacted for transaction synchronisation. The number of contact
7    * attempts is stored alongside the party.
8    */
9   public class SyncableParty {
10  
11      private final Party party;
12  
13      private final int attempts;
14  
15      public SyncableParty(final Party party, final int attempts) {
16          this.party = party;
17          this.attempts = attempts;
18      }
19  
20      public Party getParty() {
21          return party;
22      }
23  
24      public int getAttempts() {
25          return attempts;
26      }
27  }
