1   package com.quorum.tessera.sync;
2   
3   import com.quorum.tessera.partyinfo.model.Party;
4   import com.quorum.tessera.sync.model.SyncableParty;
5   
6   import java.util.Collection;
7   import java.util.Optional;
8   
9   /**
10   * Stores nodes that need to be contacted for resending existing transactions
11   *
12   * <p>Also stores auxiliary information such as the number of attempts already tried for that node
13   */
14  public interface ResendPartyStore {
15  
16      int MAX_ATTEMPTS = 20;
17  
18      /**
19       * Adds a new set of parties to the list of all parties that are outstanding to be contacted.
20       *
21       * <p>Sets the number of attempts for each party to 0, even if the party already exists in the list with a higher
22       * attempt count.
23       *
24       * <p>Duplicates are allowed and will be treated individually when making requests. i.e. adding the same url twice
25       * will allow twice the limit of maximum attempts
26       *
27       * @param partiesToRequestFrom the parties to add to the contact list
28       */
29      void addUnseenParties(Collection<Party> partiesToRequestFrom);
30  
31      /**
32       * Retrieves the next party to contact in the queue
33       *
34       * @return Returns {@link Optional#empty()} if there are no parties to contact in the request queue else an Optional
35       *     containing a party to contact
36       */
37      Optional<SyncableParty> getNextParty();
38  
39      /**
40       * Submit a party that had been attempted to contact but failed This will increase its attempt count or discard it
41       * if the maximum number of attempts have been reached
42       *
43       * @param attemptedParty the party that has failed to be contacted
44       */
45      void incrementFailedAttempt(SyncableParty attemptedParty);
46  }
