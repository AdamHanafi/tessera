3   import com.quorum.tessera.config.AppType;
4   
5   import java.util.List;
6   
7   public class PrometheusProtocolFormatter {
8   
9       public String format(final List<MBeanMetric> metrics, AppType appType) {
10          StringBuilder formattedMetrics = new StringBuilder();
11  
12          for (final MBeanMetric metric : metrics) {
13              final MBeanResourceMetric resourceMetric = (MBeanResourceMetric) metric;
14  
15              formattedMetrics.append("tessera_")
16                  .append(appType)
17                  .append("_")
18                  .append(sanitize(resourceMetric.getResourceMethod()))
19                  .append("_")
20                  .append(sanitize(resourceMetric.getName()))
21                  .append(" ")
22                  .append(resourceMetric.getValue())
23                  .append("\n");
24          }
25  
26          return formattedMetrics.toString().trim();
27      }
28  
29      private String sanitize(final String input) {
30          return input
31              .replaceAll("(#.*)|(_total)|\\(\\)|\\)|\\[\\]|\\]|;", "")
32              .replaceAll("->|\\(|\\[", "_");
33      }
34  
35  }
