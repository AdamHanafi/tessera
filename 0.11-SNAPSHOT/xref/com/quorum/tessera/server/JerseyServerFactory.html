1   package com.quorum.tessera.server;
2   
3   import com.quorum.tessera.config.CommunicationType;
4   import com.quorum.tessera.config.ServerConfig;
5   import com.quorum.tessera.config.apps.TesseraApp;
6   
7   import java.util.Set;
8   import javax.ws.rs.core.Application;
9   
10  /** Creates Grizzly and Jersey implementations of the {@link TesseraServer} */
11  public class JerseyServerFactory implements TesseraServerFactory<Object> {
12  
13      @Override
14      public TesseraServer createServer(ServerConfig serverConfig, Set<Object> services) {
15          Application application =
16                  services.stream()
17                          .filter(TesseraApp.class::isInstance)
18                          .filter(Application.class::isInstance)
19                          .map(TesseraApp.class::cast)
20                          .filter(a -> a.getAppType().equals(serverConfig.getApp()))
21                          .map(Application.class::cast)
22                          .findFirst()
23                          .get();
24  
25          return new JerseyServer(serverConfig, application);
26      }
27  
28      @Override
29      public CommunicationType communicationType() {
30          return CommunicationType.REST;
31      }
32  }
