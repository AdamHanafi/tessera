3   import java.util.List;
4   import java.util.Objects;
5   import java.util.function.Predicate;
6   import java.util.stream.Collectors;
7   
8   public class OriginMatchUtil {
9   
10      private final List<String> tokens;
11  
12      private final Predicate<String> wildcardMatch = s -> s.equals("*");
13  
14      public OriginMatchUtil(List<String> tokens) {
15          this.tokens = tokens.stream()
16              .map(String::toLowerCase)
17              .map(s -> ("\\Q" + s + "\\E"))
18              .map(s -> s.replace("*", "\\E.*\\Q"))
19              .collect(Collectors.toList());
20      }
21  
22      public boolean matches(String origin) {
23          
24          if(Objects.isNull(origin) || Objects.equals("", origin)) {
25              return false;
26          }
27  
28          Predicate<String> subdomainMatch = s -> origin.toLowerCase().matches(s);
29          Predicate<String> matchingCritera = wildcardMatch.or(subdomainMatch);
30  
31          return tokens.stream().anyMatch(matchingCritera);
32      }
33  
34  }
