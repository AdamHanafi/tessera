1   package com.quorum.tessera.config.apps;

3   import com.quorum.tessera.ServiceLoaderUtil;
4   import com.quorum.tessera.config.AppType;
5   import com.quorum.tessera.config.CommunicationType;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  import java.util.Optional;
12  
13  public class TesseraAppFactory {
14  
15      private static final Logger LOGGER = LoggerFactory.getLogger(TesseraAppFactory.class);
16  
17      private final List<TesseraApp> cache = new ArrayList<>();
18  
19      private static final TesseraAppFactory INSTANCE = new TesseraAppFactory();
20  
21      public static Optional<TesseraApp> create(CommunicationType communicationType, AppType appType) {
22          return INSTANCE.createApp(communicationType, appType);
23      }
24  
25      private TesseraAppFactory() {
26          ServiceLoaderUtil.loadAll(TesseraApp.class).peek(app -> LOGGER.info("Loaded app {}", app)).forEach(cache::add);
27      }
28  
29      private Optional<TesseraApp> createApp(CommunicationType communicationType, AppType appType) {
30          LOGGER.info("Creating application type {} for {}", appType, communicationType);
31          return cache.stream()
32                  .filter(a -> a.getAppType() == appType)
33                  .filter(a -> a.getCommunicationType() == communicationType)
34                  .findAny();
35      }
36  }
