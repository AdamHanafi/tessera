26      public ConfigBuilder create(InputStream configData, ArgonOptions options, String... filenames) {
27          Objects.requireNonNull(configData, "No config data provided. ");
28          if (filenames.length != 0) {
29              throw new UnsupportedOperationException("keyConfigData arg is not implemented for TomlConfigFactory");
30          }
31  
32          Toml toml = new Toml().read(configData);
33          toml.toMap().forEach((key, value) -> LOGGER.debug("Found entry in toml file : {} {}", key, value));
34  
35          final String urlWithoutPort = Optional
36              .ofNullable(toml.getString("url"))
37              .map(url -> {
38                  try {
39                      return new URL(url);
40                  } catch (final MalformedURLException e) {
41                      throw new RuntimeException("Bad server url given: " + e.getMessage());
42                  }
43              }).map(uri -> uri.getProtocol() + "://" + uri.getHost())
44              .orElse(null);
45  
46          final Integer port = Optional.ofNullable(toml.getLong("port"))
47              .map(Long::intValue)
48              .orElse(null);
49  
50          final String workdir = toml.getString("workdir", "");
51          final String socket = toml.getString("socket");
52  
53          final String tls = toml.getString("tls", "off").toUpperCase();
54  
55          final List<String> othernodes = toml.getList("othernodes", emptyList());
56  
57          final List<String> alwaysSendToKeyPaths = toml.getList("alwayssendto", emptyList());
58  
59          final String storage = toml.getString("storage", "memory");
60  
61          final List<String> ipwhitelist = toml.getList("ipwhitelist", emptyList());
62          final boolean useWhiteList = !ipwhitelist.isEmpty();
63  
64          //Server side
65          final String tlsservertrust = toml.getString("tlsservertrust", "tofu");
66          final Optional<String> tlsserverkey = Optional.ofNullable(toml.getString("tlsserverkey"));
67          final Optional<String> tlsservercert = Optional.ofNullable(toml.getString("tlsservercert"));
68          final Optional<List<String>> tlsserverchainnames = Optional.ofNullable(toml.getList("tlsserverchain", emptyList()));
69          final Optional<String> tlsknownclients = Optional.ofNullable(toml.getString("tlsknownclients"));
70  
71          //Client side
72          final String tlsclienttrust = toml.getString("tlsclienttrust", "tofu");
73          final Optional<String> tlsclientkey = Optional.ofNullable(toml.getString("tlsclientkey"));
74          final Optional<String> tlsclientcert = Optional.ofNullable(toml.getString("tlsclientcert"));
75          final Optional<List<String>> tlsclientchainnames = Optional.ofNullable(toml.getList("tlsclientchain", emptyList()));
76          final Optional<String> tlsknownservers = Optional.ofNullable(toml.getString("tlsknownservers"));
77  
78          ConfigBuilder configBuilder = ConfigBuilder.create()
79              .serverPort(port)
80              .serverHostname(urlWithoutPort)
81              .unixSocketFile(socket)
82              .sslAuthenticationMode(SslAuthenticationMode.valueOf(tls))
83              .sslServerTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsservertrust))
84              .sslClientTrustMode(SslTrustModeFactory.resolveByLegacyValue(tlsclienttrust))
85              .peers(othernodes)
86              .alwaysSendTo(alwaysSendToKeyPaths)
87              .useWhiteList(useWhiteList)
88              .workdir(workdir);
89  
90          tlsserverkey.ifPresent(configBuilder::sslServerTlsKeyPath);
91          tlsservercert.ifPresent(configBuilder::sslServerTlsCertificatePath);
92          tlsserverchainnames.ifPresent(configBuilder::sslServerTrustCertificates);
93          tlsknownclients.ifPresent(configBuilder::sslKnownClientsFile);
94          tlsclientkey.ifPresent(configBuilder::sslClientTlsKeyPath);
95          tlsclientcert.ifPresent(configBuilder::sslClientTlsCertificatePath);
96          tlsclientchainnames.ifPresent(configBuilder::sslClientTrustCertificates);
97          tlsknownservers.ifPresent(configBuilder::sslKnownServersFile);
98  
99          Optional.ofNullable(storage)
100             .map(JdbcConfigFactory::fromLegacyStorageString)
101             .ifPresent(configBuilder::jdbcConfig);
102 
103         return configBuilder;
104     }

106     KeyDataBuilder createKeyDataBuilder(InputStream configData) {
107         final Toml toml = new Toml().read(configData);
108 
109         final List<String> publicKeyList = toml.getList("publickeys", emptyList());
110 
111         final List<String> privateKeyList = toml.getList("privatekeys", emptyList());
112 
113         final String pwd = toml.getString("passwords");
114 
115         final String workdir = toml.getString("workdir");
116 
117         return KeyDataBuilder.create()
118             .withPublicKeys(publicKeyList)
119             .withPrivateKeys(privateKeyList)
120             .withPrivateKeyPasswordFile(pwd)
121             .withWorkingDirectory(workdir);
122     }
