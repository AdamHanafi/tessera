8   import java.io.IOException;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.nio.file.Paths;
12  import java.util.Optional;
13  
14  public final class ConfigSecretReader {
15  
16      private static final Logger LOGGER = LoggerFactory.getLogger(ConfigSecretReader.class);
17  
18      private ConfigSecretReader() {}
19  
20      public static Optional<String> readSecretFromFile() {
21  
22          final EnvironmentVariableProvider envProvider = new EnvironmentVariableProvider();
23  
24          if (envProvider.hasEnv(EnvironmentVariables.CONFIG_SECRET_PATH)) {
25              final Path secretPath = Paths.get(envProvider.getEnv(EnvironmentVariables.CONFIG_SECRET_PATH));
26              if (Files.exists(secretPath)) {
27                  try {
28                      return Optional.of(new String(Files.readAllBytes(secretPath)).trim());
29                  } catch (IOException ex) {
30                      LOGGER.error("Error while reading secret from file");
31                  }
32              }
33          }
34  
35          LOGGER.warn("Not able to find or read any secret for decrypting sensitive values in config.");
36  
37          return Optional.empty();
38      }
39  
40      public static String readSecretFromConsole() {
41          SystemAdapter.INSTANCE.out().println("Please enter the secret/password used to decrypt config value");
42          return PasswordReaderFactory.create().readPasswordFromConsole();
43      }
44  }
