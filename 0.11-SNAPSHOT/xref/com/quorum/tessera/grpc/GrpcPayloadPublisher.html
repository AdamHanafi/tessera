2   package com.quorum.tessera.grpc;
3   
4   import com.quorum.tessera.client.GrpcP2pClient;
5   import com.quorum.tessera.enclave.EncodedPayload;
6   import com.quorum.tessera.enclave.PayloadEncoder;
7   import com.quorum.tessera.partyinfo.PayloadPublisher;
8   import java.util.Objects;
9   
10  
11  public class GrpcPayloadPublisher implements PayloadPublisher {
12      
13      private final com.quorum.tessera.client.GrpcP2pClient p2pClient;
14  
15      private final PayloadEncoder payloadEncoder;
16  
17      protected GrpcPayloadPublisher(GrpcP2pClient p2pClient, PayloadEncoder payloadEncoder) {
18          this.p2pClient = Objects.requireNonNull(p2pClient);
19          this.payloadEncoder = Objects.requireNonNull(payloadEncoder);
20      }
21  
22      public GrpcPayloadPublisher() {
23          this(new GrpcP2pClient(),PayloadEncoder.create());
24      }
25  
26      @Override
27      public void publishPayload(EncodedPayload payload, String targetUrl) {
28          p2pClient.push(targetUrl, payloadEncoder.encode(payload));
29      }
30   
31  }
