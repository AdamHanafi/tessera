1   package com.quorum.tessera.key.vault.azure;
2   
3   import com.microsoft.azure.keyvault.models.SecretBundle;
4   import com.microsoft.azure.keyvault.requests.SetSecretRequest;
5   import com.quorum.tessera.config.AzureKeyVaultConfig;
6   import com.quorum.tessera.config.vault.data.AzureGetSecretData;
7   import com.quorum.tessera.config.vault.data.AzureSetSecretData;
8   import com.quorum.tessera.key.vault.KeyVaultService;
9   import com.quorum.tessera.key.vault.VaultSecretNotFoundException;
10  
11  import java.util.Objects;
12  
13  public class AzureKeyVaultService implements KeyVaultService<AzureSetSecretData, AzureGetSecretData> {
14  
15      private String vaultUrl;
16  
17      private AzureKeyVaultClientDelegate azureKeyVaultClientDelegate;
18  
19      AzureKeyVaultService(AzureKeyVaultConfig keyVaultConfig, AzureKeyVaultClientDelegate azureKeyVaultClientDelegate) {
20          this.vaultUrl = Objects.requireNonNull(keyVaultConfig).getUrl();
21          this.azureKeyVaultClientDelegate = Objects.requireNonNull(azureKeyVaultClientDelegate);
22      }
23  
24      @Override
25      public String getSecret(AzureGetSecretData azureGetSecretData) {
26          SecretBundle secretBundle;
27  
28          if (azureGetSecretData.getSecretVersion() != null) {
29              secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName(), azureGetSecretData.getSecretVersion());
30          } else {
31              secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName());
32          }
33  
34          if (secretBundle == null) {
35              throw new VaultSecretNotFoundException("Azure Key Vault secret " + azureGetSecretData.getSecretName() + " was not found in vault " + vaultUrl);
36          }
37  
38          return secretBundle.value();
39      }
40  
41      @Override
42      public Object setSecret(AzureSetSecretData azureSetSecretData) {
43          SetSecretRequest setSecretRequest = new SetSecretRequest.Builder(vaultUrl, azureSetSecretData.getSecretName(), azureSetSecretData.getSecret()).build();
44  
45          return this.azureKeyVaultClientDelegate.setSecret(setSecretRequest);
46      }
47  }
