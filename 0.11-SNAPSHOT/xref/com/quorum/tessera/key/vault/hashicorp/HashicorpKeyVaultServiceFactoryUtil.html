25      SslConfiguration configureSsl(HashicorpKeyVaultConfig keyVaultConfig, EnvironmentVariableProvider envProvider) {
26          if(keyVaultConfig.getTlsKeyStorePath() != null && keyVaultConfig.getTlsTrustStorePath() != null) {
27  
28              Resource clientKeyStore = new FileSystemResource(keyVaultConfig.getTlsKeyStorePath().toFile());
29              Resource clientTrustStore = new FileSystemResource(keyVaultConfig.getTlsTrustStorePath().toFile());
30  
31              SslConfiguration.KeyStoreConfiguration keyStoreConfiguration = SslConfiguration.KeyStoreConfiguration.of(
32                  clientKeyStore,
33                  envProvider.getEnvAsCharArray(HASHICORP_CLIENT_KEYSTORE_PWD)
34              );
35  
36              SslConfiguration.KeyStoreConfiguration trustStoreConfiguration = SslConfiguration.KeyStoreConfiguration.of(
37                  clientTrustStore,
38                  envProvider.getEnvAsCharArray(HASHICORP_CLIENT_TRUSTSTORE_PWD)
39              );
40  
41              return new SslConfiguration(keyStoreConfiguration, trustStoreConfiguration);
42  
43          } else if (keyVaultConfig.getTlsTrustStorePath() != null) {
44  
45              Resource clientTrustStore = new FileSystemResource(keyVaultConfig.getTlsTrustStorePath().toFile());
46  
47              return SslConfiguration.forTrustStore(clientTrustStore, envProvider.getEnvAsCharArray(HASHICORP_CLIENT_TRUSTSTORE_PWD));
48  
49          } else {
50              return SslConfiguration.unconfigured();
51          }
52      }
53  
54      ClientHttpRequestFactory createClientHttpRequestFactory(ClientOptions clientOptions, SslConfiguration sslConfiguration) {
55          return ClientHttpRequestFactoryFactory.create(clientOptions, sslConfiguration);
56      }
57  
58      ClientAuthentication configureClientAuthentication(HashicorpKeyVaultConfig keyVaultConfig, EnvironmentVariableProvider envProvider, ClientHttpRequestFactory clientHttpRequestFactory, VaultEndpoint vaultEndpoint) {
59  
60          final String roleId = envProvider.getEnv(HASHICORP_ROLE_ID);
61          final String secretId = envProvider.getEnv(HASHICORP_SECRET_ID);
62          final String authToken = envProvider.getEnv(HASHICORP_TOKEN);
63  
64          if(roleId != null && secretId != null) {
65  
66              AppRoleAuthenticationOptions appRoleAuthenticationOptions = AppRoleAuthenticationOptions.builder()
67                  .path(keyVaultConfig.getApprolePath())
68                  .roleId(AppRoleAuthenticationOptions.RoleId.provided(roleId))
69                  .secretId(AppRoleAuthenticationOptions.SecretId.provided(secretId))
70                  .build();
71  
72              RestOperations restOperations = VaultClients.createRestTemplate(vaultEndpoint, clientHttpRequestFactory);
73  
74              return new AppRoleAuthentication(appRoleAuthenticationOptions, restOperations);
75  
76          } else if (Objects.isNull(roleId) != Objects.isNull(secretId)) {
77  
78              throw new HashicorpCredentialNotSetException("Both " + HASHICORP_ROLE_ID + " and " + HASHICORP_SECRET_ID + " environment variables must be set to use the AppRole authentication method");
79  
80          } else if (authToken == null){
81  
82              throw new HashicorpCredentialNotSetException("Both " + HASHICORP_ROLE_ID + " and " + HASHICORP_SECRET_ID + " environment variables must be set to use the AppRole authentication method.  Alternatively set " + HASHICORP_TOKEN + " to authenticate using the Token method");
83          }
84  
85          return new TokenAuthentication(authToken);
86      }
