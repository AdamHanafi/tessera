23  import static com.quorum.tessera.config.PrivateKeyType.LOCKED;
24  import static com.quorum.tessera.config.PrivateKeyType.UNLOCKED;
25  import static java.nio.charset.StandardCharsets.UTF_8;
26  import static java.nio.file.StandardOpenOption.CREATE_NEW;
27  
28  public class FileKeyGenerator implements KeyGenerator {
29  
30      private static final Logger LOGGER = LoggerFactory.getLogger(FileKeyGenerator.class);
31  
32      private static final String EMPTY_FILENAME = "";
33  
34      private final NaclFacade nacl;
35  
36      private final KeyEncryptor keyEncryptor;
37  
38      private final PasswordReader passwordReader;
39  
40      public FileKeyGenerator(final NaclFacade nacl, final KeyEncryptor keyEncryptor, final PasswordReader passwordReader) {
41          this.nacl = Objects.requireNonNull(nacl);
42          this.keyEncryptor = Objects.requireNonNull(keyEncryptor);
43          this.passwordReader = Objects.requireNonNull(passwordReader);
44      }
45  
46      @Override
47      public FilesystemKeyPair generate(final String filename, final ArgonOptions encryptionOptions, final KeyVaultOptions keyVaultOptions) {
48  
49          final String password = this.passwordReader.requestUserPassword();
50  
51          final KeyPair generated = this.nacl.generateNewKeys();
52  
53          final String publicKeyBase64 = Base64.getEncoder().encodeToString(generated.getPublicKey().getKeyBytes());
54  
55          final KeyData finalKeys = new KeyData();
56  
57          if (!password.isEmpty()) {
58  
59              final PrivateKeyData encryptedPrivateKey = this.keyEncryptor.encryptPrivateKey(
60                  generated.getPrivateKey(), password, encryptionOptions
61              );
62  
63              finalKeys.setConfig(
64                  new KeyDataConfig(
65                      new PrivateKeyData(
66                          null,
67                          encryptedPrivateKey.getSnonce(),
68                          encryptedPrivateKey.getAsalt(),
69                          encryptedPrivateKey.getSbox(),
70                          encryptedPrivateKey.getArgonOptions()
71                      ),
72                      LOCKED
73                  )
74              );
75  
76              LOGGER.info("Newly generated private key has been encrypted");
77  
78          } else {
79              
80              String keyData = Base64.getEncoder().encodeToString(generated.getPrivateKey().getKeyBytes());
81  
82              finalKeys.setConfig(new KeyDataConfig(new PrivateKeyData(keyData, null, null, null, null), UNLOCKED));
83  
84          }
85  
86          finalKeys.setPrivateKey(generated.getPrivateKey().encodeToBase64());
87          finalKeys.setPublicKey(publicKeyBase64);
88  
89          final String privateKeyJson = JaxbUtil.marshalToString(finalKeys.getConfig());
90  
91          final Path resolvedPath = Paths.get(filename).toAbsolutePath();
92          final Path parentPath;
93  
94          if(EMPTY_FILENAME.equals(filename)) {
95              parentPath = resolvedPath;
96          } else {
97              parentPath = resolvedPath.getParent();
98          }
99  
100         final Path publicKeyPath = parentPath.resolve(filename + ".pub");
101         final Path privateKeyPath = parentPath.resolve(filename + ".key");
102 
103         IOCallback.execute(() -> Files.write(publicKeyPath, publicKeyBase64.getBytes(UTF_8), CREATE_NEW));
104         IOCallback.execute(() -> Files.write(privateKeyPath, privateKeyJson.getBytes(UTF_8), CREATE_NEW));
105 
106         LOGGER.info("Saved public key to {}", publicKeyPath.toAbsolutePath().toString());
107         LOGGER.info("Saved private key to {}", privateKeyPath.toAbsolutePath().toString());
108 
109         final FilesystemKeyPair keyPair = new FilesystemKeyPair(publicKeyPath, privateKeyPath);
110 
111         keyPair.withPassword(password);
112 
113         return keyPair;
114     }
115 
116 }
