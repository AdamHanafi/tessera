1   package com.quorum.tessera.nacl.jnacl;
2   
3   import com.neilalexander.jnacl.crypto.curve25519xsalsa20poly1305;
4   
5   /**
6    * An interface to make testing of the static methods provided by
7    * {@link curve25519xsalsa20poly1305} easier.
8    */
9   public interface SecretBox {
10  
11      /**
12       * Computes a shared key between the given public and private key
13       *
14       * @param output The output array that contains the shared key
15       * @param publicKey The bytes of the public key
16       * @param privateKey The bytes of the private key
17       * @return Whether the operation was successful (0 for success, -1 for failure)
18       */
19      int cryptoBoxBeforenm(byte[] output, byte[] publicKey, byte[] privateKey);
20  
21      /**
22       * Seal a message into ciphertext using a shared key
23       *
24       * @param output the output ciphertext that contains the required 0 padding from NaCL
25       * @param message the input message with the require 0 padding from NaCL
26       * @param messageLength the length of the padding input message
27       * @param nonce a unique nonce to encrypt the message with
28       * @param sharedKey the shared key to encrypt the message with
29       * @return Whether the operation was successful (0 for success, -1 for failure)
30       */
31      int cryptoBoxAfternm(byte[] output, byte[] message, int messageLength, byte[] nonce, byte[] sharedKey);
32  
33      /**
34       * Opens a sealed box using the provided shared key
35       *
36       * @param output the output array for the unencrpyted message
37       * @param message the cipher text with the require 0 padding from NaCL
38       * @param messageLength the length of the message + the required 0 padding from NaCL
39       * @param nonce the nonce that was used to seal the box
40       * @param sharedKey the shared key
41       * @return Whether the operation was successful (0 for success, -1 for failure)
42       */
43      int cryptoBoxOpenAfternm(byte[] output, byte[] message, int messageLength, byte[] nonce, byte[] sharedKey);
44  
45      /**
46       * Generates a new keypair
47       * The input arrays must be at least 32 bytes long (and only the first 32 bytes are populated
48       * if the array is longer)
49       *
50       * @param publicKey The output array for the public key
51       * @param privateKey The output array for the private key
52       * @return Whether the operation was successful (0 for success, -1 for failure)
53       */
54      int cryptoBoxKeypair(byte[] publicKey, byte[] privateKey);
55  
56  }
