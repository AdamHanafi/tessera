1   package com.quorum.tessera.q2t;
2   
3   import com.quorum.tessera.api.filter.GlobalFilter;
4   import com.quorum.tessera.api.filter.IPWhitelistFilter;
5   import com.quorum.tessera.app.TesseraRestApplication;
6   import com.quorum.tessera.config.AppType;
7   import com.quorum.tessera.service.locator.ServiceLocator;
8   import java.util.Set;
9   import java.util.stream.Collectors;
10  import java.util.stream.Stream;
11  
12  import javax.ws.rs.ApplicationPath;
13  
14  /**
15   * The main application that is submitted to the HTTP server Contains all the service classes created by the service
16   * locator
17   */
18  @GlobalFilter
19  @ApplicationPath("/")
20  public class Q2TRestApp extends TesseraRestApplication {
21  
22      private final ServiceLocator serviceLocator;
23  
24      public Q2TRestApp() {
25          this(ServiceLocator.create());
26      }
27  
28      public Q2TRestApp(ServiceLocator serviceLocator) {
29          this.serviceLocator = serviceLocator;
30      }
31  
32      @Override
33      public Set<Object> getSingletons() {
34  
35          IPWhitelistFilter iPWhitelistFilter = new IPWhitelistFilter();
36          TransactionResource transactionResource = new TransactionResource();
37  
38          return Stream.of(iPWhitelistFilter, transactionResource).collect(Collectors.toSet());
39      }
40  
41      @Override
42      public AppType getAppType() {
43          return AppType.Q2T;
44      }
45  }
