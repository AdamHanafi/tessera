22  public class CliKeyPasswordResolver implements KeyPasswordResolver {
23  
24      private static final int MAX_PASSWORD_ATTEMPTS = 2;
25  
26      private final PasswordReader passwordReader;
27  
28      public CliKeyPasswordResolver() {
29          this(PasswordReaderFactory.create());
30      }
31  
32      public CliKeyPasswordResolver(final PasswordReader passwordReader) {
33          this.passwordReader = Objects.requireNonNull(passwordReader);
34      }
35  
36      public void resolveKeyPasswords(final Config config) {
37  
38          final KeyConfiguration input = config.getKeys();
39          if (input == null) {
40              //invalid config, but gets picked up by validation later
41              return;
42          }
43  
44          final List<String> allPasswords = new ArrayList<>();
45          if (input.getPasswords() != null) {
46              allPasswords.addAll(input.getPasswords());
47          } else if (input.getPasswordFile() != null) {
48              try {
49                  allPasswords.addAll(Files.readAllLines(input.getPasswordFile(), StandardCharsets.UTF_8));
50              } catch (final IOException ex) {
51                  //dont do anything, if any keys are locked validation will complain that
52                  //locked keys were provided without passwords
53                  SystemAdapter.INSTANCE.err().println("Could not read the password file");
54              }
55          }
56  
57          IntStream
58              .range(0, input.getKeyData().size())
59              .forEachOrdered(i -> {
60                  if(i < allPasswords.size()) {
61                      input.getKeyData().get(i).withPassword(allPasswords.get(i));
62                  }
63              });
64  
65          //decrypt the keys, either using provided passwords or read from CLI
66          IntStream
67              .range(0, input.getKeyData().size())
68              .forEachOrdered(keyNumber -> getSingleKeyPassword(keyNumber, input.getKeyData().get(keyNumber)));
69      }
70  
71      //TODO: make private
72      //@VisibleForTesting
73      public void getSingleKeyPassword(final int keyNumber, final ConfigKeyPair keyPair) {
74          final boolean isInline = keyPair instanceof InlineKeypair;
75          final boolean isFilesystem = keyPair instanceof FilesystemKeyPair;
76  
77          if (!isInline && !isFilesystem) {
78              //some other key type that doesn't use passwords, skip
79              return;
80          }
81  
82          final InlineKeypair inlineKey = isInline ? (InlineKeypair)keyPair : ((FilesystemKeyPair)keyPair).getInlineKeypair();
83  
84          if(inlineKey == null) {
85              //filesystem key pair that couldn't load the keys, catch in validation later
86              return;
87          }
88  
89          final boolean isLocked = inlineKey.getPrivateKeyConfig().getType() == PrivateKeyType.LOCKED;
90  
91          if (isLocked) {
92              int currentAttemptNumber = MAX_PASSWORD_ATTEMPTS;
93              while (currentAttemptNumber > 0) {
94                  if (StringUtils.isEmpty(keyPair.getPassword()) || keyPair.getPrivateKey() == null || keyPair.getPrivateKey().contains("NACL_FAILURE")) {
95                      final String attemptOutput = "Attempt " + (MAX_PASSWORD_ATTEMPTS-currentAttemptNumber+1) + " of " + MAX_PASSWORD_ATTEMPTS + ".";
96                      System.out.println("Password for key[" + keyNumber + "] missing or invalid.");
97                      System.out.println(attemptOutput + " Enter a password for the key");
98                      final String pass = passwordReader.readPasswordFromConsole();
99                      keyPair.withPassword(pass);
100                 }
101                 currentAttemptNumber--;
102             }
103         }
104     }
