001package com.quorum.tessera.data.migration;
002
003import java.net.URL;
004import java.nio.file.Path;
005import java.sql.SQLException;
006import java.util.Map;
007
008public class H2DataExporter implements DataExporter {
009
010    private static final String INSERT_ROW = "INSERT INTO ENCRYPTED_TRANSACTION (HASH,ENCODED_PAYLOAD) VALUES (?,?)";
011
012    @Override
013    public void export(final Map<byte[], byte[]> data,
014                       final Path output,
015                       final String username,
016                       final String password) throws SQLException {
017
018        final String connectionString = "jdbc:h2:" + output.toString();
019
020        final URL sqlFile = getClass().getResource("/ddls/h2-ddl.sql");
021
022        final JdbcDataExporter jdbcDataExporter = new JdbcDataExporter(connectionString, INSERT_ROW, sqlFile);
023
024        jdbcDataExporter.export(data, output, username, password);
025
026    }
027
028}
