001package com.quorum.tessera.data.migration;
002
003import java.net.URL;
004import java.nio.file.Files;
005import java.nio.file.Path;
006import java.nio.file.Paths;
007import java.sql.*;
008import java.util.List;
009import java.util.Map;
010import java.util.Map.Entry;
011
012public class JdbcDataExporter implements DataExporter {
013
014    private final String jdbcUrl;
015
016    private final String insertRow;
017
018    private final List<String> createTables;
019
020    public JdbcDataExporter(String jdbcUrl, String insertRow, URL ddl) {
021        this.jdbcUrl = jdbcUrl;
022        this.insertRow = insertRow;
023        this.createTables = UriCallback.execute(() -> Files.readAllLines(Paths.get(ddl.toURI())));
024    }
025
026    @Override
027    public void export(Map<byte[], byte[]> data, Path output, String username, String password) throws SQLException {
028
029        try (Connection conn = DriverManager.getConnection(jdbcUrl, username, password)) {
030
031            try (Statement stmt = conn.createStatement()) {
032                for (String createTable : createTables) {
033                    stmt.executeUpdate(createTable);
034                }
035            }
036
037            try (PreparedStatement insertStatement = conn.prepareStatement(insertRow)) {
038                for (Entry<byte[], byte[]> values : data.entrySet()) {
039                    insertStatement.setBytes(1, values.getKey());
040                    insertStatement.setBytes(2, values.getValue());
041                    insertStatement.execute();
042                }
043            }
044
045        }
046    }
047
048}
