001package com.quorum.tessera.data.migration;
002
003import org.apache.commons.cli.*;
004
005import java.io.InputStream;
006import java.nio.file.Files;
007import java.nio.file.Path;
008import java.nio.file.Paths;
009import java.util.*;
010
011import static java.util.Collections.singletonList;
012
013public class CmdLineExecutor {
014
015    private CmdLineExecutor() {
016        throw new UnsupportedOperationException("");
017    }
018
019    protected static int execute(String... args) throws Exception {
020
021        Options options = new Options();
022
023        options.addOption(
024                Option.builder()
025                        .longOpt("storetype")
026                        .desc("Store type i.e. bdb, dir, sqlite")
027                        .hasArg(true)
028                        .optionalArg(false)
029                        .numberOfArgs(1)
030                        .argName("TYPE").valueSeparator('=')
031                        .required()
032                        .build());
033
034        options.addOption(
035                Option.builder()
036                        .longOpt("inputpath")
037                        .desc("Path to input file or directory")
038                        .hasArg(true)
039                        .optionalArg(false)
040                        .numberOfArgs(1)
041                        .argName("PATH")
042                        .required()
043                        .build());
044
045
046
047        options.addOption(
048                Option.builder()
049                        .longOpt("exporttype")
050                        .desc("Export DB type i.e. h2, sqlite")
051                        .hasArg(true)
052                        .optionalArg(false)
053                        .numberOfArgs(1)
054                        .argName("TYPE")
055                        .required()
056                        .build());
057
058        options.addOption(
059                Option.builder()
060                        .longOpt("dbconfig")
061                        .desc("Properties file with create table, insert row and jdbc url")
062                        .hasArg(true)
063                        .optionalArg(false)
064                        .numberOfArgs(1)
065                        .argName("PATH")
066                        .build());
067
068        options.addOption(
069                Option.builder()
070                        .longOpt("outputfile")
071                        .desc("Path to output file")
072                        .hasArg(true)
073                        .optionalArg(false)
074                        .numberOfArgs(1)
075                        .argName("PATH")
076                        .required()
077                        .build());
078
079        options.addOption(
080                Option.builder()
081                        .longOpt("dbuser")
082                        .desc("Database username to use")
083                        .hasArg(true)
084                        .optionalArg(true)
085                        .numberOfArgs(1)
086                        .argName("USER")
087                        .required()
088                        .build());
089
090        options.addOption(
091                Option.builder()
092                        .longOpt("dbpass")
093                        .desc("Database password to use")
094                        .hasArg(true)
095                        .optionalArg(true)
096                        .numberOfArgs(1)
097                        .argName("PASS")
098                        .required()
099                        .build());
100
101
102
103        if (Arrays.asList(args).contains("help")) {
104            HelpFormatter formatter = new HelpFormatter();
105            formatter.printHelp("tessera-data-migration", options);
106            return 0;
107        }
108
109        final CommandLineParser parser = new DefaultParser();
110        final CommandLine line = parser.parse(options, args);
111
112        final StoreType storeType = StoreType.valueOf(line.getOptionValue("storetype").toUpperCase());
113        final StoreLoader storeLoader = StoreLoader.create(storeType);
114
115        final Path inputpath = Paths.get(line.getOptionValue("inputpath"));
116        final Map<byte[], byte[]> data = storeLoader.load(inputpath);
117
118        final String username = line.getOptionValue("dbuser");
119        final String password = line.getOptionValue("dbpass");
120
121        final String exportTypeStr = line.getOptionValue("exporttype");
122
123        final ExportType exportType = Optional.ofNullable(exportTypeStr)
124                .map(String::toUpperCase)
125                .map(ExportType::valueOf).get();
126
127        final Path outputFile = Paths.get(line.getOptionValue("outputfile")).toAbsolutePath();
128
129        final DataExporter dataExporter;
130        if (exportType == ExportType.JDBC) {
131            if (!line.hasOption("dbconfig")) {
132                throw new MissingOptionException("dbconfig file path is required when no export type is defined.");
133            }
134
135            String dbconfig = line.getOptionValue("dbconfig");
136
137            Properties properties = new Properties();
138            try (InputStream inStream = Files.newInputStream(Paths.get(dbconfig))) {
139                properties.load(inStream);
140            }
141
142            String insertRow = Objects.requireNonNull(properties.getProperty("insertRow",null),
143                    "No insertRow value defined in config file. ");
144
145            String createTable = Objects.requireNonNull(properties.getProperty("createTable",null),
146                    "No createTable value defined in config file. ");
147
148            String jdbcUrl = Objects.requireNonNull(properties.getProperty("jdbcUrl",null),
149                    "No jdbcUrl value defined in config file. ");
150
151            Path sqlFile = Files.createTempFile(UUID.randomUUID().toString(), ".txt");
152
153            Files.write(sqlFile, singletonList(createTable));
154
155            dataExporter = new JdbcDataExporter(jdbcUrl, insertRow, sqlFile.toUri().toURL());
156
157        } else {
158            dataExporter = DataExporterFactory.create(exportType);
159        }
160
161        dataExporter.export(data, outputFile, username, password);
162
163        System.out.printf("Exported data to %s", Objects.toString(outputFile));
164        System.out.println();
165
166        return 0;
167    }
168}
