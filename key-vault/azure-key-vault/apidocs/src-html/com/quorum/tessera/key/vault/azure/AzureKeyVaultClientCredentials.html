001package com.quorum.tessera.key.vault.azure;
002
003import com.microsoft.aad.adal4j.AuthenticationContext;
004import com.microsoft.aad.adal4j.ClientCredential;
005import com.microsoft.azure.keyvault.authentication.KeyVaultCredentials;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import javax.annotation.PreDestroy;
010import java.net.MalformedURLException;
011import java.util.Objects;
012import java.util.concurrent.ExecutionException;
013import java.util.concurrent.ExecutorService;
014
015import static com.quorum.tessera.config.util.EnvironmentVariables.AZURE_CLIENT_ID;
016import static com.quorum.tessera.config.util.EnvironmentVariables.AZURE_CLIENT_SECRET;
017
018public class AzureKeyVaultClientCredentials extends KeyVaultCredentials {
019
020    private static final Logger LOGGER = LoggerFactory.getLogger(AzureKeyVaultClientCredentials.class);
021
022    private final String clientId;
023
024    private final String clientSecret;
025
026    private AuthenticationContext authenticationContext;
027
028    private final ExecutorService executorService;
029
030    AzureKeyVaultClientCredentials(String clientId, String clientSecret, ExecutorService executorService) {
031        this.clientId = clientId;
032        this.clientSecret = clientSecret;
033        this.executorService = Objects.requireNonNull(executorService);
034    }
035
036    void setAuthenticationContext(AuthenticationContext authenticationContext) {
037        this.authenticationContext = authenticationContext;
038    }
039
040    @Override
041    public String doAuthenticate(String authorization, String resource, String scope) {
042        if(clientId == null || clientSecret == null) {
043            throw new AzureCredentialNotSetException(AZURE_CLIENT_ID + " and " + AZURE_CLIENT_SECRET + " environment variables must be set");
044        }
045        try {
046            if(Objects.isNull(authenticationContext)) {
047                this.authenticationContext = new AuthenticationContext(authorization, false, executorService);
048            }
049            ClientCredential credential = new ClientCredential(clientId,clientSecret);
050
051            return authenticationContext.acquireToken(resource, credential, null).get().getAccessToken();
052
053        } catch (ExecutionException | MalformedURLException ex) {
054            throw new RuntimeException(ex);
055        } catch(InterruptedException ex) {
056            LOGGER.warn("Key vault executor executorService interrupted");
057            throw new RuntimeException(ex);
058        }
059    }
060
061    @PreDestroy
062    public void onDestroy() {
063        executorService.shutdown();
064    }
065}
