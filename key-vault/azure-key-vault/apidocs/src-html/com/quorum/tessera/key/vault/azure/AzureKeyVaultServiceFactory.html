001package com.quorum.tessera.key.vault.azure;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.util.EnvironmentVariableProvider;
005import com.quorum.tessera.key.vault.KeyVaultService;
006import com.quorum.tessera.key.vault.KeyVaultServiceFactory;
007
008import java.util.Objects;
009import java.util.Optional;
010import java.util.concurrent.Executors;
011
012import static com.quorum.tessera.config.util.EnvironmentVariables.AZURE_CLIENT_ID;
013import static com.quorum.tessera.config.util.EnvironmentVariables.AZURE_CLIENT_SECRET;
014
015public class AzureKeyVaultServiceFactory implements KeyVaultServiceFactory {
016
017    @Override
018    public KeyVaultService create(Config config, EnvironmentVariableProvider envProvider) {
019        Objects.requireNonNull(config);
020        Objects.requireNonNull(envProvider);
021
022        String clientId = envProvider.getEnv(AZURE_CLIENT_ID);
023        String clientSecret = envProvider.getEnv(AZURE_CLIENT_SECRET);
024
025        if(clientId == null || clientSecret == null) {
026            throw new AzureCredentialNotSetException(AZURE_CLIENT_ID + " and " + AZURE_CLIENT_SECRET + " environment variables must be set");
027        }
028
029        AzureKeyVaultConfig keyVaultConfig = Optional.ofNullable(config.getKeys())
030            .map(KeyConfiguration::getAzureKeyVaultConfig)
031            .orElseThrow(() -> new ConfigException(new RuntimeException("Trying to create Azure key vault connection but no Azure configuration provided")));
032
033        return new AzureKeyVaultService(
034            keyVaultConfig,
035            new AzureKeyVaultClientDelegate(
036                new AzureKeyVaultClientFactory(
037                    new AzureKeyVaultClientCredentials(
038                        clientId,
039                        clientSecret,
040                        Executors.newFixedThreadPool(1)
041                    )
042                ).getAuthenticatedClient()
043            )
044        );
045    }
046
047    @Override
048    public KeyVaultType getType() {
049        return KeyVaultType.AZURE;
050    }
051}
