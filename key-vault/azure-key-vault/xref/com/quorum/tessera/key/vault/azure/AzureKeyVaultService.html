1   package com.quorum.tessera.key.vault.azure;
2   
3   import com.microsoft.azure.keyvault.models.SecretBundle;
4   import com.microsoft.azure.keyvault.requests.SetSecretRequest;
5   import com.quorum.tessera.config.AzureKeyVaultConfig;
6   import com.quorum.tessera.config.vault.data.AzureGetSecretData;
7   import com.quorum.tessera.config.vault.data.AzureSetSecretData;
8   import com.quorum.tessera.config.vault.data.GetSecretData;
9   import com.quorum.tessera.config.vault.data.SetSecretData;
10  import com.quorum.tessera.key.vault.KeyVaultException;
11  import com.quorum.tessera.key.vault.KeyVaultService;
12  import com.quorum.tessera.key.vault.VaultSecretNotFoundException;
13  
14  import java.util.Objects;
15  
16  public class AzureKeyVaultService implements KeyVaultService {
17      private String vaultUrl;
18      private AzureKeyVaultClientDelegate azureKeyVaultClientDelegate;
19  
20      AzureKeyVaultService(AzureKeyVaultConfig keyVaultConfig, AzureKeyVaultClientDelegate azureKeyVaultClientDelegate) {
21          Objects.requireNonNull(keyVaultConfig);
22          Objects.requireNonNull(azureKeyVaultClientDelegate);
23  
24          this.vaultUrl = keyVaultConfig.getUrl();
25          this.azureKeyVaultClientDelegate = azureKeyVaultClientDelegate;
26      }
27  
28      @Override
29      public String getSecret(GetSecretData getSecretData) {
30          if(!(getSecretData instanceof AzureGetSecretData)) {
31              throw new KeyVaultException("Incorrect data type passed to AzureKeyVaultService.  Type was " + getSecretData.getType());
32          }
33  
34          AzureGetSecretData azureGetSecretData = (AzureGetSecretData) getSecretData;
35  
36          SecretBundle secretBundle;
37  
38          if(azureGetSecretData.getSecretVersion() != null) {
39              secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName(), azureGetSecretData.getSecretVersion());
40          }
41          else {
42              secretBundle = azureKeyVaultClientDelegate.getSecret(vaultUrl, azureGetSecretData.getSecretName());
43          }
44  
45          if(secretBundle == null) {
46              throw new VaultSecretNotFoundException("Azure Key Vault secret " + azureGetSecretData.getSecretName() + " was not found in vault " + vaultUrl);
47          }
48  
49          return secretBundle.value();
50      }
51  
52      @Override
53      public Object setSecret(SetSecretData setSecretData) {
54          if(!(setSecretData instanceof AzureSetSecretData)) {
55              throw new KeyVaultException("Incorrect data type passed to AzureKeyVaultService.  Type was " + setSecretData.getType());
56          }
57  
58          AzureSetSecretData azureSetSecretData = (AzureSetSecretData) setSecretData;
59  
60          SetSecretRequest setSecretRequest = new SetSecretRequest.Builder(vaultUrl, azureSetSecretData.getSecretName(), azureSetSecretData.getSecret()).build();
61  
62          return this.azureKeyVaultClientDelegate.setSecret(setSecretRequest);
63      }
64  }
