001package com.quorum.tessera.key.vault.hashicorp;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.util.EnvironmentVariableProvider;
005import com.quorum.tessera.key.vault.KeyVaultService;
006import com.quorum.tessera.key.vault.KeyVaultServiceFactory;
007import org.springframework.http.client.ClientHttpRequestFactory;
008import org.springframework.vault.authentication.ClientAuthentication;
009import org.springframework.vault.authentication.SessionManager;
010import org.springframework.vault.authentication.SimpleSessionManager;
011import org.springframework.vault.client.VaultEndpoint;
012import org.springframework.vault.core.VaultOperations;
013import org.springframework.vault.core.VaultTemplate;
014import org.springframework.vault.support.ClientOptions;
015import org.springframework.vault.support.SslConfiguration;
016
017import java.net.URI;
018import java.net.URISyntaxException;
019import java.util.Objects;
020import java.util.Optional;
021
022import static com.quorum.tessera.config.util.EnvironmentVariables.*;
023
024public class HashicorpKeyVaultServiceFactory implements KeyVaultServiceFactory {
025
026    @Override
027    public KeyVaultService create(Config config, EnvironmentVariableProvider envProvider) {
028        Objects.requireNonNull(config);
029        Objects.requireNonNull(envProvider);
030
031        HashicorpKeyVaultServiceFactoryUtil util = new HashicorpKeyVaultServiceFactoryUtil();
032
033        return this.create(config, envProvider, util);
034    }
035
036    //This method should not be called directly. It has been left package-private to enable injection of util during testing
037    KeyVaultService create(Config config, EnvironmentVariableProvider envProvider, HashicorpKeyVaultServiceFactoryUtil util) {
038        Objects.requireNonNull(config);
039        Objects.requireNonNull(envProvider);
040        Objects.requireNonNull(util);
041
042        final String roleId = envProvider.getEnv(HASHICORP_ROLE_ID);
043        final String secretId = envProvider.getEnv(HASHICORP_SECRET_ID);
044        final String authToken = envProvider.getEnv(HASHICORP_TOKEN);
045
046        if(roleId == null && secretId == null && authToken == null) {
047            throw new HashicorpCredentialNotSetException("Environment variables must be set to authenticate with Hashicorp Vault.  Set the " + HASHICORP_ROLE_ID + " and " + HASHICORP_SECRET_ID + " environment variables if using the AppRole authentication method.  Set the " + HASHICORP_TOKEN + " environment variable if using another authentication method.");
048        }
049        else if(isOnlyOneInputNull(roleId, secretId)) {
050            throw new HashicorpCredentialNotSetException("Only one of the " + HASHICORP_ROLE_ID + " and " + HASHICORP_SECRET_ID + " environment variables to authenticate with Hashicorp Vault using the AppRole method has been set");
051        }
052
053        HashicorpKeyVaultConfig keyVaultConfig = Optional.ofNullable(config.getKeys())
054            .map(KeyConfiguration::getHashicorpKeyVaultConfig)
055            .orElseThrow(() -> new ConfigException(new RuntimeException("Trying to create Hashicorp Vault connection but no Vault configuration provided")));
056
057        VaultEndpoint vaultEndpoint;
058
059        try {
060            vaultEndpoint = VaultEndpoint.from(new URI(keyVaultConfig.getUrl()));
061        } catch (URISyntaxException | IllegalArgumentException e) {
062            throw new ConfigException(new RuntimeException("Provided Hashicorp Vault url is incorrectly formatted", e));
063        }
064
065        SslConfiguration sslConfiguration = util.configureSsl(keyVaultConfig, envProvider);
066
067        ClientOptions clientOptions = new ClientOptions();
068
069        ClientHttpRequestFactory clientHttpRequestFactory = util.createClientHttpRequestFactory(clientOptions, sslConfiguration);
070
071        ClientAuthentication clientAuthentication = util.configureClientAuthentication(keyVaultConfig, envProvider, clientHttpRequestFactory, vaultEndpoint);
072
073        SessionManager sessionManager = new SimpleSessionManager(clientAuthentication);
074        VaultOperations vaultOperations = new VaultTemplate(vaultEndpoint, clientHttpRequestFactory, sessionManager);
075
076        return new HashicorpKeyVaultService(
077            new KeyValueOperationsDelegateFactory(vaultOperations)
078        );
079    }
080
081    @Override
082    public KeyVaultType getType() {
083        return KeyVaultType.HASHICORP;
084    }
085
086    private boolean isOnlyOneInputNull(Object obj1, Object obj2) {
087        return Objects.isNull(obj1) ^ Objects.isNull(obj2);
088    }
089}
