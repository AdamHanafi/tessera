3   import com.quorum.tessera.service.locator.ServiceLocator;

5   import javax.ws.rs.core.Application;
6   import java.util.Objects;
7   import java.util.Set;
8   import java.util.stream.Collectors;
9   
10  /**
11   * The main application that is submitted to the HTTP server
12   * Contains all the service classes created by the service locator
13   */
14  public class RestApp extends Application {
15  
16      private final ServiceLocator serviceLocator;
17  
18      private final String contextName;
19  
20      public RestApp(final ServiceLocator serviceLocator, final String contextName) {
21          this.serviceLocator = Objects.requireNonNull(serviceLocator);
22          this.contextName = Objects.requireNonNull(contextName);
23      }
24  
25      @Override
26      public Set<Object> getSingletons() {
27          final String apiPackageName = getClass().getPackage().getName();
28  
29          //TODO find a nicer way for rest apps to share exception mappers / filters
30  
31          return serviceLocator.getServices(contextName).stream()
32              .filter(Objects::nonNull)
33              .filter(o -> Objects.nonNull(o.getClass()))
34              .filter(o -> Objects.nonNull(o.getClass().getPackage()))
35              .filter(o -> o.getClass().getPackage().getName().startsWith(apiPackageName) ||
36                  o.getClass().getPackage().getName().startsWith("com.quorum.tessera.api.exception") ||
37                  o.getClass().getPackage().getName().startsWith("com.quorum.tessera.api.filter") ||
38                  o.getClass().getPackage().getName().startsWith("com.quorum.tessera.api.common"))
39              .collect(Collectors.toSet());
40      }
41  
42  }
