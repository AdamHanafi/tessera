20  public class RestAppTest {
21  
22      private static final String CONTEXT_NAME = "context";
23  
24      private ServiceLocator serviceLocator;
25  
26      private P2PRestApp p2PRestApp;
27  
28      @Before
29      public void setUp() {
30          serviceLocator = mock(ServiceLocator.class);
31          p2PRestApp = new P2PRestApp(serviceLocator, CONTEXT_NAME);
32      }
33  
34      @After
35      public void tearDown() {
36          verifyNoMoreInteractions(serviceLocator);
37      }
38  
39      @Test
40      public void getSingletons() {
41          p2PRestApp.getSingletons();
42          verify(serviceLocator).getServices(CONTEXT_NAME);
43      }
44  
45      @Test
46      public void createWithNoServiceLocator() {
47  
48          final Throwable throwable = catchThrowable(() -> new P2PRestApp(null, CONTEXT_NAME));
49          assertThat(throwable).isInstanceOf(NullPointerException.class);
50  
51          final Throwable throwableName = catchThrowable(() -> new P2PRestApp(serviceLocator, null));
52          assertThat(throwableName).isInstanceOf(NullPointerException.class);
53  
54      }
55  
56      @Test
57      public void onCreateApiObjects() {
58          ApiResource apiObject = new ApiResource();
59          DefaultExceptionMapper nestedApiObject = new DefaultExceptionMapper();
60          Object nonApiObject = new HashMap<>();
61  
62          when(serviceLocator.getServices(CONTEXT_NAME))
63              .thenReturn(Stream.of(apiObject, nestedApiObject, nonApiObject)
64                  .collect(Collectors.toSet()));
65  
66          Set<Object> result = p2PRestApp.getSingletons();
67          assertThat(result).containsOnly(apiObject, nestedApiObject);
68          verify(serviceLocator).getServices(CONTEXT_NAME);
69      }
