001package com.quorum.tessera.admin;
002
003import com.quorum.tessera.config.Peer;
004import com.quorum.tessera.config.apps.AdminApp;
005import com.quorum.tessera.core.config.ConfigService;
006import com.quorum.tessera.encryption.PublicKey;
007import com.quorum.tessera.node.PartyInfoService;
008import com.quorum.tessera.node.model.Party;
009import com.quorum.tessera.node.model.PartyInfo;
010
011import javax.validation.Valid;
012import javax.ws.rs.*;
013import javax.ws.rs.core.GenericEntity;
014import javax.ws.rs.core.MediaType;
015import javax.ws.rs.core.Response;
016import javax.ws.rs.core.UriBuilder;
017import java.net.URI;
018import java.util.Base64;
019import java.util.List;
020import java.util.Objects;
021import java.util.Set;
022import java.util.stream.Collectors;
023
024import static java.util.Collections.emptySet;
025import static java.util.Collections.singleton;
026
027@Path("/config")
028@Consumes(MediaType.APPLICATION_JSON)
029@Produces(MediaType.APPLICATION_JSON)
030public class ConfigResource implements AdminApp {
031
032    private final ConfigService configService;
033
034    private final PartyInfoService partyInfoService;
035
036    public ConfigResource(final ConfigService configService, final PartyInfoService partyInfoService) {
037        this.configService = Objects.requireNonNull(configService);
038        this.partyInfoService = Objects.requireNonNull(partyInfoService);
039    }
040
041    @PUT
042    @Path("/peers")
043    public Response addPeer(@Valid final Peer peer) {
044
045        final boolean existing = configService.getPeers().contains(peer);
046
047        if (!existing) {
048            this.configService.addPeer(peer.getUrl());
049
050            this.partyInfoService.updatePartyInfo(
051                new PartyInfo(peer.getUrl(), emptySet(), singleton(new Party(peer.getUrl())))
052            );
053        }
054
055        final int index = this.configService.getPeers().indexOf(peer);
056
057        final URI uri = UriBuilder.fromPath("config")
058            .path("peers")
059            .path(String.valueOf(index))
060            .build();
061
062        if (!existing) {
063            return Response.created(uri).build();
064        } else {
065            return Response.ok().location(uri).build();
066        }
067
068    }
069
070    @GET
071    @Path("/peers/{index}")
072    public Response getPeer(@PathParam("index") final Integer index) {
073
074        final List<Peer> peers = this.configService.getPeers();
075
076        if (peers.size() <= index) {
077            throw new NotFoundException("No peer found at index " + index);
078        }
079
080        return Response.ok(peers.get(index)).build();
081    }
082
083    @GET
084    @Path("/peers")
085    public Response getPeers() {
086        final List<Peer> peers = this.configService.getPeers();
087
088        return Response.ok(new GenericEntity<List<Peer>>(peers) {
089        }).build();
090    }
091
092    @GET
093    @Path("/keypairs/{publicKey}")
094    public Response getKeyPair(@PathParam("publicKey") String base64PublicKey) {
095
096        Base64.Decoder base64Decoder = Base64.getDecoder();
097
098        PublicKey publicKey = PublicKey.from(base64Decoder.decode(base64PublicKey));
099
100        Set<PublicKey> publicKeys = configService.getPublicKeys();
101
102        if(!publicKeys.contains(publicKey)) {
103            throw new NotFoundException("No key pair found with public key " + base64PublicKey);
104        }
105
106        PublicKeyResponse responseData = new PublicKeyResponse(base64PublicKey);
107
108        return Response.ok(responseData).build();
109    }
110
111    @GET
112    @Path("/keypairs")
113    public Response getKeyPairs() {
114        Set<PublicKey> publicKeys = configService.getPublicKeys();
115
116        List<PublicKeyResponse> responseData = publicKeys.stream()
117            .map(PublicKey::encodeToBase64)
118            .map(PublicKeyResponse::new)
119            .collect(Collectors.toList());
120
121        return Response.ok(new GenericEntity<List<PublicKeyResponse>>(responseData){}).build();
122    }
123
124}
