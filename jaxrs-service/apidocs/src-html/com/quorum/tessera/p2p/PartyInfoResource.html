001package com.quorum.tessera.p2p;
002
003import com.quorum.tessera.node.PartyInfoParser;
004import com.quorum.tessera.node.PartyInfoService;
005import com.quorum.tessera.node.model.PartyInfo;
006import io.swagger.annotations.ApiOperation;
007import io.swagger.annotations.ApiParam;
008import io.swagger.annotations.ApiResponse;
009import io.swagger.annotations.ApiResponses;
010
011import javax.json.Json;
012import javax.json.JsonArrayBuilder;
013import javax.json.JsonObjectBuilder;
014import javax.ws.rs.*;
015import javax.ws.rs.core.MediaType;
016import javax.ws.rs.core.Response;
017import javax.ws.rs.core.StreamingOutput;
018
019import static java.util.Objects.requireNonNull;
020
021/**
022 * Defines endpoints for requesting node discovery (partyinfo) information
023 */
024@Path("/partyinfo")
025public class PartyInfoResource {
026
027    private final PartyInfoParser partyInfoParser;
028
029    private final PartyInfoService partyInfoService;
030
031    public PartyInfoResource(final PartyInfoService partyInfoService, final PartyInfoParser partyInfoParser) {
032        this.partyInfoService = requireNonNull(partyInfoService, "partyInfoService must not be null");
033        this.partyInfoParser = requireNonNull(partyInfoParser, "partyInfoParser must not be null");
034    }
035
036    /**
037     * Allows node information to be retrieved in a specific encoded form
038     * including other node URLS and public key to URL mappings
039     *
040     * @param payload The encoded node information from the requester
041     * @return the merged node information from this node, which may contain new information
042     */
043    @POST
044    @Consumes(MediaType.APPLICATION_OCTET_STREAM)
045    @Produces(MediaType.APPLICATION_OCTET_STREAM)
046    @ApiOperation(value = "Request public key/url of other nodes", produces = "public keylist/url")
047    @ApiResponses({@ApiResponse(code = 200, message = "Encoded PartyInfo Data", response = byte[].class)})
048    public Response partyInfo(@ApiParam(required = true) final byte[] payload) {
049
050        final PartyInfo partyInfo = partyInfoParser.from(payload);
051
052        final PartyInfo updatedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
053
054        final byte[] encoded = partyInfoParser.to(updatedPartyInfo);
055
056        final StreamingOutput streamingOutput = out -> out.write(encoded);
057
058        return Response.status(Response.Status.OK).entity(streamingOutput).build();
059    }
060
061    @GET
062    @Produces(MediaType.APPLICATION_JSON)
063    @ApiOperation(value = "Fetch network/peer information", produces = "public list of peers/publickey mappings")
064    @ApiResponses({@ApiResponse(code = 200, message = "Peer/Network information", response = PartyInfo.class)})
065    public Response getPartyInfo() {
066
067        final PartyInfo current = this.partyInfoService.getPartyInfo();
068
069        //TODO: remove the filter when URIs don't need to end with a /
070        final JsonArrayBuilder peersBuilder = Json.createArrayBuilder();
071        current.getParties()
072            .stream()
073            .filter(p -> p.getUrl().endsWith("/"))
074            .map(party -> {
075                final JsonObjectBuilder builder = Json.createObjectBuilder();
076                builder.add("url", party.getUrl());
077                if (party.getLastContacted() != null) {
078                    builder.add("lastContact", party.getLastContacted().toString());
079                } else {
080                    builder.addNull("lastContact");
081                }
082                return builder.build();
083            })
084            .forEach(peersBuilder::add);
085
086        final JsonArrayBuilder recipientBuilder = Json.createArrayBuilder();
087        current.getRecipients()
088            .stream()
089            .map(recipient -> Json
090                .createObjectBuilder()
091                .add("key", recipient.getKey().encodeToBase64())
092                .add("url", recipient.getUrl())
093                .build()
094            ).forEach(recipientBuilder::add);
095
096        final String output = Json
097            .createObjectBuilder()
098            .add("url", current.getUrl())
099            .add("peers", peersBuilder.build())
100            .add("keys", recipientBuilder.build())
101            .build()
102            .toString();
103
104        return Response.status(Response.Status.OK).entity(output).build();
105    }
106
107}
