001package com.quorum.tessera.jaxrs.client;
002
003import com.quorum.tessera.config.ServerConfig;
004import com.quorum.tessera.reflect.ReflectCallback;
005import com.quorum.tessera.ssl.context.SSLContextFactory;
006import java.lang.reflect.Method;
007import java.net.URI;
008import java.util.Arrays;
009
010import javax.net.ssl.SSLContext;
011import javax.ws.rs.client.Client;
012import javax.ws.rs.client.ClientBuilder;
013import java.util.Objects;
014import javax.ws.rs.core.Configuration;
015
016/**
017 * Creates HTTP clients that can make requests
018 * <p>
019 * Makes the client SSL enabled if specified
020 */
021public class ClientFactory {
022
023    private final SSLContextFactory sslContextFactory;
024
025    public ClientFactory(final SSLContextFactory sslContextFactory) {
026        this.sslContextFactory = Objects.requireNonNull(sslContextFactory);
027    }
028
029    public ClientFactory() {
030        this(com.quorum.tessera.ssl.context.ClientSSLContextFactory.create());
031    }
032
033    private static Configuration createUnixServerSocketConfig() {
034
035        return ReflectCallback.execute(() -> {
036            Class configType = Class.forName("org.glassfish.jersey.client.ClientConfig");
037            Class providerClass = Class.forName("com.quorum.tessera.jaxrs.unixsocket.JerseyUnixSocketConnectorProvider");
038            Object config = configType.newInstance();
039            Object provider = providerClass.newInstance();
040            Method connectorProviderMethod = Arrays.stream(configType.getDeclaredMethods())
041                    .filter(m -> m.getName().equals("connectorProvider")).findFirst().get();
042                    
043            connectorProviderMethod.invoke(config, provider);
044            
045            return Configuration.class.cast(config);
046        });
047    }
048    
049    /**
050     * Creates a new client, which may or may not be SSL enabled 
051     * or a unix socket enabled depending on
052     * the configuration.
053     *
054     * @param config
055     * @return 
056     * @see Client
057     */
058    public Client buildFrom(final ServerConfig config) {
059  
060        if (config.isUnixSocket()) {
061            Configuration clientConfig = createUnixServerSocketConfig();
062            URI unixfile = config.getServerUri();
063            return ClientBuilder.newClient(clientConfig)
064                    .property("unixfile", unixfile);
065
066        } else if (config.isSsl()) {
067            final SSLContext sslContext = sslContextFactory.from(
068                    config.getServerUri().toString(),
069                    config.getSslConfig());
070            return ClientBuilder.newBuilder()
071                    .sslContext(sslContext)
072                    .build();
073        } else {
074            return ClientBuilder.newClient();
075        }
076    }
077
078
079}
