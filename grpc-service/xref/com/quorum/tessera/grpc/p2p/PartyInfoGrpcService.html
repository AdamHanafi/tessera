17  import static java.util.Objects.requireNonNull;
18  import static java.util.stream.Collectors.toMap;

20  public class PartyInfoGrpcService extends PartyInfoGrpc.PartyInfoImplBase {
21  
22      private final PartyInfoParser partyInfoParser;
23  
24      private final PartyInfoService partyInfoService;
25  
26      public PartyInfoGrpcService(final PartyInfoService partyInfoService, final PartyInfoParser partyInfoParser) {
27          this.partyInfoService = requireNonNull(partyInfoService, "partyInfoService must not be null");
28          this.partyInfoParser = requireNonNull(partyInfoParser, "partyInfoParser must not be null");
29      }
30  
31      @Override
32      public void getPartyInfo(final PartyInfoMessage request, final StreamObserver<PartyInfoMessage> responseObserver) {
33  
34          final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
35  
36          template.handle(() -> {
37              
38              final PartyInfo partyInfo = partyInfoParser.from(request.getPartyInfo().toByteArray());
39  
40              final PartyInfo updatedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
41  
42              return PartyInfoMessage.newBuilder()
43                      .setPartyInfo(ByteString.copyFrom(partyInfoParser.to(updatedPartyInfo)))
44                      .build();
45  
46          });
47      }
48  
49      @Override
50      public void getPartyInfoMessage(final Empty request, final StreamObserver<PartyInfoJson> responseObserver) {
51  
52          final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
53  
54          template.handle(() -> {
55  
56              final PartyInfo partyInfo = partyInfoService.getPartyInfo();
57              final Set<Peer> peers = partyInfo.getParties()
58                  .stream()
59                  .filter(p -> p.getUrl().endsWith("/"))
60                  .map(party -> {
61                      final Peer.Builder builder = Peer.newBuilder().setUrl(party.getUrl());
62  
63                      if (party.getLastContacted() != null) {
64                          final Timestamp timestamp = Timestamp.newBuilder()
65                              .setSeconds(party.getLastContacted().getEpochSecond())
66                              .build();
67  
68                          builder.setUtcTimestamp(timestamp);
69                      }
70  
71                      return builder.build();
72                  }).collect(Collectors.toSet());
73  
74              final Map<String, String> keys = partyInfo.getRecipients()
75                  .stream()
76                  .collect(toMap(r -> r.getKey().encodeToBase64(), Recipient::getUrl));
77  
78              return PartyInfoJson.newBuilder()
79                  .setUrl(partyInfo.getUrl())
80                  .addAllPeers(peers)
81                  .putAllKeys(keys)
82                  .build();
83  
84          });
85      }
