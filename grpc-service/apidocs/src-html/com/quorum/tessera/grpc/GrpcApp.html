001package com.quorum.tessera.grpc;
002
003import com.quorum.tessera.service.locator.ServiceLocator;
004import io.grpc.BindableService;
005
006import java.util.Objects;
007import java.util.Set;
008import java.util.stream.Collectors;
009
010public class GrpcApp {
011    private final ServiceLocator serviceLocator;
012
013    private final String contextName;
014
015    public GrpcApp(final ServiceLocator serviceLocator, final String contextName) {
016        this.serviceLocator = Objects.requireNonNull(serviceLocator);
017        this.contextName = Objects.requireNonNull(contextName);
018    }
019
020    public Set<BindableService> getBindableServices() {
021        final String apiPackageName = getClass().getPackage().getName();
022        return serviceLocator.getServices(contextName).stream()
023            .filter(Objects::nonNull)
024            .filter(BindableService.class::isInstance)
025            .filter(o -> Objects.nonNull(o.getClass()))
026            .filter(o -> Objects.nonNull(o.getClass().getPackage()))
027            .filter(o -> o.getClass().getPackage().getName().startsWith(apiPackageName))
028            .map(o -> (BindableService) o)
029            .collect(Collectors.toSet());
030    }
031}
