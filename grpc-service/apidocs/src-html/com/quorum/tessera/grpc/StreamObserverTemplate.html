001package com.quorum.tessera.grpc;
002
003import com.quorum.tessera.node.AutoDiscoveryDisabledException;
004import io.grpc.stub.StreamObserver;
005import java.util.Objects;
006import javax.validation.ConstraintViolationException;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010public class StreamObserverTemplate {
011
012    private static final Logger LOGGER = LoggerFactory.getLogger(StreamObserverTemplate.class);
013
014    private final StreamObserver observer;
015
016    public StreamObserverTemplate(StreamObserver observer) {
017        this.observer = Objects.requireNonNull(observer);
018    }
019
020    public void handle(StreamObserverCallback callback) {
021
022        try {
023            Object r = callback.execute();
024            observer.onNext(r);
025            observer.onCompleted();
026
027        } catch(AutoDiscoveryDisabledException ex) {
028            observer.onError(io.grpc.Status.PERMISSION_DENIED
029                    .withDescription(ex.getMessage()).asRuntimeException());
030        } catch (ConstraintViolationException validationError) {
031            observer.onError(io.grpc.Status.INVALID_ARGUMENT
032                    .withCause(validationError)
033                    .asRuntimeException());
034        } catch (Throwable ex) {
035            LOGGER.error(null, ex);
036            observer.onError(ex);
037        }
038    }
039
040}
