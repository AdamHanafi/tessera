001package com.quorum.tessera.grpc.p2p;
002
003import com.google.protobuf.ByteString;
004import com.google.protobuf.Empty;
005import com.google.protobuf.Timestamp;
006import com.quorum.tessera.grpc.StreamObserverTemplate;
007import com.quorum.tessera.node.PartyInfoParser;
008import com.quorum.tessera.node.PartyInfoService;
009import com.quorum.tessera.node.model.PartyInfo;
010import com.quorum.tessera.node.model.Recipient;
011import io.grpc.stub.StreamObserver;
012
013import java.util.Map;
014import java.util.Set;
015import java.util.stream.Collectors;
016
017import static java.util.Objects.requireNonNull;
018import static java.util.stream.Collectors.toMap;
019
020public class PartyInfoGrpcService extends PartyInfoGrpc.PartyInfoImplBase {
021
022    private final PartyInfoParser partyInfoParser;
023
024    private final PartyInfoService partyInfoService;
025
026    public PartyInfoGrpcService(final PartyInfoService partyInfoService, final PartyInfoParser partyInfoParser) {
027        this.partyInfoService = requireNonNull(partyInfoService, "partyInfoService must not be null");
028        this.partyInfoParser = requireNonNull(partyInfoParser, "partyInfoParser must not be null");
029    }
030
031    @Override
032    public void getPartyInfo(final PartyInfoMessage request, final StreamObserver<PartyInfoMessage> responseObserver) {
033
034        final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
035
036        template.handle(() -> {
037            
038            final PartyInfo partyInfo = partyInfoParser.from(request.getPartyInfo().toByteArray());
039
040            final PartyInfo updatedPartyInfo = partyInfoService.updatePartyInfo(partyInfo);
041
042            return PartyInfoMessage.newBuilder()
043                    .setPartyInfo(ByteString.copyFrom(partyInfoParser.to(updatedPartyInfo)))
044                    .build();
045
046        });
047    }
048
049    @Override
050    public void getPartyInfoMessage(final Empty request, final StreamObserver<PartyInfoJson> responseObserver) {
051
052        final StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
053
054        template.handle(() -> {
055
056            final PartyInfo partyInfo = partyInfoService.getPartyInfo();
057            final Set<Peer> peers = partyInfo.getParties()
058                .stream()
059                .filter(p -> p.getUrl().endsWith("/"))
060                .map(party -> {
061                    final Peer.Builder builder = Peer.newBuilder().setUrl(party.getUrl());
062
063                    if (party.getLastContacted() != null) {
064                        final Timestamp timestamp = Timestamp.newBuilder()
065                            .setSeconds(party.getLastContacted().getEpochSecond())
066                            .build();
067
068                        builder.setUtcTimestamp(timestamp);
069                    }
070
071                    return builder.build();
072                }).collect(Collectors.toSet());
073
074            final Map<String, String> keys = partyInfo.getRecipients()
075                .stream()
076                .collect(toMap(r -> r.getKey().encodeToBase64(), Recipient::getUrl));
077
078            return PartyInfoJson.newBuilder()
079                .setUrl(partyInfo.getUrl())
080                .addAllPeers(peers)
081                .putAllKeys(keys)
082                .build();
083
084        });
085    }
086
087}
