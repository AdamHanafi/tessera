001package com.quorum.tessera.grpc.p2p;
002
003import com.google.protobuf.Empty;
004import com.quorum.tessera.grpc.p2p.UpCheckMessage;
005import com.quorum.tessera.grpc.p2p.VersionMessage;
006import com.quorum.tessera.grpc.StreamObserverTemplate;
007import io.grpc.stub.StreamObserver;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011public class TesseraGrpcService extends TesseraGrpc.TesseraImplBase {
012
013    private static final Logger LOGGER = LoggerFactory.getLogger(TesseraGrpcService.class);
014
015    private static final String UPCHECK_RESPONSE = "I'm up!";
016
017    private static final String VERSION = "0.6";
018
019    @Override
020    public void getVersion(Empty request, StreamObserver<VersionMessage> responseObserver) {
021        LOGGER.info("GET version");
022
023        StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
024
025        template.handle(() -> {
026            return VersionMessage.newBuilder()
027                    .setVersion(VERSION)
028                    .build();
029        });
030
031    }
032
033    @Override
034    public void getUpCheck(Empty request, StreamObserver<UpCheckMessage> responseObserver) {
035        LOGGER.info("GET upcheck");
036        StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
037
038        template.handle(() -> {
039            return UpCheckMessage.newBuilder()
040                    .setUpCheck(UPCHECK_RESPONSE)
041                    .build();
042        });
043
044    }
045
046}
