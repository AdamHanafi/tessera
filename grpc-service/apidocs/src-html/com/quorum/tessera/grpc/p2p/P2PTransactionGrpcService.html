001package com.quorum.tessera.grpc.p2p;
002
003import com.google.protobuf.ByteString;
004import com.quorum.tessera.grpc.StreamObserverTemplate;
005import com.quorum.tessera.transaction.TransactionManager;
006
007import io.grpc.stub.StreamObserver;
008import java.util.Objects;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012import java.util.Optional;
013import java.util.Set;
014import javax.validation.ConstraintViolation;
015import javax.validation.ConstraintViolationException;
016import javax.validation.Validation;
017import javax.validation.Validator;
018
019public class P2PTransactionGrpcService extends P2PTransactionGrpc.P2PTransactionImplBase{
020
021    private static final Logger LOGGER = LoggerFactory.getLogger(P2PTransactionGrpcService.class);
022
023    private final Validator validator = Validation.byDefaultProvider()
024            .configure().ignoreXmlConfiguration().buildValidatorFactory().getValidator();
025
026    private final TransactionManager transactionManager;
027
028    public P2PTransactionGrpcService(TransactionManager transactionManager) {
029        this.transactionManager = Objects.requireNonNull(transactionManager);
030    }
031
032    @Override
033    public void delete(DeleteRequest grpcRequest, StreamObserver<DeleteRequest> responseObserver) {
034        LOGGER.debug("Received delete key request");
035
036        StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
037
038        template.handle(() -> {
039
040            com.quorum.tessera.api.model.DeleteRequest request = Convertor.toModel(grpcRequest);
041
042            Set<ConstraintViolation<com.quorum.tessera.api.model.DeleteRequest>> violations = validator.validate(request);
043            if (!violations.isEmpty()) {
044                throw new ConstraintViolationException(violations);
045            }
046
047            transactionManager.delete(request);
048            return grpcRequest;
049        });
050
051    }
052
053    @Override
054    public void resend(ResendRequest grpcRequest, StreamObserver<ResendResponse> responseObserver) {
055        LOGGER.debug("Received resend request");
056
057        StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
058
059        template.handle(() -> {
060            com.quorum.tessera.api.model.ResendRequest request = Convertor.toModel(grpcRequest);
061
062            Optional<byte[]> result = transactionManager.resend(request).getPayload();
063
064            ResendResponse.Builder builder = ResendResponse.newBuilder();
065            result.map(ByteString::copyFrom).ifPresent(builder::setData);
066            return builder.build();
067        });
068
069    }
070
071    @Override
072    public void push(PushRequest request, StreamObserver<PushRequest> responseObserver) {
073        LOGGER.debug("Received push request");
074
075        StreamObserverTemplate template = new StreamObserverTemplate(responseObserver);
076
077        template.handle(() -> {
078            transactionManager.storePayload(request.getData().toByteArray());
079            return request;
080        });
081
082    }
083
084}
