11      public void constructUnSupported() throws Exception {
12          Constructor c = Convertor.class.getDeclaredConstructor();
13          c.setAccessible(true);
14          try {
15              c.newInstance();
16              failBecauseExceptionWasNotThrown(InvocationTargetException.class);
17          } catch (InvocationTargetException invocationTargetException) {
18              assertThat(invocationTargetException).hasCauseExactlyInstanceOf(UnsupportedOperationException.class);
19          }
20      }
