3   import com.quorum.tessera.encryption.PublicKey;
4   import org.junit.Test;
5   
6   import java.util.Arrays;
7   
8   import static org.assertj.core.api.Assertions.assertThat;
9   
10  public class EncodedPayloadBuilderTest {
11  
12      @Test
13      public void build() {
14  
15          final PublicKey senderKey = PublicKey.from("SENDER_KEY".getBytes());
16  
17          final PublicKey recipientKey = PublicKey.from("RECIPIENT_KEY".getBytes());
18  
19          final byte[] cipherText = "cipherText".getBytes();
20          final byte[] cipherTextNonce = "cipherTextNonce".getBytes();
21  
22          final byte[] recipientNonce = "recipientNonce".getBytes();
23          final byte[] recipientBox = "recipientBox".getBytes();
24  
25          final EncodedPayload sample = EncodedPayloadBuilder.create()
26              .withSenderKey(senderKey)
27              .withCipherText(cipherText)
28              .withCipherTextNonce(cipherTextNonce)
29              .withRecipientBoxes(Arrays.asList(recipientBox))
30              .withRecipientNonce(recipientNonce)
31              .withRecipientKeys(recipientKey)
32              .build();
33  
34          assertThat(sample.getSenderKey()).isEqualTo(senderKey);
35          assertThat(sample.getCipherText()).isEqualTo("cipherText".getBytes());
36          assertThat(sample.getCipherTextNonce().getNonceBytes()).isEqualTo(cipherTextNonce);
37          assertThat(sample.getRecipientNonce().getNonceBytes()).isEqualTo(recipientNonce);
38          assertThat(sample.getRecipientBoxes()).hasSize(1).containsExactlyInAnyOrder(recipientBox);
39          assertThat(sample.getRecipientKeys()).hasSize(1).containsExactlyInAnyOrder(recipientKey);
40      }
41  
42  }
