1   package com.quorum.tessera.enclave;
2   
3   import com.quorum.tessera.encryption.PublicKey;
4   import com.quorum.tessera.service.Service;
5   
6   import java.util.List;
7   import java.util.Set;
8   
9   /**
10   * An {@link Enclave} provides encryption/decryption functions and keeps hold
11   * of all the nodes private keys so the do not leak into other services.
12   */
13  public interface Enclave extends Service {
14  
15      /**
16       * Retrieves the public key to use if no key is specified for an operation
17       * There is no guarantee this key remains the same between runs of the Enclave.
18       *
19       * @return the public key that has been assigned as the default
20       */
21      PublicKey defaultPublicKey();
22  
23      /**
24       * Returns a set of public keys that should be included as recipients to
25       * all transactions produced by this node. The keys are not be managed by
26       * this node.
27       *
28       * @return the set of public keys to be added to transactions
29       */
30      Set<PublicKey> getForwardingKeys();
31  
32      /**
33       * Returns all the public keys that are managed by this Enclave.
34       *
35       * @return all public keys managed by this Enclave
36       */
37      Set<PublicKey> getPublicKeys();
38  
39      /**
40       * Encrypts a message using the specified sender and a list of recipients.
41       * Returns a {@link EncodedPayload} which contains all the encrypted
42       * information, including the recipients and their encrypted master keys.
43       *
44       * @param message             the message to be encrypted
45       * @param senderPublicKey     the public key which this enclave manages
46       * @param recipientPublicKeys the recipients to encrypt this message for
47       * @return the encrypted information, represented by an {@link EncodedPayload}
48       */
49      EncodedPayload encryptPayload(byte[] message, PublicKey senderPublicKey, List<PublicKey> recipientPublicKeys);
50  
51      /**
52       * Decrypts a {@link RawTransaction} so that it can be re-encrypted into a
53       * {@link EncodedPayload} with the given recipient list
54       *
55       * @param rawTransaction      the transactiopn to decrypt and re-encrypt with recipients
56       * @param recipientPublicKeys the recipients to encrypt the transaction for
57       * @return the encrypted information, represented by an {@link EncodedPayload}
58       */
59      EncodedPayload encryptPayload(RawTransaction rawTransaction, List<PublicKey> recipientPublicKeys);
60  
61      /**
62       * Encrypt a payload without any recipients that can be retrieved later.
63       * The payload is encrypted using the private key that is related to the
64       * given public key.
65       *
66       * @param message the message to be encrypted
67       * @param sender  the sender's public key to encrypt the transaction with
68       * @return the encrypted transaction
69       */
70      RawTransaction encryptRawPayload(byte[] message, PublicKey sender);
71  
72      /**
73       * Decrypt a transaction and fetch the original message using the given
74       * public key. Throws an {@link com.quorum.tessera.nacl.NaclException} if
75       * the provided public key OR one of the Enclave's managed keys cannot be
76       * used to decrypt the payload
77       *
78       * @param payload     the encrypted payload
79       * @param providedKey the key to use for decryption, if the payload wasn't sent by this Enclave
80       * @return the original, decrypted message
81       */
82      byte[] unencryptTransaction(EncodedPayload payload, PublicKey providedKey);
83  
84      /**
85       * Creates a new recipient box for the payload, for which we must be the originator.
86       * At least one recipient must already be available to be able to decrypt the master key.
87       *
88       * @param payload      the payload to add a recipient to
89       * @param recipientKey the new recipient key to add
90       */
91      byte[] createNewRecipientBox(EncodedPayload payload, PublicKey recipientKey);
92  
93      @Override
94      default void start() {
95      }
96  
97      @Override
98      default void stop() {
99      }
100 
101 }
