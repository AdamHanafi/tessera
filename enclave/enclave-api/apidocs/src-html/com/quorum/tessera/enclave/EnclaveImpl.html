001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.encryption.*;
004import com.quorum.tessera.nacl.NaclFacade;
005import com.quorum.tessera.nacl.Nonce;
006
007import java.util.List;
008import java.util.Objects;
009import java.util.Set;
010import java.util.stream.Collectors;
011
012import static java.util.Collections.singletonList;
013
014public class EnclaveImpl implements Enclave {
015
016    private final NaclFacade nacl;
017
018    private final KeyManager keyManager;
019
020    public EnclaveImpl(NaclFacade nacl, KeyManager keyManager) {
021        this.nacl = Objects.requireNonNull(nacl);
022        this.keyManager = Objects.requireNonNull(keyManager);
023    }
024
025    @Override
026    public EncodedPayload encryptPayload(final byte[] message,
027                                         final PublicKey senderPublicKey,
028                                         final List<PublicKey> recipientPublicKeys) {
029        final MasterKey masterKey = nacl.createMasterKey();
030        final Nonce nonce = nacl.randomNonce();
031        final Nonce recipientNonce = nacl.randomNonce();
032
033        final byte[] cipherText = nacl.sealAfterPrecomputation(message, nonce, masterKey);
034
035        final List<byte[]> encryptedMasterKeys = buildRecipientMasterKeys(senderPublicKey, recipientPublicKeys, recipientNonce, masterKey);
036
037        return new EncodedPayload(
038            senderPublicKey, cipherText, nonce, encryptedMasterKeys, recipientNonce, recipientPublicKeys
039        );
040    }
041
042    @Override
043    public byte[] createNewRecipientBox(final EncodedPayload payload, final PublicKey publicKey) {
044
045        if (payload.getRecipientKeys().isEmpty() || payload.getRecipientBoxes().isEmpty()) {
046            throw new RuntimeException("No key or recipient-box to use");
047        }
048
049        final MasterKey master = this.getMasterKey(
050            payload.getRecipientKeys().get(0), payload.getSenderKey(),
051            payload.getRecipientNonce(), payload.getRecipientBoxes().get(0)
052        );
053
054        final List<byte[]> sealedMasterKeyList = this.buildRecipientMasterKeys(
055            payload.getSenderKey(), singletonList(publicKey), payload.getRecipientNonce(), master
056        );
057
058        return sealedMasterKeyList.get(0);
059    }
060
061    @Override
062    public EncodedPayload encryptPayload(final RawTransaction rawTransaction,
063                                         final List<PublicKey> recipientPublicKeys) {
064        final MasterKey masterKey = this.getMasterKey(
065            rawTransaction.getFrom(), rawTransaction.getFrom(),
066            rawTransaction.getNonce(), rawTransaction.getEncryptedKey()
067        );
068
069        final Nonce recipientNonce = nacl.randomNonce();
070        final List<byte[]> encryptedMasterKeys
071            = buildRecipientMasterKeys(rawTransaction.getFrom(), recipientPublicKeys, recipientNonce, masterKey);
072
073        return new EncodedPayload(
074            rawTransaction.getFrom(), rawTransaction.getEncryptedPayload(),
075            rawTransaction.getNonce(), encryptedMasterKeys, recipientNonce, recipientPublicKeys
076        );
077    }
078
079    private List<byte[]> buildRecipientMasterKeys(final PublicKey senderPublicKey,
080                                                  final List<PublicKey> recipientPublicKeys,
081                                                  final Nonce recipientNonce,
082                                                  final MasterKey masterKey){
083        final PrivateKey privateKey = keyManager.getPrivateKeyForPublicKey(senderPublicKey);
084
085        return recipientPublicKeys
086            .stream()
087            .map(publicKey -> nacl.computeSharedKey(publicKey, privateKey))
088            .map(sharedKey -> nacl.sealAfterPrecomputation(masterKey.getKeyBytes(), recipientNonce, sharedKey))
089            .collect(Collectors.toList());
090    }
091
092    @Override
093    public RawTransaction encryptRawPayload(byte[] message, PublicKey sender) {
094        final MasterKey masterKey = nacl.createMasterKey();
095        final Nonce nonce = nacl.randomNonce();
096
097        final byte[] cipherText = nacl.sealAfterPrecomputation(message, nonce, masterKey);
098
099        final PrivateKey privateKey = keyManager.getPrivateKeyForPublicKey(sender);
100
101        // TODO NL - check if it makes sense to compute a shared key from the public and private parts of the same key
102        SharedKey sharedKey = nacl.computeSharedKey(sender, privateKey);
103        final byte[] encryptedMasterKey = nacl.sealAfterPrecomputation(masterKey.getKeyBytes(), nonce, sharedKey);
104
105        return new RawTransaction(cipherText, encryptedMasterKey, nonce, sender);
106    }
107
108    @Override
109    public byte[] unencryptTransaction(EncodedPayload payload, final PublicKey providedSenderKey) {
110
111        final PublicKey senderPubKey;
112
113        final PublicKey recipientPubKey;
114
115        if (!this.getPublicKeys().contains(payload.getSenderKey())) {
116            // This is a payload originally sent to us by another node
117            senderPubKey = providedSenderKey;
118            recipientPubKey = payload.getSenderKey();
119        } else {
120            // This is a payload that originated from us
121            senderPubKey = payload.getSenderKey();
122            recipientPubKey = payload.getRecipientKeys().get(0);
123        }
124
125        final PrivateKey senderPrivKey = keyManager.getPrivateKeyForPublicKey(senderPubKey);
126
127        final SharedKey sharedKey = nacl.computeSharedKey(recipientPubKey, senderPrivKey);
128
129        final byte[] recipientBox = payload.getRecipientBoxes().iterator().next();
130
131        final Nonce recipientNonce = payload.getRecipientNonce();
132
133        final byte[] masterKeyBytes = nacl.openAfterPrecomputation(recipientBox, recipientNonce, sharedKey);
134
135        final MasterKey masterKey = MasterKey.from(masterKeyBytes);
136
137        final byte[] cipherText = payload.getCipherText();
138        final Nonce cipherTextNonce = payload.getCipherTextNonce();
139
140        return nacl.openAfterPrecomputation(cipherText, cipherTextNonce, masterKey);
141
142    }
143
144    private MasterKey getMasterKey(PublicKey recipient, PublicKey sender, Nonce nonce, byte[] encryptedKey) {
145
146        final SharedKey sharedKey = nacl.computeSharedKey(recipient, keyManager.getPrivateKeyForPublicKey(sender));
147
148        final byte[] masterKeyBytes = nacl.openAfterPrecomputation(encryptedKey, nonce, sharedKey);
149
150        return MasterKey.from(masterKeyBytes);
151    }
152
153    @Override
154    public PublicKey defaultPublicKey() {
155        return keyManager.defaultPublicKey();
156    }
157
158    @Override
159    public Set<PublicKey> getForwardingKeys() {
160        return keyManager.getForwardingKeys();
161    }
162
163    @Override
164    public Set<PublicKey> getPublicKeys() {
165        return keyManager.getPublicKeys();
166    }
167
168    @Override
169    public Status status() {
170        return Status.STARTED;
171    }
172}
