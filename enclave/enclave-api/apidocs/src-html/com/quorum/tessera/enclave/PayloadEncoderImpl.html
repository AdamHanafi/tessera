001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.encryption.PublicKey;
004import com.quorum.tessera.nacl.Nonce;
005
006import java.nio.ByteBuffer;
007import java.util.ArrayList;
008import java.util.List;
009
010import static java.util.Collections.emptyList;
011import static java.util.Collections.singletonList;
012import static java.util.stream.Collectors.toList;
013
014public class PayloadEncoderImpl implements PayloadEncoder, BinaryEncoder {
015
016    @Override
017    public byte[] encode(final EncodedPayload payload) {
018
019        final byte[] senderKey = encodeField(payload.getSenderKey().getKeyBytes());
020        final byte[] cipherText = encodeField(payload.getCipherText());
021        final byte[] nonce = encodeField(payload.getCipherTextNonce().getNonceBytes());
022        final byte[] recipientNonce = encodeField(payload.getRecipientNonce().getNonceBytes());
023        final byte[] recipients = encodeArray(payload.getRecipientBoxes());
024        final byte[] recipientBytes
025            = encodeArray(payload.getRecipientKeys().stream().map(PublicKey::getKeyBytes).collect(toList()));
026
027        return ByteBuffer
028            .allocate(senderKey.length + cipherText.length + nonce.length + recipients.length + recipientNonce.length + recipientBytes.length)
029            .put(senderKey)
030            .put(cipherText)
031            .put(nonce)
032            .put(recipients)
033            .put(recipientNonce)
034            .put(recipientBytes)
035            .array();
036    }
037
038    @Override
039    public EncodedPayload decode(final byte[] input) {
040        final ByteBuffer buffer = ByteBuffer.wrap(input);
041
042        final long senderSize = buffer.getLong();
043        final byte[] senderKey = new byte[Math.toIntExact(senderSize)];
044        buffer.get(senderKey);
045
046        final long cipherTextSize = buffer.getLong();
047        final byte[] cipherText = new byte[Math.toIntExact(cipherTextSize)];
048        buffer.get(cipherText);
049
050        final long nonceSize = buffer.getLong();
051        final byte[] nonce = new byte[Math.toIntExact(nonceSize)];
052        buffer.get(nonce);
053
054        final long numberOfRecipients = buffer.getLong();
055        final List<byte[]> recipientBoxes = new ArrayList<>();
056        for (long i = 0; i < numberOfRecipients; i++) {
057            final long boxSize = buffer.getLong();
058            final byte[] box = new byte[Math.toIntExact(boxSize)];
059            buffer.get(box);
060            recipientBoxes.add(box);
061        }
062
063        final long recipientNonceSize = buffer.getLong();
064        final byte[] recipientNonce = new byte[Math.toIntExact(recipientNonceSize)];
065        buffer.get(recipientNonce);
066
067        //this means there are no recipients in the payload (which we receive when we are a participant)
068        if (!buffer.hasRemaining()) {
069            return new EncodedPayload(
070                PublicKey.from(senderKey), cipherText, new Nonce(nonce),
071                recipientBoxes, new Nonce(recipientNonce), emptyList()
072            );
073        }
074
075        final long recipientLength = buffer.getLong();
076
077        final List<byte[]> recipientKeys = new ArrayList<>();
078        for (long i = 0; i < recipientLength; i++) {
079            final long boxSize = buffer.getLong();
080            final byte[] box = new byte[Math.toIntExact(boxSize)];
081            buffer.get(box);
082            recipientKeys.add(box);
083        }
084
085        return new EncodedPayload(
086            PublicKey.from(senderKey), cipherText, new Nonce(nonce), recipientBoxes, new Nonce(recipientNonce),
087            recipientKeys.stream().map(PublicKey::from).collect(toList())
088        );
089    }
090
091    @Override
092    public EncodedPayload forRecipient(final EncodedPayload payload, final PublicKey recipient) {
093
094        if (!payload.getRecipientKeys().contains(recipient)) {
095            throw new InvalidRecipientException("Recipient " + recipient.encodeToBase64() + " is not a recipient of transaction ");
096        }
097
098        final int recipientIndex = payload.getRecipientKeys().indexOf(recipient);
099        final byte[] recipientBox = payload.getRecipientBoxes().get(recipientIndex);
100
101        return new EncodedPayload(
102            payload.getSenderKey(), payload.getCipherText(), payload.getCipherTextNonce(),
103            singletonList(recipientBox), payload.getRecipientNonce(), emptyList()
104        );
105    }
106
107}
