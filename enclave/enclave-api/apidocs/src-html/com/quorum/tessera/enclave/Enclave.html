001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.encryption.PublicKey;
004import com.quorum.tessera.service.Service;
005
006import java.util.List;
007import java.util.Set;
008
009/**
010 * An {@link Enclave} provides encryption/decryption functions and keeps hold
011 * of all the nodes private keys so the do not leak into other services.
012 */
013public interface Enclave extends Service {
014
015    /**
016     * Retrieves the public key to use if no key is specified for an operation
017     * There is no guarantee this key remains the same between runs of the Enclave.
018     *
019     * @return the public key that has been assigned as the default
020     */
021    PublicKey defaultPublicKey();
022
023    /**
024     * Returns a set of public keys that should be included as recipients to
025     * all transactions produced by this node. The keys are not be managed by
026     * this node.
027     *
028     * @return the set of public keys to be added to transactions
029     */
030    Set<PublicKey> getForwardingKeys();
031
032    /**
033     * Returns all the public keys that are managed by this Enclave.
034     *
035     * @return all public keys managed by this Enclave
036     */
037    Set<PublicKey> getPublicKeys();
038
039    /**
040     * Encrypts a message using the specified sender and a list of recipients.
041     * Returns a {@link EncodedPayload} which contains all the encrypted
042     * information, including the recipients and their encrypted master keys.
043     *
044     * @param message             the message to be encrypted
045     * @param senderPublicKey     the public key which this enclave manages
046     * @param recipientPublicKeys the recipients to encrypt this message for
047     * @return the encrypted information, represented by an {@link EncodedPayload}
048     */
049    EncodedPayload encryptPayload(byte[] message, PublicKey senderPublicKey, List<PublicKey> recipientPublicKeys);
050
051    /**
052     * Decrypts a {@link RawTransaction} so that it can be re-encrypted into a
053     * {@link EncodedPayload} with the given recipient list
054     *
055     * @param rawTransaction      the transactiopn to decrypt and re-encrypt with recipients
056     * @param recipientPublicKeys the recipients to encrypt the transaction for
057     * @return the encrypted information, represented by an {@link EncodedPayload}
058     */
059    EncodedPayload encryptPayload(RawTransaction rawTransaction, List<PublicKey> recipientPublicKeys);
060
061    /**
062     * Encrypt a payload without any recipients that can be retrieved later.
063     * The payload is encrypted using the private key that is related to the
064     * given public key.
065     *
066     * @param message the message to be encrypted
067     * @param sender  the sender's public key to encrypt the transaction with
068     * @return the encrypted transaction
069     */
070    RawTransaction encryptRawPayload(byte[] message, PublicKey sender);
071
072    /**
073     * Decrypt a transaction and fetch the original message using the given
074     * public key. Throws an {@link com.quorum.tessera.nacl.NaclException} if
075     * the provided public key OR one of the Enclave's managed keys cannot be
076     * used to decrypt the payload
077     *
078     * @param payload     the encrypted payload
079     * @param providedKey the key to use for decryption, if the payload wasn't sent by this Enclave
080     * @return the original, decrypted message
081     */
082    byte[] unencryptTransaction(EncodedPayload payload, PublicKey providedKey);
083
084    /**
085     * Creates a new recipient box for the payload, for which we must be the originator.
086     * At least one recipient must already be available to be able to decrypt the master key.
087     *
088     * @param payload      the payload to add a recipient to
089     * @param recipientKey the new recipient key to add
090     */
091    byte[] createNewRecipientBox(EncodedPayload payload, PublicKey recipientKey);
092
093    @Override
094    default void start() {
095    }
096
097    @Override
098    default void stop() {
099    }
100
101}
