001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.ServiceLoaderUtil;
004import com.quorum.tessera.config.AppType;
005import com.quorum.tessera.config.Config;
006import com.quorum.tessera.config.ServerConfig;
007import com.quorum.tessera.config.util.EnvironmentVariableProvider;
008import com.quorum.tessera.encryption.KeyManagerImpl;
009import com.quorum.tessera.encryption.KeyPair;
010import com.quorum.tessera.encryption.PublicKey;
011import com.quorum.tessera.keypairconverter.KeyPairConverter;
012import com.quorum.tessera.nacl.NaclFacadeFactory;
013
014import java.util.Collection;
015import java.util.Optional;
016
017/**
018 * Creates {@link Enclave} instances, which may point to remote services or
019 * local, in-app instances.
020 */
021public interface EnclaveFactory {
022
023    default Enclave createLocal(Config config) {
024        return createServer(config);
025    }
026
027    static Enclave createServer(Config config) {
028
029        final KeyPairConverter keyPairConverter = new KeyPairConverter(config, new EnvironmentVariableProvider());
030        final Collection<KeyPair> keys = keyPairConverter.convert(config.getKeys().getKeyData());
031
032        final Collection<PublicKey> forwardKeys = keyPairConverter.convert(config.getAlwaysSendTo());
033
034        return new EnclaveImpl(NaclFacadeFactory.newFactory().create(), new KeyManagerImpl(keys, forwardKeys));
035    }
036
037    /**
038     * Determines from the provided configuration whether to construct a client
039     * to a remote service, or to create a local instance.
040     * <p>
041     * If a remote instance is requested, it is constructed from a
042     * {@link EnclaveClientFactory}.
043     *
044     * @param config the global configuration to use to create a remote enclave connection
045     * @return the {@link Enclave}, which may be either local or remote
046     */
047    default Enclave create(Config config) {
048        final Optional<ServerConfig> enclaveServerConfig = config
049            .getServerConfigs()
050            .stream()
051            .filter(sc -> sc.getApp() == AppType.ENCLAVE)
052            .findAny();
053
054        if (enclaveServerConfig.isPresent()) {
055            return EnclaveClientFactory.create().create(config);
056        }
057
058        return createServer(config);
059
060    }
061
062    static EnclaveFactory create() {
063        return ServiceLoaderUtil.load(EnclaveFactory.class).orElse(new EnclaveFactory() {
064        });
065    }
066
067}
