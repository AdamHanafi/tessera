001package com.quorum.tessera.enclave;
002
003import java.nio.ByteBuffer;
004import java.util.List;
005
006/**
007 * Encodes bytes to the binary form expected by other nodes on the network
008 */
009public interface BinaryEncoder {
010
011    /**
012     * Converts a byte array to its binary form, which takes the following:
013     * 8 bytes to describe the number of bytes long the data is
014     * the bytes of the data
015     *
016     * @param data the input array to encode
017     * @return the encoded byte array
018     */
019    default byte[] encodeField(final byte[] data) {
020        return ByteBuffer
021            .allocate(Long.BYTES + data.length)
022            .putLong(data.length)
023            .put(data)
024            .array();
025    }
026
027    /**
028     * Converts a list of byte arrays to its binary form, which takes the following:
029     *
030     * 8 bytes to describe the number of elements
031     * and for each element:
032     * - 8 bytes to describe the number of bytes long the element is
033     * - the bytes of the element
034     *
035     * @param data the input array to encode
036     * @return the encoded byte array
037     */
038    default byte[] encodeArray(final List<byte[]> data) {
039        final byte[] elements = data
040            .stream()
041            .map(this::encodeField)
042            .reduce(new byte[0], (a, b) -> {
043                final byte[] concat = new byte[a.length + b.length];
044                System.arraycopy(a, 0, concat, 0, a.length);
045                System.arraycopy(b, 0, concat, a.length, b.length);
046                return concat;
047            });
048
049        return ByteBuffer
050            .allocate(Long.BYTES + elements.length)
051            .putLong(data.size())
052            .put(elements)
053            .array();
054    }
055
056}
