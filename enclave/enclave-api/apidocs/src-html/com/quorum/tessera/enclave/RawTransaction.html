001package com.quorum.tessera.enclave;
002
003import com.quorum.tessera.encryption.PublicKey;
004import com.quorum.tessera.nacl.Nonce;
005
006import java.util.Arrays;
007import java.util.Objects;
008
009public class RawTransaction {
010
011    private final byte[] encryptedPayload;
012
013    private final byte[] encryptedKey;
014
015    private final Nonce nonce;
016
017    private final PublicKey from;
018
019    public RawTransaction(final byte[] encryptedPayload, final byte[] encryptedKey, final Nonce nonce, final PublicKey from) {
020        this.encryptedPayload = Objects.requireNonNull(encryptedPayload);
021        this.encryptedKey = Objects.requireNonNull(encryptedKey);
022        this.nonce = Objects.requireNonNull(nonce);
023        this.from = Objects.requireNonNull(from);
024    }
025
026    public byte[] getEncryptedPayload() {
027        return encryptedPayload;
028    }
029
030    public byte[] getEncryptedKey() {
031        return encryptedKey;
032    }
033
034    public Nonce getNonce() {
035        return nonce;
036    }
037
038    public PublicKey getFrom() {
039        return from;
040    }
041
042    @Override
043    public int hashCode() {
044        int hash = 5;
045        hash = 61 * hash + Arrays.hashCode(this.encryptedPayload);
046        hash = 61 * hash + Arrays.hashCode(this.encryptedKey);
047        hash = 61 * hash + Objects.hashCode(this.nonce);
048        hash = 61 * hash + Objects.hashCode(this.from);
049        return hash;
050    }
051
052    @Override
053    public boolean equals(final Object obj) {
054        if (!(obj instanceof RawTransaction)) {
055            return false;
056        }
057
058        final RawTransaction other = (RawTransaction) obj;
059
060        return Objects.equals(this.nonce, other.nonce) &&
061            Objects.equals(this.from, other.from) &&
062            Arrays.equals(this.encryptedKey, other.encryptedKey) &&
063            Arrays.equals(this.encryptedPayload, other.encryptedPayload);
064    }
065
066}
