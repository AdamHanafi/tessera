25  @Configuration

31      @BeforeClass

35      }

37      @Inject
38      private Enclave enclave;
39  
40      private JerseyTest jersey;
41  
42      private RestfulEnclaveClient enclaveClient;
43  
44      @Before
45      public void setUp() throws Exception {
46          assertThat(enclave).isInstanceOf(EnclaveImpl.class);
47          jersey = Util.create(enclave);
48          jersey.setUp();
49  
50          enclaveClient = new RestfulEnclaveClient(jersey.client(), jersey.target().getUri());
51      }
52  
53      @After
54      public void tearDown() throws Exception {
55          jersey.tearDown();
56      }
57  
58      @Test
59      public void defaultPublicKey() {
60          PublicKey result = enclaveClient.defaultPublicKey();
61  
62          assertThat(result).isNotNull();
63          assertThat(result.encodeToBase64()).isEqualTo("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=");
64      }
65  
66      @Test
67      public void forwardingKeys() {
68          Set<PublicKey> result = enclaveClient.getForwardingKeys();
69  
70          assertThat(result).isEmpty();
71      }
72  
73      @Test
74      public void getPublicKeys() {
75          Set<PublicKey> result = enclaveClient.getPublicKeys();
76  
77          assertThat(result).hasSize(1);
78          assertThat(result.iterator().next().encodeToBase64()).isEqualTo("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=");
79      }
80      
81      @Test
82      public void status() {
83          assertThat(enclaveClient.status())

86      }
