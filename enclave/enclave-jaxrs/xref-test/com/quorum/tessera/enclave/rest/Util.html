14      public static JerseyTest create(Enclave enclave) {
15  
16          SLF4JBridgeHandler.removeHandlersForRootLogger();
17          SLF4JBridgeHandler.install();
18  
19          return new JerseyTest() {
20              @Override
21              protected Application configure() {
22                  
23                  enable(TestProperties.LOG_TRAFFIC);
24                  enable(TestProperties.DUMP_ENTITY);
25                  set(TestProperties.CONTAINER_PORT, SocketUtils.findAvailableTcpPort());
26                  EnclaveApplication application = new EnclaveApplication(new EnclaveResource(enclave));
27  
28                  ResourceConfig config = ResourceConfig.forApplication(application);
29                  config.packages("com.quorum.tessera.enclave.rest");
30                  return config;
31              }
32  
33          };
34      }
