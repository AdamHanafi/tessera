001package com.quorum.tessera.enclave.rest;
002
003import com.quorum.tessera.config.CommunicationType;
004import com.quorum.tessera.config.Config;
005import com.quorum.tessera.config.ServerConfig;
006import com.quorum.tessera.config.cli.CliDelegate;
007import com.quorum.tessera.server.TesseraServer;
008import com.quorum.tessera.service.locator.ServiceLocator;
009import java.util.Set;
010import com.quorum.tessera.server.TesseraServerFactory;
011import java.util.Collections;
012import java.util.concurrent.CountDownLatch;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016public class Main {
017
018    private static final Logger LOGGER = LoggerFactory.getLogger(Main.class);
019    
020    public static void main(String[] args) throws Exception {
021
022        System.setProperty("javax.xml.bind.JAXBContextFactory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
023        System.setProperty("javax.xml.bind.context.factory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
024        CliDelegate.INSTANCE.execute(args);
025        ServiceLocator serviceLocator = ServiceLocator.create();
026 
027        Set<Object> services = serviceLocator.getServices("tessera-enclave-jaxrs-spring.xml");
028
029        TesseraServerFactory restServerFactory = TesseraServerFactory.create(CommunicationType.REST);
030
031        EnclaveResource enclaveResource = services.stream()
032                .filter(EnclaveResource.class::isInstance)
033                .map(EnclaveResource.class::cast)
034                .findAny().get();
035
036        final EnclaveApplication application = new EnclaveApplication(enclaveResource);
037
038        Config config = services.stream()
039                .filter(Config.class::isInstance)
040                .map(Config.class::cast)
041                .findAny().get();
042
043        final ServerConfig serverConfig = config.getServerConfigs().iterator().next();
044
045        TesseraServer server = restServerFactory.createServer(serverConfig, Collections.singleton(application));
046
047        server.start();
048        
049        CountDownLatch latch = new CountDownLatch(1);
050        
051        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
052            
053            try{
054                server.stop();
055            } catch (Exception ex) {
056                LOGGER.error(null, ex);
057            } finally {
058                
059            }
060        }));
061        
062        latch.await();
063
064
065    }
066
067}
