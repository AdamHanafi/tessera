001package com.quorum.tessera.enclave.rest;
002
003import com.quorum.tessera.config.CommunicationType;
004import com.quorum.tessera.config.Config;
005import com.quorum.tessera.config.ServerConfig;
006import com.quorum.tessera.config.cli.CliDelegate;
007import com.quorum.tessera.config.cli.CliResult;
008import com.quorum.tessera.server.TesseraServer;
009import com.quorum.tessera.service.locator.ServiceLocator;
010import java.util.Set;
011import com.quorum.tessera.server.TesseraServerFactory;
012import java.util.Collections;
013import java.util.concurrent.CountDownLatch;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017public class Main {
018
019    private static final Logger LOGGER = LoggerFactory.getLogger(Main.class);
020    
021    public static void main(String[] args) throws Exception {
022
023        System.setProperty("javax.xml.bind.JAXBContextFactory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
024        System.setProperty("javax.xml.bind.context.factory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
025        CliResult cliResult = CliDelegate.INSTANCE.execute(args);
026        if(!cliResult.getConfig().isPresent()) {
027            System.exit(cliResult.getStatus());
028        }
029        
030        ServiceLocator serviceLocator = ServiceLocator.create();
031 
032        Set<Object> services = serviceLocator.getServices("tessera-enclave-jaxrs-spring.xml");
033
034        TesseraServerFactory restServerFactory = TesseraServerFactory.create(CommunicationType.REST);
035
036        EnclaveResource enclaveResource = services.stream()
037                .filter(EnclaveResource.class::isInstance)
038                .map(EnclaveResource.class::cast)
039                .findAny().get();
040
041        final EnclaveApplication application = new EnclaveApplication(enclaveResource);
042
043        Config config = services.stream()
044                .filter(Config.class::isInstance)
045                .map(Config.class::cast)
046                .findAny().get();
047
048        final ServerConfig serverConfig = config.getServerConfigs().iterator().next();
049
050        TesseraServer server = restServerFactory.createServer(serverConfig, Collections.singleton(application));
051
052        server.start();
053        
054        CountDownLatch latch = new CountDownLatch(1);
055        
056        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
057            
058            try{
059                server.stop();
060            } catch (Exception ex) {
061                LOGGER.error(null, ex);
062            } finally {
063                
064            }
065        }));
066        
067        latch.await();
068
069
070    }
071
072}
