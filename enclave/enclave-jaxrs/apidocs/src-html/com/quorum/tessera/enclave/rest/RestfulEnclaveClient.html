031
032public class RestfulEnclaveClient implements EnclaveClient {
033
034    private static final Logger LOGGER = LoggerFactory.getLogger(RestfulEnclaveClient.class);
035
036    private final Client client;
037
038    private final URI uri;
039
040    private final ExecutorService executorService;
041
042    public RestfulEnclaveClient(Client client, URI uri) {
043        this(client, uri, Executors.newSingleThreadExecutor());
044    }
045
046    public RestfulEnclaveClient(Client client, URI uri, ExecutorService executorService) {
047        this.client = Objects.requireNonNull(client);
048        this.uri = Objects.requireNonNull(uri);
049        this.executorService = executorService;
050    }
051
052    @Override
053    public PublicKey defaultPublicKey() {
054
055        return ClientCallback.execute(() -> {
056            Response response = client.target(uri)
057                .path("default")
058                .request()
059                .get();
060
061            validateResponseIsOk(response);
062
063            byte[] data = response.readEntity(byte[].class);
064
065            return PublicKey.from(data);
066        });
067
068    }
069
070    @Override
071    public Set<PublicKey> getForwardingKeys() {
072        return ClientCallback.execute(() -> {
073            Response response = client.target(uri)
074                .path("forwarding")
075                .request()
076                .get();
077
078            validateResponseIsOk(response);
079
080            JsonArray results = response.readEntity(JsonArray.class);
081
082            return IntStream.range(0, results.size())
083                .mapToObj(i -> results.getString(i))
084                .map(s -> Base64.getDecoder().decode(s))
085                .map(PublicKey::from)
086                .collect(Collectors.toSet());
087        });
088    }
089
090    @Override
091    public Set<PublicKey> getPublicKeys() {
092        return ClientCallback.execute(() -> {
093            Response response = client.target(uri)
094                .path("public")
095                .request()
096                .get();
097
098            validateResponseIsOk(response);
099
100            JsonArray results = response.readEntity(JsonArray.class);
101
102            return IntStream.range(0, results.size())
103                .mapToObj(i -> results.getString(i))
104                .map(s -> Base64.getDecoder().decode(s))
105                .map(PublicKey::from)
106                .collect(Collectors.toSet());
107        });
108    }
109
110    @Override
111    public EncodedPayload encryptPayload(byte[] message, PublicKey senderPublicKey, List<PublicKey> recipientPublicKeys) {
112
113        return ClientCallback.execute(() -> {
114
115            EnclavePayload enclavePayload = new EnclavePayload();
116            enclavePayload.setData(message);
117            enclavePayload.setSenderKey(senderPublicKey.getKeyBytes());
118            enclavePayload.setRecipientPublicKeys(recipientPublicKeys.stream()
119                .map(PublicKey::getKeyBytes)
120                .collect(Collectors.toList()));
121
122            Response response = client.target(uri)
123                .path("encrypt")
124                .request()
125                .post(Entity.json(enclavePayload));
126
127            validateResponseIsOk(response);
128
129            byte[] result = response.readEntity(byte[].class);
130
131            return PayloadEncoder.create().decode(result);
132        });
133    }
134
135    @Override
136    public EncodedPayload encryptPayload(RawTransaction rawTransaction, List<PublicKey> recipientPublicKeys) {
137
138        return ClientCallback.execute(() -> {
139
140            EnclaveRawPayload enclaveRawPayload = new EnclaveRawPayload();
141            enclaveRawPayload.setNonce(rawTransaction.getNonce().getNonceBytes());
142            enclaveRawPayload.setFrom(rawTransaction.getFrom().getKeyBytes());
143            enclaveRawPayload.setRecipientPublicKeys(
144                recipientPublicKeys.stream()
145                    .map(PublicKey::getKeyBytes)
146                    .collect(Collectors.toList())
147            );
148            enclaveRawPayload.setEncryptedPayload(rawTransaction.getEncryptedPayload());
149            enclaveRawPayload.setEncryptedKey(rawTransaction.getEncryptedKey());
150
151            Response response = client.target(uri)
152                .path("encrypt")
153                .path("raw")
154                .request()
155                .post(Entity.json(enclaveRawPayload));
156
157            validateResponseIsOk(response);
158
159            byte[] body = response.readEntity(byte[].class);
160
161            return PayloadEncoder.create().decode(body);
162        });
163    }
164
165    @Override
166    public RawTransaction encryptRawPayload(byte[] message, PublicKey sender) {
167
168        return ClientCallback.execute(() -> {
169
170            EnclavePayload enclavePayload = new EnclavePayload();
171            enclavePayload.setData(message);
172            enclavePayload.setSenderKey(sender.getKeyBytes());
173
174            Response response = client.target(uri)
175                .path("encrypt")
176                .path("toraw")
177                .request()
178                .post(Entity.json(enclavePayload));
179
180            validateResponseIsOk(response);
181
182            EnclaveRawPayload enclaveRawPayload = response.readEntity(EnclaveRawPayload.class);
183
184            byte[] encryptedPayload = enclaveRawPayload.getEncryptedPayload();
185            byte[] encryptedKey = enclaveRawPayload.getEncryptedKey();
186            Nonce nonce = new Nonce(enclaveRawPayload.getNonce());
187            PublicKey senderKey = PublicKey.from(enclaveRawPayload.getFrom());
188            return new RawTransaction(encryptedPayload, encryptedKey, nonce, senderKey);
189        });
190    }
191
192    @Override
193    public byte[] unencryptTransaction(EncodedPayload payload, PublicKey providedKey) {
194
195        return ClientCallback.execute(() -> {
196
197            EnclaveUnencryptPayload dto = new EnclaveUnencryptPayload();
198
199            byte[] body = PayloadEncoder.create().encode(payload);
200
201            dto.setData(body);
202
203            if (providedKey != null) {
204                dto.setProvidedKey(providedKey.getKeyBytes());
205            }
206            Response response = client.target(uri)
207                .path("unencrypt")
208                .request()
209                .post(Entity.json(dto));
210
211            validateResponseIsOk(response);
212
213            return response.readEntity(byte[].class);
214        });
215    }
216
217    @Override
218    public byte[] createNewRecipientBox(final EncodedPayload payload, final PublicKey recipientKey) {
219
220        return ClientCallback.execute(() -> {
221
222            final byte[] body = PayloadEncoder.create().encode(payload);
223
224            final EnclaveUnencryptPayload dto = new EnclaveUnencryptPayload();
225            dto.setData(body);
226            dto.setProvidedKey(recipientKey.getKeyBytes());
227
228            final Response response = client.target(uri)
229                .path("addRecipient")
230                .request()
231                .post(Entity.json(dto));
232
233            validateResponseIsOk(response);
234
235            return response.readEntity(byte[].class);
236        });
237    }

239    /**
240     * In the case of a stateless client there is no start/stop all the run
241     * status logic is handled in the status command itself
242     *
243     * @return Status
244     */
245    @Override
246    public Service.Status status() {
247
248        Future<Service.Status> outcome = executorService.submit(() -> {
249            Response response = client
250                .target(uri)
251                .path("ping")
252                .request().get();
253
254            if (response.getStatus() == 200) {
255                return Service.Status.STARTED;
256            }
257            return Service.Status.STOPPED;
258        });
259
260        try {
261            //TODO: 2 seconds is arguably a long time
262            return outcome.get(2, TimeUnit.SECONDS);
263        } catch (InterruptedException | ExecutionException | TimeoutException ex) {
264            LOGGER.trace(null, ex);
265            return Service.Status.STOPPED;
266        } 
267    }
268
269    private static void validateResponseIsOk(Response response) {
270        if (response.getStatus() != 200) {
271            Response.StatusType statusInfo = response.getStatusInfo();
272            String message = String.format("Remote enclave instance threw an error %d  %s",
273                statusInfo.getStatusCode(), statusInfo.getReasonPhrase());
274
275            throw new EnclaveNotAvailableException(message);
276        }
277
278    }
