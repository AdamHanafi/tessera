001package com.quorum.tessera;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.cli.CliDelegate;
005import com.quorum.tessera.config.cli.CliException;
006import com.quorum.tessera.config.cli.CliResult;
007import com.quorum.tessera.server.TesseraServer;
008import com.quorum.tessera.server.TesseraServerFactory;
009import com.quorum.tessera.service.locator.ServiceLocator;
010import org.apache.commons.lang3.exception.ExceptionUtils;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import javax.json.JsonException;
015import javax.validation.ConstraintViolation;
016import javax.validation.ConstraintViolationException;
017import java.util.*;
018import java.util.stream.Collectors;
019
020/**
021 * The main entry point for the application. This just starts up the application
022 * in the embedded container.
023 */
024public class Launcher {
025
026    private static final Logger LOGGER = LoggerFactory.getLogger(Launcher.class);
027
028    public static void main(final String... args) throws Exception {
029
030        System.setProperty("javax.xml.bind.JAXBContextFactory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
031        System.setProperty("javax.xml.bind.context.factory", "org.eclipse.persistence.jaxb.JAXBContextFactory");
032
033        try {
034            final CliResult cliResult = CliDelegate.instance().execute(args);
035
036            if (cliResult.isSuppressStartup()) {
037                System.exit(0);
038            }
039
040            if (cliResult.getStatus() != 0) {
041                System.exit(cliResult.getStatus());
042            }
043
044            final Config config = cliResult.getConfig()
045                .orElseThrow(() -> new NoSuchElementException("No config found. Tessera will not run."));
046
047            runWebServer(config);
048
049        } catch (final ConstraintViolationException ex) {
050            for (final ConstraintViolation<?> violation : ex.getConstraintViolations()) {
051                System.out.println("Config validation issue: " + violation.getPropertyPath() + " " + violation.getMessage());
052            }
053            System.exit(1);
054        } catch (final ConfigException ex) {
055            final Throwable cause = ExceptionUtils.getRootCause(ex);
056
057            if (JsonException.class.isInstance(cause)) {
058                System.err.println("Invalid json, error is " + cause.getMessage());
059            } else {
060                System.err.println(Objects.toString(cause));
061            }
062            System.exit(3);
063        } catch (final CliException ex) {
064            System.err.println(ex.getMessage());
065            System.exit(4);
066        } catch (final Throwable ex) {
067            Optional.ofNullable(ex.getMessage()).ifPresent(System.err::println);
068            System.exit(2);
069        }
070
071    }
072
073    private static void runWebServer(final Config config) throws Exception {
074
075        ServiceLocator serviceLocator = ServiceLocator.create();
076
077        Set<Object> services = serviceLocator.getServices("tessera-spring.xml");
078
079        final List<TesseraServer> servers = config.getServerConfigs()
080            .stream()
081            .filter(server -> !AppType.ENCLAVE.equals(server.getApp()))
082            .map(conf -> TesseraServerFactory.create(conf.getCommunicationType()).createServer(conf, services))
083            .filter(Objects::nonNull)
084            .collect(Collectors.toList());
085
086        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
087            try {
088                for (TesseraServer ts : servers) {
089                    ts.stop();
090                }
091            } catch (Exception ex) {
092                LOGGER.error(null, ex);
093            }
094        }));
095
096        for (TesseraServer ts : servers) {
097            ts.start();
098        }
099
100    }
101
102}
