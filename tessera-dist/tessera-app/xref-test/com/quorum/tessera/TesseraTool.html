1   package com.quorum.tessera;
2   
3   import org.bouncycastle.util.encoders.Hex;
4   
5   import java.util.Base64;
6   
7   /**
8    *
9    * This tool takes in a transaction hash and produces the cURL call needed to
10   * retrieve that transaction via Quorum.
11   *
12   * It converts the Base64 hash to its Hex format, and then formats a cmdline
13   * call to be pasted.
14   *
15   */
16  public class TesseraTool {
17  
18      private static final String VERSION = "\"jsonrpc\":\"2.0\"";
19  
20      private static final String METHOD = "\"method\":\"eth_getQuorumPayload\"";
21  
22      private static final String ID = "\"id\":67";
23  
24      private static final String TEMPLATE = "curl -X POST %s --data '{%s, %s, \"params\":[\"0x%s\"], %s}'";
25  
26      public static void main(String[] args) {
27  
28          //put the transaction hash here
29          final String inputHash = "whA0hI7i3RC60lqlwJacz8qJ6pkDU9JqSCVRs8CTGr/pps9b8bBbk7OWUgq05tZ6PbGsjLTh9z5e+ccRjg+nLw==";
30          final String url = "http://127.0.0.1:22000";
31  
32          final byte[] decodedBase64 = Base64.getDecoder().decode(inputHash);
33          final String hexString = Hex.toHexString(decodedBase64);
34  
35          System.err.println(hexString);
36  
37          final String curlCall = String.format(TEMPLATE, url, VERSION, METHOD, hexString, ID);
38          System.err.println(curlCall);
39  
40      }
41  
42  }
