001package com.quorum.tessera.key.generation;
002
003import com.quorum.tessera.config.*;
004import com.quorum.tessera.config.keys.KeyEncryptorFactory;
005import com.quorum.tessera.config.util.EnvironmentVariableProvider;
006import com.quorum.tessera.config.util.PasswordReaderFactory;
007import com.quorum.tessera.key.vault.KeyVaultService;
008import com.quorum.tessera.key.vault.KeyVaultServiceFactory;
009import com.quorum.tessera.nacl.NaclFacadeFactory;
010
011public class DefaultKeyGeneratorFactory implements KeyGeneratorFactory {
012
013    @Override
014    public KeyGenerator create(KeyVaultConfig keyVaultConfig) {
015
016        if(keyVaultConfig != null) {
017            final KeyVaultServiceFactory keyVaultServiceFactory = KeyVaultServiceFactory.getInstance(keyVaultConfig.getKeyVaultType());
018
019            final Config config = new Config();
020            final KeyConfiguration keyConfiguration = new KeyConfiguration();
021
022            if(keyVaultConfig.getKeyVaultType().equals(KeyVaultType.AZURE)) {
023                keyConfiguration.setAzureKeyVaultConfig((AzureKeyVaultConfig) keyVaultConfig);
024
025                config.setKeys(keyConfiguration);
026
027                final KeyVaultService keyVaultService = keyVaultServiceFactory.create(config, new EnvironmentVariableProvider());
028
029                return new AzureVaultKeyGenerator(NaclFacadeFactory.newFactory().create(), keyVaultService);
030
031            } else {
032                keyConfiguration.setHashicorpKeyVaultConfig((HashicorpKeyVaultConfig) keyVaultConfig);
033
034                config.setKeys(keyConfiguration);
035
036                final KeyVaultService keyVaultService = keyVaultServiceFactory.create(config, new EnvironmentVariableProvider());
037
038                return new HashicorpVaultKeyGenerator(NaclFacadeFactory.newFactory().create(), keyVaultService);
039            }
040        }
041
042        return new FileKeyGenerator(
043            NaclFacadeFactory.newFactory().create(), KeyEncryptorFactory.create(), PasswordReaderFactory.create()
044        );
045    }
046}
