31  public class RawSteps implements En {
32  
33      private RestUtils restUtils = new RestUtils();
34  
35      private PartyHelper partyHelper = PartyHelper.create();
36  
37      private Party getSender(Collection<String> senderHolder) {
38          return partyHelper.findByAlias(senderHolder.stream().findAny().get());
39      }
40  
41      private Set<Party> getRecipientParties(Set<String> recipientAliases) {
42          return recipientAliases.stream().map(partyHelper::findByAlias).collect(Collectors.toSet());
43      }
44  
45      public RawSteps() {
46  
47          final Collection<String> senderHolder = new ArrayList<>();
48  
49          final Set<String> recipients = new HashSet<>();
50  
51          final byte[] transactionData = restUtils.createTransactionData();
52  
53          final Set<String> storedHashes = new TreeSet<>();
54  
55          Given("^Sender party (.+)$", (String pty) -> {
56              senderHolder.add(pty);
57          });
58  
59          And("^Recipient part(?:y|ies) (.+)$", (String alias) -> {
60              parseAliases(alias).stream()
61                      .forEach(recipients::add);
62  
63              assertThat(recipients).isNotEmpty();
64          });
65  
66          And("^all parties are running$", () -> {
67  
68              Client client = ClientBuilder.newClient();
69  
70              assertThat(partyHelper.getParties()
71                      .map(Party::getP2PUri)
72                      .map(client::target)
73                      .map(t -> t.path("upcheck"))
74                      .map(WebTarget::request)
75                      .map(Invocation.Builder::get)
76                      .allMatch(r -> r.getStatus() == 200)).describedAs("All serers are up ")
77                      .isTrue();
78          });
79  
80          When("^sender party receives transaction from Quorum peer$", () -> {
81  
82              Party sender = getSender(senderHolder);
83  
84              Response response = restUtils.sendRaw(sender,
85                      transactionData,
86                      getRecipientParties(recipients).toArray(new Party[0]));
87  
88              assertThat(response.getStatus()).isEqualTo(200);
89  
90              String persistedKey = response.readEntity(String.class);
91              assertThat(persistedKey).isNotNull();
92  
93              storedHashes.add(persistedKey);
94  
95          });
96  
97          When("sender party receives transaction with no sender key defined from Quorum peer", () -> {
98              Party sender = getSender(senderHolder);
99  
100             final Response response = sender.getRestClientWebTarget()
101                     .path("sendraw")
102                     .request()
103                     .header(RECIPIENTS, recipients.stream()
104                             .map(partyHelper::findByAlias)
105                             .map(Party::getPublicKey)
106                             .collect(Collectors.joining(","))
107                     )
108                     .post(Entity.entity(transactionData, MediaType.APPLICATION_OCTET_STREAM));
109 
110             assertThat(response).isNotNull();
111             assertThat(response.getStatus()).isEqualTo(200);
112 
113             String persistedKey = response.readEntity(String.class);
114             assertThat(persistedKey).isNotNull();
115             storedHashes.add(persistedKey);
116 
117             URI location = response.getLocation();
118 
119             recipients.stream()
120                     .map(partyHelper::findByAlias)
121                     .map(Party::getRestClient)
122                     .forEach(client -> {
123                         
124                         final Response checkPersistedTxnResponse = client.target(location)
125                                 .request()
126                                 .get();
127 
128                         assertThat(checkPersistedTxnResponse.getStatus()).isEqualTo(200);
129 
130                         ReceiveResponse receiveResponse = checkPersistedTxnResponse.readEntity(ReceiveResponse.class);
131 
132                         assertThat(receiveResponse.getPayload()).isEqualTo(transactionData);
133 
134                         final Set<Party> recipientParties = getRecipientParties(recipients);
135                         
136                         
137                         List<Response> responses = restUtils.findTransaction(persistedKey,recipientParties)
138                                 .collect(Collectors.toList());
139 
140                         responses
141                                 .forEach(r -> {
142                                     
143                             assertThat(r.getStatus())
144                                     .describedAs("find transaction for "+ recipients +". "+ r)
145                                     .isEqualTo(200);
146                             
147                         });
148                     });
149 
150             restUtils.findTransaction(persistedKey, partyHelper.findByAlias("C"), partyHelper.findByAlias("B")).forEach(r -> {
151                 assertThat(r.getStatus()).isEqualTo(404);
152             });
153         });
154 
155         When("sender party receives transaction with no payload from Quorum peer", () -> {
156             Party sender = getSender(senderHolder);
157 
158             Response response = restUtils.sendRaw(sender, null, getRecipientParties(recipients));
159 
160             assertThat(response).isNotNull();
161             assertThat(response.getStatus()).isEqualTo(400);
162         });
163 
164         When("sender party receives transaction with an unknown party from Quorum peer", () -> {
165             Party sender = getSender(senderHolder);
166 
167             final Response response = sender.getRestClientWebTarget()
168                     .path("sendraw")
169                     .request()
170                     .header(SENDER, sender.getPublicKey())
171                     .header(RECIPIENTS, "8SjRHlUBe4hAmTk3KDeJ96RhN+s10xRrHDrxEi1O5W0=")
172                     .post(Entity.entity(transactionData, MediaType.APPLICATION_OCTET_STREAM));
173 
174             assertThat(response).isNotNull();
175             assertThat(response.getStatus()).isEqualTo(400);
176         });
177 
178         Then("an invalid request error is raised", () -> {
179             //FIXME: validated in sending function
180         });
181 
182         Then("^sender party stores the transaction$", () -> {
183             Party sender = getSender(senderHolder);
184             try (PreparedStatement statement
185                     = sender.getDatabaseConnection()
186                             .prepareStatement("SELECT COUNT(*) FROM ENCRYPTED_TRANSACTION WHERE HASH = ?")){
187                         statement.setBytes(1, Base64.getDecoder().decode(storedHashes.iterator().next()));
188                         try (ResultSet results = statement.executeQuery()){
189                             assertThat(results.next()).isTrue();
190                             assertThat(results.getLong(1)).isEqualTo(1);
191                         }
192 
193                     }
194         });
195 
196         Then("^forwards the transaction to recipient part(?:y|ies)$", () -> {
197 
198             recipients.stream().map(partyHelper::findByAlias).forEach(rec -> {
199                 String storedHash = storedHashes.stream().findAny().get();
200                 Response response = restUtils.receiveRaw(storedHash, rec);
201 
202                 assertThat(response.getStatus()).isEqualTo(200);
203                 final byte[] result = response.readEntity(byte[].class);
204                 assertThat(result).isEqualTo(transactionData);
205             });
206 
207         });
208 
209         Then("^.*does not forward transaction to any recipients?$", () -> {
210             partyHelper.getParties()
211                     .filter(p -> !senderHolder.contains(p.getAlias()))
212                     .forEach(p -> {
213                         String storedHash = storedHashes.stream().findAny().get();
214                         Response response = restUtils.receiveRaw(storedHash, p);
215                         assertThat(response.getStatus()).isEqualTo(404);
216                     });
217 
218         });
219     }
220 
221     static List<String> parseAliases(String alias) {
222         return Arrays.asList(alias.split(",| and "));
223     }
