36      private static final Logger LOGGER = LoggerFactory.getLogger(WhitelistSteps.class);
37  
38      private static String jarPath = System.getProperty("application.jar", "../../tessera-app/target/tessera-app-0.9-SNAPSHOT-app.jar");
39  
40      private final URL logbackConfigFile = WhitelistSteps.class.getResource("/logback-node.xml");
41  
42      public WhitelistSteps() {
43  
44          try {
45              final Path pid = Files.createTempFile("whitelist", ".pid");
46  
47              ExecutorService executorService = Executors.newCachedThreadPool();
48  
49              Given("Node at port {int}", (Integer port) -> {
50  
51                  ExecutionContext executionContext = ExecutionContext.Builder.create()
52                      .with(CommunicationType.REST)
53                      .with(DBType.H2)
54                      .with(EnclaveType.LOCAL)
55                      .with(SocketType.HTTP)
56                      .build();
57  
58                  ConfigBuilder whiteListConfigBuilder = new ConfigBuilder()
59                      .withNodeId("whitelist")
60                      .withNodeNumbber(5)
61                      .withQ2TSocketType(SocketType.HTTP)
62                      .withExecutionContext(executionContext)
63                      .withP2pPort(port)
64                      .withPeer("http://localhost:7000")
65                      .withKeys("WxsJ4souK0mptNx1UGw6hb1WNNIbPhLPvW9GoaXau3Q=", "YbOOFA4mwSSdGH6aFfGl2M7N1aiPOj5nHpD7GzJKSiA=");
66  
67                  Config whiteListConfig = whiteListConfigBuilder.build();
68                  whiteListConfig.setUseWhiteList(true);
69  
70                  Path configFile = Paths.get("target").resolve("white-list-config.json");
71  
72                  try (OutputStream out = Files.newOutputStream(configFile)) {
73                      JaxbUtil.marshalWithNoValidation(whiteListConfig, out);
74                  } catch (IOException ex) {
75                      throw new UncheckedIOException(ex);
76                  }
77  
78                  List<String> cmd = new ExecArgsBuilder()
79                      .withMainClass(Launcher.class)
80                      .withClassPathItem(Paths.get(jarPath))
81                      .withConfigFile(configFile)
82                      .withJvmArg("-Dlogback.configurationFile=" + logbackConfigFile)
83                      .withJvmArg("-Dnode.number=whitelist")
84                      .withPidFile(pid)
85                      .build();
86  
87                  ProcessBuilder processBuilder = new ProcessBuilder(cmd);
88                  processBuilder.redirectErrorStream(true);
89  
90                  Process process = processBuilder.start();
91  
92                  executorService.submit(() -> {
93                      try (BufferedReader reader = Stream.of(process.getInputStream())
94                          .map(InputStreamReader::new)
95                          .map(BufferedReader::new)
96                          .findAny().get()) {
97  
98                          String line;
99                          while ((line = reader.readLine()) != null) {
100                             LOGGER.info("Exec line Whitelist : {}", line);
101                         }
102 
103                     } catch (IOException ex) {
104                         throw new UncheckedIOException(ex);
105                     }
106                 });
107 
108                 executorService.submit(() -> {
109                     try {
110                         process.waitFor();
111                     } catch (InterruptedException ex) {
112 
113                     }
114                 });
115 
116                 ServerStatusCheck serverStatusCheck = ServerStatusCheck
117                     .create(whiteListConfig.getServerConfigs().stream()
118                         .filter(s -> s.getApp() == AppType.P2P)
119                         .findAny()
120                         .get());
121 
122                 Boolean started = executorService.submit(new ServerStatusCheckExecutor(serverStatusCheck))
123                     .get(20, TimeUnit.SECONDS);
124 
125                 assertThat(started).isTrue();
126 
127             });
128 
129             List<Response> responseHolder = new ArrayList<>();
130             When("a request is made against the node", () -> {
131 
132                 Client client = ClientBuilder.newClient();
133                 Response response = client
134                     .target("http://localhost:7000")
135                     .path("upcheck")
136                     .request()
137                     .get();
138 
139                 responseHolder.add(response);
140 
141             });
142 
143             Then("the response code is UNAUTHORIZED", () -> {
144                 assertThat(responseHolder.get(0).getStatus()).isEqualTo(401);
145             });
146 
147             Then("the node is stopped", () -> {
148                 Files.lines(pid).findAny()
149                     .ifPresent(p -> {
150                         try {
151                             ExecUtils.kill(p);
152                         } catch (IOException | InterruptedException ex) {
153                         }
154                     });
155             });
156 
157         } catch (Exception ex) {
158             throw new RuntimeException(ex);
159         }
160     }
