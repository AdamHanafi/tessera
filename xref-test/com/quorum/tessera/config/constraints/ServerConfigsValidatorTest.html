1   package com.quorum.tessera.config.constraints;
2   
3   import com.quorum.tessera.config.AppType;
4   import com.quorum.tessera.config.CommunicationType;
5   import com.quorum.tessera.config.ServerConfig;
6   import org.junit.After;
7   import org.junit.Before;
8   import org.junit.Test;
9   
10  import javax.validation.ConstraintValidatorContext;
11  import java.util.ArrayList;
12  import java.util.Arrays;
13  import java.util.List;
14  
15  import static org.assertj.core.api.Assertions.assertThat;
16  import static org.mockito.ArgumentMatchers.anyString;
17  import static org.mockito.Mockito.*;
18  
19  public class ServerConfigsValidatorTest {
20  
21      private List<ServerConfig> serverConfigs;
22  
23      private ServerConfig p2pServerConfig;
24      private ServerConfig q2tServerConfig;
25      private ServerConfig thirdPartyServerConfig;
26  
27      private ConstraintValidatorContext cvc;
28  
29      private ServerConfigsValidator validator;
30  
31      @Before
32      public void onSetUp() {
33          cvc = mock(ConstraintValidatorContext.class);
34          p2pServerConfig = new ServerConfig();
35          p2pServerConfig.setApp(AppType.P2P);
36          p2pServerConfig.setEnabled(true);
37          p2pServerConfig.setServerAddress("localhost:123");
38          p2pServerConfig.setCommunicationType(CommunicationType.REST);
39          p2pServerConfig.setSslConfig(null);
40          p2pServerConfig.setInfluxConfig(null);
41          p2pServerConfig.setBindingAddress(null);
42  
43          q2tServerConfig = new ServerConfig(AppType.Q2T, true,
44              "localhost:1234", CommunicationType.REST,
45              null, null, null);
46          thirdPartyServerConfig = new ServerConfig(AppType.THIRD_PARTY, true,
47              "localhost:12345", CommunicationType.REST,
48              null, null, null);
49  
50          serverConfigs = new ArrayList<>(Arrays.asList(p2pServerConfig, q2tServerConfig, thirdPartyServerConfig));
51  
52          validator = new ServerConfigsValidator();
53  
54          when(cvc.buildConstraintViolationWithTemplate(anyString()))
55              .thenReturn(mock(ConstraintValidatorContext.ConstraintViolationBuilder.class));
56      }
57  
58      @After
59      public void onTearDown() {
60          verifyNoMoreInteractions(cvc);
61      }
62  
63      @Test
64      public void isValidWhenServerConfigsIsNull() {
65          assertThat(validator.isValid(null, cvc)).isTrue();
66      }
67  
68      @Test
69      public void isValidWhenValidDataIsSupplied() {
70          thirdPartyServerConfig.setEnabled(false);
71          assertThat(validator.isValid(serverConfigs, cvc)).isTrue();
72      }
73  
74      @Test
75      public void isNotValidWhenNoP2PServersAreEnabled() {
76          p2pServerConfig.setEnabled(false);
77          assertThat(validator.isValid(serverConfigs, cvc)).isFalse();
78          verify(cvc).disableDefaultConstraintViolation();
79          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
80      }
81  
82      @Test
83      public void isNotValidWhenNoP2PServersAreDefined() {
84          serverConfigs.remove(p2pServerConfig);
85          assertThat(validator.isValid(serverConfigs, cvc)).isFalse();
86          verify(cvc).disableDefaultConstraintViolation();
87          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
88      }
89  
90      @Test
91      public void isNotValidWhenTwoOrMoreP2PServersAreDefinedAndEnabled() {
92          serverConfigs.add(p2pServerConfig);
93          assertThat(validator.isValid(serverConfigs, cvc)).isFalse();
94          verify(cvc).disableDefaultConstraintViolation();
95          verify(cvc).buildConstraintViolationWithTemplate(eq("Only one P2P server must be configured and enabled."));
96      }
97  
98      @Test
99      public void isNotValidWhenNoQ2TServersAreEnabled() {
100         q2tServerConfig.setEnabled(false);
101         assertThat(validator.isValid(serverConfigs, cvc)).isFalse();
102         verify(cvc).disableDefaultConstraintViolation();
103         verify(cvc).buildConstraintViolationWithTemplate(eq("At least one Q2T server must be configured and enabled."));
104     }
105 
106     @Test
107     public void isNotValidWhenNoQ2TServersAreDefined() {
108         serverConfigs.remove(q2tServerConfig);
109         assertThat(validator.isValid(serverConfigs, cvc)).isFalse();
110         verify(cvc).disableDefaultConstraintViolation();
111         verify(cvc).buildConstraintViolationWithTemplate(eq("At least one Q2T server must be configured and enabled."));
112     }
113 }
