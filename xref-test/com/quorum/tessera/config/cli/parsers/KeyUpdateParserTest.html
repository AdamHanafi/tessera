282     @Test
283     public void keyGetsUpdated() throws IOException, ParseException {
284         final KeyDataConfig startingKey = JaxbUtil.unmarshal(
285             getClass().getResourceAsStream("/lockedprivatekey.json"), KeyDataConfig.class
286         );

288         final Path key = Files.createTempFile("key", ".key");

299 
300         final KeyDataConfig endingKey = JaxbUtil.unmarshal(Files.newInputStream(key), KeyDataConfig.class);

303         assertThat(endingKey.getSnonce()).isNotEqualTo(startingKey.getSnonce());
304         assertThat(endingKey.getAsalt()).isNotEqualTo(startingKey.getAsalt());
