29      @Test
30      public void createRemote() {
31          final Config config = new Config();
32  
33          ServerConfig serverConfig = new ServerConfig();
34          serverConfig.setEnabled(true);
35          serverConfig.setApp(AppType.ENCLAVE);
36          serverConfig.setCommunicationType(CommunicationType.REST);
37          serverConfig.setServerAddress("http://bogus:9898");
38  
39          config.setServerConfigs(singletonList(serverConfig));
40  
41          Enclave result = enclaveFactory.create(config);
42  
43          assertThat(result).isInstanceOf(EnclaveClient.class);
44      }

46      @Test
47      public void dontCreateRemoteWhenNoEnclaveServer() {
48  
49          Stream.of(AppType.values()).filter(t -> t != AppType.ENCLAVE).forEach(t -> {
50  
51              final Config config = new Config();
52  
53              ServerConfig serverConfig = new ServerConfig();
54              serverConfig.setApp(t);
55              serverConfig.setCommunicationType(CommunicationType.REST);
56              serverConfig.setServerAddress("http://bogus:9898");
57  
58              config.setServerConfigs(singletonList(serverConfig));
59  
60              KeyConfiguration keyConfiguration = new KeyConfiguration();
61              ConfigKeyPair pair = new DirectKeyPair("/+UuD63zItL1EbjxkKUljMgG8Z1w0AJ8pNOR4iq2yQc=", "yAWAJjwPqUtNVlqGjSrBmr1/iIkghuOh1803Yzx9jLM=");
62              keyConfiguration.setKeyData(singletonList(pair));
63              config.setKeys(keyConfiguration);
64  
65              config.setAlwaysSendTo(new ArrayList<>());
66  
67              Enclave result = enclaveFactory.create(config);
68  
69              assertThat(result).isInstanceOf(EnclaveImpl.class);
