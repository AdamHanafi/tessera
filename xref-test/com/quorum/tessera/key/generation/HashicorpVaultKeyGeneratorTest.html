20  public class HashicorpVaultKeyGeneratorTest {
21  
22      private final String pubStr = "public";
23      private final String privStr = "private";
24      private final PublicKey pub = PublicKey.from(pubStr.getBytes());
25      private final PrivateKey priv = PrivateKey.from(privStr.getBytes());
26  
27      private NaclFacade naclFacade;
28      private KeyVaultService keyVaultService;
29      private HashicorpVaultKeyGenerator hashicorpVaultKeyGenerator;
30  
31      @Before
32      public void setUp() {
33          this.naclFacade = mock(NaclFacade.class);
34          this.keyVaultService = mock(KeyVaultService.class);
35  
36          final KeyPair keyPair = new KeyPair(pub, priv);
37          when(naclFacade.generateNewKeys()).thenReturn(keyPair);
38  
39          this.hashicorpVaultKeyGenerator = new HashicorpVaultKeyGenerator(naclFacade, keyVaultService);
40  
41      }
42  
43      @Test(expected = NullPointerException.class)
44      public void nullFilenameThrowsException() {
45          KeyVaultOptions keyVaultOptions = mock(KeyVaultOptions.class);
46          when(keyVaultOptions.getSecretEngineName()).thenReturn("secretEngine");
47  
48          hashicorpVaultKeyGenerator.generate(null, null, keyVaultOptions);
49      }
50  
51      @Test(expected = NullPointerException.class)
52      public void nullKeyVaultOptionsThrowsException() {
53          hashicorpVaultKeyGenerator.generate("filename", null, null);
54      }
55  
56      @Test(expected = NullPointerException.class)
57      public void nullSecretEngineNameThrowsException() {
58          KeyVaultOptions keyVaultOptions = mock(KeyVaultOptions.class);
59          when(keyVaultOptions.getSecretEngineName()).thenReturn(null);
60  
61          hashicorpVaultKeyGenerator.generate("filename", null, keyVaultOptions);
62      }
63  
64      @Test
65      public void generatedKeyPairIsSavedToSpecifiedPathInVaultWithIds() {
66          String secretEngine = "secretEngine";
67          String filename = "secretName";
68  
69          KeyVaultOptions keyVaultOptions = mock(KeyVaultOptions.class);
70          when(keyVaultOptions.getSecretEngineName()).thenReturn(secretEngine);
71  
72          HashicorpVaultKeyPair result = hashicorpVaultKeyGenerator.generate(filename, null, keyVaultOptions);
73  
74          HashicorpVaultKeyPair expected = new HashicorpVaultKeyPair("publicKey", "privateKey", secretEngine, filename, null);
75          assertThat(result).isEqualToComparingFieldByField(expected);
76  
77          final ArgumentCaptor<HashicorpSetSecretData> captor = ArgumentCaptor.forClass(HashicorpSetSecretData.class);
78          verify(keyVaultService).setSecret(captor.capture());
79  
80          assertThat(captor.getAllValues()).hasSize(1);
81          HashicorpSetSecretData capturedArg = captor.getValue();
82  
83          Map<String, Object> expectedNameValuePairs = new HashMap<>();
84          expectedNameValuePairs.put("publicKey", pub.encodeToBase64());
85          expectedNameValuePairs.put("privateKey", priv.encodeToBase64());
86  
87          HashicorpSetSecretData expectedData = new HashicorpSetSecretData(secretEngine, filename, expectedNameValuePairs);
88  
89          assertThat(capturedArg).isEqualToComparingFieldByFieldRecursively(expectedData);
90  
91          verifyNoMoreInteractions(keyVaultService);
92  
93      }
