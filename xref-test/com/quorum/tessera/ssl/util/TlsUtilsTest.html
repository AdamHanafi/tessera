7   import java.io.IOException;

12  import java.security.*;
13  import java.security.cert.Certificate;

15  import java.security.cert.X509Certificate;

17  import static org.assertj.core.api.Java6Assertions.assertThat;
18  
19  public class TlsUtilsTest {
20  
21      private static final String FILE = "test-keystore";
22      private static final String PASSWORD ="quorum";
23      private static final String ALIAS = "tessera";
24  
25      Path privateKeyFile = Paths.get(FILE);
26  
27      @After
28      public void tearDown() throws IOException {
29          Files.deleteIfExists(privateKeyFile);
30          assertThat(Files.exists(privateKeyFile)).isFalse();
31      }
32  
33      @Test
34      public void testGenerateKeys() throws OperatorCreationException, InvalidKeyException, NoSuchAlgorithmException, IOException, SignatureException, NoSuchProviderException, CertificateException, KeyStoreException {
35  
36          assertThat(Files.exists(privateKeyFile)).isFalse();
37  
38          TlsUtils.create().generateKeyStoreWithSelfSignedCertificate("https://localhost:8080", privateKeyFile,PASSWORD);
39  
40          assertThat(Files.exists(privateKeyFile)).isTrue();
41  
42          //Read keystore from created file
43          final KeyStore keyStore = KeyStore.getInstance("JKS");
44  
45          try (InputStream in = Files.newInputStream(privateKeyFile)) {
46              keyStore.load(in, PASSWORD.toCharArray());
47          }
48  
49          assertThat(keyStore.containsAlias(ALIAS)).isTrue();
50  
51          Certificate certificate = keyStore.getCertificate(ALIAS);
52  
53          assertThat(certificate).isNotNull();
54          assertThat(certificate).isInstanceOf(X509Certificate.class);
55          assertThat(((X509Certificate) certificate))
56              .extracting("info").isNotNull()
57              .extracting("issuer").isNotNull()
58              .extracting("names").size().isEqualTo(1);
59  
60      }
61  }
