1   package db;
2   
3   import java.io.IOException;
4   import org.hsqldb.persist.HsqlProperties;
5   import org.hsqldb.server.Server;
6   import org.hsqldb.server.ServerAcl;
7   
8   public class HsqlDatabaseServer implements DatabaseServer {
9   
10      private final Server hsqlServer = new Server();
11  
12      private final String nodeId;
13  
14      public HsqlDatabaseServer(String nodeId) {
15          this.nodeId = nodeId;
16      }
17  
18      @Override
19      public void start() {
20  
21          HsqlProperties properties = new HsqlProperties();
22          for (int i = 0; i < 4; i++) {
23              String db = nodeId + (i + 1);
24              properties.setProperty("server.database." + i, "file:target/hsql/" + db);
25              properties.setProperty("server.dbname." + i, db);
26          }
27          
28          properties.setProperty("server.database.4", "file:target/hsql/rest-httpwhitelist5");
29          properties.setProperty("server.dbname.4", "rest-httpwhitelist5");
30  
31          hsqlServer.setPort(9189);
32          hsqlServer.setSilent(true);
33          hsqlServer.setTrace(false);
34          try{
35              hsqlServer.setProperties(properties);
36          } catch (IOException | ServerAcl.AclFormatException ex) {
37              throw new RuntimeException(ex);
38          }
39          hsqlServer.start();
40          if(hsqlServer.isNotRunning()) {
41              throw new IllegalStateException("HSQL DB not started. ");
42          }
43  
44      }
45  
46      @Override
47      public void stop() {
48          hsqlServer.shutdown();
49      }
50  
51  }
